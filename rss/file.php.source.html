 173  /**
 174   * Sends an error formatted as an rss file and then exits
 175   *
 176   * @package core_rss
 177   * @category rss
 178   *
 179   * @param string $error the error type, default is rsserror
 180   * @param string $filename the name of the file to create (NOT USED)

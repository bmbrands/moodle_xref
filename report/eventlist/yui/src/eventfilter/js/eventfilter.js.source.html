 100      _createTable: function(tableData) {
 101  
 102          var table = new Y.DataTable({
 103              columns: [
 104                  {
 105                      key: "fulleventname",
 106                      label: M.util.get_string('eventname', 'report_eventlist'),
 107                      allowHTML: true,
 108                      sortable: true,
 109                      /**
 110                       * Custom sort of the fulleventname column.
 111                       * This will sort via the event name rather than the event path.
 112                       *
 113                       * @param {object} eventDataListA Event data record module A.
 114                       * @param {object} eventDataListB Event data record module B.
 115                       * @param {boolean} desc True sorts list in descending order and false sorts in Ascending order.
 116                       * @return {number} order for which the column should be sorted.
 117                       * @method sortFn
 118                       */
 119                      sortFn: function(eventDataListA, eventDataListB, desc) {
 120                          var rawEventDataA = eventDataListA.getAttrs().raweventname,
 121                              rawEventDataB = eventDataListB.getAttrs().raweventname,
 122                              order = (rawEventDataA > rawEventDataB ? 1 : -1);
 123                          return desc ? -order : order;
 124                      },
 125                      title: M.util.get_string('eventname', 'report_eventlist')
 126                  }, {
 127                      key: "component",
 128                      label: M.util.get_string('component', 'report_eventlist'),
 129                      allowHTML: true,
 130                      sortable: true,
 131                      title: M.util.get_string('component', 'report_eventlist')
 132                  }, {
 133                      key:  "edulevel",

 173      /**
 174       * Filters the entries being displayed in the table.
 175       *
 176       * @method totalFilter
 177       * @private
 178       */
 179      _totalFilter: function() {
 180          // Get all of the details of the filter elements
 181          var eventNameFilter = this._eventName.get('value').toLowerCase(),
 182              // Component selected text.
 183              componentFilter = this._component.get('options').item(this._component.get('selectedIndex')).get('text').toLowerCase(),
 184              // Component selected value.
 185              componentValue = this._component.get('value'),
 186              // Education level selected text.
 187              eduLevelFilter = this._eduLevel.get('options').item(this._eduLevel.get('selectedIndex')).get('text').toLowerCase(),
 188              // Education level selected value.
 189              eduLevelValue = this._eduLevel.get('value'),
 190              // CRUD selected text.
 191              crudFilter = this._crud.get('options').item(this._crud.get('selectedIndex')).get('text').toLowerCase(),
 192              // CRUD selected value.
 193              crudValue = this._crud.get('value'),
 194              i,
 195              filtered = [];
 196  
 197          // Loop through the rows and put the ones we want into the filter.
 198          for (i = 0; i < this.get('tabledata').length; i++) {
 199              // These variables will either be false or true depending on the statement outcome.
 200              var fullEventText = Y.Node.create(this.get('tabledata')[i].fulleventname).get('text'),
 201                  eventNameValue = fullEventText.toLowerCase().indexOf(eventNameFilter) >= 0,
 202                  componentFilterValue = this.get('tabledata')[i].component.toLowerCase().indexOf(componentFilter) >= 0,
 203                  eduLevelFilterValue = this.get('tabledata')[i].edulevel.toLowerCase().indexOf(eduLevelFilter) >= 0,
 204                  crudFilterValue = this.get('tabledata')[i].crud.toLowerCase().indexOf(crudFilter) >= 0;
 205              // If the name field is empty then add to the filter.
 206              if (eventNameFilter === '') {
 207                  eventNameValue = true;
 208              }
 209              // If the component is set to 'all' then add to the filter.
 210              if (componentValue === '0') {
 211                  componentFilterValue = true;
 212              }
 213              // If the education level is set to 'all' then add to the filter.
 214              if (eduLevelValue === '0') {
 215                  eduLevelFilterValue = true;
 216              }
 217              // If the CRUD field is set to 'all' then add to the filter.
 218              if (crudValue === '0') {
 219                  crudFilterValue = true;
 220              }
 221              // If any of the Values here is false then don't add to the filter (all must be true).
 222              if (eventNameValue && componentFilterValue && eduLevelFilterValue && crudFilterValue) {
 223                  filtered.push(this.get('tabledata')[i]);
 224              }
 225          }
 226          // Display the table again with the new data.
 227          this._table.set('data', filtered);
 228      },

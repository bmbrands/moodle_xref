  47      /**
  48       * Render log report page.
  49       *
  50       * @param report_log_renderable $reportlog object of report_log.
  51       */
  52      protected function render_report_log(report_log_renderable $reportlog) {
  53          if (empty($reportlog->selectedlogreader)) {
  54              echo $this->output->notification(get_string('nologreaderenabled', 'report_log'), 'notifyproblem');
  55              return;
  56          }
  57          if ($reportlog->showselectorform) {
  58              $this->report_selector_form($reportlog);
  59          }
  60  
  61          if ($reportlog->showreport) {
  62              $reportlog->tablelog->out($reportlog->perpage, true);
  63          }
  64      }
  65  
  66      /**
  67       * Prints/return reader selector
  68       *
  69       * @param report_log_renderable $reportlog log report.
  70       */
  71      public function reader_selector(report_log_renderable $reportlog) {
  72          $readers = $reportlog->get_readers(true);
  73          if (empty($readers)) {
  74              $readers = array(get_string('nologreaderenabled', 'report_log'));
  75          }
  76          $url = fullclone ($reportlog->url);
  77          $url->remove_params(array('logreader'));
  78          $select = new single_select($url, 'logreader', $readers, $reportlog->selectedlogreader, null);
  79          $select->set_label(get_string('selectlogreader', 'report_log'));
  80          echo $this->output->render($select);
  81      }

  83      /**
  84       * This function is used to generate and display selector form
  85       *
  86       * @param report_log_renderable $reportlog log report.
  87       */
  88      public function report_selector_form(report_log_renderable $reportlog) {
  89          echo html_writer::start_tag('form', array('class' => 'logselecform', 'action' => $reportlog->url, 'method' => 'get'));
  90          echo html_writer::start_div();
  91          echo html_writer::empty_tag('input', array('type' => 'hidden', 'name' => 'chooselog', 'value' => '1'));
  92          echo html_writer::empty_tag('input', array('type' => 'hidden', 'name' => 'showusers', 'value' => $reportlog->showusers));
  93          echo html_writer::empty_tag('input', array('type' => 'hidden', 'name' => 'showcourses',
  94              'value' => $reportlog->showcourses));
  95  
  96          $selectedcourseid = empty($reportlog->course) ? 0 : $reportlog->course->id;
  97  
  98          // Add course selector.
  99          $sitecontext = context_system::instance();
 100          $courses = $reportlog->get_course_list();
 101          if (!empty($courses) && $reportlog->showcourses) {
 102              echo html_writer::label(get_string('selectacourse'), 'menuid', false, array('class' => 'accesshide'));
 103              echo html_writer::select($courses, "id", $selectedcourseid, null);
 104          } else {
 105              $courses = array();
 106              $courses[$selectedcourseid] = get_course_display_name_for_list($reportlog->course) . (($selectedcourseid == SITEID) ?
 107                  ' (' . get_string('site') . ') ' : '');
 108              echo html_writer::label(get_string('selectacourse'), 'menuid', false, array('class' => 'accesshide'));
 109              echo html_writer::select($courses, "id", $selectedcourseid, false);
 110              // Check if user is admin and this came because of limitation on number of courses to show in dropdown.
 111              if (has_capability('report/log:view', $sitecontext)) {
 112                  $a = new stdClass();
 113                  $a->url = new moodle_url('/report/log/index.php', array('chooselog' => 0,
 114                      'group' => $reportlog->get_selected_group(), 'user' => $reportlog->userid,
 115                      'id' => $selectedcourseid, 'date' => $reportlog->date, 'modid' => $reportlog->modid,
 116                      'showcourses' => 1, 'showusers' => $reportlog->showusers));
 117                  $a->url = $a->url->out(false);
 118                  print_string('logtoomanycourses', 'moodle', $a);
 119              }
 120          }
 121  
 122          // Add group selector.
 123          $groups = $reportlog->get_group_list();
 124          if (!empty($groups)) {
 125              echo html_writer::label(get_string('selectagroup'), 'menugroup', false, array('class' => 'accesshide'));
 126              echo html_writer::select($groups, "group", $reportlog->groupid, get_string("allgroups"));
 127          }
 128  
 129          // Add user selector.
 130          $users = $reportlog->get_user_list();
 131  
 132          if ($reportlog->showusers) {
 133              echo html_writer::label(get_string('selctauser'), 'menuuser', false, array('class' => 'accesshide'));
 134              echo html_writer::select($users, "user", $reportlog->userid, get_string("allparticipants"));
 135          } else {
 136              $users = array();
 137              if (!empty($reportlog->userid)) {
 138                  $users[$reportlog->userid] = $reportlog->get_selected_user_fullname();
 139              } else {
 140                  $users[0] = get_string('allparticipants');
 141              }
 142              echo html_writer::label(get_string('selctauser'), 'menuuser', false, array('class' => 'accesshide'));
 143              echo html_writer::select($users, "user", $reportlog->userid, false);
 144              $a = new stdClass();
 145              $a->url = new moodle_url('/report/log/index.php', array('chooselog' => 0,
 146                  'group' => $reportlog->get_selected_group(), 'user' => $reportlog->userid,
 147                  'id' => $selectedcourseid, 'date' => $reportlog->date, 'modid' => $reportlog->modid,
 148                  'showusers' => 1, 'showcourses' => $reportlog->showcourses));
 149              $a->url = $a->url->out(false);
 150              print_string('logtoomanyusers', 'moodle', $a);
 151          }
 152  
 153          // Add date selector.
 154          $dates = $reportlog->get_date_options();
 155          echo html_writer::label(get_string('date'), 'menudate', false, array('class' => 'accesshide'));
 156          echo html_writer::select($dates, "date", $reportlog->date, get_string("alldays"));
 157  
 158          // Add activity selector.
 159          $activities = $reportlog->get_activities_list();
 160          echo html_writer::label(get_string('activities'), 'menumodid', false, array('class' => 'accesshide'));
 161          echo html_writer::select($activities, "modid", $reportlog->modid, get_string("allactivities"));
 162  
 163          // Add actions selector.
 164          echo html_writer::label(get_string('actions'), 'menumodaction', false, array('class' => 'accesshide'));
 165          echo html_writer::select($reportlog->get_actions(), 'modaction', $reportlog->action, get_string("allactions"));
 166  
 167          // Add edulevel.
 168          $edulevel = $reportlog->get_edulevel_options();
 169          echo html_writer::label(get_string('edulevel'), 'menuedulevel', false, array('class' => 'accesshide'));
 170          echo html_writer::select($edulevel, 'edulevel', $reportlog->edulevel, false);
 171  
 172          // Add reader option.
 173          // If there is some reader available then only show submit button.
 174          $readers = $reportlog->get_readers(true);
 175          if (!empty($readers)) {
 176              if (count($readers) == 1) {
 177                  $attributes = array('type' => 'hidden', 'name' => 'logreader', 'value' => key($readers));
 178                  echo html_writer::empty_tag('input', $attributes);
 179              } else {
 180                  echo html_writer::label(get_string('selectlogreader', 'report_log'), 'menureader', false,
 181                          array('class' => 'accesshide'));
 182                  echo html_writer::select($readers, 'logreader', $reportlog->selectedlogreader, false);
 183              }
 184              echo html_writer::empty_tag('input', array('type' => 'submit', 'value' => get_string('gettheselogs')));
 185          }
 186          echo html_writer::end_div();
 187          echo html_writer::end_tag('form');
 188      }

 651              $rest->data = $input['data'];
 652  
 653          // Content-Length (required)
 654          if (isset($input['size']) && $input['size'] >= 0)

 679              else

 681          }
 682

 684              $rest->setAmzHeader('x-amz-storage-class', $storageClass);

 687              $rest->setAmzHeader('x-amz-server-side-encryption', $serverSideEncryption);

 690          if ($rest->size >= 0 && ($rest->fp !== false || $rest->data !== false))

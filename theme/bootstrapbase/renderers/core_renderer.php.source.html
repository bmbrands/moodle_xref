  31       * This renders the navbar.
  32       * Uses bootstrap compatible html.
  33       */
  34      public function navbar() {
  35          $items = $this->page->navbar->get_items();
  36          if (empty($items)) {
  37              return '';
  38          }
  39  
  40          $breadcrumbs = array();
  41          foreach ($items as $item) {
  42              $item->hideicon = true;
  43              $breadcrumbs[] = $this->render($item);
  44          }
  45          $divider = '<span class="divider">'.get_separator().'</span>';
  46          $list_items = '<li>'.join(" $divider</li><li>", $breadcrumbs).'</li>';
  47          $title = '<span class="accesshide">'.get_string('pagepath').'</span>';
  48          return $title . "<ul class=\"breadcrumb\">$list_items</ul>";
  49      }

  67       * This renders the bootstrap top menu.
  68       *
  69       * This renderer is needed to enable the Bootstrap style navigation.

  71      protected function render_custom_menu(custom_menu $menu) {
  72          global $CFG;
  73  
  74          // TODO: eliminate this duplicated logic, it belongs in core, not
  75          // here. See MDL-39565.
  76          $addlangmenu = true;
  77          $langs = get_string_manager()->get_list_of_translations();
  78          if (count($langs) < 2
  79              or empty($CFG->langmenu)
  80              or ($this->page->course != SITEID and !empty($this->page->course->lang))) {
  81              $addlangmenu = false;
  82          }
  83  
  84          if (!$menu->has_children() && $addlangmenu === false) {
  85              return '';
  86          }
  87  
  88          if ($addlangmenu) {
  89              $strlang =  get_string('language');
  90              $currentlang = current_language();
  91              if (isset($langs[$currentlang])) {
  92                  $currentlang = $langs[$currentlang];
  93              } else {
  94                  $currentlang = $strlang;
  95              }
  96              $this->language = $menu->add($currentlang, new moodle_url('#'), $strlang, 10000);
  97              foreach ($langs as $langtype => $langname) {
  98                  $this->language->add($langname, new moodle_url($this->page->url, array('lang' => $langtype)), $langname);
  99              }
 100          }
 101  
 102          $content = '<ul class="nav">';
 103          foreach ($menu->get_children() as $item) {
 104              $content .= $this->render_custom_menu_item($item, 1);
 105          }
 106  
 107          return $content.'</ul>';
 108      }

 110      /*
 111       * This code renders the custom menu items for the
 112       * bootstrap dropdown menu.

 114      protected function render_custom_menu_item(custom_menu_item $menunode, $level = 0 ) {
 115          static $submenucount = 0;
 116  
 117          $content = '';
 118          if ($menunode->has_children()) {
 119  
 120              if ($level == 1) {
 121                  $class = 'dropdown';
 122              } else {
 123                  $class = 'dropdown-submenu';
 124              }
 125  
 126              if ($menunode === $this->language) {
 127                  $class .= ' langmenu';
 128              }
 129              $content = html_writer::start_tag('li', array('class' => $class));
 130              // If the child has menus render it as a sub menu.
 131              $submenucount++;
 132              if ($menunode->get_url() !== null) {
 133                  $url = $menunode->get_url();
 134              } else {
 135                  $url = '#cm_submenu_'.$submenucount;
 136              }
 137              $content .= html_writer::start_tag('a', array('href'=>$url, 'class'=>'dropdown-toggle', 'data-toggle'=>'dropdown', 'title'=>$menunode->get_title()));
 138              $content .= $menunode->get_text();
 139              if ($level == 1) {
 140                  $content .= '<b class="caret"></b>';
 141              }
 142              $content .= '</a>';
 143              $content .= '<ul class="dropdown-menu">';
 144              foreach ($menunode->get_children() as $menunode) {
 145                  $content .= $this->render_custom_menu_item($menunode, 0);
 146              }
 147              $content .= '</ul>';
 148          } else {
 149              // The node doesn't have children so produce a final menuitem.
 150              // Also, if the node's text matches '####', add a class so we can treat it as a divider.
 151              if (preg_match("/^#+$/", $menunode->get_text())) {
 152                  // This is a divider.
 153                  $content = '<li class="divider">&nbsp;</li>';
 154              } else {
 155                  $content = '<li>';
 156                  if ($menunode->get_url() !== null) {
 157                      $url = $menunode->get_url();
 158                  } else {
 159                      $url = '#';
 160                  }
 161                  $content .= html_writer::link($url, $menunode->get_text(), array('title' => $menunode->get_title()));
 162                  $content .= '</li>';
 163              }
 164          }
 165          return $content;
 166      }

 168      /**
 169       * Renders tabtree
 170       *
 171       * @param tabtree $tabtree
 172       * @return string
 173       */
 174      protected function render_tabtree(tabtree $tabtree) {
 175          if (empty($tabtree->subtree)) {
 176              return '';
 177          }
 178          $firstrow = $secondrow = '';
 179          foreach ($tabtree->subtree as $tab) {
 180              $firstrow .= $this->render($tab);
 181              if (($tab->selected || $tab->activated) && !empty($tab->subtree) && $tab->subtree !== array()) {
 182                  $secondrow = $this->tabtree($tab->subtree);
 183              }
 184          }
 185          return html_writer::tag('ul', $firstrow, array('class' => 'nav nav-tabs')) . $secondrow;
 186      }

 188      /**
 189       * Renders tabobject (part of tabtree)
 190       *
 191       * This function is called from {@link core_renderer::render_tabtree()}
 192       * and also it calls itself when printing the $tabobject subtree recursively.
 193       *
 194       * @param tabobject $tabobject
 195       * @return string HTML fragment
 196       */
 197      protected function render_tabobject(tabobject $tab) {
 198          if (($tab->selected and (!$tab->linkedwhenselected)) or $tab->activated) {
 199              return html_writer::tag('li', html_writer::tag('a', $tab->text), array('class' => 'active'));
 200          } else if ($tab->inactive) {
 201              return html_writer::tag('li', html_writer::tag('a', $tab->text), array('class' => 'disabled'));
 202          } else {
 203              if (!($tab->link instanceof moodle_url)) {
 204                  // backward compartibility when link was passed as quoted string
 205                  $link = "<a href=\"$tab->link\" title=\"$tab->title\">$tab->text</a>";
 206              } else {
 207                  $link = html_writer::link($tab->link, $tab->text, array('title' => $tab->title));
 208              }
 209              $params = $tab->selected ? array('class' => 'active') : null;
 210              return html_writer::tag('li', $link, $params);
 211          }
 212      }

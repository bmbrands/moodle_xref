  58       * @Given /^I create a calendar event:$/
  59       * @param TableNode $data
  60       */
  61      public function i_create_a_calendar_event($data) {
  62          // Get the event name.
  63          $eventname = $data->getRow(1);
  64          $eventname = $eventname[1];
  65  
  66          // Click to create new event.
  67          $this->execute("behat_general::i_click_on", array(get_string('newevent', 'calendar'), "button"));
  68  
  69          // Set form fields.
  70          $this->execute("behat_forms::i_set_the_following_fields_to_these_values", $data);
  71  
  72          // Save event.
  73          $this->execute("behat_forms::press_button", get_string('savechanges'));
  74  
  75          // Check if event is created. Being last step, don't need to wait or check for exceptions.
  76          $this->execute("behat_general::assert_page_contains_text", $eventname);

  78      }

  80      /**
  81       * Hover over a specific day in the calendar.
  82       *
  83       * @Given /^I hover over day "(?P<dayofmonth>\d+)" of this month in the calendar$/
  84       * @param int $day The day of the current month
  85       */
  86      public function i_hover_over_day_of_this_month_in_calendar($day) {
  87          $summarytitle = get_string('calendarheading', 'calendar', userdate(time(), get_string('strftimemonthyear')));
  88          // The current month table.
  89          $currentmonth = "table[contains(concat(' ', normalize-space(@summary), ' '), ' {$summarytitle} ')]";
  90  
  91          // Strings for the class cell match.
  92          $cellclasses  = "contains(concat(' ', normalize-space(@class), ' '), ' day ')";
  93          $daycontains  = "text()[contains(concat(' ', normalize-space(.), ' '), ' {$day} ')]";
  94          $daycell      = "td[{$cellclasses}]";
  95          $dayofmonth   = "a[{$daycontains}]";
  96  
  97          $xpath = '//' . $currentmonth . '/descendant::' . $daycell . '/' . $dayofmonth;
  98          $this->execute("behat_general::i_hover", array($xpath, "xpath_element"));
  99  
 100      }

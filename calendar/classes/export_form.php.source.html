  52          $mform->addGroup($export, 'events', get_string('export', 'calendar'), '<br/>');
  53          $mform->addGroupRule('events', get_string('required'), 'required');
  54          $mform->setDefault('events', 'all');

  56          $range = array();

  58              $range[] = $mform->createElement('radio', 'timeperiod', '', get_string('weekthis', 'calendar'), 'weeknow');
  59          }
  60          if ($this->_customdata['allownextweek']) {
  61              $range[] = $mform->createElement('radio', 'timeperiod', '', get_string('weeknext', 'calendar'), 'weeknext');
  62          }
  63          $range[] = $mform->createElement('radio', 'timeperiod', '', get_string('monththis', 'calendar'), 'monthnow');
  64          if ($this->_customdata['allownextmonth']) {
  65              $range[] = $mform->createElement('radio', 'timeperiod', '', get_string('monthnext', 'calendar'), 'monthnext');
  66          }
  67          $range[] = $mform->createElement('radio', 'timeperiod', '', get_string('recentupcoming', 'calendar'), 'recentupcoming');
  68  
  69          if ($CFG->calendar_customexport) {
  70              $a = new stdClass();
  71              $now = time();
  72              $time = $now - $CFG->calendar_exportlookback * DAYSECS;
  73              $a->timestart = userdate($time, get_string('strftimedatefullshort', 'langconfig'));
  74              $time = $now + $CFG->calendar_exportlookahead * DAYSECS;
  75              $a->timeend = userdate($time, get_string('strftimedatefullshort', 'langconfig'));
  76  
  77              $range[] = $mform->createElement('radio', 'timeperiod', '', get_string('customexport', 'calendar', $a), 'custom');
  78          }

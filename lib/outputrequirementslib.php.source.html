 434      /**
 435       * Request inclusion of jQuery plugin.
 436       *
 437       * NOTE: this should not be used in official Moodle distribution!
 438       *
 439       * jQuery plugins are located in plugin/jquery/* subdirectory,
 440       * plugin/jquery/plugins.php lists all available plugins.
 441       *
 442       * Included core plugins:
 443       *   - jQuery UI
 444       *   - jQuery Migrate (useful for code written for previous UI version)
 445       *
 446       * Add-ons may include extra jQuery plugins in jquery/ directory,
 447       * plugins.php file defines the mapping between plugin names and
 448       * necessary page includes.
 449       *
 450       * Examples:
 451       * <code>
 452       *   // file: mod/xxx/view.php
 453       *   $PAGE->requires->jquery();
 454       *   $PAGE->requires->jquery_plugin('ui');
 455       *   $PAGE->requires->jquery_plugin('ui-css');
 456       * </code>
 457       *
 458       * <code>
 459       *   // file: theme/yyy/lib.php
 460       *   function theme_yyy_page_init(moodle_page $page) {
 461       *       $page->requires->jquery();
 462       *       $page->requires->jquery_plugin('ui');
 463       *       $page->requires->jquery_plugin('ui-css');
 464       *   }
 465       * </code>
 466       *
 467       * <code>
 468       *   // file: blocks/zzz/block_zzz.php
 469       *   public function get_required_javascript() {
 470       *       parent::get_required_javascript();
 471       *       $this->page->requires->jquery();
 472       *       $page->requires->jquery_plugin('ui');
 473       *       $page->requires->jquery_plugin('ui-css');
 474       *   }
 475       * </code>
 476       *
 477       * {@see http://docs.moodle.org/dev/jQuery}
 478       *
 479       * @param string $plugin name of the jQuery plugin as defined in jquery/plugins.php
 480       * @param string $component name of the component
 481       * @return bool success
 482       */
 483      public function jquery_plugin($plugin, $component = 'core') {
 484          global $CFG;
 485  
 486          if ($this->headdone) {
 487              debugging('Can not add jQuery plugins after starting page output!');
 488              return false;
 489          }
 490  
 491          if ($component !== 'core' and in_array($plugin, array('jquery', 'ui', 'ui-css', 'migrate'))) {
 492              debugging("jQuery plugin '$plugin' is included in Moodle core, other components can not use the same name.", DEBUG_DEVELOPER);
 493              $component = 'core';
 494          } else if ($component !== 'core' and strpos($component, '_') === false) {
 495              // Let's normalise the legacy activity names, Frankenstyle rulez!
 496              $component = 'mod_' . $component;
 497          }
 498  
 499          if (empty($this->jqueryplugins) and ($component !== 'core' or $plugin !== 'jquery')) {
 500              // Make sure the jQuery itself is always loaded first,
 501              // the order of all other plugins depends on order of $PAGE_>requires->.
 502              $this->jquery_plugin('jquery', 'core');
 503          }
 504  
 505          if (isset($this->jqueryplugins[$plugin])) {
 506              // No problem, we already have something, first Moodle plugin to register the jQuery plugin wins.
 507              return true;
 508          }
 509  
 510          $componentdir = core_component::get_component_directory($component);
 511          if (!file_exists($componentdir) or !file_exists("$componentdir/jquery/plugins.php")) {
 512              debugging("Can not load jQuery plugin '$plugin', missing plugins.php in component '$component'.", DEBUG_DEVELOPER);
 513              return false;
 514          }
 515  
 516          $plugins = array();
 517          require("$componentdir/jquery/plugins.php");
 518  
 519          if (!isset($plugins[$plugin])) {
 520              debugging("jQuery plugin '$plugin' can not be found in component '$component'.", DEBUG_DEVELOPER);
 521              return false;
 522          }
 523  
 524          $this->jqueryplugins[$plugin] = new stdClass();
 525          $this->jqueryplugins[$plugin]->plugin    = $plugin;
 526          $this->jqueryplugins[$plugin]->component = $component;
 527          $this->jqueryplugins[$plugin]->urls      = array();
 528  
 529          foreach ($plugins[$plugin]['files'] as $file) {
 530              if ($CFG->debugdeveloper) {
 531                  if (!file_exists("$componentdir/jquery/$file")) {
 532                      debugging("Invalid file '$file' specified in jQuery plugin '$plugin' in component '$component'");
 533                      continue;
 534                  }
 535                  $file = str_replace('.min.css', '.css', $file);
 536                  $file = str_replace('.min.js', '.js', $file);
 537              }
 538              if (!file_exists("$componentdir/jquery/$file")) {
 539                  debugging("Invalid file '$file' specified in jQuery plugin '$plugin' in component '$component'");
 540                  continue;
 541              }
 542              if (!empty($CFG->slasharguments)) {
 543                  $url = new moodle_url("$CFG->httpswwwroot/theme/jquery.php");
 544                  $url->set_slashargument("/$component/$file");
 545  
 546              } else {
 547                  // This is not really good, we need slasharguments for relative links, this means no caching...
 548                  $path = realpath("$componentdir/jquery/$file");
 549                  if (strpos($path, $CFG->dirroot) === 0) {
 550                      $url = $CFG->httpswwwroot.preg_replace('/^'.preg_quote($CFG->dirroot, '/').'/', '', $path);
 551                      $url = new moodle_url($url);
 552                  } else {
 553                      // Bad luck, fix your server!
 554                      debugging("Moodle jQuery integration requires 'slasharguments' setting to be enabled.");
 555                      continue;
 556                  }
 557              }
 558              $this->jqueryplugins[$plugin]->urls[] = $url;
 559          }
 560  
 561          return true;
 562      }

 567       * This is useful when themes want to replace the jQuery UI theme,

 569       *
 570       * Example:
 571       *  1/ generate new jQuery UI theme and place it into theme/yourtheme/jquery/

 583       * This code prevents loading of standard 'ui-css' which my be requested by other plugins,
 584       * the 'yourtheme-ui-css' gets loaded only if some other code requires jquery.

 603      protected function get_jquery_headcode() {
 604          if (empty($this->jqueryplugins['jquery'])) {
 605              // If nobody requested jQuery then do not bother to load anything.
 606              // This may be useful for themes that want to override 'ui-css' only if requested by something else.
 607              return '';
 608          }
 609  
 610          $included = array();
 611          $urls = array();
 612  
 613          foreach ($this->jqueryplugins as $name => $unused) {
 614              if (isset($included[$name])) {
 615                  continue;
 616              }
 617              if (array_key_exists($name, $this->jquerypluginoverrides)) {
 618                  // The following loop tries to resolve the replacements,
 619                  // use max 100 iterations to prevent infinite loop resulting
 620                  // in blank page.
 621                  $cyclic = true;
 622                  $oldname = $name;
 623                  for ($i=0; $i<100; $i++) {
 624                      $name = $this->jquerypluginoverrides[$name];
 625                      if (!array_key_exists($name, $this->jquerypluginoverrides)) {
 626                          $cyclic = false;
 627                          break;
 628                      }
 629                  }
 630                  if ($cyclic) {
 631                      // We can not do much with cyclic references here, let's use the old plugin.
 632                      $name = $oldname;
 633                      debugging("Cyclic overrides detected for jQuery plugin '$name'");
 634  
 635                  } else if (empty($name)) {
 636                      // Developer requested removal of the plugin.
 637                      continue;
 638  
 639                  } else if (!isset($this->jqueryplugins[$name])) {
 640                      debugging("Unknown jQuery override plugin '$name' detected");
 641                      $name = $oldname;
 642  
 643                  } else if (isset($included[$name])) {
 644                      // The plugin was already included, easy.
 645                      continue;
 646                  }
 647              }
 648  
 649              $plugin = $this->jqueryplugins[$name];
 650              $urls = array_merge($urls, $plugin->urls);
 651              $included[$name] = true;
 652          }
 653  
 654          $output = '';
 655          $attributes = array('rel' => 'stylesheet', 'type' => 'text/css');
 656          foreach ($urls as $url) {
 657              if (preg_match('/\.js$/', $url)) {
 658                  $output .= html_writer::script('', $url);
 659              } else if (preg_match('/\.css$/', $url)) {
 660                  $attributes['href'] = $url;
 661                  $output .= html_writer::empty_tag('link', $attributes) . "\n";
 662              }
 663          }
 664  
 665          return $output;
 666      }

1104      /**
1105       * Make a language string available to JavaScript.
1106       *
1107       * All the strings will be available in a M.str object in the global namespace.
1108       * So, for example, after a call to $PAGE->requires->string_for_js('course', 'moodle');
1109       * then the JavaScript variable M.str.moodle.course will be 'Course', or the
1110       * equivalent in the current language.
1111       *
1112       * The arguments to this function are just like the arguments to get_string
1113       * except that $component is not optional, and there are some aspects to consider
1114       * when the string contains {$a} placeholder.
1115       *
1116       * If the string does not contain any {$a} placeholder, you can simply use
1117       * M.str.component.identifier to obtain it. If you prefer, you can call
1118       * M.util.get_string(identifier, component) to get the same result.
1119       *
1120       * If you need to use {$a} placeholders, there are two options. Either the
1121       * placeholder should be substituted in PHP on server side or it should
1122       * be substituted in Javascript at client side.
1123       *
1124       * To substitute the placeholder at server side, just provide the required
1125       * value for the placeholder when you require the string. Because each string
1126       * is only stored once in the JavaScript (based on $identifier and $module)
1127       * you cannot get the same string with two different values of $a. If you try,
1128       * an exception will be thrown. Once the placeholder is substituted, you can
1129       * use M.str or M.util.get_string() as shown above:
1130       *
1131       *     // Require the string in PHP and replace the placeholder.
1132       *     $PAGE->requires->string_for_js('fullnamedisplay', 'moodle', $USER);
1133       *     // Use the result of the substitution in Javascript.
1134       *     alert(M.str.moodle.fullnamedisplay);
1135       *
1136       * To substitute the placeholder at client side, use M.util.get_string()
1137       * function. It implements the same logic as {@link get_string()}:
1138       *
1139       *     // Require the string in PHP but keep {$a} as it is.
1140       *     $PAGE->requires->string_for_js('fullnamedisplay', 'moodle');
1141       *     // Provide the values on the fly in Javascript.
1142       *     user = { firstname : 'Harry', lastname : 'Potter' }
1143       *     alert(M.util.get_string('fullnamedisplay', 'moodle', user);
1144       *
1145       * If you do need the same string expanded with different $a values in PHP
1146       * on server side, then the solution is to put them in your own data structure
1147       * (e.g. and array) that you pass to JavaScript with {@link data_for_js()}.
1148       *
1149       * @param string $identifier the desired string.
1150       * @param string $component the language file to look in.
1151       * @param mixed $a any extra data to add into the string (optional).
1152       */
1153      public function string_for_js($identifier, $component, $a = null) {

1199          }

1968      /**
1969       * Define YUI modules which we have been required to patch between releases.
1970       *
1971       * We must do this because we aggressively cache content on the browser, and we must also override use of the
1972       * external CDN which will serve the true authoritative copy of the code without our patches.
1973       *
1974       * @param String combobase The local combobase
1975       * @param String yuiversion The current YUI version
1976       * @param Int patchlevel The patch level we're working to for YUI
1977       * @param Array patchedmodules An array containing the names of the patched modules
1978       * @return void
1979       */
1980      public function define_patched_core_modules($combobase, $yuiversion, $patchlevel, $patchedmodules) {
1981          // The version we use is suffixed with a patchlevel so that we can get additional revisions between YUI releases.
1982          $subversion = $yuiversion . '_' . $patchlevel;
1983  
1984          if ($this->comboBase == $combobase) {
1985              // If we are using the local combobase in the loader, we can add a group and still make use of the combo
1986              // loader. We just need to specify a different root which includes a slightly different YUI version number
1987              // to include our patchlevel.
1988              $patterns = array();
1989              $modules = array();
1990              foreach ($patchedmodules as $modulename) {
1991                  // We must define the pattern and module here so that the loader uses our group configuration instead of
1992                  // the standard module definition. We may lose some metadata provided by upstream but this will be
1993                  // loaded when the module is loaded anyway.
1994                  $patterns[$modulename] = array(
1995                      'group' => 'yui-patched',
1996                  );
1997                  $modules[$modulename] = array();
1998              }
1999  
2000              // Actually add the patch group here.
2001              $this->add_group('yui-patched', array(
2002                  'combine' => true,
2003                  'root' => $subversion . '/',
2004                  'patterns' => $patterns,
2005                  'modules' => $modules,
2006              ));
2007  
2008          } else {
2009              // The CDN is in use - we need to instead use the local combobase for this module and override the modules
2010              // definition. We cannot use the local base - we must use the combobase because we cannot invalidate the
2011              // local base in browser caches.
2012              $fullpathbase = $combobase . $subversion . '/';
2013              foreach ($patchedmodules as $modulename) {
2014                  $this->modules[$modulename] = array(
2015                      'fullpath' => $fullpathbase . $modulename . '/' . $modulename . '-min.js'
2016                  );
2017              }
2018          }
2019      }

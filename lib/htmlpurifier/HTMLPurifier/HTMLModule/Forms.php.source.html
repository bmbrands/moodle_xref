  65                  'value' => 'CDATA',

  69  
  70          $this->addElement(

  80                  'tabindex' => 'Number',
  81              )
  82          );
  83  
  84          $this->addElement(
  85              'option',
  86              false,
  87              'Optional: #PCDATA',
  88              'Common',
  89              array(
  90                  'disabled' => 'Bool#disabled',

  93                  'value' => 'CDATA',

  95          );

 152  
 153          $label = $this->addElement(
 154              'label',

 164  
 165          $this->addElement(

 168              'Optional: #PCDATA | Inline',
 169              'Common',
 170              array(
 171                  'accesskey' => 'Character',
 172              )
 173          );
 174  
 175          $this->addElement(
 176              'optgroup',
 177              false,
 178              'Required: option',
 179              'Common',
 180              array(

 182                  'label*' => 'Text',
 183              )
 184          );
 185          // Don't forget an injector for <isindex>. This one's a little complex

 187      }

  73              @($num = array_pop($parent->GetCol("SELECT id FROM {$seq}")));

  75                  @$parent->Execute(sprintf($this->_genSeqSQL ,$seq));

  82                  if (!$ok) return false;
  83              }
  84              $parent->Execute(sprintf($this->_genIDSQL,$seq,$num));
  85

  87                              $num += 1;

  89                          return intval($num);
  90              }
  91          }
  92          if ($fn = $parent->raiseErrorFn) {
  93              $fn($parent->databaseType,'GENID',-32000,"Unable to generate unique id after $MAXLOOPS attempts",$seq,$num);

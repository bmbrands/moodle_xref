  74          if ($rs === false) return false;

  82              // split them out here.

  85               $fld->max_length = trim($first, "()");

  91               $fld->type = substr($rs->fields[1], 0, $first);
  92              } else {
  93               $fld->type = $rs->fields[1];

  95  
  96              switch ($fld->type) {

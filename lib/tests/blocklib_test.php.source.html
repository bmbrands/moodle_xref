 482      public function test_delete_instances() {
 483          global $DB;
 484          $this->purge_blocks();
 485          $this->setAdminUser();
 486  
 487          $regionname = 'a-region';
 488          $blockname = $this->get_a_known_block_type();
 489          $context = context_system::instance();
 490  
 491          list($page, $blockmanager) = $this->get_a_page_and_block_manager(array($regionname),
 492              $context, 'page-type');
 493  
 494          $blockmanager->add_blocks(array($regionname => array($blockname, $blockname, $blockname)), null, null, false, 3);
 495          $blockmanager->load_blocks();
 496  
 497          $blocks = $blockmanager->get_blocks_for_region($regionname);
 498          $blockids = array();
 499          $preferences = array();
 500  
 501          // Create block related data.
 502          foreach ($blocks as $block) {
 503              $instance = $block->instance;
 504              $pref = 'block' . $instance->id . 'hidden';
 505              set_user_preference($pref, '123', 123);
 506              $preferences[] = $pref;
 507              $pref = 'docked_block_instance_' . $instance->id;
 508              set_user_preference($pref, '123', 123);
 509              $preferences[] = $pref;
 510              blocks_set_visibility($instance, $page, 1);
 511              $blockids[] = $instance->id;
 512          }
 513  
 514          // Confirm what has been set.
 515          $this->assertCount(3, $blockids);
 516          list($insql, $inparams) = $DB->get_in_or_equal($blockids);
 517          $this->assertEquals(3, $DB->count_records_select('block_positions', "blockinstanceid $insql", $inparams));
 518          list($insql, $inparams) = $DB->get_in_or_equal($preferences);
 519          $this->assertEquals(6, $DB->count_records_select('user_preferences', "name $insql", $inparams));
 520  
 521          // Keep a block on the side.
 522          $allblockids = $blockids;
 523          $tokeep = array_pop($blockids);
 524  
 525          // Delete and confirm what should have happened.
 526          blocks_delete_instances($blockids);
 527  
 528          // Reload the manager.
 529          list($page, $blockmanager) = $this->get_a_page_and_block_manager(array($regionname),
 530              $context, 'page-type');
 531          $blockmanager->load_blocks();
 532          $blocks = $blockmanager->get_blocks_for_region($regionname);
 533  
 534          $this->assertCount(1, $blocks);
 535          list($insql, $inparams) = $DB->get_in_or_equal($allblockids);
 536          $this->assertEquals(1, $DB->count_records_select('block_positions', "blockinstanceid $insql", $inparams));
 537          list($insql, $inparams) = $DB->get_in_or_equal($preferences);
 538          $this->assertEquals(2, $DB->count_records_select('user_preferences', "name $insql", $inparams));
 539  
 540          $this->assertFalse(context_block::instance($blockids[0], IGNORE_MISSING));
 541          $this->assertFalse(context_block::instance($blockids[1], IGNORE_MISSING));
 542          context_block::instance($tokeep);   // Would throw an exception if it was deleted.
 543      }

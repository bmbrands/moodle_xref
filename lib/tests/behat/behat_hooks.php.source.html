 211              self::$timings[$new] = round($v, 1);
 212              unset(self::$timings[$k]);
 213          }

 215              self::$timings = array_merge($existing, self::$timings);
 216          }
 217          arsort(self::$timings);
 218          @file_put_contents(BEHAT_FEATURE_TIMING_FILE, json_encode(self::$timings, JSON_PRETTY_PRINT));

 221      /**
 222       * Resets the test environment.
 223       *
 224       * @param OutlineExampleEvent|ScenarioEvent $event event fired before scenario.
 225       * @throws coding_exception If here we are not using the test database it should be because of a coding error
 226       * @BeforeScenario
 227       */
 228      public function before_scenario($event) {
 229          global $DB, $SESSION, $CFG;
 230  
 231          // As many checks as we can.
 232          if (!defined('BEHAT_TEST') ||
 233                 !defined('BEHAT_SITE_RUNNING') ||
 234                 php_sapi_name() != 'cli' ||
 235                 !behat_util::is_test_mode_enabled() ||
 236                 !behat_util::is_test_site()) {
 237              throw new coding_exception('Behat only can modify the test database and the test dataroot!');
 238          }
 239  
 240          $moreinfo = 'More info in ' . behat_command::DOCS_URL . '#Running_tests';
 241          $driverexceptionmsg = 'Selenium server is not running, you need to start it to run tests that involve Javascript. ' . $moreinfo;
 242          try {
 243              $session = $this->getSession();
 244          } catch (CurlExec $e) {
 245              // Exception thrown by WebDriver, so only @javascript tests will be caugth; in
 246              // behat_util::is_server_running() we already checked that the server is running.
 247              throw new Exception($driverexceptionmsg);
 248          } catch (DriverException $e) {
 249              throw new Exception($driverexceptionmsg);
 250          } catch (UnknownError $e) {
 251              // Generic 'I have no idea' Selenium error. Custom exception to provide more feedback about possible solutions.
 252              $this->throw_unknown_exception($e);
 253          }
 254  
 255  
 256          // We need the Mink session to do it and we do it only before the first scenario.
 257          if (self::is_first_scenario()) {
 258              behat_selectors::register_moodle_selectors($session);
 259              behat_context_helper::set_session($session);
 260          }
 261  
 262          // Reset mink session between the scenarios.
 263          $session->reset();
 264  
 265          // Reset $SESSION.
 266          \core\session\manager::init_empty_session();
 267  
 268          behat_util::reset_all_data();
 269  
 270          // Assign valid data to admin user (some generator-related code needs a valid user).
 271          $user = $DB->get_record('user', array('username' => 'admin'));
 272          \core\session\manager::set_user($user);
 273  
 274          // Reset the browser if specified in config.php.
 275          if (!empty($CFG->behat_restart_browser_after) && $this->running_javascript()) {
 276              $now = time();
 277              if (self::$lastbrowsersessionstart + $CFG->behat_restart_browser_after < $now) {
 278                  $session->restart();
 279                  self::$lastbrowsersessionstart = $now;
 280              }
 281          }
 282  
 283          // Start always in the the homepage.
 284          try {
 285              // Let's be conservative as we never know when new upstream issues will affect us.
 286              $session->visit($this->locate_path('/'));
 287          } catch (UnknownError $e) {
 288              $this->throw_unknown_exception($e);
 289          }
 290  
 291  
 292          // Checking that the root path is a Moodle test site.
 293          if (self::is_first_scenario()) {
 294              $notestsiteexception = new Exception('The base URL (' . $CFG->wwwroot . ') is not a behat test site, ' .
 295                  'ensure you started the built-in web server in the correct directory or your web server is correctly started and set up');
 296              $this->find("xpath", "//head/child::title[normalize-space(.)='" . behat_util::BEHATSITENAME . "']", $notestsiteexception);
 297  
 298              self::$initprocessesfinished = true;
 299          }
 300          // Run all test with medium (1024x768) screen size, to avoid responsive problems.
 301          $this->resize_window('medium');
 302      }

 362                  // Catching the generic one as we never know how drivers reacts here.
 363              }
 364          } catch (Exception $e) {
 365              self::$currentstepexception = $e;
 366          }
 367      }

 426      /**
 427       * Determine the full pathname to store a failure-related dump.
 428       *
 429       * This is used for content such as the DOM, and screenshots.
 430       *
 431       * @param StepEvent $event
 432       * @param String $filetype The file suffix to use. Limited to 4 chars.
 433       */
 434      protected function get_faildump_filename(StepEvent $event, $filetype) {
 435          global $CFG;
 436  
 437          // All the contentdumps should be in the same parent dir.
 438          if (!$faildumpdir = self::get_run_faildump_dir()) {
 439              $faildumpdir = self::$faildumpdirname = date('Ymd_His');
 440  
 441              $dir = $CFG->behat_faildump_path . DIRECTORY_SEPARATOR . $faildumpdir;
 442  
 443              if (!is_dir($dir) && !mkdir($dir, $CFG->directorypermissions, true)) {
 444                  // It shouldn't, we already checked that the directory is writable.
 445                  throw new Exception('No directories can be created inside $CFG->behat_faildump_path, check the directory permissions.');
 446              }
 447          } else {
 448              // We will always need to know the full path.
 449              $dir = $CFG->behat_faildump_path . DIRECTORY_SEPARATOR . $faildumpdir;
 450          }
 451  
 452          // The scenario title + the failed step text.
 453          // We want a i-am-the-scenario-title_i-am-the-failed-step.$filetype format.
 454          $filename = $event->getStep()->getParent()->getTitle() . '_' . $event->getStep()->getText();
 455          $filename = preg_replace('/([^a-zA-Z0-9\_]+)/', '-', $filename);
 456  
 457          // File name limited to 255 characters. Leaving 4 chars for the file
 458          // extension as we allow .png for images and .html for DOM contents.
 459          $filename = substr($filename, 0, 250) . '.' . $filetype;
 460  
 461          return array($dir, $filename);
 462      }

 464      /**
 465       * Waits for all the JS to be loaded.
 466       *
 467       * @throws \Exception
 468       * @throws NoSuchWindow
 469       * @throws UnknownError
 470       * @return bool True or false depending whether all the JS is loaded or not.
 471       */
 472      protected function wait_for_pending_js() {
 473  
 474          // We don't use behat_base::spin() here as we don't want to end up with an exception
 475          // if the page & JSs don't finish loading properly.
 476          for ($i = 0; $i < self::EXTENDED_TIMEOUT * 10; $i++) {
 477              $pending = '';
 478              try {
 479                  $jscode =
 480                      'if (typeof M === "undefined") {
 481                          if (document.readyState === "complete") {
 482                              return "";
 483                          } else {
 484                              return "incomplete";
 485                          }
 486                      } else if (' . self::PAGE_READY_JS . ') {
 487                          return "";
 488                      } else {
 489                          return M.util.pending_js.join(":");
 490                      }';
 491                  $pending = $this->getSession()->evaluateScript($jscode);
 492              } catch (NoSuchWindow $nsw) {
 493                  // We catch an exception here, in case we just closed the window we were interacting with.
 494                  // No javascript is running if there is no window right?
 495                  $pending = '';
 496              } catch (UnknownError $e) {
 497                  // M is not defined when the window or the frame don't exist anymore.
 498                  if (strstr($e->getMessage(), 'M is not defined') != false) {
 499                      $pending = '';
 500                  }
 501              }
 502  
 503              // If there are no pending JS we stop waiting.
 504              if ($pending === '') {
 505                  return true;
 506              }
 507  
 508              // 0.1 seconds.
 509              usleep(100000);
 510          }
 511  
 512          // Timeout waiting for JS to complete. It will be catched and forwarded to behat_hooks::i_look_for_exceptions().
 513          // It is unlikely that Javascript code of a page or an AJAX request needs more than self::EXTENDED_TIMEOUT seconds
 514          // to be loaded, although when pages contains Javascript errors M.util.js_complete() can not be executed, so the
 515          // number of JS pending code and JS completed code will not match and we will reach this point.
 516          throw new \Exception('Javascript code and/or AJAX requests are not ready after ' . self::EXTENDED_TIMEOUT .
 517              ' seconds. There is a Javascript error or the code is extremely slow.');
 518      }

 520      /**
 521       * Internal step definition to find exceptions, debugging() messages and PHP debug messages.
 522       *
 523       * Part of behat_hooks class as is part of the testing framework, is auto-executed
 524       * after each step so no features will splicitly use it.
 525       *
 526       * @Given /^I look for exceptions$/
 527       * @throw Exception Unknown type, depending on what we caught in the hook or basic \Exception.
 528       * @see Moodle\BehatExtension\Tester\MoodleStepTester
 529       */
 530      public function i_look_for_exceptions() {
 531  
 532          // If the step already failed in a hook throw the exception.
 533          if (!is_null(self::$currentstepexception)) {
 534              throw self::$currentstepexception;
 535          }
 536  
 537          // Wrap in try in case we were interacting with a closed window.
 538          try {
 539  
 540              // Exceptions.
 541              $exceptionsxpath = "//div[@data-rel='fatalerror']";
 542              // Debugging messages.
 543              $debuggingxpath = "//div[@data-rel='debugging']";
 544              // PHP debug messages.
 545              $phperrorxpath = "//div[@data-rel='phpdebugmessage']";
 546              // Any other backtrace.
 547              $othersxpath = "(//*[contains(., ': call to ')])[1]";
 548  
 549              $xpaths = array($exceptionsxpath, $debuggingxpath, $phperrorxpath, $othersxpath);
 550              $joinedxpath = implode(' | ', $xpaths);
 551  
 552              // Joined xpath expression. Most of the time there will be no exceptions, so this pre-check
 553              // is faster than to send the 4 xpath queries for each step.
 554              if (!$this->getSession()->getDriver()->find($joinedxpath)) {
 555                  return;
 556              }
 557  
 558              // Exceptions.
 559              if ($errormsg = $this->getSession()->getPage()->find('xpath', $exceptionsxpath)) {
 560  
 561                  // Getting the debugging info and the backtrace.
 562                  $errorinfoboxes = $this->getSession()->getPage()->findAll('css', 'div.alert-error');
 563                  // If errorinfoboxes is empty, try find notifytiny (original) class.
 564                  if (empty($errorinfoboxes)) {
 565                      $errorinfoboxes = $this->getSession()->getPage()->findAll('css', 'div.notifytiny');
 566                  }
 567                  $errorinfo = $this->get_debug_text($errorinfoboxes[0]->getHtml()) . "\n" .
 568                      $this->get_debug_text($errorinfoboxes[1]->getHtml());
 569  
 570                  $msg = "Moodle exception: " . $errormsg->getText() . "\n" . $errorinfo;
 571                  throw new \Exception(html_entity_decode($msg));
 572              }
 573  
 574              // Debugging messages.
 575              if ($debuggingmessages = $this->getSession()->getPage()->findAll('xpath', $debuggingxpath)) {
 576                  $msgs = array();
 577                  foreach ($debuggingmessages as $debuggingmessage) {
 578                      $msgs[] = $this->get_debug_text($debuggingmessage->getHtml());
 579                  }
 580                  $msg = "debugging() message/s found:\n" . implode("\n", $msgs);
 581                  throw new \Exception(html_entity_decode($msg));
 582              }
 583  
 584              // PHP debug messages.
 585              if ($phpmessages = $this->getSession()->getPage()->findAll('xpath', $phperrorxpath)) {
 586  
 587                  $msgs = array();
 588                  foreach ($phpmessages as $phpmessage) {
 589                      $msgs[] = $this->get_debug_text($phpmessage->getHtml());
 590                  }
 591                  $msg = "PHP debug message/s found:\n" . implode("\n", $msgs);
 592                  throw new \Exception(html_entity_decode($msg));
 593              }
 594  
 595              // Any other backtrace.
 596              // First looking through xpath as it is faster than get and parse the whole page contents,
 597              // we get the contents and look for matches once we found something to suspect that there is a backtrace.
 598              if ($this->getSession()->getDriver()->find($othersxpath)) {
 599                  $backtracespattern = '/(line [0-9]* of [^:]*: call to [\->&;:a-zA-Z_\x7f-\xff][\->&;:a-zA-Z0-9_\x7f-\xff]*)/';
 600                  if (preg_match_all($backtracespattern, $this->getSession()->getPage()->getContent(), $backtraces)) {
 601                      $msgs = array();
 602                      foreach ($backtraces[0] as $backtrace) {
 603                          $msgs[] = $backtrace . '()';
 604                      }
 605                      $msg = "Other backtraces found:\n" . implode("\n", $msgs);
 606                      throw new \Exception(htmlentities($msg));
 607                  }
 608              }
 609  
 610          } catch (NoSuchWindow $e) {
 611              // If we were interacting with a popup window it will not exists after closing it.
 612          }
 613      }

  45      /**
  46       * Helper function to get a navigation nodes text element given its text from within the navigation block.
  47       *
  48       * This function finds the node with the given text from within the navigation block.
  49       * It checks to make sure the node is visible, and then returns it.
  50       *
  51       * @param string $text
  52       * @param bool $branch Set this true if you're only interested in the node if its a branch.
  53       * @param null|bool $collapsed Set this to true or false if you want the node to either be collapsed or not.
  54       *    If its left as null then we don't worry about it.
  55       * @param null|string|Exception|false $exception The exception to throw if the node is not found.
  56       * @return \Behat\Mink\Element\NodeElement
  57       */
  58      protected function get_node_text_node($text, $branch = false, $collapsed = null, $exception = null) {
  59          if ($exception === null) {
  60              $exception = new ExpectationException('The "' . $text . '" node could not be found', $this->getSession());
  61          } else if (is_string($exception)) {
  62              $exception = new ExpectationException($exception, $this->getSession());
  63          }
  64  
  65          $nodetextliteral = $this->getSession()->getSelectorsHandler()->xpathLiteral($text);
  66          $hasblocktree = "[contains(concat(' ', normalize-space(@class), ' '), ' block_tree ')]";
  67          $hasbranch = "[contains(concat(' ', normalize-space(@class), ' '), ' branch ')]";
  68          $hascollapsed = "li[contains(concat(' ', normalize-space(@class), ' '), ' collapsed ') or @data-expandable='1']";
  69          $notcollapsed = "li[not(contains(concat(' ', normalize-space(@class), ' '), ' collapsed '))]";
  70          $match = "[normalize-space(.)={$nodetextliteral}]";
  71  
  72          // Avoid problems with quotes.
  73          $isbranch = ($branch) ? $hasbranch : '';
  74          if ($collapsed === true) {
  75              $iscollapsed = $hascollapsed;
  76          } else if ($collapsed === false) {
  77              $iscollapsed = $notcollapsed;
  78          } else {
  79              $iscollapsed = 'li';
  80          }
  81  
  82          // First check root nodes.
  83          $xpath  = "//ul{$hasblocktree}/$hascollapsed/p{$isbranch}/span{$match}|";
  84          // Next search for the node containing the text within a link.
  85          $xpath .= "//ul{$hasblocktree}//{$notcollapsed}/ul/{$iscollapsed}/p{$isbranch}/a{$match}|";
  86          // Finally search for the node containing the text within a span.
  87          $xpath .= "//ul{$hasblocktree}//{$notcollapsed}/ul/{$iscollapsed}/p{$isbranch}/span{$match}";
  88  
  89          $node = $this->find('xpath', $xpath, $exception);
  90          $this->ensure_node_is_visible($node);
  91          return $node;
  92      }

  95       * Returns true if the navigation node with the given text is expandable.
  96       *
  97       * @Given /^navigation node "([^"]*)" should be expandable$/
  98       *
  99       * @throws ExpectationException
 100       * @param string $nodetext
 101       * @return bool
 102       */
 103      public function navigation_node_should_be_expandable($nodetext) {
 104          if (!$this->running_javascript()) {
 105              // Nodes are only expandable when JavaScript is enabled.
 106              return false;
 107          }
 108  
 109          $node = $this->get_node_text_node($nodetext, true);
 110          $node = $node->getParent();
 111          if ($node->hasAttribute('data-expandable') && $node->getAttribute('data-expandable')) {
 112              return true;
 113          }
 114          throw new ExpectationException('The "' . $nodetext . '" node is not expandable', $this->getSession());
 115      }

 167      /**
 168       * Expands the selected node of the navigation tree that matches the text.
 169       * @Given /^I expand "(?P<nodetext_string>(?:[^"]|\\")*)" node$/
 170       *
 171       * @throws ExpectationException
 172       * @param string $nodetext
 173       * @return bool|void
 174       */
 175      public function i_expand_node($nodetext) {
 176  
 177          // This step is useless with Javascript disabled as Moodle auto expands
 178          // all of tree's nodes; adding this because of scenarios that shares the
 179          // same steps with and without Javascript enabled.
 180          if (!$this->running_javascript()) {
 181              if ($nodetext === get_string('administrationsite')) {
 182                  // Administration menu is not loaded by default any more. Click the link to expand.
 183                  return new Given('I click on "'.$nodetext.'" "link" in the "'.get_string('administration').'" "block"');
 184              }
 185              return true;
 186          }
 187  
 188          $node = $this->get_node_text_node($nodetext, true, true, 'The "' . $nodetext . '" node can not be expanded');
 189          // Check if the node is a link AND a branch.
 190          if (strtolower($node->getTagName()) === 'a') {
 191              // We just want to expand the node, we don't want to follow it.
 192              $node = $node->getParent();
 193          }
 194          $node->click();
 195      }

 221      /**
 222       * Click link in navigation tree that matches the text in parentnode/s (seperated using greater-than character if more than one)
 223       *
 224       * @Given /^I navigate to "(?P<nodetext_string>(?:[^"]|\\")*)" node in "(?P<parentnodes_string>(?:[^"]|\\")*)"$/
 225       *
 226       * @throws ExpectationException
 227       * @param string $nodetext navigation node to click.
 228       * @param string $parentnodes comma seperated list of parent nodes.
 229       * @return void
 230       */
 231      public function i_navigate_to_node_in($nodetext, $parentnodes) {
 232  
 233          // Site admin is different and needs special treatment.
 234          $siteadminstr = get_string('administrationsite');
 235  
 236          // Create array of all parentnodes.
 237          $parentnodes = array_map('trim', explode('>', $parentnodes));
 238          $countparentnode = count($parentnodes);
 239  
 240          // If JS is disabled and Site administration is not expanded we
 241          // should follow it, so all the lower-level nodes are available.
 242          if (!$this->running_javascript()) {
 243              if ($parentnodes[0] === $siteadminstr) {
 244                  // We don't know if there if Site admin is already expanded so
 245                  // don't wait, it is non-JS and we already waited for the DOM.
 246                  if ($siteadminlink = $this->getSession()->getPage()->find('named', array('link', "'" . $siteadminstr . "'"))) {
 247                      $siteadminlink->click();
 248                  }
 249              }
 250          }
 251  
 252          // Expand first node, and get it.
 253          $node = $this->get_top_navigation_node($parentnodes[0]);
 254  
 255          // Expand parent, sub-parent nodes in navigation if js enabled.
 256          if ($node->hasClass('collapsed') || ($node->hasAttribute('data-loaded') && $node->getAttribute('data-loaded') == 0)) {
 257              $xpath = "/p[contains(concat(' ', normalize-space(@class), ' '), ' tree_item ')]/span";
 258              $nodetoexpand = $node->find('xpath', $xpath);
 259  
 260              if ($this->running_javascript()) {
 261                  $this->ensure_node_is_visible($nodetoexpand);
 262                  $nodetoexpand->click();
 263  
 264                  // Site administration node needs to be expanded.
 265                  if ($parentnodes[0] === $siteadminstr) {
 266                      $this->getSession()->wait(self::TIMEOUT * 1000, self::PAGE_READY_JS);
 267                  }
 268              }
 269          }
 270  
 271          // If sub-parent nodes then get to the last one.
 272          if ($countparentnode > 1) {
 273              for ($i = 1; $i < $countparentnode; $i++) {
 274                  $node = $this->get_navigation_node($parentnodes[$i], $node);
 275  
 276                  // Keep expanding all sub-parents if js enabled.
 277                  if ($this->running_javascript()) {
 278                      $xpath = "/p[contains(concat(' ', normalize-space(@class), ' '), ' tree_item ')]";
 279                      if ($node->hasClass('collapsed')) {
 280                          $nodetoexpand = $node->find('xpath', $xpath);
 281                          if ($this->running_javascript()) {
 282                              $this->ensure_node_is_visible($nodetoexpand);
 283                              $nodetoexpand->click();
 284                          }
 285                      }
 286                  }
 287              }
 288          }
 289  
 290          // Finally, click on requested node under navigation.
 291          $nodetextliteral = $this->getSession()->getSelectorsHandler()->xpathLiteral($nodetext);
 292          $xpath = "/ul/li/p[contains(concat(' ', normalize-space(@class), ' '), ' tree_item ')]" .
 293                  "/a[normalize-space(.)=" . $nodetextliteral . "]";
 294          $node = $node->find('xpath', $xpath);
 295  
 296          // Throw exception if no node found.
 297          if (!$node) {
 298              throw new ExpectationException('Navigation node "' . $nodetext . '" not found under "' .
 299                  implode($parentnodes, ' > ') . '"', $this->getSession());
 300          }
 301  
 302          if ($this->running_javascript()) {
 303              $this->ensure_node_is_visible($node);
 304          }
 305  
 306          $node->click();
 307      }

 309      /**
 310       * Helper function to get top navigation node in tree.
 311       *
 312       * @throws ExpectationException if note not found.
 313       * @param string $nodetext name of top navigation node in tree.
 314       * @return NodeElement
 315       */
 316      protected function get_top_navigation_node($nodetext) {
 317  
 318          // Avoid problems with quotes.
 319          $nodetextliteral = $this->getSession()->getSelectorsHandler()->xpathLiteral($nodetext);
 320          $exception = new ExpectationException('Top navigation node "' . $nodetext . ' not found in "', $this->getSession());
 321  
 322          // First find in navigation block.
 323          $xpath = "//div[contains(concat(' ', normalize-space(@class), ' '), ' content ')]" .
 324              "/ul[contains(concat(' ', normalize-space(@class), ' '), ' block_tree ')]" .
 325              "/li[contains(concat(' ', normalize-space(@class), ' '), ' contains_branch ')]" .
 326              "/ul/li[contains(concat(' ', normalize-space(@class), ' '), ' contains_branch ')]" .
 327              "[p[contains(concat(' ', normalize-space(@class), ' '), ' branch ')]" .
 328              "/span[normalize-space(.)=" . $nodetextliteral ."]]" .
 329              "|" .
 330              "//div[contains(concat(' ', normalize-space(@class), ' '), ' content ')]/div" .
 331              "/ul[contains(concat(' ', normalize-space(@class), ' '), ' block_tree ')]" .
 332              "/li[contains(concat(' ', normalize-space(@class), ' '), ' contains_branch ')]" .
 333              "/ul/li[contains(concat(' ', normalize-space(@class), ' '), ' contains_branch ')]" .
 334              "[p[contains(concat(' ', normalize-space(@class), ' '), ' branch ')]" .
 335              "/span[normalize-space(.)=" . $nodetextliteral ."]]" .
 336              "|" .
 337              "//div[contains(concat(' ', normalize-space(@class), ' '), ' content ')]/div" .
 338              "/ul[contains(concat(' ', normalize-space(@class), ' '), ' block_tree ')]" .
 339              "/li[p[contains(concat(' ', normalize-space(@class), ' '), ' branch ')]" .
 340              "/span[normalize-space(.)=" . $nodetextliteral ."]]";
 341              $node = $this->find('xpath', $xpath, $exception);

 346      /**
 347       * Helper function to get sub-navigation node.
 348       *
 349       * @throws ExpectationException if note not found.
 350       * @param string $nodetext node to find.
 351       * @param NodeElement $parentnode parent navigation node.
 352       * @return NodeElement.
 353       */
 354      protected function get_navigation_node($nodetext, $parentnode = null) {
 355  
 356          // Avoid problems with quotes.
 357          $nodetextliteral = $this->getSession()->getSelectorsHandler()->xpathLiteral($nodetext);
 358  
 359          $xpath = "/ul/li[contains(concat(' ', normalize-space(@class), ' '), ' contains_branch ')]" .
 360              "[child::p[contains(concat(' ', normalize-space(@class), ' '), ' branch ')]" .
 361              "/child::span[normalize-space(.)=" . $nodetextliteral ."]]";
 362          $node = $parentnode->find('xpath', $xpath);
 363          if (!$node) {
 364              $xpath = "/ul/li[contains(concat(' ', normalize-space(@class), ' '), ' contains_branch ')]" .
 365                  "[child::p[contains(concat(' ', normalize-space(@class), ' '), ' branch ')]" .
 366                  "/child::a[normalize-space(.)=" . $nodetextliteral ."]]";
 367              $node = $parentnode->find('xpath', $xpath);
 368          }
 369  
 370          if (!$node) {
 371              throw new ExpectationException('Sub-navigation node "' . $nodetext . '" not found under "' .
 372                  $parentnode->getText() . '"', $this->getSession());
 373          }
 374          return $node;
 375      }

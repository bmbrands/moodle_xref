  71      public function test_message_get_providers_for_user() {
  72          global $CFG, $DB;
  73  
  74          $this->resetAfterTest();
  75  
  76          $generator = $this->getDataGenerator();
  77  
  78          // Create a course category and course.
  79          $cat = $generator->create_category(array('parent' => 0));
  80          $course = $generator->create_course(array('category' => $cat->id));
  81          $quiz = $generator->create_module('quiz', array('course' => $course->id));
  82          $user = $generator->create_user();
  83  
  84          $coursecontext = context_course::instance($course->id);
  85          $quizcontext = context_module::instance($quiz->cmid);
  86          $frontpagecontext = context_course::instance(SITEID);
  87  
  88          $studentrole = $DB->get_record('role', array('shortname' => 'student'));
  89  
  90          // The user is a student in a course, and has the capability for quiz
  91          // confirmation emails in one quiz in that course.
  92          role_assign($studentrole->id, $user->id, $coursecontext->id);
  93          assign_capability('mod/quiz:emailconfirmsubmission', CAP_ALLOW, $studentrole->id, $quizcontext->id);
  94  
  95          // Give this message type to the front page role.
  96          assign_capability('mod/quiz:emailwarnoverdue', CAP_ALLOW, $CFG->defaultfrontpageroleid, $frontpagecontext->id);
  97  
  98          $providers = message_get_providers_for_user($user->id);
  99          $this->assertTrue($this->message_type_present('mod_forum', 'posts', $providers));
 100          $this->assertTrue($this->message_type_present('mod_quiz', 'confirmation', $providers));
 101          $this->assertTrue($this->message_type_present('mod_quiz', 'attempt_overdue', $providers));
 102          $this->assertFalse($this->message_type_present('mod_quiz', 'submission', $providers));
 103  
 104          // A user is a student in a different course, they should not get confirmation.
 105          $course2 = $generator->create_course(array('category' => $cat->id));
 106          $user2 = $generator->create_user();
 107          $coursecontext2 = context_course::instance($course2->id);
 108          role_assign($studentrole->id, $user2->id, $coursecontext2->id);
 109          accesslib_clear_all_caches_for_unit_testing();
 110          $providers = message_get_providers_for_user($user2->id);
 111          $this->assertTrue($this->message_type_present('mod_forum', 'posts', $providers));
 112          $this->assertFalse($this->message_type_present('mod_quiz', 'confirmation', $providers));
 113  
 114          // Now remove the frontpage role id, and attempt_overdue message should go away.
 115          unset_config('defaultfrontpageroleid');
 116          accesslib_clear_all_caches_for_unit_testing();
 117  
 118          $providers = message_get_providers_for_user($user->id);
 119          $this->assertTrue($this->message_type_present('mod_quiz', 'confirmation', $providers));
 120          $this->assertFalse($this->message_type_present('mod_quiz', 'attempt_overdue', $providers));
 121          $this->assertFalse($this->message_type_present('mod_quiz', 'submission', $providers));
 122      }

 124      public function test_message_get_providers_for_user_more() {
 125          global $DB;
 126  
 127          $this->resetAfterTest();
 128  
 129          // Create a course.
 130          $course = $this->getDataGenerator()->create_course();
 131          $coursecontext = context_course::instance($course->id);
 132  
 133          // It would probably be better to use a quiz instance as it has capability controlled messages
 134          // however mod_quiz doesn't have a data generator.
 135          // Instead we're going to use backup notifications and give and take away the capability at various levels.
 136          $assign = $this->getDataGenerator()->create_module('assign', array('course'=>$course->id));
 137          $modulecontext = context_module::instance($assign->cmid);
 138  
 139          // Create and enrol a teacher.
 140          $teacherrole = $DB->get_record('role', array('shortname'=>'editingteacher'), '*', MUST_EXIST);
 141          $teacher = $this->getDataGenerator()->create_user();
 142          role_assign($teacherrole->id, $teacher->id, $coursecontext);
 143          $enrolplugin = enrol_get_plugin('manual');
 144          $enrolplugin->add_instance($course);
 145          $enrolinstances = enrol_get_instances($course->id, false);
 146          foreach ($enrolinstances as $enrolinstance) {
 147              if ($enrolinstance->enrol === 'manual') {
 148                  break;
 149              }
 150          }
 151          $enrolplugin->enrol_user($enrolinstance, $teacher->id);
 152  
 153          // Make the teacher the current user.
 154          $this->setUser($teacher);
 155  
 156          // Teacher shouldn't have the required capability so they shouldn't be able to see the backup message.
 157          $this->assertFalse(has_capability('moodle/site:config', $modulecontext));
 158          $providers = message_get_providers_for_user($teacher->id);
 159          $this->assertFalse($this->message_type_present('moodle', 'backup', $providers));
 160  
 161          // Give the user the required capability in an activity module.
 162          // They should now be able to see the backup message.
 163          assign_capability('moodle/site:config', CAP_ALLOW, $teacherrole->id, $modulecontext->id, true);
 164          accesslib_clear_all_caches_for_unit_testing();
 165          $modulecontext = context_module::instance($assign->cmid);
 166          $this->assertTrue(has_capability('moodle/site:config', $modulecontext));
 167  
 168          $providers = message_get_providers_for_user($teacher->id);
 169          $this->assertTrue($this->message_type_present('moodle', 'backup', $providers));
 170  
 171          // Prohibit the capability for the user at the course level.
 172          // This overrules the CAP_ALLOW at the module level.
 173          // They should not be able to see the backup message.
 174          assign_capability('moodle/site:config', CAP_PROHIBIT, $teacherrole->id, $coursecontext->id, true);
 175          accesslib_clear_all_caches_for_unit_testing();
 176          $modulecontext = context_module::instance($assign->cmid);
 177          $this->assertFalse(has_capability('moodle/site:config', $modulecontext));
 178  
 179          $providers = message_get_providers_for_user($teacher->id);
 180          // Actually, handling PROHIBITs would be too expensive. We do not
 181          // care if users with PROHIBITs see a few more preferences than they should.
 182          // $this->assertFalse($this->message_type_present('moodle', 'backup', $providers));
 183      }

  36  class core_event_user_password_updated_testcase extends advanced_testcase {
  37      /**
  38       * Test the event.
  39       */
  40      public function test_event() {
  41          $this->resetAfterTest();
  42  
  43          $user1 = $this->getDataGenerator()->create_user();
  44          $context1 = context_user::instance($user1->id);
  45          $user2 = $this->getDataGenerator()->create_user();
  46          $context2 = context_user::instance($user2->id);
  47  
  48          $this->setUser($user1);
  49  
  50          // Changing own password.
  51          $event = \core\event\user_password_updated::create_from_user($user1);
  52          $this->assertEventContextNotUsed($event);
  53          $this->assertEquals($user1->id, $event->relateduserid);
  54          $this->assertSame($context1, $event->get_context());
  55          $this->assertEventLegacyLogData(null, $event);
  56          $this->assertFalse($event->other['forgottenreset']);
  57          $event->trigger();
  58  
  59          // Changing password of other user.
  60          $event = \core\event\user_password_updated::create_from_user($user2);
  61          $this->assertEventContextNotUsed($event);
  62          $this->assertEquals($user2->id, $event->relateduserid);
  63          $this->assertSame($context2, $event->get_context());
  64          $this->assertEventLegacyLogData(null, $event);
  65          $this->assertFalse($event->other['forgottenreset']);
  66          $event->trigger();
  67  
  68          // Password reset.
  69          $event = \core\event\user_password_updated::create_from_user($user1, true);
  70          $this->assertEventContextNotUsed($event);
  71          $this->assertEquals($user1->id, $event->relateduserid);
  72          $this->assertSame($context1, $event->get_context());
  73          $this->assertEventLegacyLogData(array(SITEID, 'user', 'set password', 'profile.php?id='.$user1->id, $user1->id), $event);
  74          $this->assertTrue($event->other['forgottenreset']);
  75          $event->trigger();
  76      }

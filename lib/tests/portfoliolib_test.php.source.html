 160      public static function file_output($file, $options = null) {
 161          if (isset($options['attributes']) && is_array($options['attributes'])) {
 162              $attributes = $options['attributes'];
 163          } else {
 164              $attributes = array();
 165          }
 166          $path = 'files/' . $file->get_filename();
 167          return self::make_tag($file, $path, $attributes);
 168      }

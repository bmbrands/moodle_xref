  38          $this->resetAfterTest();
  39          filter_set_global_state('emoticon', TEXTFILTER_ON);
  40          $this->assertRegExp('~^<p><img class="emoticon" alt="smile" ([^>]+)></p>$~',
  41                  format_text('<p>:-)</p>', FORMAT_HTML));
  42      }

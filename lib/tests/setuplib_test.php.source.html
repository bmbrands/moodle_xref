 257          $firstdir = get_request_storage_directory();
 258          $seconddir = get_request_storage_directory();
 259          $this->assertTrue(is_dir($firstdir));
 260          $this->assertEquals($firstdir, $seconddir);
 261  
 262          // Removing the directory and calling get_request_storage_directory() again should cause a new directory to be created.
 263          remove_dir($firstdir);
 264          $this->assertFalse(file_exists($firstdir));
 265          $this->assertFalse(is_dir($firstdir));
 266  
 267          $thirddir = get_request_storage_directory();
 268          $this->assertTrue(is_dir($thirddir));
 269          $this->assertNotEquals($firstdir, $thirddir);
 270  
 271          // Removing it and replacing it with a file should cause it to be regenerated again.
 272          remove_dir($thirddir);
 273          $this->assertFalse(file_exists($thirddir));
 274          $this->assertFalse(is_dir($thirddir));
 275          touch($thirddir);
 276          $this->assertTrue(file_exists($thirddir));
 277          $this->assertFalse(is_dir($thirddir));
 278  
 279          $fourthdir = get_request_storage_directory();
 280          $this->assertTrue(is_dir($fourthdir));
 281          $this->assertNotEquals($thirddir, $fourthdir);

 364              'course' => '2',
 365              'action' => 'create',
 366              'grade' => array(
 367                  0 => 'a',
 368                  1 => 'b',
 369                  2 => 'c',
 370                  3 => 'e',
 371                  4 => 'f',
 372                  5 => 'g',
 373              ),
 374              'items' => array(
 375                  'a' => 'aa',
 376                  'b' => 'bb',
 377              ),
 378              'mix' => 'mix',
 379              'numerical' => array(
 380                  '2' => array('a' => 'b', 'd' => 'e'),
 381                  '1' => '2',
 382                  '0' => 'z',
 383              ),
 384              'next' => '2',
 385          );
 386  
 387          $array = $original;
 388          merge_query_params($array, $chunk);

 109      /**

 117      process_fieldset: function(fieldset) {
 118          // Get legend element.
 119          var legendelement = fieldset.one(SELECTORS.LEGENDFTOGGLER);
 120  
 121          // Turn headers to links for accessibility.
 122          var headerlink = Y.Node.create('<a href="#"></a>');
 123          headerlink.addClass(CSS.FHEADER);
 124          headerlink.appendChild(legendelement.get('firstChild'));
 125          headerlink.setAttribute('role', 'button');
 126          headerlink.setAttribute('aria-controls', fieldset.generateID());
 127          if (legendelement.ancestor(SELECTORS.COLLAPSED)) {
 128              headerlink.setAttribute('aria-expanded', 'false');
 129          } else {
 130              headerlink.setAttribute('aria-expanded', 'true');
 131          }
 132          legendelement.prepend(headerlink);

 137      /**

 145      set_state: function(fieldset, collapsed) {
 146          var headerlink = fieldset.one(SELECTORS.FHEADER);
 147          if (collapsed) {
 148              fieldset.addClass(CSS.COLLAPSED);
 149              if (headerlink) {

 151              }
 152          } else {
 153              fieldset.removeClass(CSS.COLLAPSED);
 154              if (headerlink) {
 155                  headerlink.setAttribute('aria-expanded', 'true');

 159          if (!statuselement) {
 160              return this;
 161          }
 162          statuselement.set('value', collapsed ? 0 : 1);
 163

 213          // Counting the number of collapsed sections.
 214          fieldlist = this.form.all(SELECTORS.FIELDSETCOLLAPSIBLE);

 232          }
 233  
 234          return this;

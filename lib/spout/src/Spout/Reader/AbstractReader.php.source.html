  64       *
  65       * @api
  66       * @param  string $filePath Path of the file to be read

  70      public function open($filePath)
  71      {
  72          if ($this->isStreamWrapper($filePath) && (!$this->doesSupportStreamWrapper() || !$this->isSupportedStreamWrapper($filePath))) {
  73              throw new IOException("Could not open $filePath for reading! Stream wrapper used is not supported for this type of file.");
  74          }
  75  
  76          if (!$this->isPhpStream($filePath)) {
  77              // we skip the checks if the provided file path points to a PHP stream
  78              if (!$this->globalFunctionsHelper->file_exists($filePath)) {
  79                  throw new IOException("Could not open $filePath for reading! File does not exist.");
  80              } else if (!$this->globalFunctionsHelper->is_readable($filePath)) {
  81                  throw new IOException("Could not open $filePath for reading! File is not readable.");
  82              }
  83          }

  86              $fileRealPath = $this->getFileRealPath($filePath);
  87              $this->openReader($fileRealPath);
  88              $this->isStreamOpened = true;

  90              throw new IOException("Could not open $filePath for reading! ({$exception->getMessage()})");
  91          }
  92      }

 101      protected function getFileRealPath($filePath)
 102      {
 103          if ($this->isSupportedStreamWrapper($filePath)) {

 111      /**

 113       * For example, php://temp => php, s3://path/to/file => s3...

 118      protected function getStreamWrapperScheme($filePath)
 119      {
 120          $streamScheme = null;
 121          if (preg_match('/^(\w+):\/\//', $filePath, $matches)) {
 122              $streamScheme = $matches[1];
 123          }
 124          return $streamScheme;

 139      /**

 141       * (like php://temp, mystream://foo/bar...).
 142       * If the given path is a local path, returns true.
 143       *
 144       * @param string $filePath Path of the file to be read
 145       * @return bool Whether the given path is an supported stream wrapper

 147      protected function isSupportedStreamWrapper($filePath)
 148      {
 149          $streamScheme = $this->getStreamWrapperScheme($filePath);
 150          return ($streamScheme !== null) ?
 151              in_array($streamScheme, $this->globalFunctionsHelper->stream_get_wrappers()) :
 152              true;
 153      }

 161      protected function isPhpStream($filePath)
 162      {
 163          $streamScheme = $this->getStreamWrapperScheme($filePath);
 164          return ($streamScheme === 'php');
 165      }

 174      public function getSheetIterator()
 175      {

 178          }
 179

 184       * Closes the reader, preventing any additional reading
 185       *
 186       * @api
 187       * @return void
 188       */
 189      public function close()
 190      {
 191          if ($this->isStreamOpened) {
 192              $this->closeReader();
 193  
 194              $sheetIterator = $this->getConcreteSheetIterator();
 195              if ($sheetIterator) {
 196                  $sheetIterator->end();
 197              }
 198  
 199              $this->isStreamOpened = false;
 200          }
 201      }

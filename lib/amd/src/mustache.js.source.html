  93    /**
  94     * Breaks up the given `template` string into a tree of tokens. If the `tags`
  95     * argument is given here it must be an array with two string values: the
  96     * opening and closing tags used in the template (e.g. [ "<%", "%>" ]). Of
  97     * course, the default is to use mustaches (i.e. mustache.tags).
  98     *
  99     * A token is an array with at least 4 elements. The first element is the
 100     * mustache symbol that was used inside the tag, e.g. "#" or "&". If the tag
 101     * did not contain a symbol (i.e. {{myValue}}) this element is "name". For
 102     * all text that appears outside a symbol this element is "text".
 103     *
 104     * The second element of a token is its "value". For mustache tags this is
 105     * whatever else was inside the tag besides the opening symbol. For text tokens
 106     * this is the text itself.
 107     *
 108     * The third and fourth elements of the token are the start and end indices,
 109     * respectively, of the token in the original template.
 110     *
 111     * Tokens that are the root node of a subtree contain two more elements: 1) an
 112     * array of tokens in the subtree and 2) the index in the original template at
 113     * which the closing tag for that section begins.
 114     */
 115    function parseTemplate(template, tags) {
 116      if (!template)
 117        return [];
 118  
 119      var sections = [];     // Stack to hold section tokens
 120      var tokens = [];       // Buffer to hold the tokens
 121      var spaces = [];       // Indices of whitespace tokens on the current line
 122      var hasTag = false;    // Is there a {{tag}} on the current line?
 123      var nonSpace = false;  // Is there a non-space char on the current line?
 124  
 125      // Strips all whitespace tokens array for the current line
 126      // if there was a {{#tag}} on it and otherwise only space.
 127      function stripSpace() {
 128        if (hasTag && !nonSpace) {
 129          while (spaces.length)
 130            delete tokens[spaces.pop()];
 131        } else {
 132          spaces = [];
 133        }
 134  
 135        hasTag = false;
 136        nonSpace = false;
 137      }
 138  
 139      var openingTagRe, closingTagRe, closingCurlyRe;
 140      function compileTags(tags) {
 141        if (typeof tags === 'string')
 142          tags = tags.split(spaceRe, 2);
 143  
 144        if (!isArray(tags) || tags.length !== 2)
 145          throw new Error('Invalid tags: ' + tags);
 146  
 147        openingTagRe = new RegExp(escapeRegExp(tags[0]) + '\\s*');
 148        closingTagRe = new RegExp('\\s*' + escapeRegExp(tags[1]));
 149        closingCurlyRe = new RegExp('\\s*' + escapeRegExp('}' + tags[1]));
 150      }
 151  
 152      compileTags(tags || mustache.tags);
 153  
 154      var scanner = new Scanner(template);
 155  
 156      var start, type, value, chr, token, openSection;
 157      while (!scanner.eos()) {
 158        start = scanner.pos;
 159  
 160        // Match any text between tags.
 161        value = scanner.scanUntil(openingTagRe);
 162  
 163        if (value) {
 164          for (var i = 0, valueLength = value.length; i < valueLength; ++i) {
 165            chr = value.charAt(i);
 166  
 167            if (isWhitespace(chr)) {
 168              spaces.push(tokens.length);
 169            } else {
 170              nonSpace = true;
 171            }
 172  
 173            tokens.push([ 'text', chr, start, start + 1 ]);
 174            start += 1;
 175  
 176            // Check for whitespace on the current line.
 177            if (chr === '\n')
 178              stripSpace();
 179          }
 180        }
 181  
 182        // Match the opening tag.
 183        if (!scanner.scan(openingTagRe))
 184          break;
 185  
 186        hasTag = true;
 187  
 188        // Get the tag type.
 189        type = scanner.scan(tagRe) || 'name';
 190        scanner.scan(whiteRe);
 191  
 192        // Get the tag value.
 193        if (type === '=') {
 194          value = scanner.scanUntil(equalsRe);
 195          scanner.scan(equalsRe);
 196          scanner.scanUntil(closingTagRe);
 197        } else if (type === '{') {
 198          value = scanner.scanUntil(closingCurlyRe);
 199          scanner.scan(curlyRe);
 200          scanner.scanUntil(closingTagRe);
 201          type = '&';
 202        } else {
 203          value = scanner.scanUntil(closingTagRe);
 204        }
 205  
 206        // Match the closing tag.
 207        if (!scanner.scan(closingTagRe))
 208          throw new Error('Unclosed tag at ' + scanner.pos);
 209  
 210        token = [ type, value, start, scanner.pos ];
 211        tokens.push(token);
 212  
 213        if (type === '#' || type === '^') {
 214          sections.push(token);
 215        } else if (type === '/') {
 216          // Check section nesting.
 217          openSection = sections.pop();
 218  
 219          if (!openSection)
 220            throw new Error('Unopened section "' + value + '" at ' + start);
 221  
 222          if (openSection[1] !== value)
 223            throw new Error('Unclosed section "' + openSection[1] + '" at ' + start);
 224        } else if (type === 'name' || type === '{' || type === '&') {
 225          nonSpace = true;
 226        } else if (type === '=') {
 227          // Set the tags for the next time around.
 228          compileTags(value);
 229        }
 230      }
 231  
 232      // Make sure there are no open sections when we're done.
 233      openSection = sections.pop();
 234  
 235      if (openSection)
 236        throw new Error('Unclosed section "' + openSection[1] + '" at ' + scanner.pos);
 237  
 238      return nestTokens(squashTokens(tokens));
 239    }

 241    /**
 242     * Combines the values of consecutive text tokens in the given `tokens` array
 243     * to a single token.
 244     */
 245    function squashTokens(tokens) {
 246      var squashedTokens = [];
 247  
 248      var token, lastToken;
 249      for (var i = 0, numTokens = tokens.length; i < numTokens; ++i) {
 250        token = tokens[i];
 251  
 252        if (token) {
 253          if (token[0] === 'text' && lastToken && lastToken[0] === 'text') {
 254            lastToken[1] += token[1];
 255            lastToken[3] = token[3];
 256          } else {
 257            squashedTokens.push(token);
 258            lastToken = token;
 259          }
 260        }
 261      }
 262  
 263      return squashedTokens;
 264    }

 266    /**
 267     * Forms the given array of `tokens` into a nested tree structure where
 268     * tokens that represent a section have two additional items: 1) an array of
 269     * all tokens that appear in that section and 2) the index in the original
 270     * template that represents the end of that section.
 271     */
 272    function nestTokens(tokens) {
 273      var nestedTokens = [];
 274      var collector = nestedTokens;
 275      var sections = [];
 276  
 277      var token, section;
 278      for (var i = 0, numTokens = tokens.length; i < numTokens; ++i) {
 279        token = tokens[i];
 280  
 281        switch (token[0]) {
 282        case '#':
 283        case '^':
 284          collector.push(token);
 285          sections.push(token);
 286          collector = token[4] = [];
 287          break;
 288        case '/':
 289          section = sections.pop();
 290          section[5] = token[2];
 291          collector = sections.length > 0 ? sections[sections.length - 1][4] : nestedTokens;
 292          break;
 293        default:
 294          collector.push(token);
 295        }
 296      }
 297  
 298      return nestedTokens;
 299    }

 318    /**
 319     * Tries to match the given regular expression at the current position.
 320     * Returns the matched text if it can match, the empty string otherwise.
 321     */
 322    Scanner.prototype.scan = function (re) {
 323      var match = this.tail.match(re);
 324  
 325      if (!match || match.index !== 0)
 326        return '';
 327  
 328      var string = match[0];
 329  
 330      this.tail = this.tail.substring(string.length);
 331      this.pos += string.length;
 332  
 333      return string;
 334    };
 335  
 336    /**
 337     * Skips all text until the given regular expression can be matched. Returns
 338     * the skipped string, which is the entire tail if no match can be made.
 339     */
 340    Scanner.prototype.scanUntil = function (re) {
 341      var index = this.tail.search(re), match;
 342  
 343      switch (index) {
 344      case -1:
 345        match = this.tail;
 346        this.tail = "";
 347        break;
 348      case 0:
 349        match = "";
 350        break;
 351      default:
 352        match = this.tail.substring(0, index);
 353        this.tail = this.tail.substring(index);
 354      }
 355  
 356      this.pos += match.length;
 357  
 358      return match;
 359    };

 379    /**
 380     * Returns the value of the given name in this context, traversing
 381     * up the context hierarchy if the value is absent in this context's view.
 382     */
 383    Context.prototype.lookup = function (name) {
 384      var cache = this.cache;
 385  
 386      var value;
 387      if (name in cache) {
 388        value = cache[name];
 389      } else {
 390        var context = this, names, index;
 391  
 392        while (context) {
 393          if (name.indexOf('.') > 0) {
 394            value = context.view;
 395            names = name.split('.');
 396            index = 0;
 397  
 398            while (value != null && index < names.length)
 399              value = value[names[index++]];
 400          } else if (typeof context.view == 'object') {
 401            value = context.view[name];
 402          }
 403  
 404          if (value != null)
 405            break;
 406  
 407          context = context.parent;
 408        }
 409  
 410        cache[name] = value;
 411      }
 412  
 413      if (isFunction(value))
 414        value = value.call(this.view);
 415  
 416      return value;
 417    };

 449    /**
 450     * High-level method that is used to render the given `template` with
 451     * the given `view`.
 452     *
 453     * The optional `partials` argument may be an object that contains the
 454     * names and templates of partials that are used in the template. It may
 455     * also be a function that is used to load partial templates on the fly
 456     * that takes a single argument: the name of the partial.
 457     */
 458    Writer.prototype.render = function (template, view, partials) {
 459      var tokens = this.parse(template);
 460      var context = (view instanceof Context) ? view : new Context(view);
 461      return this.renderTokens(tokens, context, partials, template);
 462    };
 463  
 464    /**
 465     * Low-level method that renders the given array of `tokens` using
 466     * the given `context` and `partials`.
 467     *
 468     * Note: The `originalTemplate` is only ever used to extract the portion
 469     * of the original template that was contained in a higher-order section.
 470     * If the template doesn't use higher-order sections, this argument may
 471     * be omitted.
 472     */
 473    Writer.prototype.renderTokens = function (tokens, context, partials, originalTemplate) {
 474      var buffer = '';
 475  
 476      // This function is used to render an arbitrary template
 477      // in the current context by higher-order sections.
 478      var self = this;
 479      function subRender(template) {
 480        return self.render(template, context, partials);
 481      }
 482  
 483      var token, value;
 484      for (var i = 0, numTokens = tokens.length; i < numTokens; ++i) {
 485        token = tokens[i];
 486  
 487        switch (token[0]) {
 488        case '#':
 489          value = context.lookup(token[1]);
 490  
 491          if (!value)
 492            continue;
 493  
 494          if (isArray(value)) {
 495            for (var j = 0, valueLength = value.length; j < valueLength; ++j) {
 496              buffer += this.renderTokens(token[4], context.push(value[j]), partials, originalTemplate);
 497            }
 498          } else if (typeof value === 'object' || typeof value === 'string') {
 499            buffer += this.renderTokens(token[4], context.push(value), partials, originalTemplate);
 500          } else if (isFunction(value)) {
 501            if (typeof originalTemplate !== 'string')
 502              throw new Error('Cannot use higher-order sections without the original template');
 503  
 504            // Extract the portion of the original template that the section contains.
 505            value = value.call(context.view, originalTemplate.slice(token[3], token[5]), subRender);
 506  
 507            if (value != null)
 508              buffer += value;
 509          } else {
 510            buffer += this.renderTokens(token[4], context, partials, originalTemplate);
 511          }
 512  
 513          break;
 514        case '^':
 515          value = context.lookup(token[1]);
 516  
 517          // Use JavaScript's definition of falsy. Include empty arrays.
 518          // See https://github.com/janl/mustache.js/issues/186
 519          if (!value || (isArray(value) && value.length === 0))
 520            buffer += this.renderTokens(token[4], context, partials, originalTemplate);
 521  
 522          break;
 523        case '>':
 524          if (!partials)
 525            continue;
 526  
 527          value = isFunction(partials) ? partials(token[1]) : partials[token[1]];
 528  
 529          if (value != null)
 530            buffer += this.renderTokens(this.parse(value), context, partials, value);
 531  
 532          break;
 533        case '&':
 534          value = context.lookup(token[1]);
 535  
 536          if (value != null)
 537            buffer += value;
 538  
 539          break;
 540        case 'name':
 541          value = context.lookup(token[1]);
 542  
 543          if (value != null)
 544            buffer += mustache.escape(value);
 545  
 546          break;
 547        case 'text':
 548          buffer += token[1];
 549          break;
 550        }
 551      }
 552  
 553      return buffer;
 554    };

  43      var noop = function() {};
  44      var undefinedType = "undefined";

  46      function realMethod(methodName) {
  47          if (typeof console === undefinedType) {
  48              return false; // We can't build a real method without a console to log to

  49          } else if (console[methodName] !== undefined) {
  50              return bindMethod(console, methodName);
  51          } else if (console.log !== undefined) {
  52              return bindMethod(console, 'log');
  53          } else {
  54              return noop;
  55          }
  56      }

  58      function bindMethod(obj, methodName) {
  59          var method = obj[methodName];
  60          if (typeof method.bind === 'function') {
  61              return method.bind(obj);
  62          } else {
  63              try {
  64                  return Function.prototype.bind.call(method, obj);
  65              } catch (e) {
  66                  // Missing bind shim or IE8 + Modernizr, fallback to wrapping

  67                  return function() {
  68                      return Function.prototype.apply.apply(method, [obj, arguments]);
  69                  };
  70              }
  71          }
  72      }

  74      function enableLoggingWhenConsoleArrives(methodName, level) {
  75          return function () {
  76              if (typeof console !== undefinedType) {
  77                  replaceLoggingMethods(level);
  78                  self[methodName].apply(self, arguments);
  79              }
  80          };
  81      }

  83      var logMethods = [
  84          "trace",
  85          "debug",
  86          "info",
  87          "warn",
  88          "error"

  91      function replaceLoggingMethods(level) {
  92          for (var i = 0; i < logMethods.length; i++) {
  93              var methodName = logMethods[i];
  94              self[methodName] = (i < level) ? noop : self.methodFactory(methodName, level);
  95          }
  96      }

  98      function persistLevelIfPossible(levelNum) {
  99          var levelName = (logMethods[levelNum] || 'silent').toUpperCase();
 100  
 101          // Use localStorage if available

 102          try {
 103              window.localStorage['loglevel'] = levelName;
 104              return;
 105          } catch (ignore) {}
 106  
 107          // Use session cookie as fallback

 108          try {
 109              window.document.cookie = "loglevel=" + levelName + ";";
 110          } catch (ignore) {}
 111      }

 113      function loadPersistedLevel() {
 114          var storedLevel;
 115  
 116          try {
 117              storedLevel = window.localStorage['loglevel'];
 118          } catch (ignore) {}
 119  
 120          if (typeof storedLevel === undefinedType) {
 121              try {
 122                  storedLevel = /loglevel=([^;]+)/.exec(window.document.cookie)[1];
 123              } catch (ignore) {}
 124          }
 125          
 126          if (self.levels[storedLevel] === undefined) {

 142      self.methodFactory = function (methodName, level) {
 143          return realMethod(methodName) ||
 144                 enableLoggingWhenConsoleArrives(methodName, level);
 145      };

 147      self.setLevel = function (level) {
 148          if (typeof level === "string" && self.levels[level.toUpperCase()] !== undefined) {
 149              level = self.levels[level.toUpperCase()];
 150          }
 151          if (typeof level === "number" && level >= 0 && level <= self.levels.SILENT) {
 152              persistLevelIfPossible(level);
 153              replaceLoggingMethods(level);
 154              if (typeof console === undefinedType && level < self.levels.SILENT) {
 155                  return "No console available for logging";
 156              }
 157          } else {
 158              throw "log.setLevel() called with invalid level: " + level;
 159          }
 160      };

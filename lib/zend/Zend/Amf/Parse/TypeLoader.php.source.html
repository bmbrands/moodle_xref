  96      public static function loadType($className)
  97      {
  98          $class    = self::getMappedClassName($className);
  99          if(!$class) {
 100              $class = str_replace('.', '_', $className);
 101          }
 102          if (!class_exists($class)) {
 103              return "stdClass";
 104          }
 105          return $class;

 108      /**

 114      public static function getMappedClassName($className)
 115      {
 116          $mappedName = array_search($className, self::$classMap);
 117  
 118          if ($mappedName) {
 119              return $mappedName;
 120          }
 121  
 122          $mappedName = array_search($className, array_flip(self::$classMap));
 123  
 124          if ($mappedName) {
 125              return $mappedName;
 126          }
 127  
 128          return false;

 188          if(self::$_resourceLoader) {

 190              $type = str_replace(" ","", ucwords($type));

 196      /**
 197       * Convert resource to a serializable object
 198       *
 199       * @param resource $resource
 200       * @return mixed
 201       */
 202      public static function handleResource($resource)
 203      {
 204          if(!self::$_resourceLoader) {
 205              require_once 'Zend/Amf/Exception.php';
 206              throw new Zend_Amf_Exception('Unable to handle resources - resource plugin loader not set');
 207          }
 208          try {
 209              while(is_resource($resource)) {
 210                  $resclass = self::getResourceParser($resource);
 211                  if(!$resclass) {
 212                      require_once 'Zend/Amf/Exception.php';
 213                      throw new Zend_Amf_Exception('Can not serialize resource type: '. get_resource_type($resource));
 214                  }
 215                  $parser = new $resclass();
 216                  if(is_callable(array($parser, 'parse'))) {
 217                      $resource = $parser->parse($resource);
 218                  } else {
 219                      require_once 'Zend/Amf/Exception.php';
 220                      throw new Zend_Amf_Exception("Could not call parse() method on class $resclass");
 221                  }
 222              }
 223              return $resource;
 224          } catch(Zend_Amf_Exception $e) {
 225              throw new Zend_Amf_Exception($e->getMessage(), $e->getCode(), $e);
 226          } catch(Exception $e) {
 227              require_once 'Zend/Amf/Exception.php';
 228              throw new Zend_Amf_Exception('Can not serialize resource type: '. get_resource_type($resource), 0, $e);
 229          }
 230      }

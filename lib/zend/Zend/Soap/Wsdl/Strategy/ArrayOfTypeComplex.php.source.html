  37  class Zend_Soap_Wsdl_Strategy_ArrayOfTypeComplex extends Zend_Soap_Wsdl_Strategy_DefaultComplexType
  38  {
  39      protected $_inProcess = array();
  40  
  41      /**
  42       * Add an ArrayOfType based on the xsd:complexType syntax if type[] is detected in return value doc comment.
  43       *
  44       * @param string $type
  45       * @return string tns:xsd-type
  46       */
  47      public function addComplexType($type)
  48      {
  49          if(in_array($type, $this->_inProcess)) {
  50              require_once "Zend/Soap/Wsdl/Exception.php";
  51              throw new Zend_Soap_Wsdl_Exception("Infinite recursion, cannot nest '".$type."' into itself.");
  52          }
  53          $this->_inProcess[$type] = $type;
  54  
  55          $nestingLevel = $this->_getNestedCount($type);
  56  
  57          if($nestingLevel > 1) {
  58              require_once "Zend/Soap/Wsdl/Exception.php";
  59              throw new Zend_Soap_Wsdl_Exception(
  60                  "ArrayOfTypeComplex cannot return nested ArrayOfObject deeper than ".
  61                  "one level. Use array object properties to return deep nested data.
  62              ");
  63          }
  64  
  65          $singularType = $this->_getSingularPhpType($type);
  66  
  67          if(!class_exists($singularType)) {
  68              require_once "Zend/Soap/Wsdl/Exception.php";
  69              throw new Zend_Soap_Wsdl_Exception(sprintf(
  70                  "Cannot add a complex type %s that is not an object or where ".
  71                  "class could not be found in 'DefaultComplexType' strategy.", $type
  72              ));
  73          }
  74  
  75          if($nestingLevel == 1) {
  76              // The following blocks define the Array of Object structure
  77              $xsdComplexTypeName = $this->_addArrayOfComplexType($singularType, $type);
  78          } else {
  79              $xsdComplexTypeName = $singularType;
  80          }
  81  
  82          // The array for the objects has been created, now build the object definition:
  83          if(!in_array($singularType, $this->getContext()->getTypes())) {
  84              parent::addComplexType($singularType);
  85          }
  86  
  87          unset($this->_inProcess[$type]);
  88          return "tns:".$xsdComplexTypeName;
  89      }
  90  
  91      protected function _addArrayOfComplexType($singularType, $type)
  92      {
  93          $dom = $this->getContext()->toDomDocument();
  94  
  95          $xsdComplexTypeName = $this->_getXsdComplexTypeName($singularType);
  96  
  97          if(!in_array($xsdComplexTypeName, $this->getContext()->getTypes())) {
  98              $complexType = $dom->createElement('xsd:complexType');
  99              $complexType->setAttribute('name', $xsdComplexTypeName);
 100  
 101              $complexContent = $dom->createElement("xsd:complexContent");
 102              $complexType->appendChild($complexContent);
 103  
 104              $xsdRestriction = $dom->createElement("xsd:restriction");
 105              $xsdRestriction->setAttribute('base', 'soap-enc:Array');
 106              $complexContent->appendChild($xsdRestriction);
 107  
 108              $xsdAttribute = $dom->createElement("xsd:attribute");
 109              $xsdAttribute->setAttribute("ref", "soap-enc:arrayType");
 110              $xsdAttribute->setAttribute("wsdl:arrayType", sprintf("tns:%s[]", $singularType));
 111              $xsdRestriction->appendChild($xsdAttribute);
 112  
 113              $this->getContext()->getSchema()->appendChild($complexType);
 114              $this->getContext()->addType($xsdComplexTypeName);
 115          }
 116  
 117          return $xsdComplexTypeName;
 118      }
 119  
 120      protected function _getXsdComplexTypeName($type)
 121      {
 122          return sprintf('ArrayOf%s', $type);
 123      }

 125      /**
 126       * From a nested definition with type[], get the singular PHP Type
 127       *
 128       * @param  string $type
 129       * @return string
 130       */
 131      protected function _getSingularPhpType($type)
 132      {
 133          return str_replace("[]", "", $type);
 134      }

 137       * Return the array nesting level based on the type name
 138       *
 139       * @param  string $type
 140       * @return integer
 141       */
 142      protected function _getNestedCount($type)
 143      {
 144          return substr_count($type, "[]");
 145      }

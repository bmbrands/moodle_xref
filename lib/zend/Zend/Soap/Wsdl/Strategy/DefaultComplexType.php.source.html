  37  class Zend_Soap_Wsdl_Strategy_DefaultComplexType extends Zend_Soap_Wsdl_Strategy_Abstract
  38  {
  39      /**
  40       * Add a complex type by recursivly using all the class properties fetched via Reflection.
  41       *
  42       * @param  string $type Name of the class to be specified
  43       * @return string XSD Type for the given PHP type
  44       */
  45      public function addComplexType($type)
  46      {
  47          if(!class_exists($type)) {
  48              require_once "Zend/Soap/Wsdl/Exception.php";
  49              throw new Zend_Soap_Wsdl_Exception(sprintf(
  50                  "Cannot add a complex type %s that is not an object or where ".
  51                  "class could not be found in 'DefaultComplexType' strategy.", $type
  52              ));
  53          }
  54  
  55          $dom = $this->getContext()->toDomDocument();
  56          $class = new ReflectionClass($type);
  57  
  58          $complexType = $dom->createElement('xsd:complexType');
  59          $complexType->setAttribute('name', $type);
  60  
  61          $all = $dom->createElement('xsd:all');
  62  
  63          foreach ($class->getProperties() as $property) {
  64              if ($property->isPublic() && preg_match_all('/@var\s+([^\s]+)/m', $property->getDocComment(), $matches)) {
  65  
  66                  /**
  67                   * @todo check if 'xsd:element' must be used here (it may not be compatible with using 'complexType'
  68                   * node for describing other classes used as attribute types for current class
  69                   */
  70                  $element = $dom->createElement('xsd:element');
  71                  $element->setAttribute('name', $property->getName());
  72                  $element->setAttribute('type', $this->getContext()->getType(trim($matches[1][0])));
  73                  $all->appendChild($element);
  74              }
  75          }
  76  
  77          $complexType->appendChild($all);
  78          $this->getContext()->getSchema()->appendChild($complexType);
  79          $this->getContext()->addType($type);
  80  
  81          return "tns:$type";
  82      }
  83  }

 290      /**
 291       * Returns the location from the selected city
 292       *
 293       * @param  string $city    City to get location for
 294       * @param  string $horizon Horizon to use :
 295       *                         default: effective
 296       *                         others are civil, nautic, astronomic
 297       * @return array
 298       * @throws Zend_Date_Exception When city is unknown
 299       */
 300      public static function City($city, $horizon = false)
 301      {
 302          foreach (self::$cities as $key => $value) {
 303              if (strtolower($key) === strtolower($city)) {
 304                  $return            = $value;
 305                  $return['horizon'] = $horizon;
 306                  return $return;
 307              }
 308          }
 309          require_once 'Zend/Date/Exception.php';
 310          throw new Zend_Date_Exception('unknown city');
 311      }

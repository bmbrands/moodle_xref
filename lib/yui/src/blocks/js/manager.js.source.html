 153       * Enables all fo the regions so that they are all visible while dragging is occuring.

 155       * @method enable_all_regions

 157      enable_all_regions : function() {
 158          var groups = Y.DD.DDM.activeDrag.get('groups');
 159  
 160          // As we're called by Y.DD.DDM, we can't be certain that the call
 161          // relates specifically to a block drag/drop operation. Test
 162          // whether the relevant group applies here.
 163          if (!groups || Y.Array.indexOf(groups, 'block') === -1) {
 164              return;

 168          for (i in this.regionobjects) {
 169              if (!this.regionobjects.hasOwnProperty(i)) {
 170                  continue;

 172              this.regionobjects[i].enable();

 192      drag_start : function(e) {

 213      drop_over : function(e) {
 214          // Get a reference to our drag and drop nodes
 215          var drag = e.drag.get('node');
 216          var drop = e.drop.get('node');
 217  
 218          // We need to fix the case when parent drop over event has determined
 219          // 'goingup' and appended the drag node after admin-block.
 220          if (drop.hasClass(CSS.REGIONCONTENT) &&
 221                  drop.one('.'+CSS.BLOCKADMINBLOCK) &&
 222                  drop.one('.'+CSS.BLOCKADMINBLOCK).next('.'+CSS.BLOCK)) {
 223              drop.prepend(drag);

 252      /**
 253       * Called by M.core.dragdrop.global_drag_hit when something has been dropped on a drop target.
 254       * @method drop_hit
 255       * @param {Event} e
 256       */
 257      drop_hit : function(e) {
 258          // Get a reference to our drag node
 259          var dragnode = e.drag.get('node');
 260          var dropnode = e.drop.get('node');
 261  
 262          // Amend existing skipnodes
 263          if (dragnode.previous() && dragnode.previous().hasClass(CSS.SKIPBLOCK)) {
 264              // the one that belongs to block below move below
 265              dragnode.insert(dragnode.previous(), 'after');
 266          }
 267          // Move original skipnodes
 268          if (this.skipnodetop) {
 269              dragnode.insert(this.skipnodetop, 'before');
 270          }
 271          if (this.skipnodebottom) {
 272              dragnode.insert(this.skipnodebottom, 'after');
 273          }
 274  
 275          // Add lightbox if it not there
 276          var lightbox = M.util.add_lightbox(Y, dragnode);
 277  
 278          // Prepare request parameters
 279          var params = {
 280              sesskey : M.cfg.sesskey,
 281              courseid : this.get('courseid'),
 282              pagelayout : this.get('pagelayout'),
 283              pagetype : this.get('pagetype'),
 284              subpage : this.get('subpage'),
 285              contextid : this.get('contextid'),
 286              action : 'move',
 287              bui_moveid : this.get_block_id(dragnode),
 288              bui_newregion : this.get_block_region(dropnode)
 289          };
 290  
 291          if (this.get('cmid')) {
 292              params.cmid = this.get('cmid');
 293          }
 294  
 295          if (dragnode.next('.'+CSS.BLOCK) && !dragnode.next('.'+CSS.BLOCK).hasClass(CSS.BLOCKADMINBLOCK)) {
 296              params.bui_beforeid = this.get_block_id(dragnode.next('.'+CSS.BLOCK));
 297          }
 298  
 299          // Do AJAX request
 300          Y.io(M.cfg.wwwroot+AJAXURL, {
 301              method: 'POST',
 302              data: params,
 303              on: {
 304                  start : function() {
 305                      lightbox.show();
 306                  },
 307                  success: function(tid, response) {
 308                      window.setTimeout(function() {
 309                          lightbox.hide();
 310                      }, 250);
 311                      try {
 312                          var responsetext = Y.JSON.parse(response.responseText);
 313                          if (responsetext.error) {
 314                              new M.core.ajaxException(responsetext);
 315                          }
 316                      } catch (e) {}
 317                  },
 318                  failure: function(tid, response) {
 319                      this.ajax_failure(response);
 320                      lightbox.hide();
 321                  },
 322                  complete : function() {
 323                      this.disable_regions_if_required();
 324                  }
 325              },
 326              context:this
 327          });
 328      }

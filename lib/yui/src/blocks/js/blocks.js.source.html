 173      drop_over : function(e) {
 174          // Get a reference to our drag and drop nodes
 175          var drag = e.drag.get('node');
 176          var drop = e.drop.get('node');
 177  
 178          // We need to fix the case when parent drop over event has determined
 179          // 'goingup' and appended the drag node after admin-block.
 180          if (drop.hasClass(this.parentnodeclass) &&
 181                  drop.one('.'+CSS.BLOCKADMINBLOCK) &&
 182                  drop.one('.'+CSS.BLOCKADMINBLOCK).next('.'+CSS.BLOCK)) {
 183              drop.prepend(drag);
 184          }
 185  
 186          // Block is moved within the same region
 187          // stop here, no need to modify anything.
 188          if (this.dragsourceregion.contains(drop)) {
 189              return false;
 190          }
 191  
 192          // TODO: Hiding-displaying block region only works for base theme blocks
 193          // (region-pre, region-post) at the moment. It should be improved
 194          // to work with custom block regions as well.
 195  
 196          // TODO: Fix this for the case when user drag block towards empty section,
 197          // then the section appears, then user chnages his mind and moving back to
 198          // original section. The opposite section remains opened and empty.
 199  
 200          var documentbody = Y.one('body');
 201          // Moving block towards hidden region-content, display it
 202          var regionname = this.get_region_id(this.dragsourceregion);
 203          if (documentbody.hasClass('side-'+regionname+'-only')) {
 204              documentbody.removeClass('side-'+regionname+'-only');
 205          }
 206  
 207          // Moving from empty region-content towards the opposite one,
 208          // hide empty one (only for region-pre, region-post areas at the moment).
 209          regionname = this.get_region_id(drop.ancestor('div.'+CSS.BLOCKREGION));
 210          if (this.dragsourceregion.all('.'+CSS.BLOCK).size() === 0 &&

 234      drop_hit : function(e) {
 235          var drag = e.drag;
 236          // Get a reference to our drag node
 237          var dragnode = drag.get('node');
 238          var dropnode = e.drop.get('node');
 239  
 240          // Amend existing skipnodes
 241          if (dragnode.previous() && dragnode.previous().hasClass(CSS.SKIPBLOCK)) {
 242              // the one that belongs to block below move below
 243              dragnode.insert(dragnode.previous(), 'after');
 244          }
 245          // Move original skipnodes
 246          if (this.skipnodetop) {
 247              dragnode.insert(this.skipnodetop, 'before');
 248          }
 249          if (this.skipnodebottom) {
 250              dragnode.insert(this.skipnodebottom, 'after');
 251          }
 252  
 253          // Add lightbox if it not there
 254          var lightbox = M.util.add_lightbox(Y, dragnode);
 255  
 256          // Prepare request parameters
 257          var params = {
 258              sesskey : M.cfg.sesskey,
 259              courseid : this.get('courseid'),
 260              pagelayout : this.get('pagelayout'),
 261              pagetype : this.get('pagetype'),
 262              subpage : this.get('subpage'),
 263              contextid : this.get('contextid'),
 264              action : 'move',
 265              bui_moveid : this.get_block_id(dragnode),
 266              bui_newregion : this.get_block_region(dropnode)
 267          };
 268  
 269          if (this.get('cmid')) {
 270              params.cmid = this.get('cmid');
 271          }
 272  
 273          if (dragnode.next('.'+this.samenodeclass) && !dragnode.next('.'+this.samenodeclass).hasClass(CSS.BLOCKADMINBLOCK)) {
 274              params.bui_beforeid = this.get_block_id(dragnode.next('.'+this.samenodeclass));
 275          }
 276  
 277          // Do AJAX request
 278          Y.io(M.cfg.wwwroot+AJAXURL, {
 279              method: 'POST',
 280              data: params,
 281              on: {
 282                  start : function() {
 283                      lightbox.show();
 284                  },
 285                  success: function(tid, response) {
 286                      window.setTimeout(function() {
 287                          lightbox.hide();
 288                      }, 250);
 289                      try {
 290                          var responsetext = Y.JSON.parse(response.responseText);
 291                          if (responsetext.error) {
 292                              new M.core.ajaxException(responsetext);
 293                          }
 294                      } catch (e) {}
 295                  },
 296                  failure: function(tid, response) {
 297                      this.ajax_failure(response);

 299                  }
 300              },
 301              context:this

 350  M.core.blockdraganddrop.is_using_blocks_render_method = function() {
 351      if (this._isusingnewblocksmethod === null) {
 352          var goodregions = Y.all('.block-region[data-blockregion]').size();
 353          var allregions = Y.all('.block-region').size();
 354          this._isusingnewblocksmethod = (allregions === goodregions);
 355          if (goodregions > 0 && allregions > 0 && goodregions !== allregions) {
 356              Y.log('Both core_renderer::blocks and core_renderer::blocks_for_region have been used.', 'warn', 'moodle-core_blocks');
 357          }
 358      }
 359      return this._isusingnewblocksmethod;
 360  };

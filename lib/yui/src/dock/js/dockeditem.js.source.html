  34          var title = this.get('title'),
  35              titlestring,
  36              type;
  37          /**
  38           * Fired before the docked item has been drawn.
  39           * @event dockeditem:drawstart
  40           */
  41          this.publish('dockeditem:drawstart', {prefix:'dockeditem'});
  42          /**
  43           * Fired after the docked item has been drawn.
  44           * @event dockeditem:drawcomplete

  46          this.publish('dockeditem:drawcomplete', {prefix:'dockeditem'});

  48           * Fired before the docked item is to be shown.
  49           * @event dockeditem:showstart

  52          /**
  53           * Fired after the docked item has been shown.
  54           * @event dockeditem:showcomplete

  56          this.publish('dockeditem:showcomplete', {prefix:'dockeditem'});
  57          /**
  58           * Fired before the docked item has been hidden.

  61          this.publish('dockeditem:hidestart', {prefix:'dockeditem'});
  62          /**
  63           * Fired after the docked item has been hidden.

  66          this.publish('dockeditem:hidecomplete', {prefix:'dockeditem'});
  67          /**
  68           * Fired when the docked item is removed from the dock.
  69           * @event dockeditem:itemremoved
  70           */
  71          this.publish('dockeditem:itemremoved', {prefix:'dockeditem'});
  72          if (title) {
  73              type = title.get('nodeName');
  74              titlestring = title.cloneNode(true);
  75              title = Y.Node.create('<'+type+'></'+type+'>');
  76              title = M.core.dock.fixTitleOrientation(title, titlestring.get('text'));
  77              this.set('title', title);
  78              this.set('titlestring', titlestring);

  81      },
  82      /**
  83       * This function draws the item on the dock.
  84       * @method draw
  85       * @return Boolean
  86       */
  87      draw : function() {
  88          var create = Y.Node.create,
  89              dock = this.get('dock'),
  90              count = dock.count,
  91              docktitle,
  92              dockitem,
  93              closeicon,
  94              closeiconimg,
  95              id = this.get('id');
  96  
  97          this.fire('dockeditem:drawstart');
  98  
  99          docktitle = create('<div id="dock_item_'+id+'_title" role="menu" aria-haspopup="true" class="'+CSS.dockedtitle+'"></div>');
 100          docktitle.append(this.get('title'));
 101          dockitem = create('<div id="dock_item_'+id+'" class="'+CSS.dockeditem+'" tabindex="0" rel="'+id+'"></div>');
 102          if (count === 1) {
 103              dockitem.addClass('firstdockitem');
 104          }
 105          dockitem.append(docktitle);
 106          dock.append(dockitem);
 107  
 108          closeiconimg = create('<img alt="' + M.util.get_string('hidepanel', 'block') +
 109                  '" title="' + M.util.get_string('hidedockpanel', 'block') + '" />');
 110          closeiconimg.setAttribute('src', M.util.image_url('t/dockclose', 'moodle'));
 111          closeicon = create('<span class="hidepanelicon" tabindex="0"></span>').append(closeiconimg);
 112          closeicon.on('forceclose|click', this.hide, this);
 113          closeicon.on('dock:actionkey',this.hide, this, {actions:{enter:true,toggle:true}});
 114          this.get('commands').append(closeicon);
 115  
 116          this.set('dockTitleNode', docktitle);
 117          this.set('dockItemNode', dockitem);
 118  
 119          this.fire('dockeditem:drawcomplete');
 120          return true;
 121      },
 122      /**
 123       * This function toggles makes the item active and shows it.
 124       * @method show
 125       * @return Boolean
 126       */
 127      show : function() {
 128          var dock = this.get('dock'),
 129              panel = dock.getPanel(),
 130              docktitle = this.get('dockTitleNode');
 131  
 132          dock.hideActive();
 133          this.fire('dockeditem:showstart');
 134          Y.log('Showing '+this._getLogDescription(), 'debug', LOGNS);
 135          panel.setHeader(this.get('titlestring'), this.get('commands'));
 136          panel.setBody(Y.Node.create('<div class="block_' + this.get('blockclass') + ' block_docked"></div>')
 137               .append(this.get('contents')));
 138          if (M.core.actionmenu !== undefined) {
 139              M.core.actionmenu.newDOMNode(panel.get('node'));
 140          }
 141          panel.show();
 142          panel.correctWidth();
 143  
 144          this.active = true;
 145          // Add active item class first up
 146          docktitle.addClass(CSS.activeitem);
 147          // Set aria-exapanded property to true.
 148          docktitle.set('aria-expanded', "true");
 149          this.fire('dockeditem:showcomplete');
 150          dock.resize();
 151          return true;
 152      },
 153      /**
 154       * This function hides the item and makes it inactive.
 155       * @method hide
 156       */
 157      hide : function() {
 158          this.fire('dockeditem:hidestart');
 159          Y.log('Hiding "'+this._getLogDescription(), 'debug', LOGNS);
 160          if (this.active) {
 161              // No longer active
 162              this.active = false;
 163              // Hide the panel
 164              this.get('dock').getPanel().hide();
 165          }
 166          // Remove the active class
 167          // Set aria-exapanded property to false
 168          this.get('dockTitleNode').removeClass(CSS.activeitem).set('aria-expanded', "false");
 169          this.fire('dockeditem:hidecomplete');
 170      },
 171      /**
 172       * A toggle between calling show and hide functions based on css.activeitem
 173       * Applies rules to key press events (dock:actionkey)
 174       * @method toggle
 175       * @param {String} action
 176       */
 177      toggle : function(action) {
 178          var docktitle = this.get('dockTitleNode');
 179          if (docktitle.hasClass(CSS.activeitem) && action !== 'expand') {
 180              this.hide();
 181          } else if (!docktitle.hasClass(CSS.activeitem) && action !== 'collapse')  {
 182              this.show();
 183          }
 184      },
 185      /**
 186       * This function removes the node and destroys it's bits.
 187       * @method remove.
 188       */
 189      remove : function () {
 190          this.hide();
 191          // Return the block to its original position.
 192          this.get('block').returnToPage();
 193          // Remove the dock item node.
 194          this.get('dockItemNode').remove();
 195          this.fire('dockeditem:itemremoved');
 196      },
 197      /**
 198       * Returns the description of this item to use for log calls.
 199       * @method _getLogDescription

 201       * @return {String}

 203      _getLogDescription : function() {

 205      }

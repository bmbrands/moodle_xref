 237      global_drag_drag: function(e) {
 238          var drag = e.target,
 239              info = e.info;
 240  
 241          // Check that drag object belongs to correct group
 242          if (!this.in_group(drag)) {
 243              return;
 244          }
 245  
 246          // Note, we test both < and > situations here. We don't want to
 247          // effect a change in direction if the user is only moving side
 248          // to side with no Y position change.
 249  
 250          // Detect changes in the position relative to the start point.
 251          if (info.start[1] < info.xy[1]) {
 252              // We are going up if our final position is higher than our start position.
 253              this.absgoingup = true;
 254  
 255          } else if (info.start[1] > info.xy[1]) {
 256              // Otherwise we're going down.
 257              this.absgoingup = false;
 258          }
 259  
 260          // Detect changes in the position relative to the last movement.
 261          if (info.delta[1] < 0) {
 262              // We are going up if our final position is higher than our start position.
 263              this.goingup = true;
 264  
 265          } else if (info.delta[1] > 0) {
 266              // Otherwise we're going down.
 267              this.goingup = false;
 268          }
 269  
 270          this.drag_drag(e);
 271      },

 273      global_drop_over: function(e) {
 274          // Check that drop object belong to correct group.
 275          if (!e.drop || !e.drop.inGroup(this.groups)) {
 276              return;
 277          }
 278  
 279          // Get a reference to our drag and drop nodes.
 280          var drag = e.drag.get('node'),
 281              drop = e.drop.get('node');
 282  
 283          // Save last drop target for the case of missed target processing.
 284          this.lastdroptarget = e.drop;
 285  
 286          // Are we dropping within the same parent node?
 287          if (drop.hasClass(this.samenodeclass)) {
 288              var where;
 289  
 290              if (this.goingup) {
 291                  where = "before";
 292              } else {
 293                  where = "after";
 294              }
 295  
 296              // Add the node contents so that it's moved, otherwise only the drag handle is moved.
 297              drop.insert(drag, where);
 298          } else if ((drop.hasClass(this.parentnodeclass) || drop.test('[data-droptarget="1"]')) && !drop.contains(drag)) {
 299              // We are dropping on parent node and it is empty
 300              if (this.goingup) {
 301                  drop.append(drag);
 302              } else {
 303                  drop.prepend(drag);
 304              }
 305          }
 306          this.drop_over(e);
 307      },

 360      /**
 361       * This is used to initiate a keyboard version of a drag and drop.
 362       * A dialog will open listing all the valid drop targets that can be selected
 363       * using tab, tab, tab, enter.
 364       * @method global_start_keyboard_drag
 365       * @param {Event} e The keydown / click event on the grab handle.
 366       * @param {Node} dragcontainer The resolved draggable node (an ancestor of the drag handle).
 367       * @param {Node} draghandle The node that triggered this action.
 368       */
 369      global_start_keyboard_drag: function(e, draghandle, dragcontainer) {
 370          M.core.dragdrop.keydragcontainer = dragcontainer;
 371          M.core.dragdrop.keydraghandle = draghandle;
 372  
 373          // Get the name of the thing to move.
 374          var nodetitle = this.find_element_text(dragcontainer);
 375          var dialogtitle = M.util.get_string('movecontent', 'moodle', nodetitle);
 376  
 377          // Build the list of drop targets.
 378          var droplist = Y.Node.create('<ul></ul>');
 379          droplist.addClass('dragdrop-keyboard-drag');
 380          var listitem,
 381              listlink,
 382              listitemtext;
 383  
 384          // Search for possible drop targets.
 385          var droptargets = Y.all('.' + this.samenodeclass + ', .' + this.parentnodeclass);
 386  
 387          droptargets.each(function (node) {
 388              var validdrop = false, labelroot = node;
 389              if (node.drop && node.drop.inGroup(this.groups) && node.drop.get('node') !== dragcontainer) {
 390                  // This is a drag and drop target with the same class as the grabbed node.
 391                  validdrop = true;
 392              } else {
 393                  var elementgroups = node.getAttribute('data-draggroups').split(' ');
 394                  var i, j;
 395                  for (i = 0; i < elementgroups.length; i++) {
 396                      for (j = 0; j < this.groups.length; j++) {
 397                          if (elementgroups[i] === this.groups[j]) {
 398                              // This is a parent node of the grabbed node (used for dropping in empty sections).
 399                              validdrop = true;
 400                              // This node will have no text - so we get the first valid text from the parent.
 401                              labelroot = node.get('parentNode');
 402                              break;
 403                          }
 404                      }
 405                      if (validdrop) {
 406                          break;
 407                      }
 408                  }
 409              }
 410  
 411              if (validdrop) {
 412                  // It is a valid drop target - create a list item for it.
 413                  listitem = Y.Node.create('<li></li>');
 414                  listlink = Y.Node.create('<a></a>');
 415                  nodetitle = this.find_element_text(labelroot);
 416  
 417                  if (this.samenodelabel && node.hasClass(this.samenodeclass)) {
 418                      listitemtext = M.util.get_string(this.samenodelabel.identifier, this.samenodelabel.component, nodetitle);
 419                  } else if (this.parentnodelabel && node.hasClass(this.parentnodeclass)) {
 420                      listitemtext = M.util.get_string(this.parentnodelabel.identifier, this.parentnodelabel.component, nodetitle);
 421                  } else {
 422                      listitemtext = M.util.get_string('tocontent', 'moodle', nodetitle);
 423                  }
 424                  listlink.setContent(listitemtext);
 425  
 426                  // Add a data attribute so we can get the real drop target.
 427                  listlink.setAttribute('data-drop-target', node.get('id'));
 428                  // Allow tabbing to the link.
 429                  listlink.setAttribute('tabindex', '0');
 430  
 431                  // Set the event listeners for enter, space or click.
 432                  listlink.on('click', this.global_keyboard_drop, this);
 433                  listlink.on('key', this.global_keyboard_drop, 'down:enter,32', this);
 434  
 435                  // Add to the list or drop targets.
 436                  listitem.append(listlink);
 437                  droplist.append(listitem);
 438              }
 439          }, this);
 440  
 441          // Create the dialog for the interaction.
 442          M.core.dragdrop.dropui = new M.core.dialogue({
 443              headerContent: dialogtitle,
 444              bodyContent: droplist,
 445              draggable: true,
 446              visible: true,
 447              center: true,
 448              modal: true
 449          });

 451          M.core.dragdrop.dropui.after('visibleChange', function(e) {
 452              // After the dialogue has been closed, we call the cancel function. This will
 453              // ensure that tidying up happens (e.g. focusing on the start Node).
 454              if (e.prevVal && !e.newVal) {
 455                  this.global_cancel_keyboard_drag();
 456              }
 457          }, this);
 458  
 459          // Focus the first drop target.
 460          if (droplist.one('a')) {
 461              droplist.one('a').focus();
 462          }

 465      /**
 466       * This is used as a simulated drag/drop event in order to prevent any
 467       * subtle bugs from creating a real instance of a drag drop event. This means
 468       * there are no state changes in the Y.DD.DDM and any undefined functions
 469       * will trigger an obvious and fatal error.
 470       * The end result is that we call all our drag/drop handlers but do not bubble the
 471       * event to anyone else.
 472       *
 473       * The functions/properties implemented in the wrapper are:
 474       * e.target
 475       * e.drag
 476       * e.drop
 477       * e.drag.get('node')
 478       * e.drop.get('node')
 479       * e.drag.addHandle()
 480       * e.drag.removeHandle()
 481       *
 482       * @method simulated_drag_drop_event
 483       * @param {Node} dragnode The drag container node
 484       * @param {Node} dropnode The node to initiate the drop on
 485       */
 486      simulated_drag_drop_event: function(dragnode, dropnode) {
 487  
 488          // Subclass for wrapping both drag and drop.
 489          var DragDropWrapper = function(node) {
 490              this.node = node;
 491          };
 492  
 493          // Method e.drag.get() - get the node.
 494          DragDropWrapper.prototype.get = function(param) {
 495              if (param === 'node' || param === 'dragNode' || param === 'dropNode') {
 496                  return this.node;
 497              }
 498              if (param === 'activeHandle') {
 499                  return this.node.one('.editing_move');
 500              }
 501              return null;
 502          };
 503  
 504          // Method e.drag.inGroup() - we have already run the group checks before triggering the event.
 505          DragDropWrapper.prototype.inGroup = function() {
 506              return true;
 507          };
 508  
 509          // Method e.drag.addHandle() - we don't want to run this.
 510          DragDropWrapper.prototype.addHandle = function() {};
 511          // Method e.drag.removeHandle() - we don't want to run this.
 512          DragDropWrapper.prototype.removeHandle = function() {};
 513  
 514          // Create instances of the DragDropWrapper.
 515          this.drop = new DragDropWrapper(dropnode);
 516          this.drag = new DragDropWrapper(dragnode);
 517          this.target = this.drop;
 518      },

 520      /**
 521       * This is used to complete a keyboard version of a drag and drop.
 522       * A drop event will be simulated based on the drag and drop nodes.
 523       * @method global_keyboard_drop
 524       * @param {Event} e The keydown / click event on the proxy drop node.
 525       */
 526      global_keyboard_drop: function(e) {
 527          // The drag node was saved.
 528          var dragcontainer = M.core.dragdrop.keydragcontainer;
 529          // The real drop node is stored in an attribute of the proxy.
 530          var droptarget = Y.one('#' + e.target.getAttribute('data-drop-target'));
 531  
 532          // Close the dialog.
 533          M.core.dragdrop.dropui.hide();
 534          // Cancel the event.
 535          e.preventDefault();
 536          // Convert to drag drop events.
 537          var dragevent = new this.simulated_drag_drop_event(dragcontainer, dragcontainer);
 538          var dropevent = new this.simulated_drag_drop_event(dragcontainer, droptarget);
 539          // Simulate the full sequence.
 540          this.drag_start(dragevent);
 541          this.global_drop_over(dropevent);
 542  
 543          if (droptarget.hasClass(this.parentnodeclass) && droptarget.contains(dragcontainer)) {
 544              // The global_drop_over function does not handle the case where an item was moved up, without the
 545              // 'goingup' variable being set, as is the case wih keyboard drag/drop. We must detect this case and
 546              // apply it after the drop_over, but before the drop_hit event in order for it to be moved to the
 547              // correct location.
 548              droptarget.prepend(dragcontainer);
 549          }
 550  
 551          this.global_drop_hit(dropevent);
 552      },

 570      /**
 571       * Process key events on the drag handles.
 572       *
 573       * @method global_keydown
 574       * @param {EventFacade} e The keydown / click event on the drag handle.
 575       */
 576      global_keydown: function(e) {
 577          var draghandle = e.target.ancestor('.' + MOVEICON.cssclass, true),
 578              dragcontainer,
 579              draggroups;
 580  
 581          if (draghandle === null) {
 582              // The element clicked did not have a a draghandle in it's lineage.
 583              return;
 584          }
 585  
 586          if (e.keyCode === 27 ) {
 587              // Escape to cancel from anywhere.
 588              this.global_cancel_keyboard_drag();
 589              e.preventDefault();
 590              return;
 591          }
 592  
 593          // Only process events on a drag handle.
 594          if (!draghandle.hasClass(MOVEICON.cssclass)) {
 595              return;
 596          }
 597  
 598          // Do nothing if not space or enter.
 599          if (e.keyCode !== 13 && e.keyCode !== 32 && e.type !== 'click') {
 600              return;
 601          }
 602  
 603          // Check the drag groups to see if we are the handler for this node.
 604          draggroups = draghandle.getAttribute('data-draggroups').split(' ');
 605          var i, j, validgroup = false;
 606  
 607          for (i = 0; i < draggroups.length; i++) {
 608              for (j = 0; j < this.groups.length; j++) {
 609                  if (draggroups[i] === this.groups[j]) {
 610                      validgroup = true;
 611                      break;
 612                  }
 613              }
 614              if (validgroup) {
 615                  break;
 616              }
 617          }
 618          if (!validgroup) {
 619              return;
 620          }
 621  
 622          // Valid event - start the keyboard drag.
 623          dragcontainer = draghandle.ancestor('.yui3-dd-drop');
 624          this.global_start_keyboard_drag(e, draghandle, dragcontainer);
 625  
 626          e.preventDefault();
 627      },

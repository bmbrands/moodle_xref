 167      moveMenuItem: function(e) {
 168          var nextFocus,
 169              menuitem = e.target.ancestor(SELECTOR.MENUITEM, true);
 170  
 171          if (e.keyCode === 37) {
 172              nextFocus = this.getMenuItem(menuitem, true);
 173          } else if (e.keyCode === 39) {
 174              nextFocus = this.getMenuItem(menuitem);
 175          }
 176  
 177          if (nextFocus) {
 178              nextFocus.focus();
 179          }
 180          return this;

 183      /**
 184       * Get the next menuitem in a menubar.
 185       *
 186       * @method getMenuItem
 187       * @param {Node} currentItem The currently focused item in the menubar
 188       * @param {Boolean} [previous=false] Move backwards in the menubar instead of forwards
 189       * @return {Node|null} The next item, or null if none was found
 190       */
 191      getMenuItem: function(currentItem, previous) {
 192          var menubar = currentItem.ancestor(SELECTOR.MENUBAR),
 193              menuitems,
 194              next;
 195  
 196          if (!menubar) {
 197              return null;
 198          }
 199  
 200          menuitems = menubar.all(SELECTOR.MENUITEMS.join(','));
 201  
 202          if (!menuitems) {
 203              return null;
 204          }
 205  
 206          var childCount = menuitems.size();
 207  
 208          if (childCount === 1) {
 209              // Only one item, exit now because we should already be on it.
 210              return null;
 211          }
 212  
 213          // Determine the next child.
 214          var index = 0,
 215              direction = 1,
 216              checkCount = 0;
 217  
 218          // Work out the index of the currently selected item.
 219          for (index = 0; index < childCount; index++) {
 220              if (menuitems.item(index) === currentItem) {
 221                  break;
 222              }
 223          }
 224  
 225          // Check that the menu item was found - otherwise return null.
 226          if (menuitems.item(index) !== currentItem) {
 227              return null;
 228          }
 229  
 230          // Reverse the direction if we want the previous item.
 231          if (previous) {
 232              direction = -1;
 233          }
 234  
 235          do {
 236              // Update the index in the direction of travel.
 237              index += direction;
 238  
 239              next = menuitems.item(index);
 240  
 241              // Check that we don't loop multiple times.
 242              checkCount++;
 243          } while (next && next.hasAttribute('hidden'));
 244  
 245          return next;
 246      },

 306      /**
 307       * Handle keyboard events when the menu is open. We respond to:
 308       * * escape (exit)
 309       * * tab (move to next menu item)
 310       * * up/down (move to previous/next menu item)
 311       *
 312       * @method handleKeyboardEvent
 313       * @param {EventFacade} e The key event
 314       */
 315      handleKeyboardEvent: function(e) {
 316          var next;
 317  
 318          // Handle when the menu is still selected.
 319          if (e.currentTarget.ancestor(SELECTOR.TOGGLE, true)) {
 320              if ((e.keyCode === 40 || (e.keyCode === 9 && !e.shiftKey)) && this.firstMenuChild) {
 321                  this.firstMenuChild.focus();
 322                  e.preventDefault();
 323              } else if (e.keyCode === 38 && this.lastMenuChild) {
 324                  this.lastMenuChild.focus();
 325                  e.preventDefault();
 326              } else if (e.keyCode === 9 && e.shiftKey) {
 327                  this.hideMenu();
 328                  e.preventDefault();
 329              }
 330              return this;
 331          }
 332  
 333          if (e.keyCode === 27) {
 334              // The escape key was pressed so close the menu.
 335              this.hideMenu();
 336              e.preventDefault();
 337  
 338          } else if (e.keyCode === 32) {
 339              // The space bar was pressed. Trigger a click.
 340              e.preventDefault();
 341              e.currentTarget.simulate('click');
 342          } else if (e.keyCode === 9) {
 343              // The tab key was pressed. Tab moves forwards, Shift + Tab moves backwards through the menu options.
 344              // We only override the Shift + Tab on the first option, and Tab on the last option to change where the
 345              // focus is moved to.
 346              if (e.target === this.firstMenuChild && e.shiftKey) {
 347                  this.hideMenu();
 348                  e.preventDefault();
 349              } else if (e.target === this.lastMenuChild && !e.shiftKey) {
 350                  if (this.hideMenu()) {
 351                      // Determine the next selector and focus on it.
 352                      next = this.menulink.next(SELECTOR.CAN_RECEIVE_FOCUS_SELECTOR);
 353                      if (next) {
 354                          next.focus();
 355                      }
 356                  }
 357              }
 358  
 359          } else if (e.keyCode === 38 || e.keyCode === 40) {
 360              // The up (38) or down (40) key was pushed.
 361              // On cursor moves we loops through the menu rather than exiting it as in the tab behaviour.
 362              var found = false,
 363                  index = 0,
 364                  direction = 1,
 365                  checkCount = 0;
 366  
 367              // Determine which menu item is currently selected.
 368              while (!found && index < this.menuChildren.size()) {
 369                  if (this.menuChildren.item(index) === e.currentTarget) {
 370                      found = true;
 371                  } else {
 372                      index++;
 373                  }
 374              }
 375  
 376              if (!found) {
 377                  Y.log("Unable to find this menu item in the list of menu children", 'debug', 'moodle-core-actionmenu');
 378                  return;
 379              }
 380  
 381              if (e.keyCode === 38) {
 382                  // Moving up so reverse the direction.
 383                  direction = -1;
 384              }
 385  
 386              // Try to find the next
 387              do {
 388                  index += direction;
 389                  if (index < 0) {
 390                      index = this.menuChildren.size() - 1;
 391                  } else if (index >= this.menuChildren.size()) {
 392                      // Handle wrapping.
 393                      index = 0;
 394                  }
 395                  next = this.menuChildren.item(index);
 396  
 397                  // Add a counter to ensure we don't get stuck in a loop if there's only one visible menu item.
 398                  checkCount++;
 399              } while (checkCount < this.menuChildren.size() && next !== e.currentTarget && next.hasClass('hidden'));
 400  
 401              if (next) {
 402                  next.focus();
 403                  e.preventDefault();
 404              }
 405          }
 406      },

 422       * Displays the menu with the given content and alignment.
 423       *
 424       * @method showMenu
 425       * @param {EventFacade} e
 426       * @param {Node} menu
 427       * @return M.core.dialogue
 428       */
 429      showMenu : function(e, menu) {
 430          Y.log('Displaying an action menu', 'debug', ACTIONMENU.NAME);
 431          var ownerselector = menu.getData('owner'),
 432              menucontent = menu.one(SELECTOR.MENUCONTENT);
 433          this.owner = (ownerselector) ? menu.ancestor(ownerselector) : null;
 434          this.dialogue = menu;
 435          menu.addClass('show');
 436          if (this.owner) {
 437              this.owner.addClass(CSS.MENUSHOWN);
 438              this.menulink = this.owner.one(SELECTOR.TOGGLE);
 439          } else {
 440              this.menulink = e.target.ancestor(SELECTOR.TOGGLE, true);
 441          }
 442          this.constrain(menucontent.set('aria-hidden', false));
 443  
 444          this.menuChildren = this.dialogue.all(SELECTOR.MENUCHILD);
 445          if (this.menuChildren) {
 446              this.firstMenuChild = this.menuChildren.item(0);
 447              this.lastMenuChild  = this.menuChildren.item(this.menuChildren.size() - 1);
 448  
 449              this.firstMenuChild.focus();
 450          }
 451  
 452          // Close the menu if the user presses escape.
 453          this.events.push(BODY.on('key', this.hideMenu, 'esc', this));
 454  
 455          // Close the menu if the user clicks outside the menu.
 456          this.events.push(BODY.on('click', this.hideIfOutside, this));
 457  
 458          // Close the menu if the user focuses outside the menu.
 459          this.events.push(BODY.delegate('focus', this.hideIfOutside, '*', this));
 460  
 461          // Check keyboard changes.
 462          this.events.push(
 463              menu.delegate('key', this.handleKeyboardEvent,
 464                            'down:9, 27, 38, 40, 32', SELECTOR.MENUCHILD + ', ' + SELECTOR.TOGGLE, this)
 465              );
 466  
 467          // Close the menu after a button was pushed.
 468          this.events.push(menu.delegate('click', function(e) {
 469              if (e.currentTarget.test(SELECTOR.KEEPOPEN)) {
 470                  return;
 471              }
 472              this.hideMenu();
 473          }, SELECTOR.MENUCHILD, this));
 474  
 475          return true;
 476      },

 478      /**
 479       * Constrains the node to its the page width.
 480       *
 481       * @method constrain
 482       * @param {Node} node
 483       */
 484      constrain : function(node) {
 485          var selector = node.getData('constraint'),
 486              nx = node.getX(),
 487              ny = node.getY(),
 488              nwidth = node.get('offsetWidth'),
 489              nheight = node.get('offsetHeight'),
 490              cx = 0,
 491              cy = 0,
 492              cwidth,
 493              cheight,
 494              coverflow = 'auto',
 495              newwidth = null,
 496              newheight = null,
 497              newleft = null,
 498              newtop = null,
 499              boxshadow = null;
 500  
 501          if (selector) {
 502              selector = node.ancestor(selector);
 503          }
 504          if (selector) {
 505              cwidth = selector.get('offsetWidth');
 506              cheight = selector.get('offsetHeight');
 507              cx = selector.getX();
 508              cy = selector.getY();
 509              coverflow = selector.getStyle('overflow') || 'auto';
 510          } else {
 511              cwidth = node.get('docWidth');
 512              cheight = node.get('docHeight');
 513          }
 514  
 515          // Constrain X.
 516          // First up if the width is more than the constrain its easily full width + full height.
 517          if (nwidth > cwidth) {
 518              // The width of the constraint.
 519              newwidth = nwidth = cwidth;
 520              // The constraints xpoint.
 521              newleft = nx = cx;
 522          } else {
 523              if (nx < cx) {
 524                  // If nx is less than cx we need to move it right.
 525                  newleft = nx = cx;
 526              } else if (nx + nwidth >= cx + cwidth) {
 527                  // The top right of the node is outside of the constraint, move it in.
 528                  newleft = cx + cwidth - nwidth;
 529              }
 530          }
 531  
 532          // Constrain Y.
 533          if (nheight > cheight && coverflow.toLowerCase() === 'hidden') {
 534              // The node extends over the constrained area and would be clipped.
 535              // Reduce the height of the node and force its overflow to scroll.
 536              newheight = nheight = cheight;
 537              node.setStyle('overflow', 'auto');
 538          }
 539          // If the node is below the top of the constraint AND
 540          //    the node is longer than the constraint allows.
 541          if (ny >= cy && ny + nheight > cy + cheight) {
 542              // Move it up.
 543              newtop = cy + cheight - nheight;
 544              try {
 545                  boxshadow = node.getStyle('boxShadow').replace(/.*? (\d+)px \d+px$/, '$1');
 546                  if (new RegExp(/^\d+$/).test(boxshadow) && newtop - cy > boxshadow) {
 547                      newtop -= boxshadow;
 548                  }
 549              } catch (ex) {
 550                  Y.log('Failed to determine box-shadow margin.', 'warn', ACTIONMENU.NAME);
 551              }
 552          }
 553  
 554          if (newleft !== null) {
 555              node.setX(newleft);
 556          }
 557          if (newtop !== null) {
 558              node.setY(newtop);
 559          }
 560          if (newwidth !== null) {
 561              node.setStyle('width', newwidth.toString() + 'px');
 562          }
 563          if (newheight !== null) {
 564              node.setStyle('height', newheight.toString() + 'px');
 565          }
 566      }

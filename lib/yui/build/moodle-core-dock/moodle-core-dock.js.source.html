 137  M.core.dock.fixTitleOrientation = function(title, text) {
 138      var dock = M.core.dock.get(),
 139          fontsize = '11px',
 140          transform = 'rotate(270deg)',
 141          test,
 142          width,
 143          height,
 144          container,
 145          verticaldirection = M.util.get_string('thisdirectionvertical', 'langconfig');
 146      title = Y.one(title);
 147  
 148      if (dock.get('orientation') !== 'vertical') {
 149          // If the dock isn't vertical don't adjust it!
 150          title.set('innerHTML', text);
 151          return title;
 152      }
 153  
 154      if (Y.UA.ie > 0 && Y.UA.ie < 8) {
 155          // IE 6/7 can't rotate text so force ver
 156          verticaldirection = 'ver';
 157      }
 158  
 159      switch (verticaldirection) {
 160          case 'ver':
 161              // Stacked is easy
 162              return title.set('innerHTML', text.split('').join('<br />'));
 163          case 'ttb':
 164              transform = 'rotate(90deg)';
 165              break;
 166          case 'btt':
 167              // Nothing to do here. transform default is good.
 168              break;
 169      }
 170  
 171      if (Y.UA.ie === 8) {
 172          // IE8 can flip the text via CSS but not handle transform. IE9+ can handle the CSS3 transform attribute.
 173          title.set('innerHTML', text);
 174          title.setAttribute('style', 'writing-mode: tb-rl; filter: flipV flipH;display:inline;');
 175          title.addClass('filterrotate');
 176          return title;
 177      }
 178  
 179      // We need to fix a font-size - sorry theme designers.
 180      test = Y.Node.create('<h2 class="transform-test-heading"><span class="transform-test-node" style="font-size:' +
 181              fontsize + ';">' + text + '</span></h2>');
 182      BODY.insert(test, 0);
 183      width = test.one('span').get('offsetWidth') * 1.2;
 184      height = test.one('span').get('offsetHeight');
 185      test.remove();
 186  
 187      title.set('innerHTML', text);
 188      title.addClass('css3transform');
 189  
 190      // Move the title into position
 191      title.setStyles({
 192          'position' : 'relative',
 193          'fontSize' : fontsize,
 194          'width' : width,
 195          'top' : (width - height)/2
 196      });
 197  
 198      // Positioning is different when in RTL mode.
 199      if (window.right_to_left()) {
 200          title.setStyle('left', width/2 - height);
 201      } else {
 202          title.setStyle('right', width/2 - height);
 203      }
 204  
 205      // Rotate the text
 206      title.setStyles({
 207          'transform' : transform,
 208          '-ms-transform' : transform,
 209          '-moz-transform' : transform,
 210          '-webkit-transform' : transform,
 211          '-o-transform' : transform
 212      });
 213  
 214      container = Y.Node.create('<div></div>');
 215      container.append(title);
 216      container.setStyles({
 217          height : width + (width / 4),
 218          position : 'relative'
 219      });
 220      return container;
 221  };

 644              target,

 646              regex = /^dock_item_(\d+)_title$/,
 647              self = this;

 649              if (!this.get('dockNode').contains(e.target)) {

 652                  }
 653              }
 654          } else {
 655              if (e.target.test(options.cssselector)) {
 656                  target = e.target;
 657              } else {
 658                  target = e.target.ancestor(options.cssselector);
 659              }
 660              if (!target) {
 661                  return true;
 662              }

 664                  return true;
 665              }
 666              if (options.preventevent) {
 667                  this.preventevent = options.preventevent;
 668                  if (options.preventdelay) {
 669                      setTimeout(function(){
 670                          self.preventevent = null;
 671                      }, options.preventdelay * 1000);
 672                  }
 673              }
 674              if (this.delayedevent && this.delayedevent.timeout) {
 675                  clearTimeout(this.delayedevent.timeout);
 676                  this.delayedevent.event.detach();
 677                  this.delayedevent = null;
 678              }
 679              if (options.delay > 0) {
 680                  return this.delayEvent(e, options, target);
 681              }
 682              targetid = target.get('id');
 683              if (targetid.match(regex)) {
 684                  item = this.dockeditems[targetid.replace(regex, '$1')];
 685                  if (item.active) {
 686                      item.hide();
 687                  } else {
 688                      item.show();
 689                  }

1508  /**
1509   * A 'dock:actionkey' Event.
1510   * The event consists of the left arrow, right arrow, enter and space keys.
1511   * More keys can be mapped to action meanings.
1512   * actions: collapse , expand, toggle, enter.
1513   *
1514   * This event is subscribed to by dockitems.
1515   * The on() method to subscribe allows specifying the desired trigger actions as JSON.
1516   *
1517   * This event can also be delegated if needed.
1518   *
1519   * @namespace M.core.dock
1520   * @class ActionKey
1521   */
1522  Y.Event.define("dock:actionkey", {
1523      // Webkit and IE repeat keydown when you hold down arrow keys.
1524      // Opera links keypress to page scroll; others keydown.
1525      // Firefox prevents page scroll via preventDefault() on either
1526      // keydown or keypress.
1527      _event: (Y.UA.webkit || Y.UA.ie) ? 'keydown' : 'keypress',
1528  
1529      /**
1530       * The keys to trigger on.
1531       * @property _keys
1532       */
1533      _keys: {
1534          //arrows
1535          '37': 'collapse',
1536          '39': 'expand',
1537          //(@todo: lrt/rtl/M.core_dock.cfg.orientation decision to assign arrow to meanings)
1538          '32': 'toggle',
1539          '13': 'enter'
1540      },
1541  
1542      /**
1543       * Handles key events
1544       * @method _keyHandler
1545       * @param {EventFacade} e
1546       * @param {SyntheticEvent.Notifier} notifier The notifier used to trigger the execution of subscribers
1547       * @param {Object} args
1548       */
1549      _keyHandler: function (e, notifier, args) {
1550          var actObj;
1551          if (!args.actions) {
1552              actObj = {collapse: true, expand: true, toggle: true, enter: true};
1553          } else {
1554              actObj = args.actions;
1555          }
1556          if (this._keys[e.keyCode] && actObj[this._keys[e.keyCode]]) {
1557              e.action = this._keys[e.keyCode];
1558              notifier.fire(e);
1559          }
1560      },
1561  
1562      /**
1563       * Subscribes to events.
1564       * @method on
1565       * @param {Node} node The node this subscription was applied to.
1566       * @param {Subscription} sub The object tracking this subscription.
1567       * @param {SyntheticEvent.Notifier} notifier The notifier used to trigger the execution of subscribers
1568       */
1569      on: function (node, sub, notifier) {
1570          // subscribe to _event and ask keyHandler to handle with given args[0] (the desired actions).
1571          if (sub.args === null) {
1572              //no actions given
1573              sub._detacher = node.on(this._event, this._keyHandler, this, notifier, {actions: false});
1574          } else {
1575              sub._detacher = node.on(this._event, this._keyHandler, this, notifier, sub.args[0]);
1576          }
1577      },
1578  
1579      /**
1580       * Detaches an event listener
1581       * @method detach
1582       * @param {Node} node The node this subscription was applied to.
1583       * @param {Subscription} sub The object tracking this subscription.
1584       * @param {SyntheticEvent.Notifier} notifier The notifier used to trigger the execution of subscribers
1585       */
1586      detach: function (node, sub) {
1587          //detach our _detacher handle of the subscription made in on()
1588          sub._detacher.detach();
1589      },
1590  
1591      /**
1592       * Creates a delegated event listener.
1593       * @method delegate
1594       * @param {Node} node The node this subscription was applied to.
1595       * @param {Subscription} sub The object tracking this subscription.
1596       * @param {SyntheticEvent.Notifier} notifier The notifier used to trigger the execution of subscribers
1597       * @param {String|function} filter Selector string or function that accpets an event object and returns null.
1598       */
1599      delegate: function (node, sub, notifier, filter) {
1600          // subscribe to _event and ask keyHandler to handle with given args[0] (the desired actions).
1601          if (sub.args === null) {
1602              //no actions given
1603              sub._delegateDetacher = node.delegate(this._event, this._keyHandler, filter, this, notifier, {actions:false});
1604          } else {
1605              sub._delegateDetacher = node.delegate(this._event, this._keyHandler, filter, this, notifier, sub.args[0]);
1606          }
1607      },

1611       * @method detachDelegate

1613       * @param {Subscription} sub The object tracking this subscription.
1614       * @param {SyntheticEvent.Notifier} notifier The notifier used to trigger the execution of subscribers
1615       * @param {String|function} filter Selector string or function that accpets an event object and returns null.
1616       */
1617      detachDelegate: function (node, sub) {
1618          sub._delegateDetacher.detach();
1619      }

  81      public function get_server_info() {

  83          $matches = null;
  84          if (preg_match('/^5\.5\.5-(10\..+)-MariaDB/i', $version, $matches)) {
  85              // Looks like MariaDB decided to use these weird version numbers for better BC with MySQL...
  86              $version = $matches[1];

  88          return array('description'=>$this->mysqli->server_info, 'version'=>$version);

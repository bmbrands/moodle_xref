  32  require_once(dirname(__FILE__) . '/../../config.php');
  33  require_once($CFG->libdir . '/externallib.php');
  34  
  35  require_login(null, true, null, true, true);
  36  
  37  $rawjson = file_get_contents('php://input');
  38  
  39  $requests = json_decode($rawjson, true);
  40  if ($requests === null) {
  41      $lasterror = json_last_error_msg();
  42      throw new coding_exception('Invalid json in request: ' . $lasterror);
  43  }
  44  $responses = array();
  45  
  46  
  47  foreach ($requests as $request) {
  48      $response = array();
  49      $methodname = clean_param($request['methodname'], PARAM_ALPHANUMEXT);
  50      $index = clean_param($request['index'], PARAM_INT);
  51      $args = $request['args'];
  52  
  53      try {
  54          $externalfunctioninfo = external_function_info($methodname);
  55  
  56          if (!$externalfunctioninfo->allowed_from_ajax) {
  57              throw new moodle_exception('servicenotavailable', 'webservice');
  58          }
  59  
  60          // Validate params, this also sorts the params properly, we need the correct order in the next part.
  61          $callable = array($externalfunctioninfo->classname, 'validate_parameters');
  62          $params = call_user_func($callable,
  63                                   $externalfunctioninfo->parameters_desc,
  64                                   $args);
  65  
  66          // Execute - gulp!
  67          $callable = array($externalfunctioninfo->classname, $externalfunctioninfo->methodname);
  68          $result = call_user_func_array($callable,
  69                                         array_values($params));
  70  
  71          $response['error'] = false;
  72          $response['data'] = $result;
  73          $responses[$index] = $response;
  74      } catch (Exception $e) {
  75          $jsonexception = get_exception_info($e);
  76          unset($jsonexception->a);
  77          if (!debugging('', DEBUG_DEVELOPER)) {
  78              unset($jsonexception->debuginfo);
  79              unset($jsonexception->backtrace);
  80          }
  81          $response['error'] = true;
  82          $response['exception'] = $jsonexception;
  83          $responses[$index] = $response;
  84          // Do not process the remaining requests.
  85          break;
  86      }
  87  }
  88  
  89  echo json_encode($responses);

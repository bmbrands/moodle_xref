  14  /**
  15   * Stream filter to analyze an IMAP string to determine how to send to the
  16   * server.
  17   *
  18   * @author    Michael Slusarz <slusarz@horde.org>
  19   * @category  Horde

  21   * @license   http://www.horde.org/licenses/lgpl21 LGPL 2.1
  22   * @package   Imap_Client
  23   */
  24  class Horde_Imap_Client_Data_Format_Filter_String extends php_user_filter
  25  {
  26      /**
  27       * @see stream_filter_register()
  28       */
  29      public function onCreate()
  30      {
  31          $this->params->binary = false;
  32          $this->params->literal = false;
  33          // no_quote_list is used below as a config option
  34          $this->params->quoted = false;
  35  
  36          return true;
  37      }
  38  
  39      /**
  40       * @see stream_filter_register()
  41       */
  42      public function filter($in, $out, &$consumed, $closing)
  43      {
  44          $p = $this->params;
  45          $skip = false;
  46  
  47          while ($bucket = stream_bucket_make_writeable($in)) {
  48              if (!$skip) {
  49                  $len = $bucket->datalen;
  50                  $str = $bucket->data;
  51  
  52                  for ($i = 0; $i < $len; ++$i) {
  53                      $chr = ord($str[$i]);
  54  
  55                      switch ($chr) {
  56                      case 0: // null
  57                          $p->binary = true;
  58                          $p->literal = true;
  59  
  60                          // No need to scan input anymore.
  61                          $skip = true;
  62                          break 2;
  63  
  64                      case 10: // LF
  65                      case 13: // CR
  66                          $p->literal = true;
  67                          break;
  68  
  69                      case 32: // SPACE
  70                      case 34: // "
  71                      case 40: // (
  72                      case 41: // )
  73                      case 92: // \
  74                      case 123: // {
  75                      case 127: // DEL
  76                          // These are all invalid ATOM characters.
  77                          $p->quoted = true;
  78                          break;
  79  
  80                      case 37: // %
  81                      case 42: // *
  82                          // These are not quoted if being used as wildcards.
  83                          if (empty($p->no_quote_list)) {
  84                              $p->quoted = true;
  85                          }
  86                          break;
  87  
  88                      default:
  89                          if ($chr < 32) {
  90                              // CTL characters must be, at a minimum, quoted.
  91                              $p->quoted = true;
  92                          } elseif ($chr > 127) {
  93                              // 8-bit chars must be in a literal.
  94                              $p->literal = true;
  95                          }
  96                          break;
  97                      }
  98                  }
  99              }
 100  
 101              $consumed += $bucket->datalen;
 102              stream_bucket_append($out, $bucket);
 103          }
 104  
 105          if ($p->literal) {
 106              $p->quoted = false;
 107          }
 108  
 109          return PSFS_PASS_ON;
 110      }

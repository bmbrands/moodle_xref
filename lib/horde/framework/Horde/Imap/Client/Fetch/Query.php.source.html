 162       * This header search work only with the base RFC 2822 message or

 164       *
 165       * @param string $label  A unique label associated with this particular

 167       * @param array $search  The search string(s) (case-insensitive).

 272       * Remove an entry under a given criteria.
 273       *
 274       * @param integer $criteria  Criteria ID.
 275       * @param string $key        The key to remove.
 276       */
 277      public function remove($criteria, $key)
 278      {
 279          if (isset($this->_data[$criteria]) &&
 280              is_array($this->_data[$criteria])) {
 281              unset($this->_data[$criteria][$key]);
 282              if (empty($this->_data[$criteria])) {
 283                  unset($this->_data[$criteria]);
 284              }
 285          }
 286      }

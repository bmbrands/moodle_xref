  35  class phpunit_message_sink {
  36      /** @var array of records from message_read table */
  37      protected $messages = array();
  38  
  39      /**
  40       * Stop message redirection.
  41       *
  42       * Use if you do not want message redirected any more.
  43       */
  44      public function close() {
  45          phpunit_util::stop_message_redirection();
  46      }
  47  
  48      /**
  49       * To be called from phpunit_util only!
  50       *
  51       * @param stdClass $message record from message_read table
  52       */
  53      public function add_message($message) {
  54          /* Number messages from 0. */
  55          $this->messages[] = $message;
  56      }
  57  
  58      /**
  59       * Returns all redirected messages.
  60       *
  61       * The instances are records form the message_read table.
  62       * The array indexes are numbered from 0 and the order is matching
  63       * the creation of events.
  64       *
  65       * @return array
  66       */
  67      public function get_messages() {
  68          return $this->messages;
  69      }
  70  
  71      /**
  72       * Return number of messages redirected to this sink.
  73       * @return int
  74       */
  75      public function count() {
  76          return count($this->messages);
  77      }
  78  
  79      /**
  80       * Removes all previously stored messages.
  81       */
  82      public function clear() {
  83          $this->messages = array();
  84      }
  85  }

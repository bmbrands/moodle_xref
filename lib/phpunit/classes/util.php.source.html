  91      /**
  92       * Reset contents of all database tables to initial values, reset caches, etc.
  93       *
  94       * Note: this is relatively slow (cca 2 seconds for pg and 7 for mysql) - please use with care!
  95       *
  96       * @static
  97       * @param bool $detectchanges
  98       *      true  - changes in global state and database are reported as errors
  99       *      false - no errors reported
 100       *      null  - only critical problems are reported as errors
 101       * @return void
 102       */
 103      public static function reset_all_data($detectchanges = false) {
 104          global $DB, $CFG, $USER, $SITE, $COURSE, $PAGE, $OUTPUT, $SESSION;
 105  
 106          // Stop any message redirection.
 107          phpunit_util::stop_message_redirection();
 108  
 109          // Stop any message redirection.
 110          phpunit_util::stop_event_redirection();
 111  
 112          // Start a new email redirection.
 113          // This will clear any existing phpmailer redirection.
 114          // We redirect all phpmailer output to this message sink which is
 115          // called instead of phpmailer actually sending the message.
 116          phpunit_util::start_phpmailer_redirection();
 117  
 118          // We used to call gc_collect_cycles here to ensure desctructors were called between tests.
 119          // This accounted for 25% of the total time running phpunit - so we removed it.
 120  
 121          // Show any unhandled debugging messages, the runbare() could already reset it.
 122          self::display_debugging_messages();
 123          self::reset_debugging();
 124  
 125          // reset global $DB in case somebody mocked it
 126          $DB = self::get_global_backup('DB');
 127  
 128          if ($DB->is_transaction_started()) {
 129              // we can not reset inside transaction
 130              $DB->force_transaction_rollback();
 131          }
 132  
 133          $resetdb = self::reset_database();
 134          $warnings = array();
 135  
 136          if ($detectchanges === true) {
 137              if ($resetdb) {
 138                  $warnings[] = 'Warning: unexpected database modification, resetting DB state';
 139              }
 140  
 141              $oldcfg = self::get_global_backup('CFG');
 142              $oldsite = self::get_global_backup('SITE');
 143              foreach($CFG as $k=>$v) {
 144                  if (!property_exists($oldcfg, $k)) {
 145                      $warnings[] = 'Warning: unexpected new $CFG->'.$k.' value';
 146                  } else if ($oldcfg->$k !== $CFG->$k) {
 147                      $warnings[] = 'Warning: unexpected change of $CFG->'.$k.' value';
 148                  }
 149                  unset($oldcfg->$k);
 150  
 151              }
 152              if ($oldcfg) {
 153                  foreach($oldcfg as $k=>$v) {
 154                      $warnings[] = 'Warning: unexpected removal of $CFG->'.$k;
 155                  }
 156              }
 157  
 158              if ($USER->id != 0) {
 159                  $warnings[] = 'Warning: unexpected change of $USER';
 160              }
 161  
 162              if ($COURSE->id != $oldsite->id) {
 163                  $warnings[] = 'Warning: unexpected change of $COURSE';
 164              }
 165  
 166              if ($CFG->ostype === 'WINDOWS') {
 167                  if (setlocale(LC_TIME, 0) !== 'English_Australia.1252') {
 168                      $warnings[] = 'Warning: unexpected change of locale';
 169                  }
 170              } else {
 171                  if (setlocale(LC_TIME, 0) !== 'en_AU.UTF-8') {
 172                      $warnings[] = 'Warning: unexpected change of locale';
 173                  }
 174              }
 175          }
 176  
 177          if (ini_get('max_execution_time') != 0) {
 178              // This is special warning for all resets because we do not want any
 179              // libraries to mess with timeouts unintentionally.
 180              // Our PHPUnit integration is not supposed to change it either.
 181  
 182              if ($detectchanges !== false) {
 183                  $warnings[] = 'Warning: max_execution_time was changed to '.ini_get('max_execution_time');
 184              }
 185              set_time_limit(0);
 186          }
 187  
 188          // restore original globals
 189          $_SERVER = self::get_global_backup('_SERVER');
 190          $CFG = self::get_global_backup('CFG');
 191          $SITE = self::get_global_backup('SITE');
 192          $_GET = array();
 193          $_POST = array();
 194          $_FILES = array();
 195          $_REQUEST = array();
 196          $COURSE = $SITE;
 197  
 198          // reinitialise following globals
 199          $OUTPUT = new bootstrap_renderer();
 200          $PAGE = new moodle_page();
 201          $FULLME = null;
 202          $ME = null;
 203          $SCRIPT = null;
 204  
 205          // Empty sessison and set fresh new not-logged-in user.
 206          \core\session\manager::init_empty_session();
 207  
 208          // reset all static caches
 209          \core\event\manager::phpunit_reset();
 210          accesslib_clear_all_caches(true);
 211          get_string_manager()->reset_caches(true);
 212          reset_text_filters_cache(true);
 213          events_get_handlers('reset');
 214          core_text::reset_caches();
 215          get_message_processors(false, true);
 216          filter_manager::reset_caches();
 217          core_filetypes::reset_caches();
 218  
 219          // Reset internal users.
 220          core_user::reset_internal_users();
 221  
 222          //TODO MDL-25290: add more resets here and probably refactor them to new core function
 223  
 224          // Reset course and module caches.
 225          if (class_exists('format_base')) {
 226              // If file containing class is not loaded, there is no cache there anyway.
 227              format_base::reset_course_cache(0);
 228          }
 229          get_fast_modinfo(0, 0, true);
 230  
 231          // Reset other singletons.
 232          if (class_exists('core_plugin_manager')) {
 233              core_plugin_manager::reset_caches(true);
 234          }
 235          if (class_exists('\core\update\checker')) {
 236              \core\update\checker::reset_caches(true);
 237          }
 238          if (class_exists('\core\update\deployer')) {
 239              \core\update\deployer::reset_caches(true);
 240          }
 241  
 242          // purge dataroot directory
 243          self::reset_dataroot();
 244  
 245          // restore original config once more in case resetting of caches changed CFG
 246          $CFG = self::get_global_backup('CFG');
 247  
 248          // inform data generator
 249          self::get_data_generator()->reset();
 250  
 251          // fix PHP settings
 252          error_reporting($CFG->debug);
 253  
 254          // Reset the date/time class.
 255          core_date::phpunit_reset();
 256  
 257          // Make sure the time locale is consistent - that is Australian English.
 258          if ($CFG->ostype === 'WINDOWS') {
 259              setlocale(LC_TIME, 'English_Australia.1252');
 260          } else {
 261              setlocale(LC_TIME, 'en_AU.UTF-8');
 262          }
 263  
 264          // verify db writes just in case something goes wrong in reset
 265          if (self::$lastdbwrites != $DB->perf_get_writes()) {
 266              error_log('Unexpected DB writes in phpunit_util::reset_all_data()');
 267              self::$lastdbwrites = $DB->perf_get_writes();
 268          }
 269  
 270          if ($warnings) {
 271              $warnings = implode("\n", $warnings);
 272              trigger_error($warnings, E_USER_WARNING);
 273          }
 274      }

 276      /**
 277       * Reset all database tables to default values.
 278       * @static
 279       * @return bool true if reset done, false if skipped
 280       */
 281      public static function reset_database() {
 282          global $DB;
 283  
 284          if (!is_null(self::$lastdbwrites) and self::$lastdbwrites == $DB->perf_get_writes()) {
 285              return false;
 286          }
 287  
 288          if (!parent::reset_database()) {
 289              return false;
 290          }
 291  
 292          self::$lastdbwrites = $DB->perf_get_writes();
 293  
 294          return true;
 295      }

 326      /**
 327       * Returns original state of global variable.
 328       * @static
 329       * @param string $name
 330       * @return mixed
 331       */
 332      public static function get_global_backup($name) {
 333          if ($name === 'DB') {
 334              // no cloning of database object,
 335              // we just need the original reference, not original state
 336              return self::$globals['DB'];
 337          }
 338          if (isset(self::$globals[$name])) {
 339              if (is_object(self::$globals[$name])) {
 340                  $return = clone(self::$globals[$name]);
 341                  return $return;
 342              } else {
 343                  return self::$globals[$name];
 344              }
 345          }
 346          return null;
 347      }

 355      public static function testing_ready_problem() {
 356          global $DB;
 357  
 358          if (!self::is_test_site()) {
 359              // dataroot was verified in bootstrap, so it must be DB
 360              return array(PHPUNIT_EXITCODE_CONFIGERROR, 'Can not use database for testing, try different prefix');
 361          }
 362  
 363          $tables = $DB->get_tables(false);
 364          if (empty($tables)) {
 365              return array(PHPUNIT_EXITCODE_INSTALL, '');
 366          }
 367  
 368          if (!self::is_test_data_updated()) {
 369              return array(PHPUNIT_EXITCODE_REINSTALL, '');
 370          }
 371  
 372          return array(0, '');
 373      }

 452      /**
 453       * Builds dirroot/phpunit.xml and dataroot/phpunit/webrunner.xml files using defaults from /phpunit.xml.dist
 454       * @static
 455       * @return bool true means main config file created, false means only dataroot file created
 456       */
 457      public static function build_config_file() {
 458          global $CFG;
 459  
 460          $template = '
 461          <testsuite name="@component@_testsuite">
 462              <directory suffix="_test.php">@dir@</directory>
 463          </testsuite>';
 464          $data = file_get_contents("$CFG->dirroot/phpunit.xml.dist");
 465  
 466          $suites = '';
 467  
 468          $plugintypes = core_component::get_plugin_types();
 469          ksort($plugintypes);
 470          foreach ($plugintypes as $type=>$unused) {
 471              $plugs = core_component::get_plugin_list($type);
 472              ksort($plugs);
 473              foreach ($plugs as $plug=>$fullplug) {
 474                  if (!file_exists("$fullplug/tests/")) {
 475                      continue;
 476                  }
 477                  $dir = substr($fullplug, strlen($CFG->dirroot)+1);
 478                  $dir .= '/tests';
 479                  $component = $type.'_'.$plug;
 480  
 481                  $suite = str_replace('@component@', $component, $template);
 482                  $suite = str_replace('@dir@', $dir, $suite);
 483  
 484                  $suites .= $suite;
 485              }
 486          }
 487          // Start a sequence between 100000 and 199000 to ensure each call to init produces
 488          // different ids in the database.  This reduces the risk that hard coded values will
 489          // end up being placed in phpunit or behat test code.
 490          $sequencestart = 100000 + mt_rand(0, 99) * 1000;
 491  
 492          $data = preg_replace('|<!--@plugin_suites_start@-->.*<!--@plugin_suites_end@-->|s', $suites, $data, 1);
 493          $data = str_replace(
 494              '<const name="PHPUNIT_SEQUENCE_START" value=""/>',
 495              '<const name="PHPUNIT_SEQUENCE_START" value="' . $sequencestart . '"/>',
 496              $data);
 497  
 498          $result = false;
 499          if (is_writable($CFG->dirroot)) {
 500              if ($result = file_put_contents("$CFG->dirroot/phpunit.xml", $data)) {
 501                  testing_fix_file_permissions("$CFG->dirroot/phpunit.xml");
 502              }
 503          }
 504  
 505          // relink - it seems that xml:base does not work in phpunit xml files, remove this nasty hack if you find a way to set xml base for relative refs
 506          $data = str_replace('lib/phpunit/', $CFG->dirroot.DIRECTORY_SEPARATOR.'lib'.DIRECTORY_SEPARATOR.'phpunit'.DIRECTORY_SEPARATOR, $data);
 507          $data = preg_replace('|<directory suffix="_test.php">([^<]+)</directory>|',
 508              '<directory suffix="_test.php">'.$CFG->dirroot.(DIRECTORY_SEPARATOR === '\\' ? '\\\\' : DIRECTORY_SEPARATOR).'$1</directory>',
 509              $data);
 510          file_put_contents("$CFG->dataroot/phpunit/webrunner.xml", $data);
 511          testing_fix_file_permissions("$CFG->dataroot/phpunit/webrunner.xml");
 512  
 513          return (bool)$result;
 514      }

 516      /**
 517       * Builds phpunit.xml files for all components using defaults from /phpunit.xml.dist
 518       *
 519       * @static
 520       * @return void, stops if can not write files
 521       */
 522      public static function build_component_config_files() {
 523          global $CFG;
 524  
 525          $template = '
 526          <testsuites>
 527              <testsuite name="@component@">
 528                  <directory suffix="_test.php">.</directory>
 529              </testsuite>
 530          </testsuites>';
 531  
 532          // Start a sequence between 100000 and 199000 to ensure each call to init produces
 533          // different ids in the database.  This reduces the risk that hard coded values will
 534          // end up being placed in phpunit or behat test code.
 535          $sequencestart = 100000 + mt_rand(0, 99) * 1000;
 536  
 537          // Use the upstream file as source for the distributed configurations
 538          $ftemplate = file_get_contents("$CFG->dirroot/phpunit.xml.dist");
 539          $ftemplate = preg_replace('|<!--All core suites.*</testsuites>|s', '<!--@component_suite@-->', $ftemplate);
 540  
 541          // Gets all the components with tests
 542          $components = tests_finder::get_components_with_tests('phpunit');
 543  
 544          // Create the corresponding phpunit.xml file for each component
 545          foreach ($components as $cname => $cpath) {
 546              // Calculate the component suite
 547              $ctemplate = $template;
 548              $ctemplate = str_replace('@component@', $cname, $ctemplate);
 549  
 550              // Apply it to the file template
 551              $fcontents = str_replace('<!--@component_suite@-->', $ctemplate, $ftemplate);
 552              $fcontents = str_replace(
 553                  '<const name="PHPUNIT_SEQUENCE_START" value=""/>',
 554                  '<const name="PHPUNIT_SEQUENCE_START" value="' . $sequencestart . '"/>',
 555                  $fcontents);
 556  
 557              // fix link to schema
 558              $level = substr_count(str_replace('\\', '/', $cpath), '/') - substr_count(str_replace('\\', '/', $CFG->dirroot), '/');
 559              $fcontents = str_replace('lib/phpunit/', str_repeat('../', $level).'lib/phpunit/', $fcontents);
 560  
 561              // Write the file
 562              $result = false;
 563              if (is_writable($cpath)) {
 564                  if ($result = (bool)file_put_contents("$cpath/phpunit.xml", $fcontents)) {
 565                      testing_fix_file_permissions("$cpath/phpunit.xml");
 566                  }
 567              }
 568              // Problems writing file, throw error
 569              if (!$result) {
 570                  phpunit_bootstrap_error(PHPUNIT_EXITCODE_CONFIGWARNING, "Can not create $cpath/phpunit.xml configuration file, verify dir permissions");
 571              }
 572          }
 573      }

 575      /**
 576       * To be called from debugging() only.
 577       * @param string $message
 578       * @param int $level
 579       * @param string $from
 580       */
 581      public static function debugging_triggered($message, $level, $from) {
 582          // Store only if debugging triggered from actual test,
 583          // we need normal debugging outside of tests to find problems in our phpunit integration.
 584          $backtrace = debug_backtrace();
 585  
 586          foreach ($backtrace as $bt) {
 587              $intest = false;
 588              if (isset($bt['object']) and is_object($bt['object'])) {
 589                  if ($bt['object'] instanceof PHPUnit_Framework_TestCase) {
 590                      if (strpos($bt['function'], 'test') === 0) {
 591                          $intest = true;
 592                          break;
 593                      }
 594                  }
 595              }
 596          }
 597          if (!$intest) {
 598              return false;
 599          }
 600  
 601          $debug = new stdClass();
 602          $debug->message = $message;
 603          $debug->level   = $level;
 604          $debug->from    = $from;
 605  
 606          self::$debuggings[] = $debug;
 607  
 608          return true;
 609      }

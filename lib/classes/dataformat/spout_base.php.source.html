  53          $this->writer->openToBrowser($filename);
  54          if ($this->sheettitle && $this->writer instanceof \Box\Spout\Writer\AbstractMultiSheetsWriter) {
  55              $sheet = $this->writer->getCurrentSheet();

  57          }

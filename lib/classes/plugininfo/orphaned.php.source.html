  63       *

  65       * @param string $typerootdir full path to the location of the plugin dir

  69      public static function get_plugins($type, $typerootdir, $typeclass) {
  70          $return = array();
  71          $manager = \core_plugin_manager::instance();
  72          $plugins = $manager->get_installed_plugins($type);
  73  
  74          foreach ($plugins as $name => $version) {
  75              $plugin              = new $typeclass();
  76              $plugin->type        = $type;
  77              $plugin->typerootdir = $typerootdir;
  78              $plugin->name        = $name;
  79              $plugin->rootdir     = null;
  80              $plugin->displayname = $name;
  81              $plugin->versiondb   = $version;
  82              $plugin->init_is_standard();
  83  
  84              $return[$name] = $plugin;
  85          }
  86  
  87          return $return;

 141        $respHttpCode = $this->getHttpResponseCode($http_response_header);

 152      }

 155  
 156      $this->client->getLogger()->debug(
 157          'Stream response',
 158          array(
 159              'code' => $respHttpCode,
 160              'headers' => $responseHeaders,
 161              'body' => $response_data,
 162          )
 163      );

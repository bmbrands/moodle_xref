  74      /**
  75       * Helper function for walking the Mustache token parse tree.
  76       *
  77       * @throws Mustache_Exception_SyntaxException upon encountering unknown token types.
  78       *
  79       * @param array $tree  Parse tree of Mustache tokens
  80       * @param int   $level (default: 0)
  81       *
  82       * @return string Generated PHP source code
  83       */
  84      private function walk(array $tree, $level = 0)
  85      {
  86          $code = '';
  87          $level++;
  88          foreach ($tree as $node) {
  89              switch ($node[Mustache_Tokenizer::TYPE]) {
  90                  case Mustache_Tokenizer::T_PRAGMA:
  91                      $this->pragmas[$node[Mustache_Tokenizer::NAME]] = true;
  92                      break;
  93  
  94                  case Mustache_Tokenizer::T_SECTION:
  95                      $code .= $this->section(
  96                          $node[Mustache_Tokenizer::NODES],
  97                          $node[Mustache_Tokenizer::NAME],
  98                          isset($node[Mustache_Tokenizer::FILTERS]) ? $node[Mustache_Tokenizer::FILTERS] : array(),
  99                          $node[Mustache_Tokenizer::INDEX],
 100                          $node[Mustache_Tokenizer::END],
 101                          $node[Mustache_Tokenizer::OTAG],
 102                          $node[Mustache_Tokenizer::CTAG],
 103                          $level
 104                      );
 105                      break;
 106  
 107                  case Mustache_Tokenizer::T_INVERTED:
 108                      $code .= $this->invertedSection(
 109                          $node[Mustache_Tokenizer::NODES],
 110                          $node[Mustache_Tokenizer::NAME],
 111                          isset($node[Mustache_Tokenizer::FILTERS]) ? $node[Mustache_Tokenizer::FILTERS] : array(),
 112                          $level
 113                      );
 114                      break;
 115  
 116                  case Mustache_Tokenizer::T_PARTIAL:
 117                      $code .= $this->partial(
 118                          $node[Mustache_Tokenizer::NAME],
 119                          isset($node[Mustache_Tokenizer::INDENT]) ? $node[Mustache_Tokenizer::INDENT] : '',
 120                          $level
 121                      );
 122                      break;
 123  
 124                  case Mustache_Tokenizer::T_PARENT:
 125                      $code .= $this->parent(
 126                          $node[Mustache_Tokenizer::NAME],
 127                          isset($node[Mustache_Tokenizer::INDENT]) ? $node[Mustache_Tokenizer::INDENT] : '',
 128                          $node[Mustache_Tokenizer::NODES],
 129                          $level
 130                      );
 131                      break;
 132  
 133                  case Mustache_Tokenizer::T_BLOCK_ARG:
 134                      $code .= $this->blockArg(
 135                          $node[Mustache_Tokenizer::NODES],
 136                          $node[Mustache_Tokenizer::NAME],
 137                          $node[Mustache_Tokenizer::INDEX],
 138                          $node[Mustache_Tokenizer::END],
 139                          $node[Mustache_Tokenizer::OTAG],
 140                          $node[Mustache_Tokenizer::CTAG],
 141                          $level
 142                      );
 143                      break;
 144  
 145                  case Mustache_Tokenizer::T_BLOCK_VAR:
 146                      $code .= $this->blockVar(
 147                          $node[Mustache_Tokenizer::NODES],
 148                          $node[Mustache_Tokenizer::NAME],
 149                          $node[Mustache_Tokenizer::INDEX],
 150                          $node[Mustache_Tokenizer::END],
 151                          $node[Mustache_Tokenizer::OTAG],
 152                          $node[Mustache_Tokenizer::CTAG],
 153                          $level
 154                      );
 155                      break;
 156  
 157                  case Mustache_Tokenizer::T_COMMENT:
 158                      break;
 159  
 160                  case Mustache_Tokenizer::T_ESCAPED:
 161                  case Mustache_Tokenizer::T_UNESCAPED:
 162                  case Mustache_Tokenizer::T_UNESCAPED_2:
 163                      $code .= $this->variable(
 164                          $node[Mustache_Tokenizer::NAME],
 165                          isset($node[Mustache_Tokenizer::FILTERS]) ? $node[Mustache_Tokenizer::FILTERS] : array(),
 166                          $node[Mustache_Tokenizer::TYPE] === Mustache_Tokenizer::T_ESCAPED,
 167                          $level
 168                      );
 169                      break;
 170  
 171                  case Mustache_Tokenizer::T_TEXT:
 172                      $code .= $this->text($node[Mustache_Tokenizer::VALUE], $level);
 173                      break;
 174  
 175                  default:
 176                      throw new Mustache_Exception_SyntaxException(sprintf('Unknown token type: %s', $node[Mustache_Tokenizer::TYPE]), $node);
 177              }
 178          }
 179  
 180          return $code;
 181      }

 324      /**
 325       * Generate Mustache Template section PHP source.
 326       *
 327       * @param array    $nodes   Array of child tokens
 328       * @param string   $id      Section name
 329       * @param string[] $filters Array of filters
 330       * @param int      $start   Section start offset
 331       * @param int      $end     Section end offset
 332       * @param string   $otag    Current Mustache opening tag
 333       * @param string   $ctag    Current Mustache closing tag
 334       * @param int      $level
 335       * @param bool     $arg     (default: false)
 336       *
 337       * @return string Generated section PHP source code

 339      private function section($nodes, $id, $filters, $start, $end, $otag, $ctag, $level, $arg = false)
 340      {
 341          $source   = var_export(substr($this->source, $start, $end - $start), true);
 342          $callable = $this->getCallable();
 343  
 344          if ($otag !== '{{' || $ctag !== '}}') {
 345              $delims = ', '.var_export(sprintf('{{= %s %s =}}', $otag, $ctag), true);
 346          } else {
 347              $delims = '';
 348          }
 349  
 350          $key = ucfirst(md5($delims."\n".$source));
 351  
 352          if (!isset($this->sections[$key])) {
 353              $this->sections[$key] = sprintf($this->prepare(self::SECTION), $key, $callable, $source, $delims, $this->walk($nodes, 2));
 354          }
 355  
 356          if ($arg === true) {
 357              return $key;
 358          } else {
 359              $method  = $this->getFindMethod($id);
 360              $id      = var_export($id, true);
 361              $filters = $this->getFilters($filters, $level);
 362  
 363              return sprintf($this->prepare(self::SECTION_CALL, $level), $id, $method, $id, $filters, $key);
 364          }
 365      }

 545       * Prepare PHP source code snippet for output.
 546       *
 547       * @param string  $text
 548       * @param int     $bonus          Additional indent level (default: 0)
 549       * @param boolean $prependNewline Prepend a newline to the snippet? (default: true)
 550       * @param boolean $appendNewline  Append a newline to the snippet? (default: false)
 551       *
 552       * @return string PHP source code snippet
 553       */
 554      private function prepare($text, $bonus = 0, $prependNewline = true, $appendNewline = false)
 555      {
 556          $text = ($prependNewline ? "\n" : '').trim($text);
 557          if ($prependNewline) {

 559          }
 560          if ($appendNewline) {

 564          return preg_replace("/\n( {8})?/", "\n".str_repeat(" ", $bonus * 4), $text);

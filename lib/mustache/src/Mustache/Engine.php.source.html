 672      /**
 673       * Instantiate and return a Mustache Template instance by source.
 674       *
 675       * Optionally provide a Mustache_Cache instance. This is used internally by Mustache_Engine::loadLambda to respect
 676       * the 'cache_lambda_templates' configuration option.
 677       *
 678       * @see Mustache_Engine::loadTemplate
 679       * @see Mustache_Engine::loadPartial
 680       * @see Mustache_Engine::loadLambda
 681       *
 682       * @param string         $source
 683       * @param Mustache_Cache $cache  (default: null)
 684       *
 685       * @return Mustache_Template
 686       */
 687      private function loadSource($source, Mustache_Cache $cache = null)
 688      {
 689          $className = $this->getTemplateClassName($source);
 690  
 691          if (!isset($this->templates[$className])) {
 692              if ($cache === null) {
 693                  $cache = $this->getCache();
 694              }
 695  
 696              if (!class_exists($className, false)) {
 697                  if (!$cache->load($className)) {
 698                      $compiled = $this->compile($source);
 699                      $cache->cache($className, $compiled);
 700                  }
 701              }
 702  
 703              $this->log(
 704                  Mustache_Logger::DEBUG,
 705                  'Instantiating template: "{className}"',
 706                  array('className' => $className)
 707              );
 708  
 709              $this->templates[$className] = new $className($this);
 710          }
 711  
 712          return $this->templates[$className];
 713      }

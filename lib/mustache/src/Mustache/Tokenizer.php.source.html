  15   * This class is responsible for turning raw template source into a set of Mustache tokens.

  88      /**
  89       * Scan and tokenize template source.
  90       *
  91       * @throws Mustache_Exception_SyntaxException when mismatched section tags are encountered.
  92       *
  93       * @param string $text       Mustache template source to tokenize
  94       * @param string $delimiters Optionally, pass initial opening and closing delimiters (default: null)
  95       *
  96       * @return array Set of Mustache tokens
  97       */
  98      public function scan($text, $delimiters = null)
  99      {
 100          // Setting mbstring.func_overload makes things *really* slow.
 101          // Let's do everyone a favor and scan this string as ASCII instead.
 102          $encoding = null;
 103          if (function_exists('mb_internal_encoding') && ini_get('mbstring.func_overload') & 2) {
 104              $encoding = mb_internal_encoding();
 105              mb_internal_encoding('ASCII');
 106          }
 107  
 108          $this->reset();
 109  
 110          if ($delimiters = trim($delimiters)) {
 111              $this->setDelimiters($delimiters);
 112          }
 113  
 114          $len = strlen($text);
 115          for ($i = 0; $i < $len; $i++) {
 116              switch ($this->state) {
 117                  case self::IN_TEXT:
 118                      if ($this->tagChange($this->otag, $this->otagLen, $text, $i)) {
 119                          $i--;
 120                          $this->flushBuffer();
 121                          $this->state = self::IN_TAG_TYPE;
 122                      } else {
 123                          $char = $text[$i];
 124                          $this->buffer .= $char;
 125                          if ($char === "\n") {
 126                              $this->flushBuffer();
 127                              $this->line++;
 128                          }
 129                      }
 130                      break;
 131  
 132                  case self::IN_TAG_TYPE:
 133                      $i += $this->otagLen - 1;
 134                      $char = $text[$i + 1];
 135                      if (isset(self::$tagTypes[$char])) {
 136                          $tag = $char;
 137                          $this->tagType = $tag;
 138                      } else {
 139                          $tag = null;
 140                          $this->tagType = self::T_ESCAPED;
 141                      }
 142  
 143                      if ($this->tagType === self::T_DELIM_CHANGE) {
 144                          $i = $this->changeDelimiters($text, $i);
 145                          $this->state = self::IN_TEXT;
 146                      } elseif ($this->tagType === self::T_PRAGMA) {
 147                          $i = $this->addPragma($text, $i);
 148                          $this->state = self::IN_TEXT;
 149                      } else {
 150                          if ($tag !== null) {
 151                              $i++;
 152                          }
 153                          $this->state = self::IN_TAG;
 154                      }
 155                      $this->seenTag = $i;
 156                      break;
 157  
 158                  default:
 159                      if ($this->tagChange($this->ctag, $this->ctagLen, $text, $i)) {
 160                          $token = array(
 161                              self::TYPE  => $this->tagType,
 162                              self::NAME  => trim($this->buffer),
 163                              self::OTAG  => $this->otag,
 164                              self::CTAG  => $this->ctag,
 165                              self::LINE  => $this->line,
 166                              self::INDEX => ($this->tagType === self::T_END_SECTION) ? $this->seenTag - $this->otagLen : $i + $this->ctagLen
 167                          );
 168  
 169                          if ($this->tagType === self::T_UNESCAPED) {
 170                              // Clean up `{{{ tripleStache }}}` style tokens.
 171                              if ($this->ctag === '}}') {
 172                                  if (($i + 2 < $len) && $text[$i + 2] === '}') {
 173                                      $i++;
 174                                  } else {
 175                                      $msg = sprintf(
 176                                          'Mismatched tag delimiters: %s on line %d',
 177                                          $token[self::NAME],
 178                                          $token[self::LINE]
 179                                      );
 180  
 181                                      throw new Mustache_Exception_SyntaxException($msg, $token);
 182                                  }
 183                              } else {
 184                                  $lastName = $token[self::NAME];
 185                                  if (substr($lastName, -1) === '}') {
 186                                      $token[self::NAME] = trim(substr($lastName, 0, -1));
 187                                  } else {
 188                                      $msg = sprintf(
 189                                          'Mismatched tag delimiters: %s on line %d',
 190                                          $token[self::NAME],
 191                                          $token[self::LINE]
 192                                      );
 193  
 194                                      throw new Mustache_Exception_SyntaxException($msg, $token);
 195                                  }
 196                              }
 197                          }
 198  
 199                          $this->buffer = '';
 200                          $i += $this->ctagLen - 1;
 201                          $this->state = self::IN_TEXT;
 202                          $this->tokens[] = $token;
 203                      } else {
 204                          $this->buffer .= $text[$i];
 205                      }
 206                      break;
 207              }
 208          }
 209  
 210          $this->flushBuffer();
 211  
 212          // Restore the user's encoding...
 213          if ($encoding) {
 214              mb_internal_encoding($encoding);
 215          }
 216  
 217          return $this->tokens;
 218      }

 220      /**
 221       * Helper function to reset tokenizer internal state.

 223      private function reset()
 224      {
 225          $this->state   = self::IN_TEXT;
 226          $this->tagType = null;
 227          $this->tag     = null;
 228          $this->buffer  = '';
 229          $this->tokens  = array();
 230          $this->seenTag = false;
 231          $this->line    = 0;
 232          $this->otag    = '{{';
 233          $this->ctag    = '}}';
 234          $this->otagLen = 2;
 235          $this->ctagLen = 2;
 236      }

 238      /**
 239       * Flush the current buffer to a token.
 240       */
 241      private function flushBuffer()
 242      {
 243          if (strlen($this->buffer) > 0) {
 244              $this->tokens[] = array(
 245                  self::TYPE  => self::T_TEXT,
 246                  self::LINE  => $this->line,
 247                  self::VALUE => $this->buffer
 248              );
 249              $this->buffer   = '';
 250          }
 251      }

 253      /**
 254       * Change the current Mustache delimiters. Set new `otag` and `ctag` values.
 255       *
 256       * @param string $text  Mustache template source
 257       * @param int    $index Current tokenizer index
 258       *
 259       * @return int New index value
 260       */
 261      private function changeDelimiters($text, $index)
 262      {
 263          $startIndex = strpos($text, '=', $index) + 1;
 264          $close      = '='.$this->ctag;
 265          $closeIndex = strpos($text, $close, $index);
 266  
 267          $this->setDelimiters(trim(substr($text, $startIndex, $closeIndex - $startIndex)));
 268  
 269          $this->tokens[] = array(
 270              self::TYPE => self::T_DELIM_CHANGE,
 271              self::LINE => $this->line,
 272          );
 273  
 274          return $closeIndex + strlen($close) - 1;
 275      }
 276  
 277      /**
 278       * Set the current Mustache `otag` and `ctag` delimiters.
 279       *
 280       * @param string $delimiters
 281       */
 282      private function setDelimiters($delimiters)
 283      {
 284          list($otag, $ctag) = explode(' ', $delimiters);
 285          $this->otag = $otag;
 286          $this->ctag = $ctag;
 287          $this->otagLen = strlen($otag);
 288          $this->ctagLen = strlen($ctag);
 289      }
 290  
 291      /**
 292       * Add pragma token.
 293       *
 294       * Pragmas are hoisted to the front of the template, so all pragma tokens
 295       * will appear at the front of the token list.
 296       *
 297       * @param string $text
 298       * @param int    $index
 299       *
 300       * @return int New index value
 301       */
 302      private function addPragma($text, $index)
 303      {
 304          $end    = strpos($text, $this->ctag, $index);
 305          $pragma = trim(substr($text, $index + 2, $end - $index - 2));
 306  
 307          // Pragmas are hoisted to the front of the template.
 308          array_unshift($this->tokens, array(
 309              self::TYPE => self::T_PRAGMA,
 310              self::NAME => $pragma,
 311              self::LINE => 0,
 312          ));
 313  
 314          return $end + $this->ctagLen - 1;
 315      }
 316  
 317      /**
 318       * Test whether it's time to change tags.
 319       *
 320       * @param string $tag    Current tag name
 321       * @param int    $tagLen Current tag name length
 322       * @param string $text   Mustache template source
 323       * @param int    $index  Current tokenizer index
 324       *
 325       * @return boolean True if this is a closing section tag
 326       */
 327      private function tagChange($tag, $tagLen, $text, $index)
 328      {
 329          return substr($text, $index, $tagLen) === $tag;
 330      }

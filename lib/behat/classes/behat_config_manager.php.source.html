  46       * Updates a config file
  47       *
  48       * The tests runner and the steps definitions list uses different
  49       * config files to avoid problems with concurrent executions.
  50       *
  51       * The steps definitions list can be filtered by component so it's
  52       * behat.yml is different from the $CFG->dirroot one.
  53       *
  54       * @param  string $component Restricts the obtained steps definitions to the specified component
  55       * @param  string $testsrunner If the config file will be used to run tests
  56       * @param  string $tags features files including tags.
  57       * @return void
  58       */
  59      public static function update_config_file($component = '', $testsrunner = true, $tags = '') {
  60          global $CFG;
  61  
  62          // Behat must have a separate behat.yml to have access to the whole set of features and steps definitions.
  63          if ($testsrunner === true) {
  64              $configfilepath = behat_command::get_behat_dir() . '/behat.yml';
  65          } else {
  66              // Alternative for steps definitions filtering, one for each user.
  67              $configfilepath = self::get_steps_list_config_filepath();
  68          }
  69  
  70          // Gets all the components with features.
  71          $features = array();
  72          $components = tests_finder::get_components_with_tests('features');
  73          if ($components) {
  74              foreach ($components as $componentname => $path) {
  75                  $path = self::clean_path($path) . self::get_behat_tests_path();
  76                  if (empty($featurespaths[$path]) && file_exists($path)) {
  77  
  78                      // Standarizes separator (some dirs. comes with OS-dependant separator).
  79                      $uniquekey = str_replace('\\', '/', $path);
  80                      $featurespaths[$uniquekey] = $path;
  81                  }
  82              }
  83              foreach ($featurespaths as $path) {
  84                  $additional = glob("$path/*.feature");
  85                  $features = array_merge($features, $additional);
  86              }
  87          }
  88  
  89          // Optionally include features from additional directories.
  90          if (!empty($CFG->behat_additionalfeatures)) {
  91              $features = array_merge($features, array_map("realpath", $CFG->behat_additionalfeatures));
  92          }
  93  
  94          // Gets all the components with steps definitions.
  95          $stepsdefinitions = array();
  96          $steps = self::get_components_steps_definitions();
  97          if ($steps) {
  98              foreach ($steps as $key => $filepath) {
  99                  if ($component == '' || $component === $key) {
 100                      $stepsdefinitions[$key] = $filepath;
 101                  }
 102              }
 103          }
 104  
 105          // We don't want the deprecated steps definitions here.
 106          if (!$testsrunner) {
 107              unset($stepsdefinitions['behat_deprecated']);
 108          }
 109  
 110          // Behat config file specifing the main context class,
 111          // the required Behat extensions and Moodle test wwwroot.
 112          $contents = self::get_config_file_contents(self::get_features_with_tags($features, $tags), $stepsdefinitions);
 113  
 114          // Stores the file.
 115          if (!file_put_contents($configfilepath, $contents)) {
 116              behat_error(BEHAT_EXITCODE_PERMISSIONS, 'File ' . $configfilepath . ' can not be created');
 117          }
 118  
 119      }

 121      /**
 122       * Search feature files for set of tags.
 123       *
 124       * @param array $features set of feature files.
 125       * @param string $tags list of tags (currently support && only.)
 126       * @return array filtered list of feature files with tags.
 127       */
 128      public static function get_features_with_tags($features, $tags) {
 129          if (empty($tags)) {
 130              return $features;
 131          }
 132          $newfeaturelist = array();
 133          // Split tags in and and or.
 134          $tags = explode('&&', $tags);
 135          $andtags = array();
 136          $ortags = array();
 137          foreach ($tags as $tag) {
 138              // Explode all tags seperated by , and add it to ortags.
 139              $ortags = array_merge($ortags, explode(',', $tag));
 140              // And tags will be the first one before comma(,).
 141              $andtags[] = preg_replace('/,.*/', '', $tag);
 142          }
 143  
 144          foreach ($features as $featurefile) {
 145              $contents = file_get_contents($featurefile);
 146              $includefeature = true;
 147              foreach ($andtags as $tag) {
 148                  // If negitive tag, then ensure it don't exist.
 149                  if (strpos($tag, '~') !== false) {
 150                      $tag = substr($tag, 1);
 151                      if ($contents && strpos($contents, $tag) !== false) {
 152                          $includefeature = false;
 153                          break;
 154                      }
 155                  } else if ($contents && strpos($contents, $tag) === false) {
 156                      $includefeature = false;
 157                      break;
 158                  }
 159              }
 160  
 161              // If feature not included then check or tags.
 162              if (!$includefeature && !empty($ortags)) {
 163                  foreach ($ortags as $tag) {
 164                      if ($contents && (strpos($tag, '~') === false) && (strpos($contents, $tag) !== false)) {
 165                          $includefeature = true;
 166                          break;
 167                      }
 168                  }
 169              }
 170  
 171              if ($includefeature) {
 172                  $newfeaturelist[] = $featurefile;
 173              }
 174          }
 175          return $newfeaturelist;
 176      }

 188      public static function get_components_steps_definitions() {
 189  
 190          $components = tests_finder::get_components_with_tests('stepsdefinitions');
 191          if (!$components) {
 192              return false;
 193          }
 194  
 195          $stepsdefinitions = array();
 196          foreach ($components as $componentname => $componentpath) {
 197              $componentpath = self::clean_path($componentpath);
 198  
 199              if (!file_exists($componentpath . self::get_behat_tests_path())) {
 200                  continue;
 201              }
 202              $diriterator = new DirectoryIterator($componentpath . self::get_behat_tests_path());
 203              $regite = new RegexIterator($diriterator, '|behat_.*\.php$|');
 204  
 205              // All behat_*.php inside behat_config_manager::get_behat_tests_path() are added as steps definitions files.
 206              foreach ($regite as $file) {
 207                  $key = $file->getBasename('.php');
 208                  $stepsdefinitions[$key] = $file->getPathname();
 209              }
 210          }
 211  
 212          return $stepsdefinitions;

 294      public final static function drop_parallel_site_links() {
 295          global $CFG;
 296  
 297          // Get parallel test runs from first run.
 298          $parallelrun = self::get_parallel_test_runs(1);
 299  
 300          if (empty($parallelrun)) {
 301              return false;
 302          }
 303  
 304          // If parallel run then remove links and original file.
 305          clearstatcache();
 306          for ($i = 1; $i <= $parallelrun; $i++) {
 307              // Don't delete links for specified sites, as they should be accessible.
 308              if (!empty($CFG->behat_parallel_run['behat_wwwroot'][$i - 1]['behat_wwwroot'])) {
 309                  continue;
 310              }
 311              $link = $CFG->dirroot . '/' . BEHAT_PARALLEL_SITE_NAME . $i;
 312              if (file_exists($link) && is_link($link)) {
 313                  @unlink($link);
 314              }
 315          }
 316          return true;

 326      public final static function create_parallel_site_links($fromrun, $torun) {
 327          global $CFG;
 328  
 329          // Create site symlink if necessary.
 330          clearstatcache();
 331          for ($i = $fromrun; $i <= $torun; $i++) {
 332              // Don't create links for specified sites, as they should be accessible.
 333              if (!empty($CFG->behat_parallel_run['behat_wwwroot'][$i - 1]['behat_wwwroot'])) {
 334                  continue;
 335              }
 336              $link = $CFG->dirroot.'/'.BEHAT_PARALLEL_SITE_NAME.$i;
 337              clearstatcache();
 338              if (file_exists($link)) {
 339                  if (!is_link($link) || !is_dir($link)) {
 340                      echo "File exists at link location ($link) but is not a link or directory!" . PHP_EOL;
 341                      return false;
 342                  }
 343              } else if (!symlink($CFG->dirroot, $link)) {
 344                  // Try create link in case it's not already present.
 345                  echo "Unable to create behat site symlink ($link)" . PHP_EOL;
 346                  return false;
 347              }
 348          }
 349          return true;
 350      }

 361          global $CFG;
 362  
 363          // We require here when we are sure behat dependencies are available.
 364          require_once($CFG->dirroot . '/vendor/autoload.php');
 365  
 366          $selenium2wdhost = array('wd_host' => 'http://localhost:4444/wd/hub');
 367  
 368          $parallelruns = self::get_parallel_test_runs();
 369          // If parallel run, then only divide features.
 370          if (!empty($CFG->behatrunprocess) && !empty($parallelruns)) {
 371              // Attempt to split into weighted buckets using timing information, if available.
 372              if ($alloc = self::profile_guided_allocate($features, max(1, $parallelruns), $CFG->behatrunprocess)) {
 373                  $features = $alloc;
 374              } else {
 375                  // Divide the list of feature files amongst the parallel runners.
 376                  srand(crc32(floor(time() / 3600 / 24) . var_export($features, true)));
 377                  shuffle($features);
 378                  // Pull out the features for just this worker.
 379                  if (count($features)) {
 380                      $features = array_chunk($features, ceil(count($features) / max(1, $parallelruns)));
 381                      // Check if there is any feature file for this process.
 382                      if (!empty($features[$CFG->behatrunprocess - 1])) {
 383                          $features = $features[$CFG->behatrunprocess - 1];

 385                          $features = null;
 386                      }
 387                  }
 388              }
 389              // Set proper selenium2 wd_host if defined.
 390              if (!empty($CFG->behat_parallel_run[$CFG->behatrunprocess - 1]['wd_host'])) {
 391                  $selenium2wdhost = array('wd_host' => $CFG->behat_parallel_run[$CFG->behatrunprocess - 1]['wd_host']);
 392              }
 393          }
 394  
 395          // It is possible that it has no value as we don't require a full behat setup to list the step definitions.

 397              $CFG->behat_wwwroot = 'http://itwillnotbeused.com';

 399  
 400          $basedir = $CFG->dirroot . DIRECTORY_SEPARATOR . 'lib' . DIRECTORY_SEPARATOR . 'behat';

 402          $config = array(

 441      /**
 442       * Attempt to split feature list into fairish buckets using timing information, if available.
 443       * Simply add each one to lightest buckets until all files allocated.
 444       * PGA = Profile Guided Allocation. I made it up just now.
 445       * CAUTION: workers must agree on allocation, do not be random anywhere!
 446       *
 447       * @param array $features Behat feature files array
 448       * @param int $nbuckets Number of buckets to divide into
 449       * @param int $instance Index number of this instance
 450       * @return array Feature files array, sorted into allocations
 451       */
 452      protected static function profile_guided_allocate($features, $nbuckets, $instance) {
 453  
 454          $behattimingfile = defined('BEHAT_FEATURE_TIMING_FILE') &&
 455              @filesize(BEHAT_FEATURE_TIMING_FILE) ? BEHAT_FEATURE_TIMING_FILE : false;
 456  
 457          if (!$behattimingfile || !$behattimingdata = @json_decode(file_get_contents($behattimingfile), true)) {
 458              // No data available, fall back to relying on steps data.
 459              $stepfile = "";
 460              if (defined('BEHAT_FEATURE_STEP_FILE') && BEHAT_FEATURE_STEP_FILE) {
 461                  $stepfile = BEHAT_FEATURE_STEP_FILE;
 462              }
 463              // We should never get this. But in case we can't do this then fall back on simple splitting.
 464              if (empty($stepfile) || !$behattimingdata = @json_decode(file_get_contents($stepfile), true)) {
 465                  return false;
 466              }
 467          }
 468  
 469          arsort($behattimingdata); // Ensure most expensive is first.
 470  
 471          $realroot = realpath(__DIR__.'/../../../').'/';
 472          $defaultweight = array_sum($behattimingdata) / count($behattimingdata);
 473          $weights = array_fill(0, $nbuckets, 0);
 474          $buckets = array_fill(0, $nbuckets, array());
 475          $totalweight = 0;
 476  
 477          // Re-key the features list to match timing data.
 478          foreach ($features as $k => $file) {
 479              $key = str_replace($realroot, '', $file);
 480              $features[$key] = $file;
 481              unset($features[$k]);
 482              if (!isset($behattimingdata[$key])) {
 483                  $behattimingdata[$key] = $defaultweight;
 484              }
 485          }
 486  
 487          // Sort features by known weights; largest ones should be allocated first.
 488          $behattimingorder = array();
 489          foreach ($features as $key => $file) {
 490              $behattimingorder[$key] = $behattimingdata[$key];
 491          }
 492          arsort($behattimingorder);
 493  
 494          // Finally, add each feature one by one to the lightest bucket.
 495          foreach ($behattimingorder as $key => $weight) {
 496              $file = $features[$key];
 497              $lightbucket = array_search(min($weights), $weights);
 498              $weights[$lightbucket] += $weight;
 499              $buckets[$lightbucket][] = $file;
 500              $totalweight += $weight;
 501          }
 502  
 503          if ($totalweight && !defined('BEHAT_DISABLE_HISTOGRAM') && $instance == $nbuckets) {
 504              echo "Bucket weightings:\n";
 505              foreach ($weights as $k => $weight) {
 506                  echo $k + 1 . ": " . str_repeat('*', 70 * $nbuckets * $weight / $totalweight) . PHP_EOL;
 507              }
 508          }
 509  
 510          // Return the features for this worker.
 511          return $buckets[$instance - 1];
 512      }

 514      /**
 515       * Overrides default config with local config values
 516       *
 517       * array_merge does not merge completely the array's values
 518       *
 519       * @param mixed $config The node of the default config
 520       * @param mixed $localconfig The node of the local config
 521       * @return mixed The merge result
 522       */
 523      protected static function merge_config($config, $localconfig) {
 524  
 525          if (!is_array($config) && !is_array($localconfig)) {
 526              return $localconfig;
 527          }
 528  
 529          // Local overrides also deeper default values.
 530          if (is_array($config) && !is_array($localconfig)) {
 531              return $localconfig;
 532          }
 533  
 534          foreach ($localconfig as $key => $value) {
 535  
 536              // If defaults are not as deep as local values let locals override.
 537              if (!is_array($config)) {
 538                  unset($config);
 539              }
 540  
 541              // Add the param if it doesn't exists or merge branches.
 542              if (empty($config[$key])) {
 543                  $config[$key] = $value;
 544              } else {
 545                  $config[$key] = self::merge_config($config[$key], $localconfig[$key]);
 546              }
 547          }
 548  
 549          return $config;
 550      }

 174      public function test_create_module() {
 175          global $CFG, $SITE, $DB;
 176          if (!file_exists("$CFG->dirroot/mod/page/")) {
 177              $this->markTestSkipped('Can not find standard Page module');
 178          }
 179  
 180          $this->resetAfterTest(true);
 181          $generator = $this->getDataGenerator();
 182  
 183          $page = $generator->create_module('page', array('course'=>$SITE->id));
 184          $this->assertNotEmpty($page);
 185          $cm = get_coursemodule_from_instance('page', $page->id, $SITE->id, true);
 186          $this->assertEquals(0, $cm->sectionnum);
 187  
 188          $page = $generator->create_module('page', array('course'=>$SITE->id), array('section'=>3));
 189          $this->assertNotEmpty($page);
 190          $cm = get_coursemodule_from_instance('page', $page->id, $SITE->id, true);
 191          $this->assertEquals(3, $cm->sectionnum);
 192  
 193          // Prepare environment to generate modules with all possible options.
 194  
 195          // Enable advanced functionality.
 196          $CFG->enablecompletion = 1;
 197          $CFG->enableavailability = 1;
 198          $CFG->enableoutcomes = 1;
 199          require_once($CFG->libdir.'/gradelib.php');
 200          require_once($CFG->libdir.'/completionlib.php');
 201          require_once($CFG->dirroot.'/rating/lib.php');
 202  
 203          // Create a course with enabled completion.
 204          $course = $generator->create_course(array('enablecompletion' => true));
 205  
 206          // Create new grading category in this course.
 207          $grade_category = new grade_category();
 208          $grade_category->courseid = $course->id;
 209          $grade_category->fullname = 'Grade category';
 210          $grade_category->insert();
 211  
 212          // Create group and grouping.
 213          $group = $generator->create_group(array('courseid' => $course->id));
 214          $grouping = $generator->create_grouping(array('courseid' => $course->id));
 215          $generator->create_grouping_group(array('groupid' => $group->id, 'groupingid' => $grouping->id));
 216  
 217          // Prepare arrays with properties that we can both use for creating modules and asserting the data in created modules.
 218  
 219          // General properties.
 220          $optionsgeneral = array(
 221              'visible' => 0, // Note: 'visibleold' will always be set to the same value as 'visible'.
 222              'section' => 3, // Note: section will be created if does not exist.
 223              // Module supports FEATURE_IDNUMBER.
 224              'cmidnumber' => 'IDNUM', // Note: alternatively can have key 'idnumber'.
 225              // Module supports FEATURE_GROUPS;
 226              'groupmode' => SEPARATEGROUPS, // Note: will be reset to 0 if course groupmodeforce is set.
 227              // Module supports FEATURE_GROUPINGS.
 228              'groupingid' => $grouping->id,
 229          );
 230  
 231          // In case completion is enabled on site and for course every module can have manual completion.
 232          $featurecompletionmanual = array(
 233              'completion' => COMPLETION_TRACKING_MANUAL, // "Students can manually mark activity as completed."
 234              'completionexpected' => time() + 7 * DAYSECS,
 235          );
 236  
 237          // Automatic completion is possible if module supports FEATURE_COMPLETION_TRACKS_VIEWS or FEATURE_GRADE_HAS_GRADE.
 238          // Note: completionusegrade is stored in DB and can be found in cm_info as 'completiongradeitemnumber' - either NULL or 0.
 239          // Note: module can have more autocompletion rules as defined in moodleform_mod::add_completion_rules().
 240          $featurecompletionautomatic = array(
 241              'completion' => COMPLETION_TRACKING_AUTOMATIC, // "Show activity as complete when conditions are met."
 242              'completionview' => 1, // "Student must view this activity to complete it"
 243              'completionusegrade' => 1, // "Student must receive a grade to complete this activity"
 244          );
 245  
 246          // Module supports FEATURE_RATE:
 247          $featurerate = array(
 248              'assessed' => RATING_AGGREGATE_AVERAGE, // "Aggregate type"
 249              'scale' => 100, // Either max grade or negative number for scale id.
 250              'ratingtime' => 1, // "Restrict ratings to items with dates in this range".
 251              'assesstimestart' => time() - DAYSECS, // Note: Will be ignored if neither 'assessed' nor 'ratingtime' is set.
 252              'assesstimefinish' => time() + DAYSECS, // Note: Will be ignored if neither 'assessed' nor 'ratingtime' is set.
 253          );
 254  
 255          // Module supports FEATURE_GRADE_HAS_GRADE:
 256          $featuregrade = array(
 257              'grade' => 10,
 258              'gradecat' => $grade_category->id, // Note: if $CFG->enableoutcomes is set, this can be set to -1 to automatically create new grade category.
 259          );
 260  
 261          // Now let's create several modules with different options.
 262          $m1 = $generator->create_module('assign',
 263              array('course' => $course->id) +
 264              $optionsgeneral);
 265          $m2 = $generator->create_module('data',
 266              array('course' => $course->id) +
 267              $featurecompletionmanual +
 268              $featurerate);
 269          $m3 = $generator->create_module('assign',
 270              array('course' => $course->id) +
 271              $featurecompletionautomatic +
 272              $featuregrade);
 273  
 274          // We need id of the grading item for the second module to create availability dependency in the 3rd module.
 275          $gradingitem = grade_item::fetch(array('courseid'=>$course->id, 'itemtype'=>'mod', 'itemmodule' => 'assign', 'iteminstance' => $m3->id));
 276  
 277          // Now prepare option to create the 4th module with an availability condition.
 278          $optionsavailability = array(
 279              'availability' => '{"op":"&","showc":[true],"c":[' .
 280                  '{"type":"date","d":">=","t":' . (time() - WEEKSECS) . '}]}',
 281          );
 282  
 283          // Create module with conditional availability.
 284          $m4 = $generator->create_module('assign',
 285                  array('course' => $course->id) +
 286                  $optionsavailability
 287          );
 288  
 289          // Verifying that everything is generated correctly.
 290          $modinfo = get_fast_modinfo($course->id);
 291          $cm1 = $modinfo->cms[$m1->cmid];
 292          $this->assertEquals($optionsgeneral['visible'], $cm1->visible);
 293          $this->assertEquals($optionsgeneral['section'], $cm1->sectionnum); // Note difference in key.
 294          $this->assertEquals($optionsgeneral['cmidnumber'], $cm1->idnumber); // Note difference in key.
 295          $this->assertEquals($optionsgeneral['groupmode'], $cm1->groupmode);
 296          $this->assertEquals($optionsgeneral['groupingid'], $cm1->groupingid);
 297  
 298          $cm2 = $modinfo->cms[$m2->cmid];
 299          $this->assertEquals($featurecompletionmanual['completion'], $cm2->completion);
 300          $this->assertEquals($featurecompletionmanual['completionexpected'], $cm2->completionexpected);
 301          $this->assertEquals(null, $cm2->completiongradeitemnumber);
 302          // Rating info is stored in the module's table (in our test {data}).
 303          $data = $DB->get_record('data', array('id' => $m2->id));
 304          $this->assertEquals($featurerate['assessed'], $data->assessed);
 305          $this->assertEquals($featurerate['scale'], $data->scale);
 306          $this->assertEquals($featurerate['assesstimestart'], $data->assesstimestart);
 307          $this->assertEquals($featurerate['assesstimefinish'], $data->assesstimefinish);
 308          // No validation for 'ratingtime'. It is only used in to enable/disable assesstime* when adding module.
 309  
 310          $cm3 = $modinfo->cms[$m3->cmid];
 311          $this->assertEquals($featurecompletionautomatic['completion'], $cm3->completion);
 312          $this->assertEquals($featurecompletionautomatic['completionview'], $cm3->completionview);
 313          $this->assertEquals(0, $cm3->completiongradeitemnumber); // Zero instead of default null since 'completionusegrade' was set.
 314          $gradingitem = grade_item::fetch(array('courseid'=>$course->id, 'itemtype'=>'mod', 'itemmodule' => 'assign', 'iteminstance' => $m3->id));
 315          $this->assertEquals(0, $gradingitem->grademin);
 316          $this->assertEquals($featuregrade['grade'], $gradingitem->grademax);
 317          $this->assertEquals($featuregrade['gradecat'], $gradingitem->categoryid);
 318  
 319          $cm4 = $modinfo->cms[$m4->cmid];
 320          $this->assertEquals($optionsavailability['availability'], $cm4->availability);
 321      }

 415      public function test_create_grade_category() {
 416          global $DB, $CFG;
 417          require_once $CFG->libdir . '/grade/constants.php';
 418  
 419          $this->resetAfterTest(true);
 420          $generator = $this->getDataGenerator();
 421          $course = $generator->create_course();
 422  
 423          // Generate category and make sure number of records in DB table increases.
 424          // Note we only count grade cats with depth > 1 because the course grade category
 425          // is lazily created.
 426          $count = $DB->count_records_select('grade_categories', 'depth <> 1');
 427          $gradecategory = $generator->create_grade_category(array('courseid'=>$course->id));
 428          $this->assertEquals($count+1, $DB->count_records_select('grade_categories', 'depth <> 1'));

 431          $this->assertEquals('Grade category 1', $gradecategory->fullname);
 432

 435                  array('courseid' => $course->id, 'aggregation' => GRADE_AGGREGATE_MEDIAN));
 436          $this->assertEquals(GRADE_AGGREGATE_MEDIAN, $gradecategory->aggregation);
 437  
 438          // Generate category and make sure parent is set.
 439          $gradecategory2 = $generator->create_grade_category(
 440                  array('courseid' => $course->id,
 441                      'parent' => $gradecategory->id));

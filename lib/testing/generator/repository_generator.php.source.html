 106      /**
 107       * Create a test repository instance.
 108       *
 109       * @param array|stdClass $record
 110       * @param array $options
 111       * @return stdClass repository instance record
 112       */
 113      public function create_instance($record = null, array $options = null) {
 114          global $CFG, $DB;
 115          require_once($CFG->dirroot . '/repository/lib.php');
 116  
 117          $this->instancecount++;
 118          $record = (array) $record;
 119  
 120          $typeid = $DB->get_field('repository', 'id', array('type' => $this->get_typename()), MUST_EXIST);
 121          $instanceoptions = repository::static_function($this->get_typename(), 'get_instance_option_names');
 122  
 123          if (empty($instanceoptions)) {
 124              // There can only be one instance of this repository, and it should have been created along with the type.
 125              $id = $DB->get_field('repository_instances', 'id', array('typeid' => $typeid), MUST_EXIST);
 126          } else {
 127              // Create the new instance, but first make sure all the required parameters are set.
 128              $record = $this->prepare_record($record);
 129  
 130              if (empty($record['contextid'])) {
 131                  throw new coding_exception('contextid must be present in testing_repository_generator::create_instance() $record');
 132              }
 133  
 134              foreach ($instanceoptions as $option) {
 135                  if (!isset($record[$option])) {
 136                      throw new coding_exception("$option must be present in testing_repository_generator::create_instance() \$record");
 137                  }
 138              }
 139  
 140              $context = context::instance_by_id($record['contextid']);
 141              unset($record['contextid']);
 142              if (!in_array($context->contextlevel, array(CONTEXT_SYSTEM, CONTEXT_COURSE, CONTEXT_USER))) {
 143                  throw new coding_exception('Wrong contextid passed in testing_repository_generator::create_instance() $record');
 144              }
 145  
 146              $id = repository::static_function($this->get_typename(), 'create', $this->get_typename(), 0, $context, $record);
 147          }
 148  
 149          return $DB->get_record('repository_instances', array('id' => $id), '*', MUST_EXIST);
 150      }

 152      /**
 153       * Create the type of repository.
 154       *
 155       * @param stdClass|array $record data to use to set up the type
 156       * @param array $options options for the set up of the type
 157       *
 158       * @return stdClass repository type record
 159       */
 160      public function create_type($record = null, array $options = null) {
 161          global $CFG, $DB;
 162          require_once($CFG->dirroot . '/repository/lib.php');
 163  
 164          $record = (array) $record;
 165          $type = $this->get_typename();
 166  
 167          $typeoptions = repository::static_function($type, 'get_type_option_names');
 168          $instanceoptions = repository::static_function($type, 'get_instance_option_names');
 169  
 170          // The type allow for user and course instances.
 171          if (!empty($instanceoptions)) {
 172              $typeoptions[] = 'enableuserinstances';
 173              $typeoptions[] = 'enablecourseinstances';
 174          }
 175  
 176          // Make sure all the parameters are set.
 177          $record = $this->prepare_type_record($record);
 178          foreach ($typeoptions as $option) {
 179              if (!isset($record[$option])) {
 180                  throw new coding_exception("$option must be present in testing::create_repository_type() for $type");
 181              }
 182          }
 183  
 184          // Limit to allowed options.
 185          $record = array_intersect_key($record, array_flip($typeoptions));
 186  
 187          // Create the type.
 188          $plugintype = new repository_type($type, $record);
 189          $plugintype->create(false);
 190  
 191          return $DB->get_record('repository', array('type' => $type));
 192      }

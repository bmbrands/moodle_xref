 291      /**
 292       * Flip string table (for index searching)
 293       *
 294       * @param     array    $stringTable    Stringtable
 295       * @return     array
 296       */
 297      public function flipStringTable($stringTable = array())
 298      {
 299          // Return value
 300          $returnValue = array();
 301  
 302          // Loop through stringtable and add flipped items to $returnValue
 303          foreach ($stringTable as $key => $value) {
 304              if (! $value instanceof PHPExcel_RichText) {
 305                  $returnValue[$value] = $key;
 306              } elseif ($value instanceof PHPExcel_RichText) {
 307                  $returnValue[$value->getHashCode()] = $key;
 308              }
 309          }
 310  
 311          return $returnValue;

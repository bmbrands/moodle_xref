 356       * Set default options for libxml loader
 357       *
 358       * @param int $options Default options for libxml loader
 359       */
 360      public static function setLibXmlLoaderOptions($options = null)
 361      {
 362          if (is_null($options) && defined(LIBXML_DTDLOAD)) {
 363              $options = LIBXML_DTDLOAD | LIBXML_DTDATTR;
 364          }
 365          if (version_compare(PHP_VERSION, '5.2.11') >= 0) {
 366              @libxml_disable_entity_loader($options == (LIBXML_DTDLOAD | LIBXML_DTDATTR));
 367          }
 368          self::$libXmlLoaderOptions = $options;
 369      }

 377      public static function getLibXmlLoaderOptions()

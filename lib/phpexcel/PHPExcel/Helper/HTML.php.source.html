 776      {

 785      protected function parseElementNode(DOMElement $element)
 786      {
 787          $callbackTag = strtolower($element->nodeName);
 788          $this->stack[] = $callbackTag;
 789  
 790          $this->handleCallback($element, $callbackTag, $this->startTagCallbacks);
 791  
 792          $this->parseElements($element);
 793          array_pop($this->stack);
 794  
 795          $this->handleCallback($element, $callbackTag, $this->endTagCallbacks);

  37      private static function getAttribute($component, $name, $format)
  38      {
  39          $attributes = $component->attributes();
  40          if (isset($attributes[$name])) {
  41              if ($format == 'string') {
  42                  return (string) $attributes[$name];
  43              } elseif ($format == 'integer') {
  44                  return (integer) $attributes[$name];
  45              } elseif ($format == 'boolean') {
  46                  return (boolean) ($attributes[$name] === '0' || $attributes[$name] !== 'true') ? false : true;
  47              } else {
  48                  return (float) $attributes[$name];
  49              }
  50          }
  51          return null;
  52      }

 329      private static function chartDataSeriesValues($seriesValueSet, $dataType = 'n')
 330      {
 331          $seriesVal = array();
 332          $formatCode = '';
 333          $pointCount = 0;
 334  
 335          foreach ($seriesValueSet as $seriesValueIdx => $seriesValue) {
 336              switch ($seriesValueIdx) {
 337                  case 'ptCount':
 338                      $pointCount = self::getAttribute($seriesValue, 'val', 'integer');
 339                      break;
 340                  case 'formatCode':
 341                      $formatCode = (string) $seriesValue;
 342                      break;
 343                  case 'pt':
 344                      $pointVal = self::getAttribute($seriesValue, 'idx', 'integer');
 345                      if ($dataType == 's') {
 346                          $seriesVal[$pointVal] = (string) $seriesValue->v;
 347                      } else {
 348                          $seriesVal[$pointVal] = (float) $seriesValue->v;
 349                      }
 350                      break;
 351              }
 352          }
 353  
 354          if (empty($seriesVal)) {
 355              $seriesVal = null;
 356          }
 357  
 358          return array(
 359              'formatCode'    => $formatCode,
 360              'pointCount'    => $pointCount,
 361              'dataValues'    => $seriesVal
 362          );

 399      private static function parseRichText($titleDetailPart = null)
 400      {
 401          $value = new PHPExcel_RichText();
 402  
 403          foreach ($titleDetailPart as $titleDetailElementKey => $titleDetailElement) {
 404              if (isset($titleDetailElement->t)) {
 405                  $objText = $value->createTextRun((string) $titleDetailElement->t);
 406              }
 407              if (isset($titleDetailElement->rPr)) {
 408                  if (isset($titleDetailElement->rPr->rFont["val"])) {
 409                      $objText->getFont()->setName((string) $titleDetailElement->rPr->rFont["val"]);
 410                  }
 411  
 412                  $fontSize = (self::getAttribute($titleDetailElement->rPr, 'sz', 'integer'));
 413                  if (!is_null($fontSize)) {
 414                      $objText->getFont()->setSize(floor($fontSize / 100));
 415                  }
 416  
 417                  $fontColor = (self::getAttribute($titleDetailElement->rPr, 'color', 'string'));
 418                  if (!is_null($fontColor)) {
 419                      $objText->getFont()->setColor(new PHPExcel_Style_Color(self::readColor($fontColor)));
 420                  }
 421  
 422                  $bold = self::getAttribute($titleDetailElement->rPr, 'b', 'boolean');
 423                  if (!is_null($bold)) {
 424                      $objText->getFont()->setBold($bold);
 425                  }
 426  
 427                  $italic = self::getAttribute($titleDetailElement->rPr, 'i', 'boolean');
 428                  if (!is_null($italic)) {
 429                      $objText->getFont()->setItalic($italic);
 430                  }
 431  
 432                  $baseline = self::getAttribute($titleDetailElement->rPr, 'baseline', 'integer');
 433                  if (!is_null($baseline)) {
 434                      if ($baseline > 0) {
 435                          $objText->getFont()->setSuperScript(true);
 436                      } elseif ($baseline < 0) {
 437                          $objText->getFont()->setSubScript(true);
 438                      }
 439                  }
 440  
 441                  $underscore = (self::getAttribute($titleDetailElement->rPr, 'u', 'string'));
 442                  if (!is_null($underscore)) {
 443                      if ($underscore == 'sng') {
 444                          $objText->getFont()->setUnderline(PHPExcel_Style_Font::UNDERLINE_SINGLE);
 445                      } elseif ($underscore == 'dbl') {
 446                          $objText->getFont()->setUnderline(PHPExcel_Style_Font::UNDERLINE_DOUBLE);
 447                      } else {
 448                          $objText->getFont()->setUnderline(PHPExcel_Style_Font::UNDERLINE_NONE);
 449                      }
 450                  }
 451  
 452                  $strikethrough = (self::getAttribute($titleDetailElement->rPr, 's', 'string'));
 453                  if (!is_null($strikethrough)) {
 454                      if ($strikethrough == 'noStrike') {
 455                          $objText->getFont()->setStrikethrough(false);
 456                      } else {
 457                          $objText->getFont()->setStrikethrough(true);
 458                      }
 459                  }
 460              }
 461          }
 462  
 463          return $value;
 464      }

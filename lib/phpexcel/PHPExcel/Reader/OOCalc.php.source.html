  75          $mimeType = 'UNKNOWN';

  77          $zip = new $zipClass;

  88                      foreach ($manifest as $manifestDataSet) {
  89                          $manifestAttributes = $manifestDataSet->attributes($namespacesContent['manifest']);
  90                          if ($manifestAttributes->{'full-path'} == '/') {

 282      /**
 283       * Lookup the metro region based on the provided DMA code.
 284       * 
 285       * @param int $dmaCode The DMA code
 286       * 
 287       * @return string Metro region name.
 288       */
 289      public static function getMetroRegion($dmaCode)
 290      {
 291          if ($dmaCode === null) {
 292              return null;
 293          }
 294          if (self::$dmaMap === null) {
 295              self::initialize();
 296          }
 297          return self::$dmaMap[$dmaCode];
 298      }
 299  
 300      /**
 301       * Reverse lookup of DMA code if [exact] metro region name is known.
 302       * 
 303       * @param string $metro Metro region name.
 304       * 
 305       * @return int DMA code, or false if not found.
 306       */
 307      public static function getDMACode($metro)    
 308      {
 309          if (self::$dmaMap === null) {
 310              self::initialize();
 311          }
 312          return array_search($metro, self::$dmaMap);
 313      }

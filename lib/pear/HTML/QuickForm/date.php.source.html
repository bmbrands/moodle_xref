 269      function HTML_QuickForm_date($elementName = null, $elementLabel = null, $options = array(), $attributes = null)
 270      {
 271          $this->HTML_QuickForm_element($elementName, $elementLabel, $attributes);
 272          $this->_persistantFreeze = true;
 273          $this->_appendName = true;
 274          $this->_type = 'date';
 275          // set the options, do not bother setting bogus ones
 276          if (is_array($options)) {
 277              foreach ($options as $name => $value) {
 278                  if ('language' == $name) {
 279                      $this->_options['language'] = isset($this->_locale[$value])? $value: 'en';
 280                  } elseif (isset($this->_options[$name])) {
 281                      if (is_array($value) && is_array($this->_options[$name])) {
 282                          $this->_options[$name] = @array_merge($this->_options[$name], $value);
 283                      } else {
 284                          $this->_options[$name] = $value;
 285                      }
 286                  }
 287              }

 294      function _createElements()
 295      {
 296          $this->_separator = $this->_elements = array();
 297          $separator =  '';
 298          $locale    =& $this->_locale[$this->_options['language']];
 299          $backslash =  false;
 300          for ($i = 0, $length = strlen($this->_options['format']); $i < $length; $i++) {
 301              $sign = $this->_options['format']{$i};
 302              if ($backslash) {
 303                  $backslash  = false;
 304                  $separator .= $sign;
 305              } else {
 306                  $loadSelect = true;
 307                  switch ($sign) {
 308                      case 'D':
 309                          // Sunday is 0 like with 'w' in date()
 310                          $options = $locale['weekdays_short'];
 311                          break;
 312                      case 'l':
 313                          $options = $locale['weekdays_long'];
 314                          break;
 315                      case 'd':
 316                          $options = $this->_createOptionList(1, 31);
 317                          break;
 318                      case 'M':
 319                          $options = $locale['months_short'];
 320                          array_unshift($options , '');
 321                          unset($options[0]);
 322                          break;
 323                      case 'm':
 324                          $options = $this->_createOptionList(1, 12);
 325                          break;
 326                      case 'F':
 327                          $options = $locale['months_long'];
 328                          array_unshift($options , '');
 329                          unset($options[0]);
 330                          break;
 331                      case 'Y':
 332                          $options = $this->_createOptionList(
 333                              $this->_options['minYear'],
 334                              $this->_options['maxYear'],
 335                              $this->_options['minYear'] > $this->_options['maxYear']? -1: 1
 336                          );
 337                          break;
 338                      case 'y':
 339                          $options = $this->_createOptionList(
 340                              $this->_options['minYear'],
 341                              $this->_options['maxYear'],
 342                              $this->_options['minYear'] > $this->_options['maxYear']? -1: 1
 343                          );
 344                          array_walk($options, create_function('&$v,$k','$v = substr($v,-2);'));
 345                          break;
 346                      case 'h':
 347                          $options = $this->_createOptionList(1, 12);
 348                          break;
 349                      case 'g':
 350                          $options = $this->_createOptionList(1, 12);
 351                          array_walk($options, create_function('&$v,$k', '$v = intval($v);'));
 352                          break;
 353                      case 'H':
 354                          $options = $this->_createOptionList(0, 23);
 355                          break;
 356                      case 'i':
 357                          $options = $this->_createOptionList(0, 59, $this->_options['optionIncrement']['i']);
 358                          break;
 359                      case 's':
 360                          $options = $this->_createOptionList(0, 59, $this->_options['optionIncrement']['s']);
 361                          break;
 362                      case 'a':
 363                          $options = array('am' => 'am', 'pm' => 'pm');
 364                          break;
 365                      case 'A':
 366                          $options = array('AM' => 'AM', 'PM' => 'PM');
 367                          break;
 368                      case 'W':
 369                          $options = $this->_createOptionList(1, 53);
 370                          break;
 371                      case '\\':
 372                          $backslash  = true;
 373                          $loadSelect = false;
 374                          break;
 375                      default:
 376                          $separator .= (' ' == $sign? '&nbsp;': $sign);
 377                          $loadSelect = false;
 378                  }
 379  
 380                  if ($loadSelect) {
 381                      if (0 < count($this->_elements)) {
 382                          $this->_separator[] = $separator;
 383                      } else {
 384                          $this->_wrap[0] = $separator;
 385                      }
 386                      $separator = '';
 387                      // Should we add an empty option to the top of the select?
 388                      if (!is_array($this->_options['addEmptyOption']) && $this->_options['addEmptyOption'] ||
 389                          is_array($this->_options['addEmptyOption']) && !empty($this->_options['addEmptyOption'][$sign])) {
 390  
 391                          // Using '+' array operator to preserve the keys
 392                          if (is_array($this->_options['emptyOptionText']) && !empty($this->_options['emptyOptionText'][$sign])) {
 393                              $options = array($this->_options['emptyOptionValue'] => $this->_options['emptyOptionText'][$sign]) + $options;
 394                          } else {
 395                              $options = array($this->_options['emptyOptionValue'] => $this->_options['emptyOptionText']) + $options;
 396                          }
 397                      }
 398                      $this->_elements[] = new HTML_QuickForm_select($sign, null, $options, $this->getAttributes());
 399                  }
 400              }
 401          }
 402          $this->_wrap[1] = $separator . ($backslash? '\\': '');
 403      }

 408     /**
 409      * Creates an option list containing the numbers from the start number to the end, inclusive
 410      *
 411      * @param    int     The start number
 412      * @param    int     The end number
 413      * @param    int     Increment by this value
 414      * @access   private
 415      * @return   array   An array of numeric options.
 416      */
 417      function _createOptionList($start, $end, $step = 1)
 418      {
 419          for ($i = $start, $options = array(); $start > $end? $i >= $end: $i <= $end; $i += $step) {
 420              $options[$i] = sprintf('%02d', $i);
 421          }
 422          return $options;
 423      }

 428      function setValue($value)
 429      {
 430          if (empty($value)) {
 431              $value = array();
 432          } elseif (is_scalar($value)) {
 433              if (!is_numeric($value)) {
 434                  $value = strtotime($value);
 435              }
 436              // might be a unix epoch, then we fill all possible values
 437              $arr = explode('-', date('w-d-n-Y-h-H-i-s-a-A-W', (int)$value));
 438              $value = array(
 439                  'D' => $arr[0],
 440                  'l' => $arr[0],
 441                  'd' => $arr[1],
 442                  'M' => $arr[2],
 443                  'm' => $arr[2],
 444                  'F' => $arr[2],
 445                  'Y' => $arr[3],
 446                  'y' => $arr[3],
 447                  'h' => $arr[4],
 448                  'g' => $arr[4],
 449                  'H' => $arr[5],
 450                  'i' => $arr[6],
 451                  's' => $arr[7],
 452                  'a' => $arr[8],
 453                  'A' => $arr[9],
 454                  'W' => $arr[10]
 455              );
 456          }
 457          parent::setValue($value);

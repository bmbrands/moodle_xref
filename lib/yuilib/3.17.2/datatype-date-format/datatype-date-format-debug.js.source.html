 250      format : function (oDate, oConfig) {
 251          oConfig = oConfig || {};
 252  
 253          if(!Y.Lang.isDate(oDate)) {
 254              Y.log("format called without a date", "WARN", "date");
 255              return Y.Lang.isValue(oDate) ? oDate : "";
 256          }
 257  
 258          var format, resources, compatMode, sLocale, LOCALE;
 259  
 260          format = oConfig.format || "%Y-%m-%d";
 261  
 262          resources = Y.Intl.get('datatype-date-format');
 263  
 264          var replace_aggs = function (m0, m1) {
 265              if (compatMode && m1 === "r") {
 266                  return resources[m1];
 267              }
 268              var f = Dt.aggregates[m1];
 269              return (f === "locale" ? resources[m1] : f);
 270          };
 271  
 272          var replace_formats = function (m0, m1) {
 273              var f = Dt.formats[m1];
 274              switch(Y.Lang.type(f)) {
 275                  case "string":                  // string => built in date function
 276                      return oDate[f]();
 277                  case "function":                // function => our own function
 278                      return f.call(oDate, oDate, resources);
 279                  case "array":                   // built in function with padding
 280                      if(Y.Lang.type(f[0]) === "string") {
 281                          return xPad(oDate[f[0]](), f[1]);
 282                      } // no break; (fall through to default:)
 283                  default:
 284                      Y.log("unrecognised replacement type, please file a bug (format: " + oConfig.format + ")", "WARN", "date");
 285                      return m1;
 286              }
 287          };
 288  
 289          // First replace aggregates (run in a loop because an agg may be made up of other aggs)
 290          while(format.match(/%[cDFhnrRtTxX]/)) {
 291              format = format.replace(/%([cDFhnrRtTxX])/g, replace_aggs);
 292          }
 293  
 294          // Now replace formats (do not run in a loop otherwise %%a will be replace with the value of %a)
 295          var str = format.replace(/%([aAbBCdegGHIjklmMpPsSuUVwWyYzZ%])/g, replace_formats);
 296  
 297          replace_aggs = replace_formats = undefined;
 298  
 299          return str;
 300      }

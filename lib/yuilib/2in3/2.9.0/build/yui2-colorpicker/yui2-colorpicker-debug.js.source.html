 790           * <ul>
 791           * <li>0 = not a number, letter in range, or special key</li>

 793           * <li>2 = a-fA-F</li>

 803  
 804              //alert(Event.getCharCode(e) + ", " + e.keyCode + ", " + e.charCode);
 805

 807              if (c === 38) { // up arrow
 808                  return 3;
 809              } else if (c === 13) { // return
 810                  return 6;
 811              } else if (c === 40) { // down array
 812                  return 4;
 813              } else if (c >= 48 && c<=57) { // 0-9
 814                  return 1;
 815              } else if (c >= 97 && c<=102) { // a-f
 816                  return 2;
 817              } else if (c >= 65 && c<=70) { // A-F

 820              //              (c >= 112 && c <=123)) { // including F-keys
 821              // tab, delete, return, escape, left, right or ctrl/meta sequences
 822              } else if ("8, 9, 13, 27, 37, 39".indexOf(c) > -1 ||
 823                         e.ctrlKey || e.metaKey) { // special chars

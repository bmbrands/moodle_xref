1008                      bp = this._getBorderSizes(el.parentNode);

1010                      if ((bp[i1] !== 0) && (bp[i2] !== 0)) { //Parent has Borders
1011                          dim = (dim - (bp[i1] + bp[i2]));
1012                      }
1013                  } else {
1014                      if ((bp[i1] === 0) && (bp[i2] === 0)) {
1015                          dim = (dim + (b[i1] + b[i2]));

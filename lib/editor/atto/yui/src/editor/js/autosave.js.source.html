  26  var SUCCESS_MESSAGE_TIMEOUT = 5000,
  27      RECOVER_MESSAGE_TIMEOUT = 60000,
  28      LOGNAME_AUTOSAVE = 'moodle-editor_atto-editor-autosave';

  30  function EditorAutosave() {}
  31  
  32  EditorAutosave.ATTRS= {
  33      /**
  34       * Enable/Disable auto save for this instance.
  35       *
  36       * @attribute autosaveEnabled
  37       * @type Boolean
  38       * @writeOnce
  39       */
  40      autosaveEnabled: {
  41          value: true,
  42          writeOnce: true
  43      },
  44  
  45      /**
  46       * The time between autosaves (in seconds).
  47       *
  48       * @attribute autosaveFrequency
  49       * @type Number
  50       * @default 60
  51       * @writeOnce
  52       */
  53      autosaveFrequency: {
  54          value: 60,
  55          writeOnce: true
  56      },
  57  
  58      /**
  59       * Unique hash for this page instance. Calculated from $PAGE->url in php.
  60       *
  61       * @attribute pageHash
  62       * @type String
  63       * @writeOnce

  65      pageHash: {
  66          value: '',
  67          writeOnce: true
  68      },

  70      /**
  71       * The relative path to the ajax script.
  72       *
  73       * @attribute autosaveAjaxScript
  74       * @type String
  75       * @default '/lib/editor/atto/autosave-ajax.php'
  76       * @readOnly

  78      autosaveAjaxScript: {
  79          value: '/lib/editor/atto/autosave-ajax.php',
  80          readOnly: true
  81      }
  82  };

  84  EditorAutosave.prototype = {
  85  
  86      /**
  87       * The text that was auto saved in the last request.
  88       *
  89       * @property lastText
  90       * @type string
  91       */
  92      lastText: "",
  93  
  94      /**
  95       * Autosave instance.
  96       *
  97       * @property autosaveInstance
  98       * @type string
  99       */
 100      autosaveInstance: null,
 101  
 102      /**
 103       * Initialize the autosave process
 104       *
 105       * @method setupAutosave
 106       * @chainable
 107       */
 108      setupAutosave: function() {
 109          var draftid = -1,
 110              form,
 111              optiontype = null,
 112              options = this.get('filepickeroptions'),
 113              params,
 114              url;
 115  
 116          if (!this.get('autosaveEnabled')) {
 117              // Autosave disabled for this instance.
 118              return;
 119          }
 120  
 121          this.autosaveInstance = Y.stamp(this);
 122          for (optiontype in options) {
 123              if (typeof options[optiontype].itemid !== "undefined") {
 124                  draftid = options[optiontype].itemid;
 125              }
 126          }
 127  
 128          // First see if there are any saved drafts.
 129          // Make an ajax request.
 130          url = M.cfg.wwwroot + this.get('autosaveAjaxScript');
 131          params = {
 132              sesskey: M.cfg.sesskey,
 133              contextid: this.get('contextid'),
 134              action: 'resume',
 135              drafttext: '',
 136              draftid: draftid,
 137              elementid: this.get('elementid'),
 138              pageinstance: this.autosaveInstance,
 139              pagehash: this.get('pageHash')
 140          };
 141  
 142          Y.io(url, {
 143              method: 'POST',
 144              data: params,
 145              context: this,
 146              on: {
 147                  success: function(id,o) {
 148                      var response_json;
 149                      if (typeof o.responseText !== "undefined" && o.responseText !== "") {
 150                          response_json = JSON.parse(o.responseText);
 151  
 152                          // Revert untouched editor contents to an empty string.
 153                          // Check for FF and Chrome.
 154                          if (response_json.result === '<p></p>' || response_json.result === '<p><br></p>' ||
 155                              response_json.result === '<br>') {
 156                              response_json.result = '';
 157                          }
 158  
 159                          // Check for IE 9 and 10.
 160                          if (response_json.result === '<p>&nbsp;</p>' || response_json.result === '<p><br>&nbsp;</p>') {
 161                              response_json.result = '';
 162                          }
 163  
 164                          if (response_json.error || typeof response_json.result === 'undefined') {
 165                              Y.log('Error occurred recovering draft text: ' + response_json.error, 'debug', LOGNAME_AUTOSAVE);
 166                              this.showMessage(M.util.get_string('errortextrecovery', 'editor_atto'),
 167                                      NOTIFY_WARNING, RECOVER_MESSAGE_TIMEOUT);
 168                          } else if (response_json.result !== this.textarea.get('value') &&
 169                                  response_json.result !== '') {
 170                              Y.log('Autosave text found - recover it.', 'debug', LOGNAME_AUTOSAVE);
 171                              this.recoverText(response_json.result);
 172                          }
 173                          this._fireSelectionChanged();
 174                      }
 175                  },
 176                  failure: function() {
 177                      this.showMessage(M.util.get_string('errortextrecovery', 'editor_atto'),
 178                              NOTIFY_WARNING, RECOVER_MESSAGE_TIMEOUT);
 179                  }
 180              }
 181          });
 182  
 183          // Now setup the timer for periodic saves.
 184  
 185          var delay = parseInt(this.get('autosaveFrequency'), 10) * 1000;
 186          Y.later(delay, this, this.saveDraft, false, true);
 187  
 188          // Now setup the listener for form submission.
 189          form = this.textarea.ancestor('form');
 190          if (form) {
 191              form.on('submit', this.resetAutosave, this);
 192          }
 193          return this;
 194      },
 195  
 196      /**
 197       * Clear the autosave text because the form was submitted normally.
 198       *
 199       * @method resetAutosave
 200       * @chainable
 201       */
 202      resetAutosave: function() {
 203          // Make an ajax request to reset the autosaved text.
 204          var url = M.cfg.wwwroot + this.get('autosaveAjaxScript');
 205          var params = {
 206              sesskey: M.cfg.sesskey,
 207              contextid: this.get('contextid'),
 208              action: 'reset',
 209              elementid: this.get('elementid'),
 210              pageinstance: this.autosaveInstance,
 211              pagehash: this.get('pageHash')
 212          };
 213  
 214          Y.io(url, {
 215              method: 'POST',
 216              data: params,
 217              sync: true
 218          });
 219          return this;
 220      },
 221  
 222  
 223      /**
 224       * Recover a previous version of this text and show a message.
 225       *
 226       * @method recoverText
 227       * @param {String} text
 228       * @chainable
 229       */
 230      recoverText: function(text) {
 231          this.editor.setHTML(text);
 232          this.saveSelection();
 233          this.updateOriginal();
 234          this.lastText = text;
 235  
 236          this.showMessage(M.util.get_string('textrecovered', 'editor_atto'),
 237                  NOTIFY_INFO, RECOVER_MESSAGE_TIMEOUT);
 238  
 239          return this;
 240      },
 241  
 242      /**
 243       * Save a single draft via ajax.
 244       *
 245       * @method saveDraft
 246       * @chainable
 247       */
 248      saveDraft: function() {
 249          var url, params;
 250          // Only copy the text from the div to the textarea if the textarea is not currently visible.
 251          if (!this.editor.get('hidden')) {
 252              this.updateOriginal();
 253          }
 254          var newText = this.textarea.get('value');
 255  
 256          if (newText !== this.lastText) {
 257              Y.log('Autosave text', 'debug', LOGNAME_AUTOSAVE);
 258  
 259              // Make an ajax request.
 260              url = M.cfg.wwwroot + this.get('autosaveAjaxScript');
 261              params = {
 262                  sesskey: M.cfg.sesskey,
 263                  contextid: this.get('contextid'),
 264                  action: 'save',
 265                  drafttext: newText,
 266                  elementid: this.get('elementid'),
 267                  pagehash: this.get('pageHash'),
 268                  pageinstance: this.autosaveInstance
 269              };
 270  
 271              // Reusable error handler - must be passed the correct context.
 272              var ajaxErrorFunction = function(code, response) {
 273                  var errorDuration = parseInt(this.get('autosaveFrequency'), 10) * 1000;
 274                  Y.log('Error while autosaving text:' + code, 'warn', LOGNAME_AUTOSAVE);
 275                  Y.log(response, 'warn', LOGNAME_AUTOSAVE);
 276                  this.showMessage(M.util.get_string('autosavefailed', 'editor_atto'), NOTIFY_WARNING, errorDuration);
 277              };
 278  
 279              Y.io(url, {
 280                  method: 'POST',
 281                  data: params,
 282                  context: this,
 283                  on: {
 284                      error: ajaxErrorFunction,
 285                      failure: ajaxErrorFunction,
 286                      success: function(code, response) {
 287                          if (response.responseText !== "") {
 288                              Y.soon(Y.bind(ajaxErrorFunction, this, [code, response]));
 289                          } else {
 290                              // All working.
 291                              this.lastText = newText;
 292                              this.showMessage(M.util.get_string('autosavesucceeded', 'editor_atto'),
 293                                      NOTIFY_INFO, SUCCESS_MESSAGE_TIMEOUT);
 294                          }
 295                      }
 296                  }
 297              });
 298          }
 299          return this;
 300      }
 301  };
 302  
 303  Y.Base.mix(Y.M.editor_atto.Editor, [EditorAutosave]);

  22  /**
  23   * Toolbar Navigation functions for the Atto editor.
  24   *
  25   * See {{#crossLink "M.editor_atto.Editor"}}{{/crossLink}} for details.
  26   *
  27   * @namespace M.editor_atto
  28   * @class EditorToolbarNav
  29   */
  30  
  31  function EditorToolbarNav() {}
  32  
  33  EditorToolbarNav.ATTRS = {
  34  };
  35  
  36  EditorToolbarNav.prototype = {
  37      /**
  38       * The current focal point for tabbing.
  39       *
  40       * @property _tabFocus
  41       * @type Node
  42       * @default null
  43       * @private
  44       */
  45      _tabFocus: null,
  46  
  47      /**
  48       * Set up the watchers for toolbar navigation.
  49       *
  50       * @method setupToolbarNavigation
  51       * @chainable
  52       */
  53      setupToolbarNavigation: function() {
  54          // Listen for Arrow left and Arrow right keys.
  55          this._wrapper.delegate('key',
  56                  this.toolbarKeyboardNavigation,
  57                  'down:37,39',
  58                  '.' + CSS.TOOLBAR,
  59                  this);
  60          this._wrapper.delegate('focus',
  61                  function(e) {
  62                      this._setTabFocus(e.currentTarget);
  63                  }, '.' + CSS.TOOLBAR + ' button', this);
  64  
  65          return this;
  66      },
  67  
  68      /**
  69       * Implement arrow key navigation for the buttons in the toolbar.
  70       *
  71       * @method toolbarKeyboardNavigation
  72       * @param {EventFacade} e - the keyboard event.
  73       */
  74      toolbarKeyboardNavigation: function(e) {
  75          // Prevent the default browser behaviour.
  76          e.preventDefault();
  77  
  78          // On cursor moves we loops through the buttons.
  79          var buttons = this.toolbar.all('button'),
  80              direction = 1,
  81              button,
  82              current = e.target.ancestor('button', true);
  83  
  84          if (e.keyCode === 37) {
  85              // Moving left so reverse the direction.
  86              direction = -1;
  87          }
  88  
  89          button = this._findFirstFocusable(buttons, current, direction);
  90          if (button) {
  91              button.focus();
  92              this._setTabFocus(button);
  93          } else {
  94              Y.log("Unable to find a button to focus on", 'debug', LOGNAME);
  95          }
  96      },
  97  
  98      /**
  99       * Find the first focusable button.
 100       *
 101       * @param {NodeList} buttons A list of nodes.
 102       * @param {Node} startAt The node in the list to start the search from.
 103       * @param {Number} direction The direction in which to search (1 or -1).
 104       * @return {Node | Undefined} The Node or undefined.
 105       * @method _findFirstFocusable
 106       * @private
 107       */
 108      _findFirstFocusable: function(buttons, startAt, direction) {
 109          var checkCount = 0,
 110              group,
 111              candidate,
 112              button,
 113              index;
 114  
 115          // Determine which button to start the search from.
 116          index = buttons.indexOf(startAt);
 117          if (index < -1) {
 118              Y.log("Unable to find the button in the list of buttons", 'debug', LOGNAME);
 119              index = 0;
 120          }
 121  
 122          // Try to find the next.
 123          while (checkCount < buttons.size()) {
 124              index += direction;
 125              if (index < 0) {
 126                  index = buttons.size() - 1;
 127              } else if (index >= buttons.size()) {
 128                  // Handle wrapping.
 129                  index = 0;
 130              }
 131  
 132              candidate = buttons.item(index);
 133  
 134              // Add a counter to ensure we don't get stuck in a loop if there's only one visible menu item.
 135              checkCount++;
 136  
 137              // Loop while:
 138              // * we haven't checked every button;
 139              // * the button is hidden or disabled;
 140              // * the group is hidden.
 141              if (candidate.hasAttribute('hidden') || candidate.hasAttribute('disabled')) {
 142                  continue;
 143              }
 144              group = candidate.ancestor('.atto_group');
 145              if (group.hasAttribute('hidden')) {
 146                  continue;
 147              }
 148  
 149              button = candidate;
 150              break;
 151          }
 152  
 153          return button;
 154      },
 155  
 156      /**
 157       * Check the tab focus.
 158       *
 159       * When we disable or hide a button, we should call this method to ensure that the
 160       * focus is not currently set on an inaccessible button, otherwise tabbing to the toolbar
 161       * would be impossible.
 162       *
 163       * @method checkTabFocus
 164       * @chainable
 165       */
 166      checkTabFocus: function() {
 167          if (this._tabFocus) {
 168              if (this._tabFocus.hasAttribute('disabled') || this._tabFocus.hasAttribute('hidden')
 169                      || this._tabFocus.ancestor('.atto_group').hasAttribute('hidden')) {
 170                  // Find first available button.
 171                  var button = this._findFirstFocusable(this.toolbar.all('button'), this._tabFocus, -1);
 172                  if (button) {
 173                      if (this._tabFocus.compareTo(document.activeElement)) {
 174                          // We should also move the focus, because the inaccessible button also has the focus.
 175                          button.focus();
 176                      }
 177                      this._setTabFocus(button);
 178                  }
 179              }
 180          }
 181          return this;
 182      },
 183  
 184      /**
 185       * Sets tab focus for the toolbar to the specified Node.
 186       *
 187       * @method _setTabFocus
 188       * @param {Node} button The node that focus should now be set to
 189       * @chainable
 190       * @private
 191       */
 192      _setTabFocus: function(button) {
 193          if (this._tabFocus) {
 194              // Unset the previous entry.
 195              this._tabFocus.setAttribute('tabindex', '-1');
 196          }
 197  
 198          // Set up the new entry.
 199          this._tabFocus = button;
 200          this._tabFocus.setAttribute('tabindex', 0);
 201  
 202          // And update the activedescendant to point at the currently selected button.
 203          this.toolbar.setAttribute('aria-activedescendant', this._tabFocus.generateID());
 204  
 205          return this;
 206      }
 207  };
 208  
 209  Y.Base.mix(Y.M.editor_atto.Editor, [EditorToolbarNav]);

  21  /**
  22   * Selection functions for the Atto editor.
  23   *
  24   * See {{#crossLink "M.editor_atto.Editor"}}{{/crossLink}} for details.
  25   *
  26   * @namespace M.editor_atto
  27   * @class EditorSelection
  28   */

  30  function EditorSelection() {}
  31  
  32  EditorSelection.ATTRS= {
  33  };

  35  EditorSelection.prototype = {
  36  
  37      /**
  38       * List of saved selections per editor instance.
  39       *
  40       * @property _selections
  41       * @private
  42       */
  43      _selections: null,
  44  
  45      /**
  46       * A unique identifier for the last selection recorded.
  47       *
  48       * @property _lastSelection
  49       * @param lastselection
  50       * @type string
  51       * @private
  52       */
  53      _lastSelection: null,
  54  
  55      /**
  56       * Whether focus came from a click event.
  57       *
  58       * This is used to determine whether to restore the selection or not.
  59       *
  60       * @property _focusFromClick
  61       * @type Boolean
  62       * @default false
  63       * @private
  64       */
  65      _focusFromClick: false,
  66  
  67      /**
  68       * Set up the watchers for selection save and restoration.
  69       *
  70       * @method setupSelectionWatchers
  71       * @chainable
  72       */
  73      setupSelectionWatchers: function() {
  74          // Save the selection when a change was made.
  75          this.on('atto:selectionchanged', this.saveSelection, this);
  76  
  77          this.editor.on('focus', this.restoreSelection, this);
  78  
  79          // Do not restore selection when focus is from a click event.
  80          this.editor.on('mousedown', function() {
  81              this._focusFromClick = true;
  82          }, this);
  83  
  84          // Copy the current value back to the textarea when focus leaves us and save the current selection.
  85          this.editor.on('blur', function() {
  86              // Clear the _focusFromClick value.
  87              this._focusFromClick = false;
  88  
  89              // Update the original text area.
  90              this.updateOriginal();
  91          }, this);
  92  
  93          this.editor.on(['keyup', 'focus'], function(e) {
  94                  Y.soon(Y.bind(this._hasSelectionChanged, this, e));
  95              }, this);
  96  
  97          // To capture both mouseup and touchend events, we need to track the gesturemoveend event in standAlone mode. Without
  98          // standAlone, it will only fire if we listened to a gesturemovestart too.
  99          this.editor.on('gesturemoveend', function(e) {
 100                  Y.soon(Y.bind(this._hasSelectionChanged, this, e));
 101              }, {
 102                  standAlone: true
 103              }, this);
 104  
 105          return this;
 106      },

 108      /**
 109       * Work out if the cursor is in the editable area for this editor instance.
 110       *
 111       * @method isActive
 112       * @return {boolean}
 113       */
 114      isActive: function() {
 115          var range = rangy.createRange(),
 116              selection = rangy.getSelection();
 117  
 118          if (!selection.rangeCount) {
 119              // If there was no range count, then there is no selection.
 120              return false;
 121          }
 122  
 123          // We can't be active if the editor doesn't have focus at the moment.
 124          if (!document.activeElement ||
 125                  !(this.editor.compareTo(document.activeElement) || this.editor.contains(document.activeElement))) {
 126              return false;
 127          }
 128  
 129          // Check whether the range intersects the editor selection.
 130          range.selectNode(this.editor.getDOMNode());
 131          return range.intersectsRange(selection.getRangeAt(0));
 132      },
 133  
 134      /**
 135       * Create a cross browser selection object that represents a YUI node.
 136       *
 137       * @method getSelectionFromNode
 138       * @param {Node} YUI Node to base the selection upon.
 139       * @return {[rangy.Range]}
 140       */
 141      getSelectionFromNode: function(node) {
 142          var range = rangy.createRange();
 143          range.selectNode(node.getDOMNode());
 144          return [range];
 145      },
 146  
 147      /**
 148       * Save the current selection to an internal property.
 149       *
 150       * This allows more reliable return focus, helping improve keyboard navigation.
 151       *
 152       * Should be used in combination with {{#crossLink "M.editor_atto.EditorSelection/restoreSelection"}}{{/crossLink}}.
 153       *
 154       * @method saveSelection
 155       */
 156      saveSelection: function() {
 157          if (this.isActive()) {
 158              this._selections = this.getSelection();
 159          }
 160      },
 161  
 162      /**
 163       * Restore any stored selection when the editor gets focus again.
 164       *
 165       * Should be used in combination with {{#crossLink "M.editor_atto.EditorSelection/saveSelection"}}{{/crossLink}}.
 166       *
 167       * @method restoreSelection
 168       */
 169      restoreSelection: function() {
 170          if (!this._focusFromClick) {
 171              if (this._selections) {
 172                  this.setSelection(this._selections);
 173              }
 174          }
 175          this._focusFromClick = false;
 176      },
 177  
 178      /**
 179       * Get the selection object that can be passed back to setSelection.
 180       *
 181       * @method getSelection
 182       * @return {array} An array of rangy ranges.
 183       */
 184      getSelection: function() {
 185          return rangy.getSelection().getAllRanges();
 186      },
 187  
 188      /**
 189       * Check that a YUI node it at least partly contained by the current selection.
 190       *
 191       * @method selectionContainsNode
 192       * @param {Node} The node to check.
 193       * @return {boolean}
 194       */
 195      selectionContainsNode: function(node) {
 196          return rangy.getSelection().containsNode(node.getDOMNode(), true);
 197      },
 198  
 199      /**
 200       * Runs a filter on each node in the selection, and report whether the
 201       * supplied selector(s) were found in the supplied Nodes.
 202       *
 203       * By default, all specified nodes must match the selection, but this
 204       * can be controlled with the requireall property.
 205       *
 206       * @method selectionFilterMatches
 207       * @param {String} selector
 208       * @param {NodeList} [selectednodes] For performance this should be passed. If not passed, this will be looked up each time.
 209       * @param {Boolean} [requireall=true] Used to specify that "any" match is good enough.
 210       * @return {Boolean}
 211       */
 212      selectionFilterMatches: function(selector, selectednodes, requireall) {
 213          if (typeof requireall === 'undefined') {
 214              requireall = true;
 215          }
 216          if (!selectednodes) {
 217              // Find this because it was not passed as a param.
 218              selectednodes = this.getSelectedNodes();
 219          }
 220          var allmatch = selectednodes.size() > 0,
 221              anymatch = false;
 222  
 223          var editor = this.editor,
 224              stopFn = function(node) {
 225                  // The function getSelectedNodes only returns nodes within the editor, so this test is safe.
 226                  return node === editor;
 227              };
 228  
 229          // If we do not find at least one match in the editor, no point trying to find them in the selection.
 230          if (!editor.one(selector)) {
 231              return false;
 232          }
 233  
 234          selectednodes.each(function(node){
 235              // Check each node, if it doesn't match the tags AND is not within the specified tags then fail this thing.
 236              if (requireall) {
 237                  // Check for at least one failure.
 238                  if (!allmatch || !node.ancestor(selector, true, stopFn)) {
 239                      allmatch = false;
 240                  }
 241              } else {
 242                  // Check for at least one match.
 243                  if (!anymatch && node.ancestor(selector, true, stopFn)) {
 244                      anymatch = true;
 245                  }
 246              }
 247          }, this);
 248          if (requireall) {
 249              return allmatch;
 250          } else {
 251              return anymatch;
 252          }
 253      },
 254  
 255      /**
 256       * Get the deepest possible list of nodes in the current selection.
 257       *
 258       * @method getSelectedNodes
 259       * @return {NodeList}
 260       */
 261      getSelectedNodes: function() {
 262          var results = new Y.NodeList(),
 263              nodes,
 264              selection,
 265              range,
 266              node,
 267              i;
 268  
 269          selection = rangy.getSelection();
 270  
 271          if (selection.rangeCount) {
 272              range = selection.getRangeAt(0);
 273          } else {
 274              // Empty range.
 275              range = rangy.createRange();
 276          }
 277  
 278          if (range.collapsed) {
 279              // We do not want to select all the nodes in the editor if we managed to
 280              // have a collapsed selection directly in the editor.
 281              // It's also possible for the commonAncestorContainer to be the document, which selectNode does not handle
 282              // so we must filter that out here too.
 283              if (range.commonAncestorContainer !== this.editor.getDOMNode()
 284                      && range.commonAncestorContainer !== Y.config.doc) {
 285                  range = range.cloneRange();
 286                  range.selectNode(range.commonAncestorContainer);
 287              }
 288          }
 289  
 290          nodes = range.getNodes();
 291  
 292          for (i = 0; i < nodes.length; i++) {
 293              node = Y.one(nodes[i]);
 294              if (this.editor.contains(node)) {
 295                  results.push(node);
 296              }
 297          }
 298          return results;
 299      },
 300  
 301      /**
 302       * Check whether the current selection has changed since this method was last called.
 303       *
 304       * If the selection has changed, the atto:selectionchanged event is also fired.
 305       *
 306       * @method _hasSelectionChanged
 307       * @private
 308       * @param {EventFacade} e
 309       * @return {Boolean}
 310       */
 311      _hasSelectionChanged: function(e) {
 312          var selection = rangy.getSelection(),
 313              range,
 314              changed = false;
 315  
 316          if (selection.rangeCount) {
 317              range = selection.getRangeAt(0);
 318          } else {
 319              // Empty range.
 320              range = rangy.createRange();
 321          }
 322  
 323          if (this._lastSelection) {
 324              if (!this._lastSelection.equals(range)) {
 325                  changed = true;
 326                  return this._fireSelectionChanged(e);
 327              }
 328          }
 329          this._lastSelection = range;
 330          return changed;
 331      },
 332  
 333      /**
 334       * Fires the atto:selectionchanged event.
 335       *
 336       * When the selectionchanged event is fired, the following arguments are provided:
 337       *   - event : the original event that lead to this event being fired.
 338       *   - selectednodes :  an array containing nodes that are entirely selected of contain partially selected content.
 339       *
 340       * @method _fireSelectionChanged
 341       * @private
 342       * @param {EventFacade} e
 343       */
 344      _fireSelectionChanged: function(e) {
 345          this.fire('atto:selectionchanged', {
 346              event: e,
 347              selectedNodes: this.getSelectedNodes()
 348          });
 349      },
 350  
 351      /**
 352       * Get the DOM node representing the common anscestor of the selection nodes.
 353       *
 354       * @method getSelectionParentNode
 355       * @return {Element|boolean} The DOM Node for this parent, or false if no seletion was made.
 356       */
 357      getSelectionParentNode: function() {
 358          var selection = rangy.getSelection();
 359          if (selection.rangeCount) {
 360              return selection.getRangeAt(0).commonAncestorContainer;
 361          }
 362          return false;
 363      },
 364  
 365      /**
 366       * Set the current selection. Used to restore a selection.
 367       *
 368       * @method selection
 369       * @param {array} ranges A list of rangy.range objects in the selection.
 370       */
 371      setSelection: function(ranges) {
 372          var selection = rangy.getSelection();
 373          selection.setRanges(ranges);
 374      },
 375  
 376      /**
 377       * Inserts the given HTML into the editable content at the currently focused point.
 378       *
 379       * @method insertContentAtFocusPoint
 380       * @param {String} html
 381       * @return {Node} The YUI Node object added to the DOM.
 382       */
 383      insertContentAtFocusPoint: function(html) {
 384          var selection = rangy.getSelection(),
 385              range,
 386              node = Y.Node.create(html);
 387          if (selection.rangeCount) {
 388              range = selection.getRangeAt(0);
 389          }
 390          if (range) {
 391              range.deleteContents();
 392              range.insertNode(node.getDOMNode());
 393          }
 394          return node;
 395      }

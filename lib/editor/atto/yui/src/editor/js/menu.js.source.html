 100      /**
 101       * Setup the Event handlers.
 102       *
 103       * @method _setupHandlers
 104       * @private
 105       */
 106      _setupHandlers: function() {
 107          var contentBox = this.get('contentBox');
 108          // Handle menu item selection.
 109          this._menuHandlers.push(
 110              // Select the menu item on space, and enter.
 111              contentBox.delegate('key', this._chooseMenuItem, '32, enter', '.atto_menuentry', this),
 112  
 113              // Move up and down the menu on up/down.
 114              contentBox.delegate('key', this._handleKeyboardEvent, 'down:38,40', '.dropdown-menu', this),
 115  
 116              // Hide the menu when clicking outside of it.
 117              contentBox.on('focusoutside', this.hide, this),
 118  
 119              // Hide the menu on left/right, and escape keys.
 120              contentBox.delegate('key', this.hide, 'down:37,39,esc', '.dropdown-menu', this)
 121          );
 122      },

 135      /**
 136       * Hide a menu, removing all of the event handlers which trigger the hide.
 137       *
 138       * @method hide
 139       * @param {EventFacade} e
 140       */
 141      hide: function(e) {
 142          if (this.get('preventHideMenu') === true) {
 143              return;
 144          }
 145  
 146          // We must prevent the default action (left/right/escape) because
 147          // there are other listeners on the toolbar which will focus on the
 148          // editor.
 149          if (e) {
 150              e.preventDefault();
 151          }
 152  
 153          return Menu.superclass.hide.call(this, arguments);
 154      },

 156      /**
 157       * Implement arrow-key navigation for the items in a toolbar menu.
 158       *
 159       * @method _handleKeyboardEvent
 160       * @param {EventFacade} e The keyboard event.
 161       * @static
 162       */
 163      _handleKeyboardEvent: function(e) {
 164          // Prevent the default browser behaviour.
 165          e.preventDefault();
 166  
 167          // Get a list of all buttons in the menu.
 168          var buttons = e.currentTarget.all('a[role="menuitem"]');
 169  
 170          // On cursor moves we loops through the buttons.
 171          var found = false,
 172              index = 0,
 173              direction = 1,
 174              checkCount = 0,
 175              current = e.target.ancestor('a[role="menuitem"]', true),
 176              next;
 177  
 178          // Determine which button is currently selected.
 179          while (!found && index < buttons.size()) {
 180              if (buttons.item(index) === current) {
 181                  found = true;
 182              } else {
 183                  index++;
 184              }
 185          }
 186  
 187          if (!found) {
 188              Y.log("Unable to find this menu item in the menu", 'debug', LOGNAME);
 189              return;
 190          }
 191  
 192          if (e.keyCode === 38) {
 193              // Moving up so reverse the direction.
 194              direction = -1;
 195          }
 196  
 197          // Try to find the next
 198          do {
 199              index += direction;
 200              if (index < 0) {
 201                  index = buttons.size() - 1;
 202              } else if (index >= buttons.size()) {
 203                  // Handle wrapping.
 204                  index = 0;
 205              }
 206              next = buttons.item(index);
 207  
 208              // Add a counter to ensure we don't get stuck in a loop if there's only one visible menu item.
 209              checkCount++;
 210              // Loop while:
 211              // * we are not in a loop and have not already checked every button; and
 212              // * we are on a different button; and
 213              // * the next menu item is not hidden.
 214          } while (checkCount < buttons.size() && next !== current && next.hasAttribute('hidden'));
 215  
 216          if (next) {
 217              next.focus();
 218          }
 219  
 220          e.preventDefault();
 221          e.stopImmediatePropagation();
 222      }

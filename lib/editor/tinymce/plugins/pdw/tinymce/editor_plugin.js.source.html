  63  (function() {
  64      var DOM = tinymce.DOM;
  65      tinymce.PluginManager.requireLangPack('pdw');
  66  
  67      tinymce.create('tinymce.plugins.pdw', {
  68          /**

  69           * Initializes the plugin, this will be executed after the plugin has been created.

  70           * This call is done before the editor instance has finished it's initialization so use the onInit event

  71           * of the editor instance to intercept that event.

  72           *

  73           * @param {tinymce.Editor} ed Editor instance that the plugin is initialized in.

  74           * @param {string} url Absolute URL to where the plugin is located.

  75           */
  76          init : function(ed, url) {
  77              var t = this, tbIds = new Array(), toolbars = new Array(), i;
  78  
  79              // Split toolbars

  80              toolbars = (ed.settings.pdw_toggle_toolbars).split(',');
  81  
  82              for(i = 0; i < toolbars.length; i++){
  83                  tbIds[i] = ed.getParam('', 'toolbar' + (toolbars[i]).replace(' ',''));
  84              }
  85  
  86              // Register the command so that it can be invoked by using tinyMCE.activeEditor.execCommand('mceExample');

  87              ed.addCommand('mcePDWToggleToolbars', function() {
  88  
  89                  var cm = ed.controlManager, id, j, Cookie = tinymce.util.Cookie, Toggle_PDW, Toggle = Cookie.getHash("TinyMCE_toggle") || new Object();
  90                  for(j = 0; j < tbIds.length; j++){
  91  
  92                      obj = ed.controlManager.get(tbIds[j]);
  93                      if(typeof obj =="undefined") {
  94                          continue;
  95                      }
  96                      id = obj.id;
  97  
  98                      if (DOM.isHidden(id)) {
  99                          Toggle_PDW = 0;
 100                          var e = document.getElementById(id);
 101                          if (e) {
 102                              e.style.display = 'table';
 103                              t._resizeIframe(ed, tbIds[j], -26);
 104                          }
 105  
 106                      } else {
 107                          Toggle_PDW = 1;
 108                          var e = document.getElementById(id);
 109                          if (e) {
 110                              e.style.display = 'none';
 111                          }
 112                          t._resizeIframe(ed, tbIds[j], 26);
 113                      }
 114                  }
 115                  cm.setActive('pdw_toggle', Toggle_PDW);
 116                  ed.settings.pdw_toggle_on = Toggle_PDW;
 117                  Toggle[ed.id] = Toggle_PDW;
 118                  Cookie.setHash("TinyMCE_toggle", Toggle);
 119              });
 120  
 121              // Register pdw_toggle button

 122              ed.addButton('pdw_toggle', {
 123                  title : ed.getLang('pdw.desc', 0),
 124                  cmd : 'mcePDWToggleToolbars',
 125                  image : url + '/img/toolbars.png'
 126              });
 127  
 128              ed.onPostRender.add(function(){
 129                  var toggle = tinymce.util.Cookie.getHash("TinyMCE_toggle") || new Object();
 130                  var run = false;
 131  
 132                  // Check if value is stored in cookie

 133                  if(toggle[ed.id] == null){
 134                      // No cookie so check if the setting pdw_toggle_on is set to 1 then hide toolbars and set button active

 135                      run = ed.settings.pdw_toggle_on == 1 ? true : false;
 136                  } else if(toggle[ed.id] == 1){
 137                      run = true;
 138                  }
 139  
 140                  if (run) {
 141  
 142                      var cm = ed.controlManager, tdId, id;
 143  
 144                      for(i = 0; i < toolbars.length; i++){
 145                          tbId = ed.getParam('', 'toolbar' + (toolbars[i]).replace(' ',''));
 146                          id = ed.controlManager.get(tbId).id;
 147                          cm.setActive('pdw_toggle', 1);
 148                          DOM.hide(id);
 149                          t._resizeIframe(ed, tbId, 26);
 150                      }
 151                  }
 152              });
 153          },
 154  
 155          // Resizes the iframe by a relative height value

 156          _resizeIframe : function(ed, tb_id, dy) {
 157              var ifr = ed.getContentAreaContainer().firstChild;
 158              var parcon = ed.getContainer(); //Parent container

 159              var tablecon = ed.getContainer().childNodes[1]; // Table container

 160              var textarea = DOM.get(ed.id);
 161              var rows = textarea ? textarea.getAttribute('rows') : 3;
 162  
 163              // This set of changes addresses MDL-42481.

 164              // Moodle collapses form sections by setting display to none on the fcontainer element.

 165              // In order to calculate the offsetWidth, the iframe must be visible within the DOM, otherwise it's offsetWidth is

 166              // calculate as 0px.

 167              // We attempt to find any collapsed element, uncollapse them, then calculate the width and height, and finally

 168              // collapse them again.

 169              var collapsedContainer = DOM.getParent(ifr, 'fieldset.collapsed');
 170              if (collapsedContainer) {
 171                  DOM.removeClass(collapsedContainer, 'collapsed');
 172              }
 173              // For very small text areas - allow the editable region to be smaller than the size of the toolbars.

 174              if (rows >= 3) {
 175                  DOM.setStyle(ifr, 'height',DOM.getSize(ifr).h + dy); // Resize iframe

 176                  ed.theme.deltaHeight += dy; // For resize cookie

 177              }
 178  
 179              // Set all the containers to the same width

 180              DOM.setStyle(textarea, 'width','100%');
 181              DOM.setStyle(parcon, 'width','100%');
 182              DOM.setStyle(tablecon, 'width','100%');
 183              DOM.setStyle(ifr, 'width','100%');
 184  
 185  
 186              if (collapsedContainer) {
 187                  // We have a collapsedContainer, so collapse it again.

 188                  DOM.addClass(collapsedContainer, 'collapsed');
 189              }
 190  
 191          },
 192  
 193          /**

 194           * Returns information about the plugin as a name/value array.

 195           * The current keys are longname, author, authorurl, infourl and version.

 196           *

 197           * @return {Object} Name/value array containing information about the plugin.

 198           */
 199          getInfo : function() {
 200              return {
 201                  longname : 'PDW Toggle Toolbars',
 202                  author : 'Guido Neele',
 203                  authorurl : 'http://www.neele.name/',
 204                  infourl : 'http://www.neele.name/pdw_toggle_toolbars',
 205                  version : "1.2"
 206              };
 207          }
 208      });
 209  
 210      // Register plugin

 211      tinymce.PluginManager.add('pdw', tinymce.plugins.pdw);
 212  })();

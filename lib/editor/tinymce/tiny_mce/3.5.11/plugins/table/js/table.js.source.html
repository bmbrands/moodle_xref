 272  function makeAttrib(attrib, value) {
 273      var formObj = document.forms[0];
 274      var valueElm = formObj.elements[attrib];
 275  
 276      if (typeof(value) == "undefined" || value == null) {
 277          value = "";
 278  
 279          if (valueElm)
 280              value = valueElm.value;
 281      }
 282  
 283      if (value == "")
 284          return "";
 285  
 286      // XML encode it

 287      value = value.replace(/&/g, '&amp;');
 288      value = value.replace(/\"/g, '&quot;');
 289      value = value.replace(/</g, '&lt;');
 290      value = value.replace(/>/g, '&gt;');
 291  
 292      return ' ' + attrib + '="' + value + '"';
 293  }

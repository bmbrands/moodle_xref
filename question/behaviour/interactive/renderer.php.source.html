  42      public function feedback(question_attempt $qa, question_display_options $options) {
  43          if (!$qa->get_state()->is_active() || !$options->readonly) {
  44              return '';
  45          }
  46  
  47          $attributes = array(
  48              'type' => 'submit',
  49              'id' => $qa->get_behaviour_field_name('tryagain'),
  50              'name' => $qa->get_behaviour_field_name('tryagain'),
  51              'value' => get_string('tryagain', 'qbehaviour_interactive'),
  52              'class' => 'submit btn',
  53          );
  54          if ($options->readonly !== qbehaviour_interactive::READONLY_EXCEPT_TRY_AGAIN) {
  55              $attributes['disabled'] = 'disabled';
  56          }
  57          $output = html_writer::empty_tag('input', $attributes);
  58          if (empty($attributes['disabled'])) {
  59              $this->page->requires->js_init_call('M.core_question_engine.init_submit_button',
  60                      array($attributes['id'], $qa->get_slot()));
  61          }
  62          return $output;
  63      }

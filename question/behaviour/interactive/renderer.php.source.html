  37  class qbehaviour_interactive_renderer extends qbehaviour_renderer {
  38      public function controls(question_attempt $qa, question_display_options $options) {
  39          if ($options->readonly === qbehaviour_interactive::READONLY_EXCEPT_TRY_AGAIN) {
  40              return '';
  41          }
  42          return $this->submit_button($qa, $options);
  43      }

  45      public function feedback(question_attempt $qa, question_display_options $options) {
  46          if (!$qa->get_state()->is_active() || !$options->readonly) {
  47              return '';
  48          }
  49  
  50          $attributes = array(
  51              'type' => 'submit',
  52              'id' => $qa->get_behaviour_field_name('tryagain'),
  53              'name' => $qa->get_behaviour_field_name('tryagain'),
  54              'value' => get_string('tryagain', 'qbehaviour_interactive'),
  55              'class' => 'submit btn',
  56          );
  57          if ($options->readonly !== qbehaviour_interactive::READONLY_EXCEPT_TRY_AGAIN) {
  58              $attributes['disabled'] = 'disabled';
  59          }
  60          $output = html_writer::empty_tag('input', $attributes);
  61          if (empty($attributes['disabled'])) {
  62              $this->page->requires->js_init_call('M.core_question_engine.init_submit_button',
  63                      array($attributes['id'], $qa->get_slot()));
  64          }
  65          return $output;
  66      }

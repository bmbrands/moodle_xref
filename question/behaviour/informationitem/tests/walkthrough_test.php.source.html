  51                  $this->get_contains_hidden_expectation(
  52                          $this->quba->get_field_prefix($this->slot) . '-seen', 1),
  53                  $this->get_does_not_contain_feedback_expectation());
  54  
  55          // Check no hidden input when read-only.

  57          $this->check_current_output($this->get_contains_question_text_expectation($description),
  58                  $this->get_does_not_contain_hidden_expectation(
  59                          $this->quba->get_field_prefix($this->slot) . '-seen', 1));
  60          $this->displayoptions->readonly = false;
  61  
  62          // Process a submission indicating this question has been seen.
  63          $this->process_submission(array('-seen' => 1));
  64  
  65          $this->check_current_state(question_state::$complete);
  66          $this->check_current_mark(null);

  68                  new question_no_pattern_expectation(
  69                  '/type=\"hidden\"[^>]*name=\"[^"]*seen\"|name=\"[^"]*seen\"[^>]*type=\"hidden\"/'),
  70                  $this->get_does_not_contain_feedback_expectation());
  71  
  72          // Finish the attempt.

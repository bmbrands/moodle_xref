 383      public function add_category($newparent, $newcategory, $newinfo, $return = false, $newinfoformat = FORMAT_HTML) {
 384          global $DB;
 385          if (empty($newcategory)) {
 386              print_error('categorynamecantbeblank', 'question');
 387          }
 388          list($parentid, $contextid) = explode(',', $newparent);
 389          //moodle_form makes sure select element output is legal no need for further cleaning
 390          require_capability('moodle/question:managecategory', context::instance_by_id($contextid));
 391  
 392          if ($parentid) {
 393              if(!($DB->get_field('question_categories', 'contextid', array('id' => $parentid)) == $contextid)) {
 394                  print_error('cannotinsertquestioncatecontext', 'question', '', array('cat'=>$newcategory, 'ctx'=>$contextid));
 395              }
 396          }
 397  
 398          $cat = new stdClass();
 399          $cat->parent = $parentid;
 400          $cat->contextid = $contextid;
 401          $cat->name = $newcategory;
 402          $cat->info = $newinfo;
 403          $cat->infoformat = $newinfoformat;
 404          $cat->sortorder = 999;
 405          $cat->stamp = make_unique_id_code();
 406          $categoryid = $DB->insert_record("question_categories", $cat);
 407  
 408          // Log the creation of this category.
 409          $params = array(
 410              'objectid' => $categoryid,
 411              'contextid' => $contextid
 412          );
 413          $event = \core\event\question_category_created::create($params);
 414          $event->trigger();
 415  
 416          if ($return) {
 417              return $categoryid;
 418          } else {
 419              redirect($this->pageurl);//always redirect after successful action
 420          }
 421      }

 423      /**
 424       * Updates an existing category with given params
 425       */
 426      public function update_category($updateid, $newparent, $newname, $newinfo, $newinfoformat = FORMAT_HTML) {
 427          global $CFG, $DB;
 428          if (empty($newname)) {
 429              print_error('categorynamecantbeblank', 'question');
 430          }
 431  
 432          // Get the record we are updating.
 433          $oldcat = $DB->get_record('question_categories', array('id' => $updateid));
 434          $lastcategoryinthiscontext = question_is_only_toplevel_category_in_context($updateid);
 435  
 436          if (!empty($newparent) && !$lastcategoryinthiscontext) {
 437              list($parentid, $tocontextid) = explode(',', $newparent);
 438          } else {
 439              $parentid = $oldcat->parent;
 440              $tocontextid = $oldcat->contextid;
 441          }
 442  
 443          // Check permissions.
 444          $fromcontext = context::instance_by_id($oldcat->contextid);
 445          require_capability('moodle/question:managecategory', $fromcontext);
 446  
 447          // If moving to another context, check permissions some more.
 448          if ($oldcat->contextid != $tocontextid) {
 449              $tocontext = context::instance_by_id($tocontextid);
 450              require_capability('moodle/question:managecategory', $tocontext);
 451          }
 452  
 453          // Update the category record.
 454          $cat = new stdClass();
 455          $cat->id = $updateid;
 456          $cat->name = $newname;
 457          $cat->info = $newinfo;
 458          $cat->infoformat = $newinfoformat;
 459          $cat->parent = $parentid;
 460          $cat->contextid = $tocontextid;
 461          $DB->update_record('question_categories', $cat);
 462  
 463          // If the category name has changed, rename any random questions in that category.
 464          if ($oldcat->name != $cat->name) {
 465              $where = "qtype = 'random' AND category = ? AND " . $DB->sql_compare_text('questiontext') . " = ?";
 466  
 467              $randomqtype = question_bank::get_qtype('random');
 468              $randomqname = $randomqtype->question_name($cat, false);
 469              $DB->set_field_select('question', 'name', $randomqname, $where, array($cat->id, '0'));
 470  
 471              $randomqname = $randomqtype->question_name($cat, true);
 472              $DB->set_field_select('question', 'name', $randomqname, $where, array($cat->id, '1'));
 473          }
 474  
 475          if ($oldcat->contextid != $tocontextid) {
 476              // Moving to a new context. Must move files belonging to questions.
 477              question_move_category_to_context($cat->id, $oldcat->contextid, $tocontextid);
 478          }
 479  
 480          // Cat param depends on the context id, so update it.
 481          $this->pageurl->param('cat', $updateid . ',' . $tocontextid);
 482          redirect($this->pageurl);
 483      }

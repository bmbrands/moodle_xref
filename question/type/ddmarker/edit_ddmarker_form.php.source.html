 157      public function data_preprocessing($question) {
 158  
 159          $question = parent::data_preprocessing($question);
 160          $question = $this->data_preprocessing_combined_feedback($question, true);
 161          $question = $this->data_preprocessing_hints($question, true, true);
 162  
 163          $dragids = array(); // Drag no -> dragid.
 164          if (!empty($question->options)) {
 165              $question->shuffleanswers = $question->options->shuffleanswers;
 166              $question->showmisplaced = $question->options->showmisplaced;
 167              $question->drags = array();
 168              foreach ($question->options->drags as $drag) {
 169                  $dragindex = $drag->no - 1;
 170                  $question->drags[$dragindex] = array();
 171                  $question->drags[$dragindex]['label'] = $drag->label;
 172                  if ($drag->infinite == 1) {
 173                      $question->drags[$dragindex]['noofdrags'] = 0;
 174                  } else {
 175                      $question->drags[$dragindex]['noofdrags'] = $drag->noofdrags;
 176                  }
 177                  $dragids[$dragindex] = $drag->id;
 178              }
 179              $question->drops = array();
 180              foreach ($question->options->drops as $drop) {
 181                  $droparray = (array)$drop;
 182                  unset($droparray['id']);
 183                  unset($droparray['no']);
 184                  unset($droparray['questionid']);
 185                  $question->drops[$drop->no - 1] = $droparray;
 186              }
 187          }
 188          // Initialise file picker for bgimage.
 189          $draftitemid = file_get_submitted_draft_itemid('bgimage');
 190  
 191          file_prepare_draft_area($draftitemid, $this->context->id, 'qtype_ddmarker',
 192                                  'bgimage', !empty($question->id) ? (int) $question->id : null,
 193                                  self::file_picker_options());
 194          $question->bgimage = $draftitemid;
 195  
 196          $this->js_call();
 197  
 198          return $question;
 199      }

 224      public function validation($data, $files) {
 225          $errors = parent::validation($data, $files);
 226          $bgimagesize = $this->get_image_size_in_draft_area($data['bgimage']);
 227          if ($bgimagesize === null) {
 228              $errors["bgimage"] = get_string('formerror_nobgimage', 'qtype_ddmarker');
 229          }
 230  
 231          $allchoices = array();
 232          for ($i = 0; $i < $data['nodropzone']; $i++) {
 233              $choice = $data['drops'][$i]['choice'];
 234              $choicepresent = ($choice !== '0');
 235  
 236              if ($choicepresent) {
 237                  // Test coords here.
 238                  if ($bgimagesize !== null) {
 239                      $shape = $data['drops'][$i]['shape'];
 240                      $coordsstring = $data['drops'][$i]['coords'];
 241                      $shapeobj = qtype_ddmarker_shape::create($shape, $coordsstring);
 242                      $interpretererror = $shapeobj->get_coords_interpreter_error();
 243                      if ($interpretererror !== false) {
 244                          $errors["drops[{$i}]"] = $interpretererror;
 245                      } else if (!$shapeobj->inside_width_height($bgimagesize)) {
 246                          $errorcode = 'shapeoutsideboundsofbgimage';
 247                          $errors["drops[{$i}]"] =
 248                                              get_string('formerror_'.$errorcode, 'qtype_ddmarker');
 249                      }
 250                  }
 251              } else {
 252                  if (trim($data['drops'][$i]['coords']) !== '') {
 253                      $errorcode = 'noitemselected';
 254                      $errors["drops[{$i}]"] = get_string('formerror_'.$errorcode, 'qtype_ddmarker');
 255                  }
 256              }
 257  
 258          }
 259          for ($dragindex = 0; $dragindex < $data['noitems']; $dragindex++) {
 260              $label = $data['drags'][$dragindex]['label'];
 261              if ($label != strip_tags($label, QTYPE_DDMARKER_ALLOWED_TAGS_IN_MARKER)) {
 262                  $errors["drags[{$dragindex}]"]
 263                      = get_string('formerror_onlysometagsallowed', 'qtype_ddmarker',
 264                                    s(QTYPE_DDMARKER_ALLOWED_TAGS_IN_MARKER));
 265              }
 266          }
 267          return $errors;
 268      }

 270      /**
 271       * Gets the width and height of a draft image.
 272       *
 273       * @param int $draftitemid ID of the draft image
 274       * @return array Return array of the width and height of the draft image.
 275       */
 276      public function get_image_size_in_draft_area($draftitemid) {
 277          global $USER;
 278          $usercontext = context_user::instance($USER->id);
 279          $fs = get_file_storage();
 280          $draftfiles = $fs->get_area_files($usercontext->id, 'user', 'draft', $draftitemid, 'id');
 281          if ($draftfiles) {
 282              foreach ($draftfiles as $file) {
 283                  if ($file->is_directory()) {
 284                      continue;
 285                  }
 286                  // Just return the data for the first good file, there should only be one.
 287                  $imageinfo = $file->get_imageinfo();
 288                  $width    = $imageinfo['width'];
 289                  $height   = $imageinfo['height'];
 290                  return array($width, $height);
 291              }
 292          }
 293          return null;
 294      }

 245      draw_shape_polygon: function(dropzoneno, coords, colour) {
 246          var coordsparts = coords.split(';');

 256                  type: "path",

 264                  }

 267              var minxy = [this.doc.bg_img().get('width'), this.doc.bg_img().get('height')];
 268              for (i = 0; i < xy.length; i++) {
 269                  // calculate min and max points to find center to show marker on
 270                  minxy[0] = Math.min(xy[i][0], minxy[0]);
 271                  minxy[1] = Math.min(xy[i][1], minxy[1]);
 272                  maxxy[0] = Math.max(xy[i][0], maxxy[0]);
 273                  maxxy[1] = Math.max(xy[i][1], maxxy[1]);
 274                  if (i === 0) {
 275                      polygon.moveTo(xy[i][0], xy[i][1]);
 276                  } else {
 277                      polygon.lineTo(xy[i][0], xy[i][1]);
 278                  }
 279              }

 281                  polygon.lineTo(xy[0][0], xy[0][1]); // Close polygon if not already closed.

 341      /**
 342       * prevent_touchmove_from_scrolling allows users of touch screen devices to
 343       * use drag and drop and normal scrolling at the same time. I.e.when
 344       * touching and dragging a draggable item, the screen does not scroll, but
 345       * you can scroll by touching other area of the screen apart from the
 346       * draggable items.

 348      prevent_touchmove_from_scrolling: function(drag) {
 349          var touchstart = (Y.UA.ie) ? 'MSPointerStart' : 'touchstart';
 350          var touchend = (Y.UA.ie) ? 'MSPointerEnd' : 'touchend';
 351          var touchmove = (Y.UA.ie) ? 'MSPointerMove' : 'touchmove';
 352  
 353          // Disable scrolling when touching the draggable items.
 354          drag.on(touchstart, function() {
 355              if (this.touchscrolldisable) {
 356                  return; // Already disabled.
 357              }
 358              this.touchscrolldisable = Y.one('body').on(touchmove, function(e) {
 359                  e = e || window.event;
 360                  e.preventDefault();
 361              });
 362          }, this);
 363  
 364          // Allow scrolling after releasing the draggable items.
 365          drag.on(touchend, function() {
 366              if (this.touchscrolldisable) {
 367                  this.touchscrolldisable.detach();
 368                  this.touchscrolldisable = null;
 369              }
 370          }, this);

 540      drop_zone_key_press: function(e) {
 541          var dragitem = e.target;
 542          var xy = dragitem.getXY();
 543          switch (e.direction) {
 544              case 'left' :
 545                  xy[0] -= 1;
 546                  break;
 547              case 'right' :
 548                  xy[0] += 1;
 549                  break;
 550              case 'down' :
 551                  xy[1] += 1;
 552                  break;
 553              case 'up' :
 554                  xy[1] -= 1;
 555                  break;
 556              case 'remove' :
 557                  xy = null;
 558                  break;
 559          }
 560          var choiceno = this.get_choiceno_for_node(dragitem);
 561          if (xy !== null) {
 562              xy = this.constrain_to_bgimg(xy);
 563          } else {

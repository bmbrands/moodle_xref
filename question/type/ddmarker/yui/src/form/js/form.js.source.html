  26      initializer: function() {

  37                  '<li><span id="xcoordpreview">X = </span></li>' +
  38                  '<li><span id="ycoordpreview">Y = </span></li>' +

  60      load_bg_image: function() {
  61          var bgimageurl = this.fp.file('bgimage').href;
  62          if (bgimageurl !== null) {
  63              this.doc.load_bg_img(bgimageurl);
  64  
  65              var drop = new Y.DD.Drop({

  69              // Listen for a drop:hit on the background image.

  74              this.afterimageloaddone = false;

  79      constrain_image_size: function(e) {
  80          var maxsize = this.get('maxsizes').bgimage;
  81          var reduceby = Math.max(e.target.get('width') / maxsize.width,
  82                                  e.target.get('height') / maxsize.height);
  83          if (reduceby > 1) {
  84              e.target.set('width', Math.floor(e.target.get('width') / reduceby));
  85          }
  86          e.target.addClass('constrained');
  87          e.target.detach('load', this.constrain_image_size);
  88      },

  90      update_drop_zones: function(pendingid) {
  91  
  92          // Set up drop zones.
  93          if (this.graphics !== null) {
  94              this.graphics.destroy();
  95          }
  96          this.restart_colours();
  97          this.graphics = new Y.Graphic({render: "div.ddarea div.dropzones"});
  98          var noofdropzones = this.form.get_form_value('nodropzone', []);
  99          for (var dropzoneno = 0; dropzoneno < noofdropzones; dropzoneno++) {
 100              var dragitemno = this.form.get_form_value('drops', [dropzoneno, 'choice']);
 101              var markertext = this.get_marker_text(dragitemno);
 102              var shape = this.form.get_form_value('drops', [dropzoneno, 'shape']);
 103              var coords = this.get_coords(dropzoneno);
 104              var colourfordropzone = this.get_next_colour();
 105              Y.one('input#id_drops_' + dropzoneno + '_coords')
 106                      .setStyle('background-color', colourfordropzone);
 107              this.draw_drop_zone(dropzoneno, markertext,
 108                      shape, coords, colourfordropzone, false);
 109          }
 110          if (this.doc.bg_img()) {
 111              Y.one('div.ddarea .grid')
 112                      .setXY(this.convert_to_window_xy([0, 0]))
 113                      .setStyle('width', this.doc.bg_img().get('width'))
 114                      .setStyle('height', this.doc.bg_img().get('height'));
 115          }
 116          M.util.js_complete(pendingid);

 119      get_coords: function(dropzoneno) {
 120          var coords = this.form.get_form_value('drops', [dropzoneno, 'coords']);
 121          return coords.replace(new RegExp("\\s*", 'g'), '');
 122      },
 123      get_marker_text: function(markerno) {
 124          if (Number(markerno) !== 0) {
 125              var label = this.form.get_form_value('drags', [markerno - 1, 'label']);
 126              return label.replace(new RegExp("^\\s*(.*)\\s*$"), "$1");
 127          } else {
 128              return '';
 129          }

 131      set_options_for_drag_item_selectors: function() {
 132          var dragitemsoptions = {'0': ''};
 133          for (var i = 1; i <= this.form.get_form_value('noitems', []); i++) {
 134              var label = this.get_marker_text(i);
 135              if (label !== "") {
 136                  dragitemsoptions[i] = Y.Escape.html(label);
 137              }
 138          }
 139          // Get all the currently selected drags for each drop.
 140          var selectedvalues = [];
 141          var selector;
 142          for (i = 0; i < this.form.get_form_value('nodropzone', []); i++) {
 143              selector = Y.one('#id_drops_' + i + '_choice');
 144              selectedvalues[i] = Number(selector.get('value'));
 145          }
 146          for (i = 0; i < this.form.get_form_value('nodropzone', []); i++) {
 147              selector = Y.one('#id_drops_' + i + '_choice');
 148              // Remove all options for drag choice.
 149              selector.all('option').remove(true);
 150              // And recreate the options.
 151              for (var value in dragitemsoptions) {
 152                  value = Number(value);
 153                  var option = '<option value="' + value + '">' + dragitemsoptions[value] + '</option>';
 154                  selector.append(option);
 155                  var optionnode = selector.one('option[value="' + value + '"]');
 156                  // Is this the currently selected value?
 157                  if (value === selectedvalues[i]) {
 158                      optionnode.set('selected', true);
 159                  } else {
 160                      // It is not the currently selected value, is it selectable?
 161                      if (value !== 0) { // The 'no item' option is always selectable.
 162                          // Variables to hold form values about this drag item.
 163                          var noofdrags = this.form.get_form_value('drags', [value - 1, 'noofdrags']);
 164                          if (Number(noofdrags) !== 0) { // 'noofdrags == 0' means infinite.
 165                              // Go through all selected values in drop downs.
 166                              for (var k in selectedvalues) {
 167                                  // Count down 'noofdrags' and if reach zero then set disabled option for this drag item.
 168                                  if (Number(selectedvalues[k]) === value) {
 169                                      if (Number(noofdrags) === 1) {
 170                                          optionnode.set('disabled', true);
 171                                          break;
 172                                      } else {
 173                                          noofdrags--;
 174                                      }
 175                                  }
 176                              }
 177                          }
 178                      }
 179                  }
 180              }
 181          }
 182      },

 210      form: {
 211          to_name_with_index: function(name, indexes) {
 212              var indexstring = name;
 213              for (var i = 0; i < indexes.length; i++) {
 214                  indexstring = indexstring + '[' + indexes[i] + ']';
 215              }
 216              return indexstring;
 217          },
 218          get_el: function(name, indexes) {
 219              var form = document.getElementById('mform1');
 220              return form.elements[this.to_name_with_index(name, indexes)];
 221          },
 222          get_form_value: function(name, indexes) {
 223              var el = this.get_el(name, indexes);
 224              if (el.type === 'checkbox') {
 225                  return el.checked;
 226              } else {
 227                  return el.value;
 228              }
 229          },
 230          set_form_value: function(name, indexes, value) {
 231              var el = this.get_el(name, indexes);
 232              if (el.type === 'checkbox') {
 233                  el.checked = value;
 234              } else {
 235                  el.value = value;
 236              }
 237          },
 238          from_name_with_index: function(name) {
 239              var toreturn = {};
 240              toreturn.indexes = [];
 241              var bracket = name.indexOf('[');
 242              toreturn.name = name.substring(0, bracket);
 243              while (bracket !== -1) {
 244                  var end = name.indexOf(']', bracket + 1);
 245                  toreturn.indexes.push(name.substring(bracket + 1, end));
 246                  bracket = name.indexOf('[', end + 1);
 247              }
 248              return toreturn;
 249          }
 250      },

 252      file_pickers: function() {
 253          var draftitemidstoname;
 254          var nametoparentnode;
 255          if (draftitemidstoname === undefined) {
 256              draftitemidstoname = {};
 257              nametoparentnode = {};
 258              var filepickers = Y.all('form.mform input.filepickerhidden');
 259              filepickers.each(function(filepicker) {
 260                  draftitemidstoname[filepicker.get('value')] = filepicker.get('name');
 261                  nametoparentnode[filepicker.get('name')] = filepicker.get('parentNode');
 262              }, this);
 263          }
 264          var toreturn = {
 265              file: function(name) {
 266                  var parentnode = nametoparentnode[name];
 267                  var fileanchor = parentnode.one('div.filepicker-filelist a');
 268                  if (fileanchor) {
 269                      return {href: fileanchor.get('href'), name: fileanchor.get('innerHTML')};
 270                  } else {
 271                      return {href: null, name: null};
 272                  }
 273              },
 274              name: function(draftitemid) {
 275                  return draftitemidstoname[draftitemid];
 276              }
 277          };
 278          return toreturn;
 279      }

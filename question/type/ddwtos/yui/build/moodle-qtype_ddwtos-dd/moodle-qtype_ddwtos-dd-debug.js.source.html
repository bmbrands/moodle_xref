 170      get_classname_numeric_suffix: function(node, prefix) {
 171          var classes = node.getAttribute('class');
 172          if (classes !== '') {
 173              var classesarr = classes.split(' ');
 174              for (var index = 0; index < classesarr.length; index++) {
 175                  var patt1 = new RegExp('^' + prefix + '([0-9])+$');
 176                  if (patt1.test(classesarr[index])) {
 177                      var patt2 = new RegExp('([0-9])+$');
 178                      var match = patt2.exec(classesarr[index]);
 179                      return Number(match[0]);
 180                  }
 181              }
 182          }

 226      /**
 227       * prevent_touchmove_from_scrolling allows users of touch screen devices to
 228       * use drag and drop and normal scrolling at the same time. I.e. when
 229       * touching and dragging a draggable item, the screen does not scroll, but
 230       * you can scroll by touching other area of the screen apart from the
 231       * draggable items.
 232       */
 233      prevent_touchmove_from_scrolling: function(drag) {
 234          var touchstart = (Y.UA.ie) ? 'MSPointerStart' : 'touchstart';
 235          var touchend = (Y.UA.ie) ? 'MSPointerEnd' : 'touchend';
 236          var touchmove = (Y.UA.ie) ? 'MSPointerMove' : 'touchmove';
 237  
 238          // Disable scrolling when touching the draggable items.
 239          drag.on(touchstart, function() {
 240              if (this.touchscrolldisable) {
 241                  return; // Already disabled.
 242              }
 243              this.touchscrolldisable = Y.one('body').on(touchmove, function(e) {
 244                  e = e || window.event;
 245                  e.preventDefault();
 246              });
 247          }, this);
 248  
 249          // Allow scrolling after releasing the draggable items.
 250          drag.on(touchend, function() {
 251              if (this.touchscrolldisable) {
 252                  this.touchscrolldisable.detach();
 253                  this.touchscrolldisable = null;
 254              }
 255          }, this);
 256      },

 258      make_drop_zones: function() {

 261      make_drop_zone: function(drop) {
 262          var dropdd = new Y.DD.Drop({
 263              node: drop,
 264              groups: [this.get_group(drop)]});
 265          dropdd.on('drop:hit', function(e) {
 266              var drag = e.drag.get('node');
 267              var drop = e.drop.get('node');
 268              if (this.get_group(drop) === this.get_group(drag)) {

 270              }
 271          }, this);
 272          if (!this.get('readonly')) {
 273              drop.on('dragchange', this.drop_zone_key_press, this);
 274          }

 276      place_drag_in_drop: function(drag, drop) {
 277          var placeno = this.get_place(drop);
 278          var inputid = this.get('inputids')[placeno];
 279          var inputnode = Y.one('input#' + inputid);
 280          if (drag !== null) {
 281              inputnode.set('value', this.get_choice(drag));
 282          } else {
 283              inputnode.set('value', '0');
 284          }
 285          for (var alreadytheredragno in this.placed) {
 286              if (this.placed[alreadytheredragno] === placeno) {
 287                  delete this.placed[alreadytheredragno];
 288                  var alreadytheredrag = Y.one(this.selectors.drag(alreadytheredragno));
 289                  if (alreadytheredrag && alreadytheredrag.dd) {
 290                      alreadytheredrag.dd.detach('drag:start');
 291                  }
 292              }
 293          }
 294          if (drag !== null) {
 295              this.placed[this.get_no(drag)] = placeno;
 296              if (drag.dd) {
 297                  drag.dd.once('drag:start', function(e, inputnode, drag) {
 298                      inputnode.set('value', 0);
 299                      delete this.placed[this.get_no(drag)];
 300                      drag.addClass('unplaced');
 301                  }, this, inputnode, drag);
 302              }
 303          }
 304      },
 305      remove_drag_from_drop: function(drop) {
 306          this.place_drag_in_drop(null, drop);

 309      /**
 310       * Postition, or reposition, all the drag items.
 311       * @param pendingid (optional) if given, then mark the js task complete after the
 312       * items are all positioned.
 313       * @param dotimeout (optional) if true, continually re-position the items so

 318         M.util.js_complete(pendingid);
 319         if (dotimeout === true || dotimeout > 0) {

 324         }

 326      position_drag_item: function(drag) {
 327          if (!drag.hasClass('yui3-dd-dragging')) {
 328              if (!this.placed[this.get_no(drag)]) {
 329                  var groupno = this.get_group(drag);
 330                  var choiceno = this.get_choice(drag);
 331                  var home = Y.one(this.selectors.drag_home(groupno, choiceno));
 332                  drag.setXY(home.getXY());
 333                  drag.addClass('unplaced');
 334              } else {
 335                  var placeno = this.placed[this.get_no(drag)];
 336                  var drop = Y.one(this.selectors.drop_for_place(placeno));
 337                  drag.setXY([drop.getX() + 2, drop.getY() + 2]);
 338                  drag.removeClass('unplaced');
 339              }
 340          }

 343      drop_zone_key_press: function(e) {
 344          switch (e.direction) {
 345              case 'next' :
 346                  this.place_next_drag_in(e.target);
 347                  break;
 348              case 'previous' :
 349                  this.place_previous_drag_in(e.target);
 350                  break;
 351              case 'remove' :
 352                  this.remove_drag_from_drop(e.target);
 353                  break;
 354          }
 355          e.preventDefault();
 356      },
 357      place_next_drag_in: function(drop) {
 358          this.choose_next_choice_for_drop(drop, 1);
 359      },
 360      place_previous_drag_in: function(drop) {
 361          this.choose_next_choice_for_drop(drop, -1);
 362      },
 363      choose_next_choice_for_drop: function(drop, direction) {
 364          var next;
 365          var groupno = this.get_group(drop);
 366          var current = this.current_choice_in_drop(drop);
 367          var unplaceddragsingroup = Y.all(this.selectors.unplaced_drags_in_group(groupno));
 368          if (0 === current) {
 369              if (direction === 1) {
 370                  next = 1;
 371              } else {
 372                  var lastdrag = unplaceddragsingroup.pop();
 373                  next = this.get_choice(lastdrag);
 374              }
 375          } else {
 376              next = current + direction;
 377          }
 378          var drag;
 379          do {
 380              drag = Y.one(this.selectors.unplaced_drags_for_choice_in_group(next, groupno));
 381              if (Y.one(this.selectors.drags_for_choice_in_group(next, groupno)) === null) {
 382                  this.remove_drag_from_drop(drop);
 383                  return;
 384              }
 385              next = next + direction;
 386          } while (drag === null);
 387          this.place_drag_in_drop(drag, drop);
 388      },
 389      current_choice_in_drop: function(drop) {
 390          var inputid = this.get('inputids')[this.get_place(drop)];
 391          var inputnode = Y.one('input#' + inputid);
 392          return Number(inputnode.get('value'));
 393      }

  41      /**
  42       * Appends the multichoice specific information to the question
  43       */
  44      public function process_question(array $data, array $raw) {
  45  
  46          // Convert and write the answers first.
  47          if (isset($data['answers'])) {
  48              $this->write_answers($data['answers'], $this->pluginname);
  49          }
  50  
  51          // Convert and write the multichoice.
  52          if (!isset($data['multichoice'])) {
  53              // This should never happen, but it can do if the 1.9 site contained
  54              // corrupt data.
  55              $data['multichoice'] = array(array(
  56                  'single'                         => 1,
  57                  'shuffleanswers'                 => 1,
  58                  'correctfeedback'                => '',
  59                  'correctfeedbackformat'          => FORMAT_HTML,
  60                  'partiallycorrectfeedback'       => '',
  61                  'partiallycorrectfeedbackformat' => FORMAT_HTML,
  62                  'incorrectfeedback'              => '',
  63                  'incorrectfeedbackformat'        => FORMAT_HTML,
  64                  'answernumbering'                => 'abc',
  65              ));
  66          }
  67          $this->write_multichoice($data['multichoice'], $data['oldquestiontextformat'], $data['id']);
  68      }
  69  
  70      /**
  71       * Converts the multichoice info and writes it into the question.xml
  72       *
  73       * @param array $multichoices the grouped structure
  74       * @param int $oldquestiontextformat - {@see moodle1_question_bank_handler::process_question()}
  75       * @param int $questionid question id
  76       */
  77      protected function write_multichoice(array $multichoices, $oldquestiontextformat, $questionid) {
  78          global $CFG;
  79  
  80          // The grouped array is supposed to have just one element - let us use foreach anyway
  81          // just to be sure we do not loose anything.
  82          foreach ($multichoices as $multichoice) {
  83              // Append an artificial 'id' attribute (is not included in moodle.xml).
  84              $multichoice['id'] = $this->converter->get_nextid();
  85  
  86              // Replay the upgrade step 2009021801.
  87              $multichoice['correctfeedbackformat']               = 0;
  88              $multichoice['partiallycorrectfeedbackformat']      = 0;
  89              $multichoice['incorrectfeedbackformat']             = 0;
  90  
  91              if ($CFG->texteditors !== 'textarea' and $oldquestiontextformat == FORMAT_MOODLE) {
  92                  $multichoice['correctfeedback']                 = text_to_html($multichoice['correctfeedback'], false, false, true);
  93                  $multichoice['correctfeedbackformat']           = FORMAT_HTML;
  94                  $multichoice['partiallycorrectfeedback']        = text_to_html($multichoice['partiallycorrectfeedback'], false, false, true);
  95                  $multichoice['partiallycorrectfeedbackformat']  = FORMAT_HTML;
  96                  $multichoice['incorrectfeedback']               = text_to_html($multichoice['incorrectfeedback'], false, false, true);
  97                  $multichoice['incorrectfeedbackformat']         = FORMAT_HTML;
  98              } else {
  99                  $multichoice['correctfeedbackformat']           = $oldquestiontextformat;
 100                  $multichoice['partiallycorrectfeedbackformat']  = $oldquestiontextformat;
 101                  $multichoice['incorrectfeedbackformat']         = $oldquestiontextformat;
 102              }
 103  
 104              $multichoice['correctfeedback'] = $this->migrate_files(
 105                      $multichoice['correctfeedback'], 'question', 'correctfeedback', $questionid);
 106              $multichoice['partiallycorrectfeedback'] = $this->migrate_files(
 107                      $multichoice['partiallycorrectfeedback'], 'question', 'partiallycorrectfeedback', $questionid);
 108              $multichoice['incorrectfeedback'] = $this->migrate_files(
 109                      $multichoice['incorrectfeedback'], 'question', 'incorrectfeedback', $questionid);
 110  
 111              $this->write_xml('multichoice', $multichoice, array('/multichoice/id'));
 112          }
 113      }
 114  }

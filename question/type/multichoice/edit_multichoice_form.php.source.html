 107      public function validation($data, $files) {
 108          $errors = parent::validation($data, $files);
 109          $answers = $data['answer'];
 110          $answercount = 0;
 111  
 112          $totalfraction = 0;
 113          $maxfraction = -1;
 114  
 115          foreach ($answers as $key => $answer) {
 116              // Check no of choices.
 117              $trimmedanswer = trim($answer['text']);
 118              $fraction = (float) $data['fraction'][$key];
 119              if ($trimmedanswer === '' && empty($fraction)) {
 120                  continue;
 121              }
 122              if ($trimmedanswer === '') {
 123                  $errors['fraction['.$key.']'] = get_string('errgradesetanswerblank', 'qtype_multichoice');
 124              }
 125  
 126              $answercount++;
 127  
 128              // Check grades.
 129              if ($data['fraction'][$key] > 0) {
 130                  $totalfraction += $data['fraction'][$key];
 131              }
 132              if ($data['fraction'][$key] > $maxfraction) {
 133                  $maxfraction = $data['fraction'][$key];
 134              }
 135          }
 136  
 137          if ($answercount == 0) {
 138              $errors['answer[0]'] = get_string('notenoughanswers', 'qtype_multichoice', 2);
 139              $errors['answer[1]'] = get_string('notenoughanswers', 'qtype_multichoice', 2);
 140          } else if ($answercount == 1) {
 141              $errors['answer[1]'] = get_string('notenoughanswers', 'qtype_multichoice', 2);
 142  
 143          }
 144  
 145          // Perform sanity checks on fractional grades.
 146          if ($data['single']) {
 147              if ($maxfraction != 1) {
 148                  $errors['fraction[0]'] = get_string('errfractionsnomax', 'qtype_multichoice',
 149                          $maxfraction * 100);
 150              }
 151          } else {
 152              $totalfraction = round($totalfraction, 2);
 153              if ($totalfraction != 1) {
 154                  $errors['fraction[0]'] = get_string('errfractionsaddwrong', 'qtype_multichoice',
 155                          $totalfraction * 100);
 156              }
 157          }
 158          return $errors;
 159      }

 344          }
 345  
 346          // Check the initial state.
 347          $this->check_current_state(question_state::$todo);
 348          $this->check_current_mark(null);
 349          $this->check_current_output(
 350                  $this->get_contains_select_expectation('sub0', $choices, null, true),
 351                  $this->get_contains_select_expectation('sub1', $choices, null, true),

  50      /** @var string regex to match HTML open tags. */
  51      private $htmltstarttagsandattributes = '~<\s*\w+\b[^>]*>~';

  53      /** @var string regex to match HTML close tags or br. */
  54      private $htmltclosetags = '~<\s*/\s*\w+\b[^>]*>~';

  56      /** @var string regex to select text like [[cat]] (including the square brackets). */
  57      private $squarebracketsregex = '/\[\[[^]]*?\]\]/';

  59      /**
  60       * Vaidate some input to make sure it does not contain any tags other than
  61       * $this->allowedhtmltags.
  62       * @param unknown_type $text the input to validate.
  63       * @return string any validation errors.
  64       */
  65      protected function get_illegal_tag_error($text) {
  66          // Remove legal tags.
  67          $strippedtext = $text;
  68          foreach ($this->allowedhtmltags as $htmltag) {
  69              $tagpair = "~<\s*/?\s*$htmltag\b\s*[^>]*>~";
  70              $strippedtext = preg_replace($tagpair, '', $strippedtext);
  71          }
  72  
  73          $textarray = array();
  74          preg_match_all($this->htmltstarttagsandattributes, $strippedtext, $textarray);
  75          if ($textarray[0]) {
  76              return $this->allowed_tags_message($textarray[0][0]);
  77          }
  78  
  79          preg_match_all($this->htmltclosetags, $strippedtext, $textarray);
  80          if ($textarray[0]) {
  81              return $this->allowed_tags_message($textarray[0][0]);
  82          }
  83  
  84          return '';
  85      }

  87      /**
  88       * Returns a message indicating what tags are allowed.
  89       *
  90       * @param string $badtag The disallowed tag that was supplied
  91       * @return string Message indicating what tags are allowed
  92       */
  93      private function allowed_tags_message($badtag) {
  94          $a = new stdClass();
  95          $a->tag = htmlspecialchars($badtag);
  96          $a->allowed = $this->get_list_of_printable_allowed_tags($this->allowedhtmltags);
  97          if ($a->allowed) {
  98              return get_string('tagsnotallowed', 'qtype_gapselect', $a);
  99          } else {
 100              return get_string('tagsnotallowedatall', 'qtype_gapselect', $a);
 101          }
 102      }

 104      /**
 105       * Returns a prinatble list of allowed HTML tags.
 106       *
 107       * @param array $allowedhtmltags An array for tag strings that are allowed
 108       * @return string A printable list of tags
 109       */
 110      private function get_list_of_printable_allowed_tags($allowedhtmltags) {
 111          $allowedtaglist = array();
 112          foreach ($allowedhtmltags as $htmltag) {
 113              $allowedtaglist[] = htmlspecialchars('<' . $htmltag . '>');
 114          }
 115          return implode(', ', $allowedtaglist);
 116      }

 133       * Defines form elements for answer choices.

 137      protected function definition_answer_choice(&$mform) {
 138          $mform->addElement('header', 'choicehdr', get_string('choices', 'qtype_gapselect'));
 139          $mform->setExpanded('choicehdr', 1);
 140  
 141          $mform->addElement('checkbox', 'shuffleanswers', get_string('shuffle', 'qtype_gapselect'));
 142          $mform->setDefault('shuffleanswers', 0);
 143  
 144          $textboxgroup = array();
 145          $textboxgroup[] = $mform->createElement('group', 'choices',
 146                  get_string('choicex', 'qtype_gapselect'), $this->choice_group($mform));
 147  
 148          if (isset($this->question->options)) {
 149              $countanswers = count($this->question->options->answers);
 150          } else {
 151              $countanswers = 0;
 152          }
 153  
 154          if ($this->question->formoptions->repeatelements) {
 155              $defaultstartnumbers = QUESTION_NUMANS_START * 2;
 156              $repeatsatstart = max($defaultstartnumbers, QUESTION_NUMANS_START,
 157                      $countanswers + QUESTION_NUMANS_ADD);
 158          } else {
 159              $repeatsatstart = $countanswers;
 160          }
 161  
 162          $repeatedoptions = $this->repeated_options();
 163          $mform->setType('answer', PARAM_RAW);
 164          $this->repeat_elements($textboxgroup, $repeatsatstart, $repeatedoptions,
 165                  'noanswers', 'addanswers', QUESTION_NUMANS_ADD,
 166                  get_string('addmorechoiceblanks', 'qtype_gapselect'), true);
 167      }

 200      public function data_preprocessing($question) {
 201          $question = parent::data_preprocessing($question);
 202          $question = $this->data_preprocessing_combined_feedback($question, true);
 203          $question = $this->data_preprocessing_hints($question, true, true);
 204  
 205          $question = $this->data_preprocessing_answers($question, true);
 206          if (!empty($question->options->answers)) {
 207              $key = 0;
 208              foreach ($question->options->answers as $answer) {
 209                  $question = $this->data_preprocessing_choice($question, $answer, $key);
 210                  $key++;
 211              }
 212          }
 213  
 214          if (!empty($question->options)) {
 215              $question->shuffleanswers = $question->options->shuffleanswers;
 216          }
 217  
 218          return $question;
 219      }

 221      protected function data_preprocessing_choice($question, $answer, $key) {
 222          // See comment in data_preprocessing_answers.
 223          unset($this->_form->_defaultValues['choices[$key][choicegroup]']);
 224          $question->choices[$key]['answer'] = $answer->answer;
 225          $question->choices[$key]['choicegroup'] = $answer->feedback;
 226          return $question;

 229      public function validation($data, $files) {
 230          $errors = parent::validation($data, $files);
 231          $questiontext = $data['questiontext'];
 232          $choices = $data['choices'];
 233  
 234          // Check the whether the slots are valid.
 235          $errorsinquestiontext = $this->validate_slots($questiontext['text'], $choices);
 236          if ($errorsinquestiontext) {
 237              $errors['questiontext'] = $errorsinquestiontext;
 238          }
 239          foreach ($choices as $key => $choice) {
 240              $answer = $choice['answer'];
 241  
 242              // Check whether the HTML tags are allowed tags.
 243              $tagerror = $this->get_illegal_tag_error($answer);
 244              if ($tagerror) {
 245                  $errors['choices['.$key.']'] = $tagerror;
 246              }
 247          }
 248          return $errors;
 249      }

 251      /**
 252       * Finds errors in question slots.
 253       *
 254       * @param string $questiontext The question text
 255       * @param array $choices Question choices
 256       * @return string|bool Error message or false if no errors
 257       */
 258      private function validate_slots($questiontext, $choices) {
 259          $error = 'Please check the Question text: ';
 260          if (!$questiontext) {
 261              return get_string('errorquestiontextblank', 'qtype_gapselect');
 262          }
 263  
 264          $matches = array();
 265          preg_match_all($this->squarebracketsregex, $questiontext, $matches);
 266          $slots = $matches[0];
 267  
 268          if (!$slots) {
 269              return get_string('errornoslots', 'qtype_gapselect');
 270          }
 271  
 272          $cleanedslots = array();
 273          foreach ($slots as $slot) {
 274              // The 2 is for'[[' and 4 is for '[[]]'.
 275              $cleanedslots[] = substr($slot, 2, (strlen($slot) - 4));
 276          }
 277          $slots = $cleanedslots;
 278  
 279          $found = false;
 280          foreach ($slots as $slot) {
 281              $found = false;
 282              foreach ($choices as $key => $choice) {
 283                  if ($slot == $key + 1) {
 284                      if ($choice['answer'] === '') {
 285                          return get_string('errorblankchoice', 'qtype_gapselect',
 286                                  html_writer::tag('b', $slot));
 287                      }
 288                      $found = true;
 289                      break;
 290                  }
 291              }
 292              if (!$found) {
 293                  return get_string('errormissingchoice', 'qtype_gapselect',
 294                          html_writer::tag('b', $slot));
 295              }
 296          }
 297          return false;
 298      }

  29  M.qtype_multianswer.init = function (Y, questiondiv) {
  30      Y.one(questiondiv).all('span.subquestion').each(function(subqspan) {
  31          var feedbackspan = subqspan.one('.feedbackspan');
  32          if (!feedbackspan) {
  33              return;
  34          }
  35  
  36          var overlay = new Y.Overlay({
  37              srcNode: feedbackspan,
  38              visible: false,
  39              align: {
  40                  node: subqspan,
  41                  points: [Y.WidgetPositionAlign.TC, Y.WidgetPositionAlign.BC]
  42              },
  43              constrain: subqspan.ancestor('div.que'),
  44              zIndex: 1,
  45              preventOverlap: true
  46          });
  47          overlay.render();
  48  
  49          Y.on('mouseover', function() { overlay.show(); }, subqspan);
  50          Y.on('mouseout', function() { overlay.hide(); }, subqspan);
  51  
  52          feedbackspan.removeClass('accesshide');
  53      });
  54  };

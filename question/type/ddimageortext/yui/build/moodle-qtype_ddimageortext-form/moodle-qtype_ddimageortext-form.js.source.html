 274      constrain_xy: function(draginstanceno, bgimgxy) {
 275          var drag = this.doc.drag_item(draginstanceno);
 276          var xleftconstrained =
 277              Math.min(bgimgxy[0], this.doc.bg_img().get('width') - drag.get('offsetWidth'));
 278          var ytopconstrained =
 279              Math.min(bgimgxy[1], this.doc.bg_img().get('height') - drag.get('offsetHeight'));
 280          xleftconstrained = Math.max(xleftconstrained, 0);
 281          ytopconstrained = Math.max(ytopconstrained, 0);
 282          return [xleftconstrained, ytopconstrained];

 320          from_name_with_index: function(name) {
 321              var toreturn = {};
 322              toreturn.indexes = [];
 323              var bracket = name.indexOf('[');
 324              toreturn.name = name.substring(0, bracket);
 325              while (bracket !== -1) {
 326                  var end = name.indexOf(']', bracket + 1);
 327                  toreturn.indexes.push(name.substring(bracket + 1, end));
 328                  bracket = name.indexOf('[', end + 1);

 270      /**
 271       * prevent_touchmove_from_scrolling allows users of touch screen devices to
 272       * use drag and drop and normal scrolling at the same time. I.e. when
 273       * touching and dragging a draggable item, the screen does not scroll, but
 274       * you can scroll by touching other area of the screen apart from the
 275       * draggable items.
 276       */
 277      prevent_touchmove_from_scrolling: function(drag) {
 278          var touchstart = (Y.UA.ie) ? 'MSPointerStart' : 'touchstart';
 279          var touchend = (Y.UA.ie) ? 'MSPointerEnd' : 'touchend';
 280          var touchmove = (Y.UA.ie) ? 'MSPointerMove' : 'touchmove';
 281  
 282          // Disable scrolling when touching the draggable items.
 283          drag.on(touchstart, function() {
 284              if (this.touchscrolldisable) {
 285                  return; // Already disabled.
 286              }
 287              this.touchscrolldisable = Y.one('body').on(touchmove, function(e) {
 288                  e = e || window.event;
 289                  e.preventDefault();
 290              });
 291          }, this);
 292  
 293          // Allow scrolling after releasing the draggable items.
 294          drag.on(touchend, function() {
 295              if (this.touchscrolldisable) {
 296                  this.touchscrolldisable.detach();
 297                  this.touchscrolldisable = null;
 298              }
 299          }, this);
 300      },
 301      create_all_drag_and_drops: function() {
 302          this.init_drops();
 303          this.update_padding_sizes_all();
 304          var i = 0;
 305          this.doc.drag_item_homes().each(function(dragitemhome) {
 306              var dragitemno = Number(this.doc.get_classname_numeric_suffix(dragitemhome, 'dragitemhomes'));
 307              var choice = +this.doc.get_classname_numeric_suffix(dragitemhome, 'choice');
 308              var group = +this.doc.get_classname_numeric_suffix(dragitemhome, 'group');
 309              var groupsize = this.doc.drop_zone_group(group).size();
 310              var dragnode = this.doc.clone_new_drag_item(i, dragitemno);
 311              i++;
 312              if (!this.get('readonly')) {
 313                  this.doc.draggable_for_question(dragnode, group, choice);
 314  
 315                  // Prevent scrolling whilst dragging on Adroid devices.
 316                  this.prevent_touchmove_from_scrolling(dragnode);
 317              }
 318              if (dragnode.hasClass('infinite')) {
 319                  var dragstocreate = groupsize - 1;
 320                  while (dragstocreate > 0) {
 321                      dragnode = this.doc.clone_new_drag_item(i, dragitemno);
 322                      i++;
 323                      if (!this.get('readonly')) {
 324                          this.doc.draggable_for_question(dragnode, group, choice);
 325  
 326                          // Prevent scrolling whilst dragging on Adroid devices.
 327                          this.prevent_touchmove_from_scrolling(dragnode);
 328                      }
 329                      dragstocreate--;
 330                  }
 331              }
 332          }, this);
 333          this.reposition_drags_for_question();
 334          if (!this.get('readonly')) {
 335              this.doc.drop_zones().set('tabIndex', 0);
 336              this.doc.drop_zones().each(
 337                  function(v) {
 338                      v.on('dragchange', this.drop_zone_key_press, this);
 339                  }, this);
 340          }
 341          M.util.js_complete(this.pendingid);
 342      },
 343      drop_zone_key_press: function(e) {
 344          switch (e.direction) {
 345              case 'next' :
 346                  this.place_next_drag_in(e.target);
 347                  break;
 348              case 'previous' :
 349                  this.place_previous_drag_in(e.target);
 350                  break;
 351              case 'remove' :
 352                  this.remove_drag_from_drop(e.target);
 353                  break;
 354          }
 355          e.preventDefault();
 356          this.reposition_drags_for_question();
 357      },
 358      place_next_drag_in: function(drop) {
 359          this.search_for_unplaced_drop_choice(drop, 1);
 360      },
 361      place_previous_drag_in: function(drop) {
 362          this.search_for_unplaced_drop_choice(drop, -1);
 363      },
 364      search_for_unplaced_drop_choice: function(drop, direction) {
 365          var next;
 366          var current = this.current_drag_in_drop(drop);
 367          if ('' === current) {
 368              if (direction === 1) {
 369                  next = 1;
 370              } else {
 371                  next = 1;
 372                  var groupno = drop.getData('group');
 373                  this.doc.drag_items_in_group(groupno).each(function(drag) {
 374                      next = Math.max(next, drag.getData('choice'));
 375                  }, this);
 376              }
 377          } else {
 378              next = +current + direction;
 379          }
 380          var drag;
 381          do {
 382              if (this.get_choices_for_drop(next, drop).size() === 0) {
 383                  this.remove_drag_from_drop(drop);
 384                  return;
 385              } else {
 386                  drag = this.get_unplaced_choice_for_drop(next, drop);
 387              }
 388              next = next + direction;
 389          } while (drag === null);
 390          this.place_drag_in_drop(drag, drop);
 391      },
 392      current_drag_in_drop: function(drop) {
 393          var inputid = drop.getData('inputid');
 394          var inputnode = Y.one('input#' + inputid);
 395          return inputnode.get('value');
 396      },
 397      remove_drag_from_drop: function(drop) {
 398          this.place_drag_in_drop(null, drop);
 399      },
 400      place_drag_in_drop: function(drag, drop) {
 401          var inputid = drop.getData('inputid');
 402          var inputnode = Y.one('input#' + inputid);
 403          if (drag !== null) {
 404              inputnode.set('value', drag.getData('choice'));
 405          } else {
 406              inputnode.set('value', '');
 407          }
 408      },
 409      reposition_drags_for_question: function(dotimeout) {
 410          this.doc.drag_items().removeClass('placed');
 411          this.doc.drag_items().each(function(dragitem) {
 412              if (dragitem.dd !== undefined) {
 413                  dragitem.dd.detachAll('drag:start');
 414              }
 415          }, this);
 416          this.doc.drop_zones().each(function(dropzone) {
 417              var relativexy = dropzone.getData('xy');
 418              dropzone.setXY(this.convert_to_window_xy(relativexy));
 419              var inputcss = 'input#' + dropzone.getData('inputid');
 420              var input = this.doc.top_node().one(inputcss);
 421              var choice = input.get('value');
 422              if (choice !== "") {
 423                  var dragitem = this.get_unplaced_choice_for_drop(choice, dropzone);
 424                  if (dragitem !== null) {
 425                      dragitem.setXY(dropzone.getXY());
 426                      dragitem.addClass('placed');
 427                      if (dragitem.dd !== undefined) {
 428                          dragitem.dd.once('drag:start', function(e, input) {
 429                              input.set('value', '');
 430                              e.target.get('node').removeClass('placed');

 432                      }

 434              }
 435          }, this);

 438                  var dragitemhome = this.doc.drag_item_home(dragitem.getData('dragitemno'));
 439                  dragitem.setXY(dragitemhome.getXY());
 440              }

 451      get_unplaced_choice_for_drop: function(choice, drop) {
 452          var dragitems = this.get_choices_for_drop(choice, drop);
 453          var dragitem = null;
 454          dragitems.some(function(d) {
 455              if (this.get('readonly') || (!d.hasClass('placed') && !d.hasClass('yui3-dd-dragging'))) {
 456                  dragitem = d;
 457                  return true;
 458              } else {
 459                  return false;
 460              }
 461          });
 462          return dragitem;

 465          var dropareas = this.doc.top_node().one('div.dropzones');

  42      public function data_preprocessing($question) {
  43          $question = parent::data_preprocessing($question);
  44          $question = $this->data_preprocessing_combined_feedback($question, true);
  45          $question = $this->data_preprocessing_hints($question, true, true);
  46  
  47          $dragids = array(); // Drag no -> dragid.
  48          if (!empty($question->options)) {
  49              $question->shuffleanswers = $question->options->shuffleanswers;
  50              $question->drags = array();
  51              foreach ($question->options->drags as $drag) {
  52                  $dragindex = $drag->no - 1;
  53                  $question->drags[$dragindex] = array();
  54                  $question->draglabel[$dragindex] = $drag->label;
  55                  $question->drags[$dragindex]['infinite'] = $drag->infinite;
  56                  $question->drags[$dragindex]['draggroup'] = $drag->draggroup;
  57                  $dragids[$dragindex] = $drag->id;
  58              }
  59              $question->drops = array();
  60              foreach ($question->options->drops as $drop) {
  61                  $question->drops[$drop->no - 1] = array();
  62                  $question->drops[$drop->no - 1]['choice'] = $drop->choice;
  63                  $question->drops[$drop->no - 1]['droplabel'] = $drop->label;
  64                  $question->drops[$drop->no - 1]['xleft'] = $drop->xleft;
  65                  $question->drops[$drop->no - 1]['ytop'] = $drop->ytop;
  66              }
  67          }
  68          // Initialise file picker for bgimage.
  69          $draftitemid = file_get_submitted_draft_itemid('bgimage');
  70  
  71          file_prepare_draft_area($draftitemid, $this->context->id, 'qtype_ddimageortext',
  72                                  'bgimage', !empty($question->id) ? (int) $question->id : null,
  73                                  self::file_picker_options());
  74          $question->bgimage = $draftitemid;
  75  
  76          // Initialise file picker for dragimages.
  77          list(, $imagerepeats) = $this->get_drag_item_repeats();
  78          $draftitemids = optional_param_array('dragitem', array(), PARAM_INT);
  79          for ($imageindex = 0; $imageindex < $imagerepeats; $imageindex++) {
  80              $draftitemid = isset($draftitemids[$imageindex]) ? $draftitemids[$imageindex] : 0;
  81              // Numbers not allowed in filearea name.
  82              $itemid = isset($dragids[$imageindex]) ? $dragids[$imageindex] : null;
  83              file_prepare_draft_area($draftitemid, $this->context->id, 'qtype_ddimageortext',
  84                                  'dragimage', $itemid, self::file_picker_options());
  85              $question->dragitem[$imageindex] = $draftitemid;
  86          }
  87          if (!empty($question->options)) {
  88              foreach ($question->options->drags as $drag) {
  89                  $dragindex = $drag->no - 1;
  90                  if (!isset($question->dragitem[$dragindex])) {
  91                      $fileexists = false;
  92                  } else {
  93                      $fileexists = self::file_uploaded($question->dragitem[$dragindex]);
  94                  }
  95                  $labelexists = (trim($question->draglabel[$dragindex]) != '');
  96                  if ($labelexists && !$fileexists) {
  97                      $question->drags[$dragindex]['dragitemtype'] = 'word';
  98                  } else {
  99                      $question->drags[$dragindex]['dragitemtype'] = 'image';
 100                  }
 101              }
 102          }
 103          $this->js_call();
 104  
 105          return $question;
 106      }

 220      public function validation($data, $files) {
 221          $errors = parent::validation($data, $files);
 222          if (!self::file_uploaded($data['bgimage'])) {
 223              $errors["bgimage"] = get_string('formerror_nobgimage', 'qtype_'.$this->qtype());
 224          }
 225  
 226          $allchoices = array();
 227          for ($i = 0; $i < $data['nodropzone']; $i++) {
 228              $ytoppresent = (trim($data['drops'][$i]['ytop']) !== '');
 229              $xleftpresent = (trim($data['drops'][$i]['xleft']) !== '');
 230              $ytopisint = (string) clean_param($data['drops'][$i]['ytop'], PARAM_INT) === trim($data['drops'][$i]['ytop']);
 231              $xleftisint = (string) clean_param($data['drops'][$i]['xleft'], PARAM_INT) === trim($data['drops'][$i]['xleft']);
 232              $labelpresent = (trim($data['drops'][$i]['droplabel']) !== '');
 233              $choice = $data['drops'][$i]['choice'];
 234              $imagechoicepresent = ($choice !== '0');
 235  
 236              if ($imagechoicepresent) {
 237                  if (!$ytoppresent) {
 238                      $errors["drops[$i]"] = get_string('formerror_noytop', 'qtype_ddimageortext');
 239                  } else if (!$ytopisint) {
 240                      $errors["drops[$i]"] = get_string('formerror_notintytop', 'qtype_ddimageortext');
 241                  }
 242                  if (!$xleftpresent) {
 243                      $errors["drops[$i]"] = get_string('formerror_noxleft', 'qtype_ddimageortext');
 244                  } else if (!$xleftisint) {
 245                      $errors["drops[$i]"] = get_string('formerror_notintxleft', 'qtype_ddimageortext');
 246                  }
 247  
 248                  if ($data['drags'][$choice - 1]['dragitemtype'] != 'word' &&
 249                                          !self::file_uploaded($data['dragitem'][$choice - 1])) {
 250                      $errors['dragitem['.($choice - 1).']'] =
 251                                      get_string('formerror_nofile', 'qtype_ddimageortext', $i);
 252                  }
 253  
 254                  if (isset($allchoices[$choice]) && !$data['drags'][$choice - 1]['infinite']) {
 255                      $errors["drops[$i]"] =
 256                              get_string('formerror_multipledraginstance', 'qtype_ddimageortext', $choice);
 257                      $errors['drops['.($allchoices[$choice]).']'] =
 258                              get_string('formerror_multipledraginstance', 'qtype_ddimageortext', $choice);
 259                      $errors['drags['.($choice - 1).']'] =
 260                              get_string('formerror_multipledraginstance2', 'qtype_ddimageortext', $choice);
 261                  }
 262                  $allchoices[$choice] = $i;
 263              } else {
 264                  if ($ytoppresent || $xleftpresent || $labelpresent) {
 265                      $errors["drops[$i]"] =
 266                              get_string('formerror_noimageselected', 'qtype_ddimageortext');
 267                  }
 268              }
 269          }
 270          for ($dragindex = 0; $dragindex < $data['noitems']; $dragindex++) {
 271              $label = $data['draglabel'][$dragindex];
 272              if ($data['drags'][$dragindex]['dragitemtype'] == 'word') {
 273                  $allowedtags = '<br><sub><sup><b><i><strong><em>';
 274                  $errormessage = get_string('formerror_disallowedtags', 'qtype_ddimageortext');
 275              } else {
 276                  $allowedtags = '';
 277                  $errormessage = get_string('formerror_noallowedtags', 'qtype_ddimageortext');
 278              }
 279              if ($label != strip_tags($label, $allowedtags)) {
 280                  $errors["drags[{$dragindex}]"] = $errormessage;
 281              }
 282  
 283          }
 284          return $errors;
 285      }

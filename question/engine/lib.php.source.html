 134      /**
 135       * Validate that the manual grade submitted for a particular question is in range.
 136       * @param int $qubaid the question_usage id.
 137       * @param int $slot the slot number within the usage.
 138       * @return bool whether the submitted data is in range.
 139       */
 140      public static function is_manual_grade_in_range($qubaid, $slot) {
 141          $prefix = 'q' . $qubaid . ':' . $slot . '_';
 142          $mark = question_utils::optional_param_mark($prefix . '-mark');
 143          $maxmark = optional_param($prefix . '-maxmark', null, PARAM_FLOAT);
 144          $minfraction = optional_param($prefix . ':minfraction', null, PARAM_FLOAT);
 145          $maxfraction = optional_param($prefix . ':maxfraction', null, PARAM_FLOAT);
 146          return is_null($mark) || ($mark >= $minfraction * $maxmark && $mark <= $maxfraction * $maxmark);
 147      }

 237      /**
 238       * Load the behaviour class(es) belonging to a particular model. That is,
 239       * include_once('/question/behaviour/' . $behaviour . '/behaviour.php'), with a bit
 240       * of checking.
 241       * @param string $qtypename the question type name. For example 'multichoice' or 'shortanswer'.
 242       */
 243      public static function load_behaviour_class($behaviour) {
 244          global $CFG;
 245          if (isset(self::$loadedbehaviours[$behaviour])) {
 246              return;
 247          }
 248          $file = $CFG->dirroot . '/question/behaviour/' . $behaviour . '/behaviour.php';
 249          if (!is_readable($file)) {
 250              throw new coding_exception('Unknown question behaviour ' . $behaviour);
 251          }
 252          include_once($file);
 253  
 254          $class = 'qbehaviour_' . $behaviour;
 255          if (!class_exists($class)) {
 256              throw new coding_exception('Question behaviour ' . $behaviour .
 257                      ' does not define the required class ' . $class . '.');
 258          }
 259  
 260          self::$loadedbehaviours[$behaviour] = 1;
 261      }

 276      public static function get_behaviour_type($behaviour) {
 277  
 278          if (array_key_exists($behaviour, self::$behaviourtypes)) {
 279              return self::$behaviourtypes[$behaviour];
 280          }
 281  
 282          self::load_behaviour_type_class($behaviour);
 283  
 284          $class = 'qbehaviour_' . $behaviour . '_type';
 285          if (class_exists($class)) {
 286              self::$behaviourtypes[$behaviour] = new $class();
 287          } else {
 288              debugging('Question behaviour ' . $behaviour .
 289                      ' does not define the required class ' . $class . '.', DEBUG_DEVELOPER);
 290              self::$behaviourtypes[$behaviour] = new question_behaviour_type_fallback($behaviour);
 291          }
 292  
 293          return self::$behaviourtypes[$behaviour];
 294      }

 342      /**
 343       * Return an array where the keys are the internal names of the behaviours
 344       * in preferred order and the values are a human-readable name.
 345       *
 346       * @param array $archetypes, array of behaviours
 347       * @param string $orderlist, a comma separated list of behaviour names
 348       * @param string $disabledlist, a comma separated list of behaviour names
 349       * @param string $current, current behaviour name
 350       * @return array model name => lang string for this behaviour name.
 351       */
 352      public static function sort_behaviours($archetypes, $orderlist, $disabledlist, $current=null) {
 353  
 354          // Get disabled behaviours
 355          if ($disabledlist) {
 356              $disabled = explode(',', $disabledlist);
 357          } else {
 358              $disabled = array();
 359          }
 360  
 361          if ($orderlist) {
 362              $order = explode(',', $orderlist);
 363          } else {
 364              $order = array();
 365          }
 366  
 367          foreach ($disabled as $behaviour) {
 368              if (array_key_exists($behaviour, $archetypes) && $behaviour != $current) {
 369                  unset($archetypes[$behaviour]);
 370              }
 371          }
 372  
 373          // Get behaviours in preferred order
 374          $behaviourorder = array();
 375          foreach ($order as $behaviour) {
 376              if (array_key_exists($behaviour, $archetypes)) {
 377                  $behaviourorder[$behaviour] = $archetypes[$behaviour];
 378              }
 379          }
 380          // Get the rest of behaviours and sort them alphabetically
 381          $leftover = array_diff_key($archetypes, $behaviourorder);
 382          asort($leftover, SORT_LOCALE_STRING);
 383  
 384          // Set up the final order to be displayed
 385          return $behaviourorder + $leftover;
 386      }

 388      /**
 389       * Return an array where the keys are the internal names of the behaviours
 390       * in preferred order and the values are a human-readable name.
 391       *
 392       * @param string $currentbehaviour
 393       * @return array model name => lang string for this behaviour name.

 395      public static function get_behaviour_options($currentbehaviour) {
 396          $config = question_bank::get_config();
 397          $archetypes = self::get_archetypal_behaviours();
 398  
 399          // If no admin setting return all behavious
 400          if (empty($config->disabledbehaviours) && empty($config->behavioursortorder)) {
 401              return $archetypes;
 402          }
 403  
 404          if (empty($config->behavioursortorder)) {
 405              $order = '';
 406          } else {
 407              $order = $config->behavioursortorder;
 408          }
 409          if (empty($config->disabledbehaviours)) {
 410              $disabled = '';
 411          } else {
 412              $disabled = $config->disabledbehaviours;
 413          }
 414  
 415          return self::sort_behaviours($archetypes, $order, $disabled, $currentbehaviour);
 416      }

 787      /**
 788       * Tests to see whether two arrays have the same keys, with the same values
 789       * (as compared by ===) for each key. However, the order of the arrays does
 790       * not have to be the same.
 791       * @param array $array1 the first array.
 792       * @param array $array2 the second array.
 793       * @return bool whether the two arrays have the same keys with the same
 794       *      corresponding values.
 795       */
 796      public static function arrays_have_same_keys_and_values(array $array1, array $array2) {
 797          if (count($array1) != count($array2)) {
 798              return false;
 799          }
 800          foreach ($array1 as $key => $value1) {
 801              if (!array_key_exists($key, $array2)) {
 802                  return false;
 803              }
 804              if (((string) $value1) !== ((string) $array2[$key])) {
 805                  return false;
 806              }
 807          }
 808          return true;
 809      }

 811      /**
 812       * Tests to see whether two arrays have the same value at a particular key.
 813       * This method will return true if:
 814       * 1. Neither array contains the key; or
 815       * 2. Both arrays contain the key, and the corresponding values compare
 816       *      identical when cast to strings and compared with ===.
 817       * @param array $array1 the first array.
 818       * @param array $array2 the second array.
 819       * @param string $key an array key.
 820       * @return bool whether the two arrays have the same value (or lack of
 821       *      one) for a given key.
 822       */
 823      public static function arrays_same_at_key(array $array1, array $array2, $key) {
 824          if (array_key_exists($key, $array1) && array_key_exists($key, $array2)) {
 825              return ((string) $array1[$key]) === ((string) $array2[$key]);
 826          }
 827          if (!array_key_exists($key, $array1) && !array_key_exists($key, $array2)) {
 828              return true;
 829          }
 830          return false;
 831      }

 837       * @param array $array1 the first array.
 838       * @param array $array2 the second array.

 863       * @param array $array2 the second array.

 980   * A {@link question_variant_selection_strategy} that is effectively random
 981   * for the first attempt, and then after that cycles through the available
 982   * variants so that the students will not get a repeated variant until they have
 983   * seen them all.

 986   * @license    http://www.gnu.org/copyleft/gpl.html GNU GPL v3 or later

 988  class question_variant_pseudorandom_no_repeats_strategy
 989          implements question_variant_selection_strategy {
 990  
 991      /** @var int the number of attempts this users has had, including the curent one. */
 992      protected $attemptno;
 993  
 994      /** @var int the user id the attempt belongs to. */
 995      protected $userid;
 996  
 997      /** @var string extra input fed into the pseudo-random code. */
 998      protected $extrarandomness = '';

1000      /**
1001       * Constructor.
1002       * @param int $attemptno The attempt number.
1003       * @param int $userid the user the attempt is for (defaults to $USER->id).
1004       */
1005      public function __construct($attemptno, $userid = null, $extrarandomness = '') {

1010          } else {

1019      public function choose_variant($maxvariants, $seed) {
1020          if ($maxvariants == 1) {
1021              return 1;
1022          }
1023  
1024          $hash = sha1($seed . '|user' . $this->userid . $this->extrarandomness);
1025          $randint = hexdec(substr($hash, 17, 7));
1026  
1027          return ($randint + $this->attemptno) % $maxvariants + 1;

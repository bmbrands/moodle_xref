  42      public function readquestions($lines) {
  43          question_bank::get_qtype('multianswer'); // Ensure the multianswer code is loaded.
  44  
  45          // For this class the method has been simplified as
  46          // there can never be more than one question for a
  47          // multianswer import.
  48          $questions = array();
  49  
  50          $questiontext = array();
  51          $questiontext['text'] = implode('', $lines);
  52          $questiontext['format'] = FORMAT_MOODLE;
  53          $questiontext['itemid'] = '';
  54          $question = qtype_multianswer_extract_question($questiontext);
  55          $question->questiontext = $question->questiontext['text'];
  56          $question->questiontextformat = 0;
  57  
  58          $question->qtype = 'multianswer';
  59          $question->generalfeedback = '';
  60          $question->generalfeedbackformat = FORMAT_MOODLE;
  61          $question->length = 1;
  62          $question->penalty = 0.3333333;
  63  
  64          if (!empty($question)) {
  65              $question->name = $this->create_default_question_name($question->questiontext, get_string('questionname', 'question'));
  66              $questions[] = $question;
  67          }
  68  
  69          return $questions;
  70      }

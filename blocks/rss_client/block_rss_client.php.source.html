  46      function get_content() {
  47          global $CFG, $DB;
  48  
  49          if ($this->content !== NULL) {
  50              return $this->content;
  51          }
  52  
  53          // initalise block content object
  54          $this->content = new stdClass;
  55          $this->content->text   = '';
  56          $this->content->footer = '';
  57  
  58          if (empty($this->instance)) {
  59              return $this->content;
  60          }
  61  
  62          if (!isset($this->config)) {
  63              // The block has yet to be configured - just display configure message in
  64              // the block if user has permission to configure it
  65  
  66              if (has_capability('block/rss_client:manageanyfeeds', $this->context)) {
  67                  $this->content->text = get_string('feedsconfigurenewinstance2', 'block_rss_client');
  68              }
  69

  71          }
  72

  74          $maxentries = 5;
  75          if ( !empty($this->config->shownumentries) ) {
  76              $maxentries = intval($this->config->shownumentries);
  77          }elseif( isset($CFG->block_rss_client_num_entries) ) {
  78              $maxentries = intval($CFG->block_rss_client_num_entries);
  79          }

  81  
  82          /* ---------------------------------
  83           * Begin Normal Display of Block Content
  84           * --------------------------------- */
  85  
  86          $output = '';
  87  
  88  
  89          if (!empty($this->config->rssid)) {
  90              list($rss_ids_sql, $params) = $DB->get_in_or_equal($this->config->rssid);
  91  
  92              $rss_feeds = $DB->get_records_select('block_rss_client', "id $rss_ids_sql", $params);
  93  
  94              $showtitle = false;
  95              if (count($rss_feeds) > 1) {
  96                  // when many feeds show the title for each feed
  97                  $showtitle = true;
  98              }
  99  
 100              foreach($rss_feeds as $feed){
 101                  $output.= $this->get_feed_html($feed, $maxentries, $showtitle);

 123      /**
 124       * Returns the html of a feed to be displaed in the block
 125       *
 126       * @param mixed feedrecord The feed record from the database
 127       * @param int maxentries The maximum number of entries to be displayed
 128       * @param boolean showtitle Should the feed title be displayed in html
 129       * @return string html representing the rss feed content
 130       */
 131      function get_feed_html($feedrecord, $maxentries, $showtitle){
 132          global $CFG;
 133          require_once($CFG->libdir.'/simplepie/moodle_simplepie.php');
 134  
 135          $feed = new moodle_simplepie($feedrecord->url);
 136  
 137          if(isset($CFG->block_rss_client_timeout)){
 138              $feed->set_cache_duration($CFG->block_rss_client_timeout*60);
 139          }
 140  
 141          if ($CFG->debugdeveloper && $feed->error()) {
 142              return '<p>'. $feedrecord->url .' Failed with code: '.$feed->error().'</p>';
 143          }
 144  
 145          $r = ''; // return string
 146  
 147          if($this->config->block_rss_client_show_channel_image){
 148              if($image = $feed->get_image_url()){
 149                  $imagetitle = s($feed->get_image_title());
 150                  $imagelink  = $feed->get_image_link();
 151  
 152                  $r.='<div class="image" title="'.$imagetitle.'">'."\n";
 153                  if($imagelink){
 154                      $r.='<a href="'.$imagelink.'">';
 155                  }
 156                  $r.='<img src="'.$image.'" alt="'.$imagetitle.'" />'."\n";
 157                  if($imagelink){
 158                      $r.='</a>';
 159                  }
 160                  $r.= '</div>';
 161              }
 162          }
 163  
 164          if(empty($feedrecord->preferredtitle)){
 165              $feedtitle = $this->format_title($feed->get_title());
 166          }else{
 167              $feedtitle = $this->format_title($feedrecord->preferredtitle);
 168          }
 169  
 170          if($showtitle){
 171              $r.='<div class="title">'.$feedtitle.'</div>';
 172          }
 173  
 174  
 175          $r.='<ul class="list no-overflow">'."\n";
 176  
 177          $feeditems = $feed->get_items(0, $maxentries);
 178          foreach($feeditems as $item){
 179              $r.= $this->get_item_html($item);
 180          }
 181  
 182          $r.='</ul>';
 183  
 184  
 185          if ($this->config->block_rss_client_show_channel_link) {
 186  
 187              $channellink = $feed->get_link();
 188  
 189              if (!empty($channellink)){
 190                  //NOTE: this means the 'last feed' display wins the block title - but
 191                  //this is exiting behaviour..
 192                  $this->content->footer = '<a href="'.htmlspecialchars(clean_param($channellink,PARAM_URL)).'">'. get_string('clientchannellink', 'block_rss_client') .'</a>';
 193              }
 194          }
 195  
 196          if (empty($this->config->title)){
 197              //NOTE: this means the 'last feed' displayed wins the block title - but
 198              //this is exiting behaviour..
 199              $this->title = strip_tags($feedtitle);
 200          }
 201  
 202          return $r;
 203      }

 206      /**
 207       * Returns the html list item of a feed item
 208       *
 209       * @param mixed item simplepie_item representing the feed item
 210       * @return string html li representing the rss feed item
 211       */
 212      function get_item_html($item){
 213  
 214          $link        = $item->get_link();
 215          $title       = $item->get_title();
 216          $description = $item->get_description();
 217  
 218  
 219          if(empty($title)){
 220              // no title present, use portion of description
 221              $title = core_text::substr(strip_tags($description), 0, 20) . '...';
 222          }else{
 223              $title = break_up_long_words($title, 30);
 224          }
 225  
 226          if(empty($link)){
 227              $link = $item->get_id();
 228          } else {
 229              try {
 230                  // URLs in our RSS cache will be escaped (correctly as theyre store in XML)
 231                  // html_writer::link() will re-escape them. To prevent double escaping unescape here.
 232                  // This can by done using htmlspecialchars_decode() but moodle_url also has that effect.
 233                  $link = new moodle_url($link);
 234              } catch (moodle_exception $e) {
 235                  // Catching the exception to prevent the whole site to crash in case of malformed RSS feed
 236                  $link = '';
 237              }
 238          }
 239  
 240          $r = html_writer::start_tag('li');
 241              $r.= html_writer::start_tag('div',array('class'=>'link'));
 242                  $r.= html_writer::link($link, s($title), array('onclick'=>'this.target="_blank"'));
 243              $r.= html_writer::end_tag('div');
 244  
 245              if($this->config->display_description && !empty($description)){
 246  
 247                  $formatoptions = new stdClass();
 248                  $formatoptions->para = false;
 249  
 250                  $r.= html_writer::start_tag('div',array('class'=>'description'));
 251                      $description = format_text($description, FORMAT_HTML, $formatoptions, $this->page->course->id);
 252                      $description = break_up_long_words($description, 30);
 253                      $r.= $description;
 254                  $r.= html_writer::end_tag('div');
 255              }
 256          $r.= html_writer::end_tag('li');
 257  
 258          return $r;
 259      }

 277      /**
 278       * cron - goes through all feeds and retrieves them with the cache
 279       * duration set to 0 in order to force the retrieval of the item and
 280       * refresh the cache
 281       *
 282       * @return boolean true if all feeds were retrieved succesfully
 283       */
 284      function cron() {
 285          global $CFG, $DB;
 286          require_once($CFG->libdir.'/simplepie/moodle_simplepie.php');
 287  
 288          // We are going to measure execution times
 289          $starttime =  microtime();
 290  
 291          // And we have one initial $status
 292          $status = true;
 293  
 294          // Fetch all site feeds.
 295          $rs = $DB->get_recordset('block_rss_client');
 296          $counter = 0;
 297          mtrace('');
 298          foreach ($rs as $rec) {
 299              mtrace('    ' . $rec->url . ' ', '');
 300              // Fetch the rss feed, using standard simplepie caching
 301              // so feeds will be renewed only if cache has expired
 302              core_php_time_limit::raise(60);
 303  
 304              $feed =  new moodle_simplepie();
 305              // set timeout for longer than normal to be agressive at
 306              // fetching feeds if possible..
 307              $feed->set_timeout(40);
 308              $feed->set_cache_duration(0);
 309              $feed->set_feed_url($rec->url);
 310              $feed->init();
 311  
 312              if ($feed->error()) {
 313                  mtrace('Error: could not load/find the RSS feed');
 314                  $status = false;
 315              } else {
 316                  mtrace ('ok');
 317              }
 318              $counter ++;
 319          }
 320          $rs->close();
 321  
 322          // Show times
 323          mtrace($counter . ' feeds refreshed (took ' . microtime_diff($starttime, microtime()) . ' seconds)');
 324  
 325          // And return $status
 326          return $status;
 327      }

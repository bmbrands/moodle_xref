  30  class moodle1_block_rss_client_handler extends moodle1_block_handler {
  31      public function process_block(array $data) {
  32          parent::process_block($data);
  33          $instanceid = $data['id'];
  34          $contextid = $this->converter->get_contextid(CONTEXT_BLOCK, $data['id']);
  35  
  36          // Moodle 1.9 backups do not include sufficient data to restore feeds, so we need an empty shell rss_client.xml
  37          // for the restore process to find
  38          $this->open_xml_writer("course/blocks/{$data['name']}_{$instanceid}/rss_client.xml");
  39          $this->xmlwriter->begin_tag('block', array('id' => $instanceid, 'contextid' => $contextid, 'blockname' => 'rss_client'));
  40          $this->xmlwriter->begin_tag('rss_client', array('id' => $instanceid));
  41          $this->xmlwriter->full_tag('feeds', '');
  42          $this->xmlwriter->end_tag('rss_client');
  43          $this->xmlwriter->end_tag('block');
  44          $this->close_xml_writer();
  45  
  46          return $data;
  47      }

  37      /**
  38       * The definition of the fields to use.
  39       *
  40       * @param MoodleQuickForm $mform
  41       */
  42      protected function specific_definition($mform) {
  43          global $DB;
  44  
  45          // Fields for editing activity_results block title and contents.
  46          $mform->addElement('header', 'configheader', get_string('blocksettings', 'block'));
  47  
  48          // Get supported modules (Only modules using grades or scales will be listed).
  49          $sql = 'SELECT id, itemname FROM {grade_items} WHERE courseid = ? and itemtype = ? and (gradetype = ? or gradetype = ?)';
  50          $params = array($this->page->course->id, 'mod', GRADE_TYPE_VALUE, GRADE_TYPE_SCALE);
  51          $activities = $DB->get_records_sql_menu($sql, $params);
  52  
  53          if (empty($activities)) {
  54              $mform->addElement('static', 'noactivitieswarning', get_string('config_select_activity', 'block_activity_results'),
  55                      get_string('config_no_activities_in_course', 'block_activity_results'));
  56          } else {
  57              foreach ($activities as $id => $name) {
  58                  $activities[$id] = strip_tags(format_string($name));
  59              }
  60              $mform->addElement('select', 'config_activitygradeitemid',
  61                      get_string('config_select_activity', 'block_activity_results'), $activities);
  62              $mform->setDefault('config_activitygradeitemid', $this->block->get_owning_activity()->id);
  63          }
  64  
  65          $mform->addElement('text', 'config_showbest',
  66                  get_string('config_show_best', 'block_activity_results'), array('size' => 3));
  67          $mform->setDefault('config_showbest', 3);
  68          $mform->setType('config_showbest', PARAM_INT);
  69  
  70          $mform->addElement('text', 'config_showworst',
  71                  get_string('config_show_worst', 'block_activity_results'), array('size' => 3));
  72          $mform->setDefault('config_showworst', 0);
  73          $mform->setType('config_showworst', PARAM_INT);
  74  
  75          $mform->addElement('selectyesno', 'config_usegroups', get_string('config_use_groups', 'block_activity_results'));
  76  
  77          $nameoptions = array(
  78              B_ACTIVITYRESULTS_NAME_FORMAT_FULL => get_string('config_names_full', 'block_activity_results'),
  79              B_ACTIVITYRESULTS_NAME_FORMAT_ID => get_string('config_names_id', 'block_activity_results'),
  80              B_ACTIVITYRESULTS_NAME_FORMAT_ANON => get_string('config_names_anon', 'block_activity_results')
  81          );
  82          $mform->addElement('select', 'config_nameformat',
  83                  get_string('config_name_format', 'block_activity_results'), $nameoptions);
  84          $mform->setDefault('config_nameformat', B_ACTIVITYRESULTS_NAME_FORMAT_FULL);
  85  
  86          $gradeeoptions = array(
  87              B_ACTIVITYRESULTS_GRADE_FORMAT_PCT => get_string('config_format_percentage', 'block_activity_results'),
  88              B_ACTIVITYRESULTS_GRADE_FORMAT_FRA => get_string('config_format_fraction', 'block_activity_results'),
  89              B_ACTIVITYRESULTS_GRADE_FORMAT_ABS => get_string('config_format_absolute', 'block_activity_results')
  90          );
  91          $mform->addElement('select', 'config_gradeformat',
  92                  get_string('config_grade_format', 'block_activity_results'), $gradeeoptions);
  93          $mform->setDefault('config_gradeformat', B_ACTIVITYRESULTS_GRADE_FORMAT_PCT);
  94  
  95          $options = array();
  96          for ($i = 0; $i <= 5; $i++) {
  97              $options[$i] = $i;
  98          }
  99          $mform->addElement('select', 'config_decimalpoints', get_string('config_decimalplaces', 'block_activity_results'),
 100                  $options);
 101          $mform->setDefault('config_decimalpoints', 2);
 102          $mform->setType('config_decimalpoints', PARAM_INT);
 103      }

  25  function ctags_getKeywords() {
  26    /*
  27    // This 'workaround' removing the xhtml strict form autocomplete="off" needs to
  28    // be added to the body onload() script to work - but decided not to include
  29    // (having the browser list might help with screen readers more than this script)
  30    // document.forms['coursetag'].setAttribute("autocomplete", "off");
  31    */
  32    var input = document.forms['coursetag'].coursetag_new_tag.value;
  33    var len = input.length;
  34    sug_disp = ""; sug = "";
  35  
  36    if (input.length) {
  37      for (ele in coursetag_tags)
  38      {
  39        if (coursetag_tags[ele].substr(0,len).toLowerCase() == input.toLowerCase())
  40        {
  41          sug_disp = input + coursetag_tags[ele].substr(len);
  42          sug = coursetag_tags[ele];
  43          break;
  44        }
  45      }
  46    }
  47    document.forms['coursetag'].coursetag_sug_keyword.value = sug_disp;
  48    if (!sug.length || input == sug_disp) {
  49      document.getElementById('coursetag_sug_btn').style.display = "none";
  50    } else {
  51      document.getElementById('coursetag_sug_btn').style.display = "block";
  52    }
  53  }

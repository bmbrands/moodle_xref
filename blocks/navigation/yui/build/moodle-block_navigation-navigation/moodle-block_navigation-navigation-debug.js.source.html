  51   * A 'actionkey' Event to help with Y.delegate().
  52   * The event consists of the left arrow, right arrow, enter and space keys.
  53   * More keys can be mapped to action meanings.
  54   * actions: collapse , expand, toggle, enter.
  55   *
  56   * This event is delegated to branches in the navigation tree.
  57   * The on() method to subscribe allows specifying the desired trigger actions as JSON.
  58   *
  59   * @namespace M.block_navigation
  60   * @class ActionKey
  61   */
  62  Y.Event.define("actionkey", {
  63      // Webkit and IE repeat keydown when you hold down arrow keys.
  64      // Opera links keypress to page scroll; others keydown.
  65      // Firefox prevents page scroll via preventDefault() on either
  66      // keydown or keypress.
  67      _event: (Y.UA.webkit || Y.UA.ie) ? 'keydown' : 'keypress',
  68  
  69      /**
  70       * The keys to trigger on.
  71       * @method _keys
  72       */
  73      _keys: {
  74          //arrows
  75          '37': 'collapse',
  76          '39': 'expand',
  77          '32': 'toggle',
  78          '13': 'enter'
  79      },
  80  
  81      /**
  82       * Handles key events
  83       * @method _keyHandler
  84       * @param {EventFacade} e
  85       * @param {SyntheticEvent.Notifier} notifier The notifier used to trigger the execution of subscribers
  86       * @param {Object} args
  87       */
  88      _keyHandler: function (e, notifier, args) {
  89          var actObj;
  90          if (!args.actions) {
  91              actObj = {collapse:true, expand:true, toggle:true, enter:true};
  92          } else {
  93              actObj = args.actions;
  94          }
  95          if (this._keys[e.keyCode] && actObj[this._keys[e.keyCode]]) {
  96              e.action = this._keys[e.keyCode];
  97              notifier.fire(e);
  98          }
  99      },
 100  
 101      /**
 102       * Subscribes to events.
 103       * @method on
 104       * @param {Node} node The node this subscription was applied to.
 105       * @param {Subscription} sub The object tracking this subscription.
 106       * @param {SyntheticEvent.Notifier} notifier The notifier used to trigger the execution of subscribers
 107       */
 108      on: function (node, sub, notifier) {
 109          // subscribe to _event and ask keyHandler to handle with given args[0] (the desired actions).
 110          if (sub.args === null) {
 111              //no actions given
 112              sub._detacher = node.on(this._event, this._keyHandler,this, notifier, {actions:false});
 113          } else {
 114              sub._detacher = node.on(this._event, this._keyHandler,this, notifier, sub.args[0]);
 115          }
 116      },
 117  
 118      /**
 119       * Detaches an event listener
 120       * @method detach
 121       */
 122      detach: function (node, sub) {
 123          //detach our _detacher handle of the subscription made in on()
 124          sub._detacher.detach();
 125      },
 126  
 127      /**
 128       * Creates a delegated event listener.
 129       * @method delegate
 130       * @param {Node} node The node this subscription was applied to.
 131       * @param {Subscription} sub The object tracking this subscription.
 132       * @param {SyntheticEvent.Notifier} notifier The notifier used to trigger the execution of subscribers
 133       * @param {String|function} filter Selector string or function that accpets an event object and returns null.
 134       */
 135      delegate: function (node, sub, notifier, filter) {
 136          // subscribe to _event and ask keyHandler to handle with given args[0] (the desired actions).
 137          if (sub.args === null) {
 138              //no actions given
 139              sub._delegateDetacher = node.delegate(this._event, this._keyHandler,filter, this, notifier, {actions:false});
 140          } else {
 141              sub._delegateDetacher = node.delegate(this._event, this._keyHandler,filter, this, notifier, sub.args[0]);
 142          }
 143      },

 154          sub._delegateDetacher.detach();

 303      /**
 304       * This is a callback function responsible for expanding and collapsing the
 305       * branches of the tree. It is delegated to rather than multiple event handles.
 306       * @method toggleExpansion
 307       * @param {EventFacade} e
 308       * @return Boolean
 309       */
 310      toggleExpansion : function(e) {
 311          // First check if they managed to click on the li iteslf, then find the closest
 312          // LI ancestor and use that
 313  
 314          if (e.target.test('a') && (e.keyCode === 0 || e.keyCode === 13)) {
 315              // A link has been clicked (or keypress is 'enter') don't fire any more events just do the default.
 316              e.stopPropagation();
 317              return;
 318          }
 319  
 320          // Makes sure we can get to the LI containing the branch.
 321          var target = e.target;
 322          if (!target.test('li')) {
 323              target = target.ancestor('li');
 324          }
 325          if (!target) {
 326              return;
 327          }
 328  
 329          // Toggle expand/collapse providing its not a root level branch.
 330          if (!target.hasClass('depth_1')) {
 331              if (e.type === 'actionkey') {
 332                  switch (e.action) {
 333                      case 'expand' :
 334                          target.removeClass('collapsed');
 335                          target.set('aria-expanded', true);
 336                          break;
 337                      case 'collapse' :
 338                          target.addClass('collapsed');
 339                          target.set('aria-expanded', false);
 340                          break;
 341                      default :
 342                          target.toggleClass('collapsed');
 343                          target.set('aria-expanded', !target.hasClass('collapsed'));
 344                  }
 345                  e.halt();
 346              } else {
 347                  target.toggleClass('collapsed');
 348                  target.set('aria-expanded', !target.hasClass('collapsed'));
 349              }
 350          }
 351  
 352          // If the accordian feature has been enabled collapse all siblings.
 353          if (this.get('accordian')) {
 354              target.siblings('li').each(function(){
 355                  if (this.get('id') !== target.get('id') && !this.hasClass('collapsed')) {
 356                      this.addClass('collapsed');
 357                      this.set('aria-expanded', false);
 358                  }
 359              });
 360          }
 361  
 362          // If this block can dock tell the dock to resize if required and check
 363          // the width on the dock panel in case it is presently in use.
 364          if (this.get('candock') && M.core.dock.notifyBlockChange) {
 365              M.core.dock.notifyBlockChange(this.id);
 366          }
 367          return true;
 368  
 369      }

 441  BRANCH.prototype = {
 442      /**
 443       * The node for this branch (p)
 444       * @property node
 445       * @type Node
 446       * @protected

 449      /**
 450       * Initialises the branch when it is first created.
 451       * @method initializer
 452       * @param {Object} config
 453       */
 454      initializer : function(config) {
 455          var i,
 456              children;
 457          if (config.branchobj !== null) {
 458              // Construct from the provided xml
 459              for (i in config.branchobj) {
 460                  this.set(i, config.branchobj[i]);
 461              }
 462              children = this.get('children');
 463              this.set('haschildren', (children.length > 0));
 464          }

 466              // Construct from the provided xml

 468                  this.set(i, config.overrides[i]);
 469              }
 470          }

 476              this.set('expandable', false);
 477              this.set('haschildren', false);
 478          }
 479      },
 480      /**
 481       * Draws the branch within the tree.
 482       *
 483       * This function creates a DOM structure for the branch and then injects
 484       * it into the navigation tree at the correct point.
 485       *
 486       * It is important that this is kept in check with block_navigation_renderer::navigation_node as that produces
 487       * the same thing as this but on the php side.
 488       *
 489       * @method draw
 490       * @chainable
 491       * @param {Node} element
 492       * @return Branch
 493       */
 494      draw : function(element) {
 495  
 496          var isbranch = (this.get('expandable') || this.get('haschildren'));
 497          var branchli = Y.Node.create('<li></li>');
 498          var link = this.get('link');
 499          var branchp = Y.Node.create('<p class="tree_item"></p>').setAttribute('id', this.get('id'));
 500          var name;
 501          if (!link) {
 502              //add tab focus if not link (so still one focus per menu node).
 503              // it was suggested to have 2 foci. one for the node and one for the link in MDL-27428.
 504              branchp.setAttribute('tabindex', '0');
 505          }
 506          if (isbranch) {
 507              branchli.addClass('collapsed').addClass('contains_branch');
 508              branchli.set('aria-expanded', false);
 509              branchp.addClass('branch');
 510          }
 511  
 512          // Prepare the icon, should be an object representing a pix_icon
 513          var branchicon = false;
 514          var icon = this.get('icon');
 515          if (icon && (!isbranch || this.get('type') === NODETYPE.ACTIVITY || this.get('type') === NODETYPE.RESOURCE)) {
 516              branchicon = Y.Node.create('<img alt="" />');
 517              branchicon.setAttribute('src', M.util.image_url(icon.pix, icon.component));
 518              branchli.addClass('item_with_icon');
 519              branchp.addClass('hasicon');
 520              if (icon.alt) {
 521                  branchicon.setAttribute('alt', icon.alt);
 522              }
 523              if (icon.title) {
 524                  branchicon.setAttribute('title', icon.title);
 525              }
 526              if (icon.classes) {
 527                  for (var i in icon.classes) {
 528                      branchicon.addClass(icon.classes[i]);
 529                  }
 530              }
 531          }
 532  
 533          if (!link) {
 534              var branchspan = Y.Node.create('<span></span>');
 535              if (branchicon) {
 536                  branchspan.appendChild(branchicon);
 537                  name = '<span class="item-content-wrap">' + this.get('name') + '</span>';
 538              } else {
 539                  name = this.get('name');
 540              }
 541              branchspan.append(name);
 542              if (this.get('hidden')) {
 543                  branchspan.addClass('dimmed_text');
 544              }
 545              branchp.appendChild(branchspan);
 546          } else {
 547              var branchlink = Y.Node.create('<a title="'+this.get('title')+'" href="'+link+'"></a>');
 548              if (branchicon) {
 549                  branchlink.appendChild(branchicon);
 550                  name = '<span class="item-content-wrap">' + this.get('name') + '</span>';
 551              } else {
 552                  name = this.get('name');
 553              }
 554              branchlink.append(name);
 555              if (this.get('hidden')) {
 556                  branchlink.addClass('dimmed');
 557              }
 558              branchp.appendChild(branchlink);
 559          }

 571       * @return {BRANCH} This function is chainable, it always returns itself.

 574          this.node = this.node || Y.one('#'+this.get('id'));

 576              return this;
 577          }

 587       * @return Node
 588       */
 589      getChildrenUL : function() {
 590          var ul = this.node.next('ul');
 591          if (!ul) {
 592              ul = Y.Node.create('<ul></ul>');
 593              this.node.ancestor().append(ul);
 594          }
 595          return ul;
 596      },
 597      /**
 598       * Load the content of the branch via AJAX.
 599       *
 600       * This function calls ajaxProcessResponse with the result of the AJAX
 601       * request made here.
 602       *
 603       * @method ajaxLoad
 604       * @param {EventFacade} e
 605       * @return Bool
 606       */
 607      ajaxLoad : function(e) {
 608          if (e.type === 'actionkey' && e.action !== 'enter') {
 609              e.halt();
 610          } else {
 611              e.stopPropagation();
 612          }
 613          if ((e.type === 'actionkey' && e.action === 'enter') || e.target.test('a')) {
 614              // No ajaxLoad for enter.
 615              this.node.setAttribute('data-expandable', '0');
 616              this.node.setAttribute('data-loaded', '1');
 617              return true;
 618          }
 619  
 620          if (this.node.hasClass('loadingbranch')) {
 621              // Already loading. Just skip.
 622              return true;
 623          }
 624  
 625          if (this.node.getAttribute('data-loaded') === '1') {
 626              // We've already loaded this stuff.
 627              return true;
 628          }
 629          Y.log('Loading navigation branch via AJAX: '+this.get('key'), 'note', 'moodle-block_navigation');
 630          this.node.addClass('loadingbranch');
 631  
 632          var params = {
 633              elementid : this.get('id'),
 634              id : this.get('key'),
 635              type : this.get('type'),
 636              sesskey : M.cfg.sesskey,
 637              instance : this.get('tree').get('instance')
 638          };
 639  
 640          var ajaxfile = '/lib/ajax/getnavbranch.php';
 641          // For siteadmin navigation get tree from getsiteadminbranch.php.
 642          if (this.get('type') === NODETYPE.SITEADMIN) {
 643              ajaxfile = '/lib/ajax/getsiteadminbranch.php';
 644          }
 645  
 646          Y.io(M.cfg.wwwroot + ajaxfile, {
 647              method:'POST',
 648              data:  params,
 649              on: {
 650                  complete: this.ajaxProcessResponse
 651              },
 652              context:this
 653          });
 654          return true;
 655      },
 656      /**
 657       * Processes an AJAX request to load the content of this branch through
 658       * AJAX.
 659       *
 660       * @method ajaxProcessResponse
 661       * @param {Int} tid The transaction id.
 662       * @param {Object} outcome
 663       * @return Boolean
 664       */
 665      ajaxProcessResponse : function(tid, outcome) {
 666          this.node.removeClass('loadingbranch');
 667          this.node.setAttribute('data-loaded', '1');
 668          try {
 669              var object = Y.JSON.parse(outcome.responseText);
 670              if (object.error) {
 671                  Y.use('moodle-core-notification-ajaxexception', function () {
 672                      return new M.core.ajaxException(object).show();
 673                  });
 674                  return false;
 675              }
 676              if (object.children && object.children.length > 0) {
 677                  var coursecount = 0;
 678                  for (var i in object.children) {
 679                      if (typeof(object.children[i])==='object') {
 680                          if (object.children[i].type === NODETYPE.COURSE) {
 681                              coursecount++;
 682                          }
 683                          this.addChild(object.children[i]);
 684                      }
 685                  }
 686                  if ((this.get('type') === NODETYPE.CATEGORY ||
 687                       this.get('type') === NODETYPE.ROOTNODE ||
 688                       this.get('type') === NODETYPE.MYCATEGORY)
 689                       && coursecount >= M.block_navigation.courselimit) {
 690                      this.addViewAllCoursesChild(this);
 691                  }
 692                  Y.log('AJAX loading complete.', 'note', 'moodle-block_navigation');
 693                  // If this block can dock tell the dock to resize if required and check

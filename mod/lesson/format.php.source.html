  66   * @param object $question Contains question data like question, type and answers.

  71  function lesson_save_question_options($question, $lesson, $contextid) {
  72      global $DB;
  73  
  74      // These lines are required to ensure that all page types have
  75      // been loaded for the following switch
  76      if (!($lesson instanceof lesson)) {
  77          $lesson = new lesson($lesson);
  78      }
  79      $manager = lesson_page_type_manager::get($lesson);
  80  
  81      $timenow = time();
  82      $result = new stdClass();
  83  
  84      // Default answer to avoid code duplication.
  85      $defaultanswer = new stdClass();
  86      $defaultanswer->lessonid   = $question->lessonid;
  87      $defaultanswer->pageid = $question->id;
  88      $defaultanswer->timecreated   = $timenow;
  89      $defaultanswer->answerformat = FORMAT_HTML;
  90      $defaultanswer->jumpto = LESSON_THISPAGE;
  91      $defaultanswer->grade = 0;
  92      $defaultanswer->score = 0;
  93  
  94      switch ($question->qtype) {
  95          case LESSON_PAGE_SHORTANSWER:
  96  
  97              $answers = array();
  98              $maxfraction = -1;
  99  
 100              // Insert all the new answers
 101              foreach ($question->answer as $key => $dataanswer) {
 102                  if ($dataanswer != "") {
 103                      $answer = clone($defaultanswer);
 104                      if ($question->fraction[$key] >=0.5) {
 105                          $answer->jumpto = LESSON_NEXTPAGE;
 106                          $answer->score = 1;
 107                      }
 108                      $answer->grade = round($question->fraction[$key] * 100);
 109                      $answer->answer   = $dataanswer;
 110                      $answer->response = $question->feedback[$key]['text'];
 111                      $answer->responseformat = $question->feedback[$key]['format'];
 112                      $answer->id = $DB->insert_record("lesson_answers", $answer);
 113                      lesson_import_question_files('response', $question->feedback[$key], $answer, $contextid);
 114                      $answers[] = $answer->id;
 115                      if ($question->fraction[$key] > $maxfraction) {
 116                          $maxfraction = $question->fraction[$key];
 117                      }
 118                  }
 119              }
 120  
 121  
 122              /// Perform sanity checks on fractional grades
 123              if ($maxfraction != 1) {
 124                  $maxfraction = $maxfraction * 100;
 125                  $result->notice = get_string("fractionsnomax", "lesson", $maxfraction);
 126                  return $result;
 127              }
 128              break;
 129  
 130          case LESSON_PAGE_NUMERICAL:   // Note similarities to shortanswer.
 131  
 132              $answers = array();
 133              $maxfraction = -1;
 134  
 135  
 136              // for each answer store the pair of min and max values even if they are the same
 137              foreach ($question->answer as $key => $dataanswer) {
 138                  if ($dataanswer != "") {
 139                      $answer = clone($defaultanswer);
 140                      if ($question->fraction[$key] >= 0.5) {
 141                          $answer->jumpto = LESSON_NEXTPAGE;
 142                          $answer->score = 1;
 143                      }
 144                      $answer->grade = round($question->fraction[$key] * 100);
 145                      $min = $question->answer[$key] - $question->tolerance[$key];
 146                      $max = $question->answer[$key] + $question->tolerance[$key];
 147                      $answer->answer   = $min.":".$max;
 148                      $answer->response = $question->feedback[$key]['text'];
 149                      $answer->responseformat = $question->feedback[$key]['format'];
 150                      $answer->id = $DB->insert_record("lesson_answers", $answer);
 151                      lesson_import_question_files('response', $question->feedback[$key], $answer, $contextid);
 152  
 153                      $answers[] = $answer->id;
 154                      if ($question->fraction[$key] > $maxfraction) {
 155                          $maxfraction = $question->fraction[$key];
 156                      }
 157                  }
 158              }
 159  
 160              /// Perform sanity checks on fractional grades
 161              if ($maxfraction != 1) {
 162                  $maxfraction = $maxfraction * 100;
 163                  $result->notice = get_string("fractionsnomax", "lesson", $maxfraction);
 164                  return $result;
 165              }
 166          break;
 167  
 168  
 169          case LESSON_PAGE_TRUEFALSE:
 170  
 171              // the truth
 172              $answer = clone($defaultanswer);
 173              $answer->answer = get_string("true", "lesson");
 174              $answer->grade = $question->correctanswer * 100;
 175              if ($answer->grade > 50 ) {
 176                  $answer->jumpto = LESSON_NEXTPAGE;
 177                  $answer->score = 1;
 178              }
 179              if (isset($question->feedbacktrue)) {
 180                  $answer->response = $question->feedbacktrue['text'];
 181                  $answer->responseformat = $question->feedbacktrue['format'];
 182              }
 183              $answer->id = $DB->insert_record("lesson_answers", $answer);
 184              lesson_import_question_files('response', $question->feedbacktrue, $answer, $contextid);
 185  
 186              // the lie
 187              $answer = clone($defaultanswer);
 188              $answer->answer = get_string("false", "lesson");
 189              $answer->grade = (1 - (int)$question->correctanswer) * 100;
 190              if ($answer->grade > 50 ) {
 191                  $answer->jumpto = LESSON_NEXTPAGE;
 192                  $answer->score = 1;
 193              }
 194              if (isset($question->feedbackfalse)) {
 195                  $answer->response = $question->feedbackfalse['text'];
 196                  $answer->responseformat = $question->feedbackfalse['format'];
 197              }
 198              $answer->id = $DB->insert_record("lesson_answers", $answer);
 199              lesson_import_question_files('response', $question->feedbackfalse, $answer, $contextid);
 200  
 201            break;
 202  
 203          case LESSON_PAGE_MULTICHOICE:
 204  
 205              $totalfraction = 0;
 206              $maxfraction = -1;
 207  
 208              $answers = array();
 209  
 210              // Insert all the new answers
 211              foreach ($question->answer as $key => $dataanswer) {
 212                  if ($dataanswer != "") {
 213                      $answer = clone($defaultanswer);
 214                      $answer->grade = round($question->fraction[$key] * 100);
 215  
 216                      if ($question->single) {
 217                          if ($answer->grade > 50) {
 218                              $answer->jumpto = LESSON_NEXTPAGE;
 219                              $answer->score = 1;
 220                          }
 221                      } else {
 222                          // If multi answer allowed, any answer with fraction > 0 is considered correct.
 223                          if ($question->fraction[$key] > 0) {
 224                              $answer->jumpto = LESSON_NEXTPAGE;
 225                              $answer->score = 1;
 226                          }
 227                      }
 228                      $answer->answer   = $dataanswer['text'];
 229                      $answer->answerformat   = $dataanswer['format'];
 230                      $answer->response = $question->feedback[$key]['text'];
 231                      $answer->responseformat = $question->feedback[$key]['format'];
 232                      $answer->id = $DB->insert_record("lesson_answers", $answer);
 233                      lesson_import_question_files('answer', $dataanswer, $answer, $contextid);
 234                      lesson_import_question_files('response', $question->feedback[$key], $answer, $contextid);
 235  
 236                      // for Sanity checks
 237                      if ($question->fraction[$key] > 0) {
 238                          $totalfraction += $question->fraction[$key];
 239                      }
 240                      if ($question->fraction[$key] > $maxfraction) {
 241                          $maxfraction = $question->fraction[$key];
 242                      }
 243                  }
 244              }
 245  
 246              /// Perform sanity checks on fractional grades
 247              if ($question->single) {
 248                  if ($maxfraction != 1) {
 249                      $maxfraction = $maxfraction * 100;
 250                      $result->notice = get_string("fractionsnomax", "lesson", $maxfraction);
 251                      return $result;
 252                  }
 253              } else {
 254                  $totalfraction = round($totalfraction,2);
 255                  if ($totalfraction != 1) {
 256                      $totalfraction = $totalfraction * 100;
 257                      $result->notice = get_string("fractionsaddwrong", "lesson", $totalfraction);
 258                      return $result;
 259                  }
 260              }
 261          break;
 262  
 263          case LESSON_PAGE_MATCHING:
 264  
 265              $subquestions = array();
 266  
 267              // The first answer should always be the correct answer
 268              $correctanswer = clone($defaultanswer);
 269              $correctanswer->answer = get_string('thatsthecorrectanswer', 'lesson');
 270              $correctanswer->jumpto = LESSON_NEXTPAGE;
 271              $correctanswer->score = 1;
 272              $DB->insert_record("lesson_answers", $correctanswer);
 273  
 274              // The second answer should always be the wrong answer
 275              $wronganswer = clone($defaultanswer);
 276              $wronganswer->answer = get_string('thatsthewronganswer', 'lesson');
 277              $DB->insert_record("lesson_answers", $wronganswer);
 278  
 279              $i = 0;
 280              // Insert all the new question+answer pairs
 281              foreach ($question->subquestions as $key => $questiontext) {
 282                  $answertext = $question->subanswers[$key];
 283                  if (!empty($questiontext) and !empty($answertext)) {
 284                      $answer = clone($defaultanswer);
 285                      $answer->answer = $questiontext['text'];
 286                      $answer->answerformat   = $questiontext['format'];
 287                      $answer->response   = $answertext;
 288                      if ($i == 0) {
 289                          // first answer contains the correct answer jump
 290                          $answer->jumpto = LESSON_NEXTPAGE;
 291                      }
 292                      $answer->id = $DB->insert_record("lesson_answers", $answer);
 293                      lesson_import_question_files('answer', $questiontext, $answer, $contextid);
 294                      $subquestions[] = $answer->id;
 295                      $i++;
 296                  }
 297              }
 298  
 299              if (count($subquestions) < 3) {
 300                  $result->notice = get_string("notenoughsubquestions", "lesson");
 301                  return $result;
 302              }
 303              break;
 304          default:
 305              $result->error = "Unsupported question type ($question->qtype)!";
 306              return $result;
 307      }
 308      return true;
 309  }

 376      /// Processes a given file.  There's probably little need to change this
 377          $timenow = time();

 548      function readdata($filename) {
 549      /// Returns complete file with an array, one item per line
 550  
 551          if (is_readable($filename)) {
 552              $filearray = file($filename);
 553  
 554              /// Check for Macintosh OS line returns (ie file on one line), and fix
 555              if (preg_match("/\r/", $filearray[0]) AND !preg_match("/\n/", $filearray[0])) {
 556                  return explode("\r", $filearray[0]);
 557              } else {
 558                  return $filearray;
 559              }
 560          }
 561          return false;
 562      }

 564      protected function readquestions($lines) {
 565      /// Parses an array of lines into an array of questions,
 566      /// where each item is a question object as defined by
 567      /// readquestion().   Questions are defined as anything
 568      /// between blank lines.
 569  
 570          $questions = array();
 571          $currentquestion = array();
 572  
 573          foreach ($lines as $line) {
 574              $line = trim($line);
 575              if (empty($line)) {
 576                  if (!empty($currentquestion)) {
 577                      if ($question = $this->readquestion($currentquestion)) {
 578                          $questions[] = $question;
 579                      }
 580                      $currentquestion = array();
 581                  }
 582              } else {
 583                  $currentquestion[] = $line;
 584              }
 585          }
 586  
 587          if (!empty($currentquestion)) {  // There may be a final question
 588              if ($question = $this->readquestion($currentquestion)) {
 589                  $questions[] = $question;
 590              }
 591          }
 592  
 593          return $questions;
 594      }

 760       * @param array xml data to read
 761       * @param array path path to node expressed as array
 762       * @param mixed default

 764       * @param string error if set value must exist, return false and issue message if not
 765       * @return mixed value
 766       */
 767      public function getpath($xml, $path, $default, $istext=false, $error='') {
 768          foreach ($path as $index) {
 769              if (!isset($xml[$index])) {
 770                  if (!empty($error)) {
 771                      $this->error($error);
 772                      return false;
 773                  } else {
 774                      return $default;
 775                  }
 776              }
 777  
 778              $xml = $xml[$index];
 779          }
 780  
 781          if ($istext) {
 782              if (!is_string($xml)) {
 783                  $this->error(get_string('invalidxml', 'qformat_xml'));
 784              }
 785              $xml = trim($xml);
 786          }

 126      public function construction_override($pageid, lesson $lesson) {
 127          global $CFG, $PAGE, $DB;
 128          require_sesskey();
 129  
 130          $timenow = time();
 131  
 132          // the new page is not the first page (end of cluster always comes after an existing page)
 133          if (!$page = $DB->get_record("lesson_pages", array("id" => $pageid))) {
 134              print_error('cannotfindpages', 'lesson');
 135          }
 136  
 137          // could put code in here to check if the user really can insert an end of cluster
 138  
 139          $newpage = new stdClass;
 140          $newpage->lessonid = $lesson->id;
 141          $newpage->prevpageid = $pageid;
 142          $newpage->nextpageid = $page->nextpageid;
 143          $newpage->qtype = $this->qtype;
 144          $newpage->timecreated = $timenow;
 145          $newpage->title = get_string("endofclustertitle", "lesson");
 146          $newpage->contents = get_string("endofclustertitle", "lesson");
 147          $newpageid = $DB->insert_record("lesson_pages", $newpage);
 148          // update the linked list...
 149          $DB->set_field("lesson_pages", "nextpageid", $newpageid, array("id" => $pageid));
 150          if ($page->nextpageid) {
 151              // the new page is not the last page
 152              $DB->set_field("lesson_pages", "prevpageid", $newpageid, array("id" => $page->nextpageid));
 153          }
 154          // ..and the single "answer"
 155          $newanswer = new stdClass;
 156          $newanswer->lessonid = $lesson->id;
 157          $newanswer->pageid = $newpageid;
 158          $newanswer->timecreated = $timenow;
 159          $newanswer->jumpto = LESSON_NEXTPAGE;
 160          $newanswerid = $DB->insert_record("lesson_answers", $newanswer);
 161          $lesson->add_message(get_string('addedendofcluster', 'lesson'), 'notifysuccess');
 162          redirect($CFG->wwwroot.'/mod/lesson/edit.php?id='.$PAGE->cm->id);
 163      }

  30  /**
  31   * Removes double CRs.
  32   *
  33   * @deprecated Since Moodle 2.9 MDL-48901 - please do not use this function any more.
  34   * @todo MDL-48985 This will be deleted in Moodle 3.1
  35   * @param string $filename
  36   * @return void
  37   */
  38  function removedoublecr($filename) {
  39  // This function will adjust a file in roughly Aiken style by replacing extra newlines with <br/> tags
  40  // so that instructors can have newlines wherever they like as long as the overall format is in Aiken
  41  
  42      $filearray = file($filename);
  43      /// Check for Macintosh OS line returns (ie file on one line), and fix
  44      if (preg_match("/\r/", $filearray[0]) AND !preg_match("/\n/", $filearray[0])) {
  45          $outfile = explode("\r", $filearray[0]);
  46      } else {
  47          $outfile = $filearray;
  48      }
  49  
  50      $outarray = array();
  51  
  52      foreach ($outfile as $line) {
  53          // remove leading and trailing whitespace
  54          trim($line);
  55          // check it's length, if 0 do not output... if it is > 0 output
  56          if ($line[0] == "\n" OR strlen($line)==0 ) {
  57              if (count($outarray) ) {
  58                  // get the last item in the outarray
  59                  $cur_pos = (count($outarray) - 1);
  60                  $outarray[$cur_pos] = trim($outarray[$cur_pos])."<br/>\n";
  61              }
  62          }
  63          else {
  64              $length=strlen($line);
  65              if ($length==0) {
  66                  // don't do anything
  67              }
  68              else {
  69                  if ($line[$length-1] == "\n") {
  70                      $outarray[] = $line;
  71                  }
  72                  else {
  73                      $outarray[] = $line."\n";
  74                  }
  75              }
  76          }
  77      }
  78      // output modified file to original
  79      if ( is_writable($filename) ) {
  80  
  81          if (! $handle =fopen ($filename ,'w' )) {
  82              echo "Cannot open file ($filename)" ;
  83              exit;
  84          }
  85          foreach ($outarray as $outline) {
  86              fwrite($handle, $outline);
  87          }
  88          fclose($handle);
  89      }
  90      else {
  91          // file not writeable
  92      }
  93  }

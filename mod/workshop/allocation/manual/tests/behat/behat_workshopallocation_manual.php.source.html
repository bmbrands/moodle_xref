  45       * Manually adds a reviewer for workshop participant.
  46       *
  47       * This step should start on manual allocation page.
  48       *
  49       * @When /^I add a reviewer "(?P<reviewer_name_string>(?:[^"]|\\")*)" for workshop participant "(?P<participant_name_string>(?:[^"]|\\")*)"$/
  50       * @param string $reviewername
  51       * @param string $participantname
  52       */
  53      public function i_add_a_reviewer_for_workshop_participant($reviewername, $participantname) {
  54          $participantnameliteral = $this->getSession()->getSelectorsHandler()->xpathLiteral($participantname);
  55          $xpathtd = "//table[contains(concat(' ', normalize-space(@class), ' '), ' allocations ')]/".
  56                  "tbody/tr[./td[contains(concat(' ', normalize-space(@class), ' '), ' peer ')]".
  57                  "[contains(.,$participantnameliteral)]]/".
  58                  "td[contains(concat(' ', normalize-space(@class), ' '), ' reviewedby ')]";
  59          $xpathselect = $xpathtd . "/descendant::select";
  60          try {
  61              $selectnode = $this->find('xpath', $xpathselect);
  62          } catch (Exception $ex) {
  63              $this->find_button(get_string('showallparticipants', 'workshopallocation_manual'))->press();
  64              $selectnode = $this->find('xpath', $xpathselect);
  65          }
  66  
  67          $selectformfield = behat_field_manager::get_form_field($selectnode, $this->getSession());
  68          $selectformfield->set_value($reviewername);
  69  
  70          if (!$this->running_javascript()) {
  71              // Without Javascript we need to press the "Go" button.
  72              $go = $this->getSession()->getSelectorsHandler()->xpathLiteral(get_string('go'));
  73              $this->find('xpath', $xpathtd."/descendant::input[@value=$go]")->click();
  74          } else {
  75              // With Javascript we just wait for the page to reload.
  76              $this->getSession()->wait(self::EXTENDED_TIMEOUT, self::PAGE_READY_JS);
  77          }
  78          // Check the success string to appear.
  79          $allocatedtext = $this->getSession()->getSelectorsHandler()->xpathLiteral(
  80              get_string('allocationadded', 'workshopallocation_manual'));
  81          $this->find('xpath', "//*[contains(.,$allocatedtext)]");
  82      }

  84      /**
  85       * Manually allocates multiple reviewers in workshop.
  86       *
  87       * @When /^I allocate submissions in workshop "(?P<workshop_name_string>(?:[^"]|\\")*)" as:"$/
  88       * @param string $workshopname
  89       * @param TableNode $table should have one column with title 'Reviewer' and another with title 'Participant' (or 'Reviewee')
  90       */
  91      public function i_allocate_submissions_in_workshop_as($workshopname, TableNode $table) {
  92  
  93          $this->find_link($workshopname)->click();
  94          $this->find_link(get_string('allocate', 'workshop'))->click();
  95          $rows = $table->getRows();
  96          $reviewer = $participant = null;
  97          for ($i = 0; $i < count($rows[0]); $i++) {
  98              if (strtolower($rows[0][$i]) === 'reviewer') {
  99                  $reviewer = $i;
 100              } else if (strtolower($rows[0][$i]) === 'reviewee' || strtolower($rows[0][$i]) === 'participant') {
 101                  $participant = $i;
 102              } else {
 103                  throw new ElementTextException('Unrecognised column "'.$rows[0][$i].'"', $this->getSession());
 104              }
 105          }
 106          if ($reviewer === null) {
 107              throw new ElementTextException('Column "Reviewer" could not be located', $this->getSession());
 108          }
 109          if ($participant === null) {
 110              throw new ElementTextException('Neither "Participant" nor "Reviewee" column could be located', $this->getSession());
 111          }
 112          for ($i = 1; $i < count($rows); $i++) {
 113              $this->i_add_a_reviewer_for_workshop_participant($rows[$i][$reviewer], $rows[$i][$participant]);
 114          }
 115      }

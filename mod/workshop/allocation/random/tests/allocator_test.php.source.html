 224       * If there are several elements with the lowest workload, one of them should be chosen randomly
 225       */
 226      public function test_get_element_with_lowest_workload_random() {
 227          // fixture setup
 228          $workload = array(4 => 6, 9 => 2, 10 => 2);
 229          // exercise SUT
 230          $elements = $this->allocator->get_element_with_lowest_workload($workload);
 231          // verify
 232          // in theory, this test can fail even if the function works well. However, the probability of getting
 233          // a row of a hundred same ids in this use case is 1/pow(2, 100)
 234          // also, this just tests that each of the two elements has been chosen at least once. this is not to
 235          // measure the quality or randomness of the algorithm
 236          $counts = array(4 => 0, 9 => 0, 10 => 0);
 237          for ($i = 0; $i < 100; $i++) {
 238              $chosen = $this->allocator->get_element_with_lowest_workload($workload);
 239              if (!in_array($chosen, array(4, 9, 10))) {
 240                  $this->fail('Invalid element ' . var_export($chosen, true) . ' chosen');
 241                  break;
 242              } else {
 243                  $counts[$this->allocator->get_element_with_lowest_workload($workload)]++;
 244              }
 245          }
 246          $this->assertTrue(($counts[9] > 0) && ($counts[10] > 0));
 247      }

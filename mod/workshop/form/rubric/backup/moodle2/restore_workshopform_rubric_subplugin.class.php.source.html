 104      /**

 169      /**

 174       *
 175       * @param mixed $elementname the name of the assessment element
 176       * @param array $data parsed xml data
 177       */
 178      private function process_dimension_grades_structure($elementname, $data) {
 179          global $DB;
 180  
 181          $data = (object)$data;
 182          $oldid = $data->id;
 183  
 184          $data->assessmentid = $this->get_new_parentid($elementname);
 185          $data->strategy = 'rubric';
 186          $data->dimensionid = $this->get_mappingid($this->get_namefor('dimension'), $data->dimensionid);

 188          $DB->insert_record('workshop_grades', $data);

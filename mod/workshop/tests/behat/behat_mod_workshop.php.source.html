 126       * Checks that the user has particular grade set by his reviewing peer in workshop
 127       *
 128       * @Then /^I should see grade "(?P<grade_string>[^"]*)" for workshop participant "(?P<participant_name_string>(?:[^"]|\\")*)" set by peer "(?P<reviewer_name_string>(?:[^"]|\\")*)"$/
 129       * @param string $grade
 130       * @param string $participant
 131       * @param string $reviewer
 132       */
 133      public function i_should_see_grade_for_workshop_participant_set_by_peer($grade, $participant, $reviewer) {
 134          $participantliteral = $this->getSession()->getSelectorsHandler()->xpathLiteral($participant);
 135          $reviewerliteral = $this->getSession()->getSelectorsHandler()->xpathLiteral($reviewer);
 136          $gradeliteral = $this->getSession()->getSelectorsHandler()->xpathLiteral($grade);
 137          $participantselector = "contains(concat(' ', normalize-space(@class), ' '), ' participant ') ".
 138                  "and contains(.,$participantliteral)";
 139          $trxpath = "//table/tbody/tr[td[$participantselector]]";
 140          $tdparticipantxpath = "//table/tbody/tr/td[$participantselector]";
 141          $tdxpath = "/td[contains(concat(' ', normalize-space(@class), ' '), ' receivedgrade ') and contains(.,$reviewerliteral)]/".
 142                  "descendant::span[contains(concat(' ', normalize-space(@class), ' '), ' grade ') and .=$gradeliteral]";
 143  
 144          $tr = $this->find('xpath', $trxpath);
 145          $rowspan = $this->find('xpath', $tdparticipantxpath)->getAttribute('rowspan');
 146  
 147          $xpath = $trxpath.$tdxpath;
 148          if (!empty($rowspan)) {
 149              for ($i = 1; $i < $rowspan; $i++) {
 150                  $xpath .= ' | '.$trxpath."/following-sibling::tr[$i]".$tdxpath;
 151              }
 152          }
 153          $this->find('xpath', $xpath);
 154      }

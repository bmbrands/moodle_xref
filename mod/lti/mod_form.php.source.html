 100          foreach (lti_get_types_for_add_instance() as $id => $type) {
 101              if (!empty($type->toolproxyid)) {
 102                  $toolproxy[] = $type->id;
 103                  $attributes = array( 'globalTool' => 1, 'toolproxy' => 1);
 104                  $enabledcapabilities = explode("\n", $type->enabledcapability);
 105                  if (!in_array('Result.autocreate', $enabledcapabilities)) {
 106                      $attributes['nogrades'] = 1;
 107                  }
 108                  if (!in_array('Person.name.full', $enabledcapabilities) && !in_array('Person.name.family', $enabledcapabilities) &&
 109                      !in_array('Person.name.given', $enabledcapabilities)) {
 110                      $attributes['noname'] = 1;
 111                  }
 112                  if (!in_array('Person.email.primary', $enabledcapabilities)) {
 113                      $attributes['noemail'] = 1;
 114                  }
 115              } else if ($type->course == $COURSE->id) {
 116                  $attributes = array( 'editable' => 1, 'courseTool' => 1, 'domain' => $type->tooldomain );
 117              } else if ($id != 0) {
 118                  $attributes = array( 'globalTool' => 1, 'domain' => $type->tooldomain);
 119              } else {
 120                  $attributes = array();
 121              }
 122  
 123              $tooltypes->addOption($type->name, $id, $attributes);
 124          }

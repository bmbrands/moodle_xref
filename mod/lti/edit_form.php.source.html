  55  class mod_lti_edit_types_form extends moodleform{
  56      public function definition() {
  57          global $CFG;
  58  
  59          $mform    =& $this->_form;
  60  
  61          $istool = $this->_customdata && $this->_customdata->istool;
  62  
  63          // Add basiclti elements.
  64          $mform->addElement('header', 'setup', get_string('tool_settings', 'lti'));
  65  
  66          $mform->addElement('text', 'lti_typename', get_string('typename', 'lti'));
  67          $mform->setType('lti_typename', PARAM_TEXT);
  68          $mform->addHelpButton('lti_typename', 'typename', 'lti');
  69          $mform->addRule('lti_typename', null, 'required', null, 'client');
  70  
  71          $mform->addElement('text', 'lti_toolurl', get_string('toolurl', 'lti'), array('size' => '64'));
  72          $mform->setType('lti_toolurl', PARAM_URL);
  73          $mform->addHelpButton('lti_toolurl', 'toolurl', 'lti');
  74  
  75          $mform->addElement('textarea', 'lti_description', get_string('tooldescription', 'lti'), array('rows' => 4, 'cols' => 60));
  76          $mform->setType('lti_description', PARAM_TEXT);
  77          $mform->addHelpButton('lti_description', 'tooldescription', 'lti');
  78          if (!$istool) {
  79              $mform->addRule('lti_toolurl', null, 'required', null, 'client');
  80          } else {
  81              $mform->disabledIf('lti_toolurl', null);
  82          }
  83  
  84          if (!$istool) {
  85              $mform->addElement('text', 'lti_resourcekey', get_string('resourcekey_admin', 'lti'));
  86              $mform->setType('lti_resourcekey', PARAM_TEXT);
  87              $mform->addHelpButton('lti_resourcekey', 'resourcekey_admin', 'lti');
  88  
  89              $mform->addElement('passwordunmask', 'lti_password', get_string('password_admin', 'lti'));
  90              $mform->setType('lti_password', PARAM_TEXT);
  91              $mform->addHelpButton('lti_password', 'password_admin', 'lti');
  92          }
  93  
  94          if ($istool) {
  95              $mform->addElement('textarea', 'lti_parameters', get_string('parameter', 'lti'), array('rows' => 4, 'cols' => 60));
  96              $mform->setType('lti_parameters', PARAM_TEXT);
  97              $mform->addHelpButton('lti_parameters', 'parameter', 'lti');
  98              $mform->disabledIf('lti_parameters', null);
  99          }
 100  
 101          $mform->addElement('textarea', 'lti_customparameters', get_string('custom', 'lti'), array('rows' => 4, 'cols' => 60));
 102          $mform->setType('lti_customparameters', PARAM_TEXT);
 103          $mform->addHelpButton('lti_customparameters', 'custom', 'lti');
 104  
 105          if (!empty($this->_customdata->isadmin)) {
 106              $options = array(
 107                  LTI_COURSEVISIBLE_NO => get_string('show_in_course_no', 'lti'),
 108                  LTI_COURSEVISIBLE_PRECONFIGURED => get_string('show_in_course_preconfigured', 'lti'),
 109                  LTI_COURSEVISIBLE_ACTIVITYCHOOSER => get_string('show_in_course_activity_chooser', 'lti'),
 110              );
 111              if ($istool) {
 112                  // LTI2 tools can not be matched by URL, they have to be either in preconfigured tools or in activity chooser.
 113                  unset($options[LTI_COURSEVISIBLE_NO]);
 114                  $stringname = 'show_in_course_lti2';
 115              } else {
 116                  $stringname = 'show_in_course_lti1';
 117              }
 118              $mform->addElement('select', 'lti_coursevisible', get_string($stringname, 'lti'), $options);
 119              $mform->addHelpButton('lti_coursevisible', $stringname, 'lti');
 120              $mform->setDefault('lti_coursevisible', '1');
 121          } else {
 122              $mform->addElement('hidden', 'lti_coursevisible', LTI_COURSEVISIBLE_PRECONFIGURED);
 123          }
 124          $mform->setType('lti_coursevisible', PARAM_INT);
 125  
 126          $mform->addElement('hidden', 'typeid');
 127          $mform->setType('typeid', PARAM_INT);
 128  
 129          $launchoptions = array();
 130          $launchoptions[LTI_LAUNCH_CONTAINER_EMBED] = get_string('embed', 'lti');
 131          $launchoptions[LTI_LAUNCH_CONTAINER_EMBED_NO_BLOCKS] = get_string('embed_no_blocks', 'lti');
 132          $launchoptions[LTI_LAUNCH_CONTAINER_REPLACE_MOODLE_WINDOW] = get_string('existing_window', 'lti');
 133          $launchoptions[LTI_LAUNCH_CONTAINER_WINDOW] = get_string('new_window', 'lti');
 134  
 135          $mform->addElement('select', 'lti_launchcontainer', get_string('default_launch_container', 'lti'), $launchoptions);
 136          $mform->setDefault('lti_launchcontainer', LTI_LAUNCH_CONTAINER_EMBED_NO_BLOCKS);
 137          $mform->addHelpButton('lti_launchcontainer', 'default_launch_container', 'lti');
 138          $mform->setType('lti_launchcontainer', PARAM_INT);
 139  
 140          $mform->addElement('hidden', 'oldicon');
 141          $mform->setType('oldicon', PARAM_URL);
 142  
 143          $mform->addElement('text', 'lti_icon', get_string('icon_url', 'lti'), array('size' => '64'));
 144          $mform->setType('lti_icon', PARAM_URL);
 145          $mform->setAdvanced('lti_icon');
 146          $mform->addHelpButton('lti_icon', 'icon_url', 'lti');
 147  
 148          $mform->addElement('text', 'lti_secureicon', get_string('secure_icon_url', 'lti'), array('size' => '64'));
 149          $mform->setType('lti_secureicon', PARAM_URL);
 150          $mform->setAdvanced('lti_secureicon');
 151          $mform->addHelpButton('lti_secureicon', 'secure_icon_url', 'lti');
 152  
 153          if (!$istool) {
 154              // Add privacy preferences fieldset where users choose whether to send their data.
 155              $mform->addElement('header', 'privacy', get_string('privacy', 'lti'));
 156  
 157              $options = array();
 158              $options[0] = get_string('never', 'lti');
 159              $options[1] = get_string('always', 'lti');
 160              $options[2] = get_string('delegate', 'lti');
 161  
 162              $mform->addElement('select', 'lti_sendname', get_string('share_name_admin', 'lti'), $options);
 163              $mform->setType('lti_sendname', PARAM_INT);
 164              $mform->setDefault('lti_sendname', '2');
 165              $mform->addHelpButton('lti_sendname', 'share_name_admin', 'lti');
 166  
 167              $mform->addElement('select', 'lti_sendemailaddr', get_string('share_email_admin', 'lti'), $options);
 168              $mform->setType('lti_sendemailaddr', PARAM_INT);
 169              $mform->setDefault('lti_sendemailaddr', '2');
 170              $mform->addHelpButton('lti_sendemailaddr', 'share_email_admin', 'lti');
 171  
 172              // LTI Extensions.
 173  
 174              // Add grading preferences fieldset where the tool is allowed to return grades.
 175              $mform->addElement('select', 'lti_acceptgrades', get_string('accept_grades_admin', 'lti'), $options);
 176              $mform->setType('lti_acceptgrades', PARAM_INT);
 177              $mform->setDefault('lti_acceptgrades', '2');
 178              $mform->addHelpButton('lti_acceptgrades', 'accept_grades_admin', 'lti');
 179  
 180              $mform->addElement('checkbox', 'lti_forcessl', '&nbsp;', ' ' . get_string('force_ssl', 'lti'), $options);
 181              $mform->setType('lti_forcessl', PARAM_BOOL);
 182              if (!empty($CFG->mod_lti_forcessl)) {
 183                  $mform->setDefault('lti_forcessl', '1');
 184                  $mform->freeze('lti_forcessl');
 185              } else {
 186                  $mform->setDefault('lti_forcessl', '0');
 187              }
 188              $mform->addHelpButton('lti_forcessl', 'force_ssl', 'lti');
 189  
 190              if (!empty($this->_customdata->isadmin)) {
 191                  // Add setup parameters fieldset.
 192                  $mform->addElement('header', 'setupoptions', get_string('miscellaneous', 'lti'));
 193  
 194                  // Adding option to change id that is placed in context_id.
 195                  $idoptions = array();
 196                  $idoptions[0] = get_string('id', 'lti');
 197                  $idoptions[1] = get_string('courseid', 'lti');
 198  
 199                  $mform->addElement('text', 'lti_organizationid', get_string('organizationid', 'lti'));
 200                  $mform->setType('lti_organizationid', PARAM_TEXT);
 201                  $mform->addHelpButton('lti_organizationid', 'organizationid', 'lti');
 202  
 203                  $mform->addElement('text', 'lti_organizationurl', get_string('organizationurl', 'lti'));
 204                  $mform->setType('lti_organizationurl', PARAM_URL);
 205                  $mform->addHelpButton('lti_organizationurl', 'organizationurl', 'lti');
 206              }
 207          }
 208  
 209          /* Suppress this for now - Chuck
 210           * mform->addElement('text', 'lti_organizationdescr', get_string('organizationdescr', 'lti'))
 211           * mform->setType('lti_organizationdescr', PARAM_TEXT)
 212           * mform->addHelpButton('lti_organizationdescr', 'organizationdescr', 'lti')
 213           */
 214  
 215          /*

 217          //$mform->addElement('hidden', 'lti_fix');
 218          */
 219  
 220          $tab = optional_param('tab', '', PARAM_ALPHAEXT);
 221          $mform->addElement('hidden', 'tab', $tab);
 222          $mform->setType('tab', PARAM_ALPHAEXT);
 223  
 224          $courseid = optional_param('course', 1, PARAM_INT);
 225          $mform->addElement('hidden', 'course', $courseid);
 226          $mform->setType('course', PARAM_INT);
 227  
 228          // Add standard buttons, common to all modules.
 229          $this->add_action_buttons();
 230  
 231      }

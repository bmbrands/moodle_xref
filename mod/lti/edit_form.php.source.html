  55  class mod_lti_edit_types_form extends moodleform{
  56      public function definition() {
  57          global $CFG;
  58  
  59          $mform    =& $this->_form;
  60  
  61          $istool = $this->_customdata && $this->_customdata->istool;
  62  
  63          // Add basiclti elements.
  64          $mform->addElement('header', 'setup', get_string('tool_settings', 'lti'));
  65  
  66          $mform->addElement('text', 'lti_typename', get_string('typename', 'lti'));
  67          $mform->setType('lti_typename', PARAM_TEXT);
  68          $mform->addHelpButton('lti_typename', 'typename', 'lti');
  69          $mform->addRule('lti_typename', null, 'required', null, 'client');
  70  
  71          $mform->addElement('text', 'lti_toolurl', get_string('toolurl', 'lti'), array('size' => '64'));
  72          $mform->setType('lti_toolurl', PARAM_TEXT);
  73          $mform->addHelpButton('lti_toolurl', 'toolurl', 'lti');
  74          if (!$istool) {
  75              $mform->addRule('lti_toolurl', null, 'required', null, 'client');
  76          } else {
  77              $mform->disabledIf('lti_toolurl', null);
  78          }
  79  
  80          if (!$istool) {
  81              $mform->addElement('text', 'lti_resourcekey', get_string('resourcekey_admin', 'lti'));
  82              $mform->setType('lti_resourcekey', PARAM_TEXT);
  83              $mform->addHelpButton('lti_resourcekey', 'resourcekey_admin', 'lti');
  84  
  85              $mform->addElement('passwordunmask', 'lti_password', get_string('password_admin', 'lti'));
  86              $mform->setType('lti_password', PARAM_TEXT);
  87              $mform->addHelpButton('lti_password', 'password_admin', 'lti');
  88          }
  89  
  90          if ($istool) {
  91              $mform->addElement('textarea', 'lti_parameters', get_string('parameter', 'lti'), array('rows' => 4, 'cols' => 60));
  92              $mform->setType('lti_parameters', PARAM_TEXT);
  93              $mform->addHelpButton('lti_parameters', 'parameter', 'lti');
  94              $mform->disabledIf('lti_parameters', null);
  95          }
  96  
  97          $mform->addElement('textarea', 'lti_customparameters', get_string('custom', 'lti'), array('rows' => 4, 'cols' => 60));
  98          $mform->setType('lti_customparameters', PARAM_TEXT);
  99          $mform->addHelpButton('lti_customparameters', 'custom', 'lti');
 100  
 101          if (!$istool && !empty($this->_customdata->isadmin)) {
 102              $mform->addElement('checkbox', 'lti_coursevisible', '&nbsp;', ' ' . get_string('show_in_course', 'lti'));
 103              $mform->addHelpButton('lti_coursevisible', 'show_in_course', 'lti');
 104          } else {
 105              $mform->addElement('hidden', 'lti_coursevisible', '1');
 106          }
 107          $mform->setType('lti_coursevisible', PARAM_BOOL);
 108  
 109          $mform->addElement('hidden', 'typeid');
 110          $mform->setType('typeid', PARAM_INT);
 111  
 112          $launchoptions = array();
 113          $launchoptions[LTI_LAUNCH_CONTAINER_EMBED] = get_string('embed', 'lti');
 114          $launchoptions[LTI_LAUNCH_CONTAINER_EMBED_NO_BLOCKS] = get_string('embed_no_blocks', 'lti');
 115          $launchoptions[LTI_LAUNCH_CONTAINER_REPLACE_MOODLE_WINDOW] = get_string('existing_window', 'lti');
 116          $launchoptions[LTI_LAUNCH_CONTAINER_WINDOW] = get_string('new_window', 'lti');
 117  
 118          $mform->addElement('select', 'lti_launchcontainer', get_string('default_launch_container', 'lti'), $launchoptions);
 119          $mform->setDefault('lti_launchcontainer', LTI_LAUNCH_CONTAINER_EMBED_NO_BLOCKS);
 120          $mform->addHelpButton('lti_launchcontainer', 'default_launch_container', 'lti');
 121          $mform->setType('lti_launchcontainer', PARAM_INT);
 122  
 123          if (!$istool) {
 124              // Add privacy preferences fieldset where users choose whether to send their data.
 125              $mform->addElement('header', 'privacy', get_string('privacy', 'lti'));
 126  
 127              $options = array();
 128              $options[0] = get_string('never', 'lti');
 129              $options[1] = get_string('always', 'lti');
 130              $options[2] = get_string('delegate', 'lti');
 131  
 132              $mform->addElement('select', 'lti_sendname', get_string('share_name_admin', 'lti'), $options);
 133              $mform->setType('lti_sendname', PARAM_INT);
 134              $mform->setDefault('lti_sendname', '2');
 135              $mform->addHelpButton('lti_sendname', 'share_name_admin', 'lti');
 136  
 137              $mform->addElement('select', 'lti_sendemailaddr', get_string('share_email_admin', 'lti'), $options);
 138              $mform->setType('lti_sendemailaddr', PARAM_INT);
 139              $mform->setDefault('lti_sendemailaddr', '2');
 140              $mform->addHelpButton('lti_sendemailaddr', 'share_email_admin', 'lti');
 141  
 142              // LTI Extensions.
 143  
 144              // Add grading preferences fieldset where the tool is allowed to return grades.
 145              $mform->addElement('select', 'lti_acceptgrades', get_string('accept_grades_admin', 'lti'), $options);
 146              $mform->setType('lti_acceptgrades', PARAM_INT);
 147              $mform->setDefault('lti_acceptgrades', '2');
 148              $mform->addHelpButton('lti_acceptgrades', 'accept_grades_admin', 'lti');
 149  
 150              $mform->addElement('checkbox', 'lti_forcessl', '&nbsp;', ' ' . get_string('force_ssl', 'lti'), $options);
 151              $mform->setType('lti_forcessl', PARAM_BOOL);
 152              if (!empty($CFG->mod_lti_forcessl)) {
 153                  $mform->setDefault('lti_forcessl', '1');
 154                  $mform->freeze('lti_forcessl');
 155              } else {
 156                  $mform->setDefault('lti_forcessl', '0');
 157              }
 158              $mform->addHelpButton('lti_forcessl', 'force_ssl', 'lti');
 159  
 160              if (!empty($this->_customdata->isadmin)) {
 161                  // Add setup parameters fieldset.
 162                  $mform->addElement('header', 'setupoptions', get_string('miscellaneous', 'lti'));
 163  
 164                  // Adding option to change id that is placed in context_id.
 165                  $idoptions = array();
 166                  $idoptions[0] = get_string('id', 'lti');
 167                  $idoptions[1] = get_string('courseid', 'lti');
 168  
 169                  $mform->addElement('text', 'lti_organizationid', get_string('organizationid', 'lti'));
 170                  $mform->setType('lti_organizationid', PARAM_TEXT);
 171                  $mform->addHelpButton('lti_organizationid', 'organizationid', 'lti');
 172  
 173                  $mform->addElement('text', 'lti_organizationurl', get_string('organizationurl', 'lti'));
 174                  $mform->setType('lti_organizationurl', PARAM_TEXT);
 175                  $mform->addHelpButton('lti_organizationurl', 'organizationurl', 'lti');
 176              }
 177          }
 178  
 179          /* Suppress this for now - Chuck
 180           * mform->addElement('text', 'lti_organizationdescr', get_string('organizationdescr', 'lti'))
 181           * mform->setType('lti_organizationdescr', PARAM_TEXT)
 182           * mform->addHelpButton('lti_organizationdescr', 'organizationdescr', 'lti')
 183           */
 184

 188          */

 190          $tab = optional_param('tab', '', PARAM_ALPHAEXT);
 191          $mform->addElement('hidden', 'tab', $tab);
 192          $mform->setType('tab', PARAM_ALPHAEXT);
 193  
 194          $courseid = optional_param('course', 1, PARAM_INT);
 195          $mform->addElement('hidden', 'course', $courseid);
 196          $mform->setType('course', PARAM_INT);
 197  
 198          // Add standard buttons, common to all modules.
 199          $this->add_action_buttons();

 201      }

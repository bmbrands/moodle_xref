  37  class mod_assign_grading_batch_operations_form extends moodleform {
  38      /**
  39       * Define this form - called by the parent constructor.
  40       */
  41      public function definition() {
  42          $mform = $this->_form;
  43          $instance = $this->_customdata;
  44  
  45          // Visible elements.
  46          $options = array();
  47          $options['lock'] = get_string('locksubmissions', 'assign');
  48          $options['unlock'] = get_string('unlocksubmissions', 'assign');
  49          $options['downloadselected'] = get_string('downloadselectedsubmissions', 'assign');
  50          if ($instance['submissiondrafts']) {
  51              $options['reverttodraft'] = get_string('reverttodraft', 'assign');
  52          }
  53          if ($instance['duedate'] && has_capability('mod/assign:grantextension', $instance['context'])) {
  54              $options['grantextension'] = get_string('grantextension', 'assign');
  55          }
  56          if ($instance['attemptreopenmethod'] == ASSIGN_ATTEMPT_REOPEN_METHOD_MANUAL) {
  57              $options['addattempt'] = get_string('addattempt', 'assign');
  58          }
  59  
  60          foreach ($instance['feedbackplugins'] as $plugin) {
  61              if ($plugin->is_visible() && $plugin->is_enabled()) {
  62                  foreach ($plugin->get_grading_batch_operations() as $action => $description) {
  63                      $operationkey = 'plugingradingbatchoperation_' . $plugin->get_type() . '_' . $action;
  64                      $options[$operationkey] = $description;
  65                  }
  66              }
  67          }
  68          if ($instance['markingworkflow']) {
  69              $options['setmarkingworkflowstate'] = get_string('setmarkingworkflowstate', 'assign');
  70          }
  71          if ($instance['markingallocation']) {
  72              $options['setmarkingallocation'] = get_string('setmarkingallocation', 'assign');
  73          }
  74  
  75          $mform->addElement('hidden', 'action', 'gradingbatchoperation');
  76          $mform->setType('action', PARAM_ALPHA);
  77          $mform->addElement('hidden', 'id', $instance['cm']);
  78          $mform->setType('id', PARAM_INT);
  79          $mform->addElement('hidden', 'selectedusers', '', array('class'=>'selectedusers'));
  80          $mform->setType('selectedusers', PARAM_SEQUENCE);
  81          $mform->addElement('hidden', 'returnaction', 'grading');
  82          $mform->setType('returnaction', PARAM_ALPHA);
  83  
  84          $objs = array();
  85          $objs[] =& $mform->createElement('select', 'operation', get_string('chooseoperation', 'assign'), $options);
  86          $objs[] =& $mform->createElement('submit', 'submit', get_string('go'));
  87          $batchdescription = get_string('batchoperationsdescription', 'assign');
  88          $mform->addElement('group', 'actionsgrp', $batchdescription, $objs, ' ', false);
  89      }
  90  }

 223      this.bound = function(points) {
 224          var minx = 0,

 227              maxy = 0,
 228              i = 0,
 229              point;
 230  
 231          for (i = 0; i < points.length; i++) {
 232              point = points[i];
 233              if (point.x < minx || i === 0) {
 234                  minx = point.x;
 235              }
 236              if (point.x > maxx || i === 0) {
 237                  maxx = point.x;
 238              }
 239              if (point.y < miny || i === 0) {
 240                  miny = point.y;
 241              }
 242              if (point.y > maxy || i === 0) {
 243                  maxy = point.y;
 244              }
 245          }
 246          this.x = minx;
 247          this.y = miny;
 248          this.width = maxx - minx;
 249          this.height = maxy - miny;
 250          // Allow chaining.
 251          return this;
 252      };

 481              x = this.nodes[i].getData('x');

 484                  this.nodes[i].setX(parseInt(x, 10) - scrollx);
 485                  this.nodes[i].setY(parseInt(y, 10) - scrolly);

 487          }

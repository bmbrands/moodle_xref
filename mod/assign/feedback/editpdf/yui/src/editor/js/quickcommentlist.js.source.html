  22  /**
  23   * Class representing a users list of quick comments.
  24   *
  25   * @namespace M.assignfeedback_editpdf
  26   * @class quickcommentlist
  27   */
  28  var QUICKCOMMENTLIST = function(editor) {
  29  
  30      /**
  31       * Reference to M.assignfeedback_editpdf.editor.
  32       * @property editor
  33       * @type M.assignfeedback_editpdf.editor
  34       * @public
  35       */
  36      this.editor = editor;
  37  
  38      /**
  39       * Array of Comments
  40       * @property shapes
  41       * @type M.assignfeedback_editpdf.quickcomment[]
  42       * @public
  43       */
  44      this.comments = [];
  45  
  46      /**
  47       * Add a comment to the users quicklist.
  48       *
  49       * @protected
  50       * @method add
  51       */
  52      this.add = function(comment) {
  53          var ajaxurl = AJAXBASE,
  54              config;
  55  
  56          // Do not save empty comments.
  57          if (comment.rawtext === '') {
  58              return;
  59          }
  60  
  61          config = {
  62              method: 'post',
  63              context: this,
  64              sync: false,
  65              data : {
  66                  'sesskey' : M.cfg.sesskey,
  67                  'action' : 'addtoquicklist',
  68                  'userid' : this.editor.get('userid'),
  69                  'commenttext' : comment.rawtext,
  70                  'width' : comment.width,
  71                  'colour' : comment.colour,
  72                  'attemptnumber' : this.editor.get('attemptnumber'),
  73                  'assignmentid' : this.editor.get('assignmentid')
  74              },
  75              on: {
  76                  success: function(tid, response) {
  77                      var jsondata, quickcomment;
  78                      try {
  79                          jsondata = Y.JSON.parse(response.responseText);
  80                          if (jsondata.error) {
  81                              return new M.core.ajaxException(jsondata);
  82                          } else {
  83                              quickcomment = new M.assignfeedback_editpdf.quickcomment(jsondata.id,
  84                                                                                       jsondata.rawtext,
  85                                                                                       jsondata.width,
  86                                                                                       jsondata.colour);
  87                              this.comments.push(quickcomment);
  88                          }
  89                      } catch (e) {
  90                          return new M.core.exception(e);
  91                      }
  92                  },
  93                  failure: function(tid, response) {
  94                      return M.core.exception(response.responseText);
  95                  }
  96              }
  97          };
  98  
  99          Y.io(ajaxurl, config);
 100      };
 101  
 102      /**
 103       * Remove a comment from the users quicklist.
 104       *
 105       * @public
 106       * @method remove
 107       */
 108      this.remove = function(comment) {
 109          var ajaxurl = AJAXBASE,
 110              config;
 111  
 112          // Should not happen.
 113          if (!comment) {
 114              return;
 115          }
 116  
 117          config = {
 118              method: 'post',
 119              context: this,
 120              sync: false,
 121              data : {
 122                  'sesskey' : M.cfg.sesskey,
 123                  'action' : 'removefromquicklist',
 124                  'userid' : this.editor.get('userid'),
 125                  'commentid' : comment.id,
 126                  'attemptnumber' : this.editor.get('attemptnumber'),
 127                  'assignmentid' : this.editor.get('assignmentid')
 128              },
 129              on: {
 130                  success: function() {
 131                      var i;
 132  
 133                      // Find and remove the comment from the quicklist.
 134                      i = this.comments.indexOf(comment);
 135                      if (i >= 0) {
 136                          this.comments.splice(i, 1);
 137                      }
 138                  },
 139                  failure: function(tid, response) {
 140                      return M.core.exception(response.responseText);
 141                  }
 142              }
 143          };
 144  
 145          Y.io(ajaxurl, config);
 146      };
 147  
 148      /**
 149       * Load the users quick comments list.
 150       *
 151       * @protected
 152       * @method load_quicklist
 153       */
 154      this.load = function() {
 155          var ajaxurl = AJAXBASE,
 156              config;
 157  
 158          config = {
 159              method: 'get',
 160              context: this,
 161              sync: false,
 162              data : {
 163                  'sesskey' : M.cfg.sesskey,
 164                  'action' : 'loadquicklist',
 165                  'userid' : this.editor.get('userid'),
 166                  'attemptnumber' : this.editor.get('attemptnumber'),
 167                  'assignmentid' : this.editor.get('assignmentid')
 168              },
 169              on: {
 170                  success: function(tid, response) {
 171                      var jsondata;
 172                      try {
 173                          jsondata = Y.JSON.parse(response.responseText);
 174                          if (jsondata.error) {
 175                              return new M.core.ajaxException(jsondata);
 176                          } else {
 177                              Y.each(jsondata, function(comment) {
 178                                  var quickcomment = new M.assignfeedback_editpdf.quickcomment(comment.id,
 179                                                                                               comment.rawtext,
 180                                                                                               comment.width,
 181                                                                                               comment.colour);
 182                                  this.comments.push(quickcomment);
 183                              }, this);
 184                          }
 185                      } catch (e) {
 186                          return new M.core.exception(e);
 187                      }
 188                  },
 189                  failure: function(tid, response) {
 190                      return M.core.exception(response.responseText);
 191                  }
 192              }
 193          };
 194  
 195          Y.io(ajaxurl, config);
 196      };
 197  };
 198  
 199  M.assignfeedback_editpdf = M.assignfeedback_editpdf || {};
 200  M.assignfeedback_editpdf.quickcommentlist = QUICKCOMMENTLIST;

  22  /**
  23   * Class representing a stamp.
  24   *
  25   * @namespace M.assignfeedback_editpdf
  26   * @class annotationstamp
  27   * @extends M.assignfeedback_editpdf.annotation
  28   */
  29  var ANNOTATIONSTAMP = function(config) {
  30      ANNOTATIONSTAMP.superclass.constructor.apply(this, [config]);
  31  };
  32  
  33  ANNOTATIONSTAMP.NAME = "annotationstamp";
  34  ANNOTATIONSTAMP.ATTRS = {};
  35  
  36  Y.extend(ANNOTATIONSTAMP, M.assignfeedback_editpdf.annotation, {
  37      /**
  38       * Draw a stamp annotation
  39       * @protected
  40       * @method draw
  41       * @return M.assignfeedback_editpdf.drawable
  42       */
  43      draw : function() {
  44          var drawable = new M.assignfeedback_editpdf.drawable(this.editor),
  45              drawingregion = Y.one(SELECTOR.DRAWINGREGION),
  46              node,
  47              position;
  48  
  49          position = this.editor.get_window_coordinates(new M.assignfeedback_editpdf.point(this.x, this.y));
  50          node = Y.Node.create('<div/>');
  51          node.setStyles({
  52              'position': 'absolute',
  53              'display': 'inline-block',
  54              'backgroundImage': 'url(' + this.editor.get_stamp_image_url(this.path) + ')',
  55              'width': (this.endx - this.x),
  56              'height': (this.endy - this.y),
  57              'backgroundSize': '100% 100%',
  58              'zIndex': 50
  59          });
  60  
  61          drawingregion.append(node);
  62          node.setX(position.x);
  63          node.setY(position.y);
  64          drawable.store_position(node, position.x, position.y);
  65  
  66          // Pass throught the event handlers on the div.
  67          node.on('gesturemovestart', this.editor.edit_start, null, this.editor);
  68          node.on('gesturemove', this.editor.edit_move, null, this.editor);
  69          node.on('gesturemoveend', this.editor.edit_end, null, this.editor);
  70  
  71          drawable.nodes.push(node);
  72  
  73          this.drawable = drawable;
  74          return ANNOTATIONSTAMP.superclass.draw.apply(this);
  75      },
  76  
  77      /**
  78       * Draw the in progress edit.
  79       *
  80       * @public
  81       * @method draw_current_edit
  82       * @param M.assignfeedback_editpdf.edit edit
  83       */
  84      draw_current_edit : function(edit) {
  85          var bounds = new M.assignfeedback_editpdf.rect(),
  86              drawable = new M.assignfeedback_editpdf.drawable(this.editor),
  87              drawingregion = Y.one(SELECTOR.DRAWINGREGION),
  88              node,
  89              position;
  90  
  91          bounds.bound([edit.start, edit.end]);
  92          position = this.editor.get_window_coordinates(new M.assignfeedback_editpdf.point(bounds.x, bounds.y));
  93  
  94          node = Y.Node.create('<div/>');
  95          node.setStyles({
  96              'position': 'absolute',
  97              'display': 'inline-block',
  98              'backgroundImage': 'url(' + this.editor.get_stamp_image_url(edit.stamp) + ')',
  99              'width': bounds.width,
 100              'height': bounds.height,
 101              'backgroundSize': '100% 100%',
 102              'zIndex': 50
 103          });
 104  
 105          drawingregion.append(node);
 106          node.setX(position.x);
 107          node.setY(position.y);
 108          drawable.store_position(node, position.x, position.y);
 109  
 110          drawable.nodes.push(node);
 111  
 112          return drawable;
 113      },
 114  
 115      /**
 116       * Promote the current edit to a real annotation.
 117       *
 118       * @public
 119       * @method init_from_edit
 120       * @param M.assignfeedback_editpdf.edit edit
 121       * @return bool if width/height is more than min. required.
 122       */
 123      init_from_edit : function(edit) {
 124          var bounds = new M.assignfeedback_editpdf.rect();
 125          bounds.bound([edit.start, edit.end]);
 126  
 127          if (bounds.width < 40) {
 128              bounds.width = 40;
 129          }
 130          if (bounds.height < 40) {
 131              bounds.height = 40;
 132          }
 133          this.gradeid = this.editor.get('gradeid');
 134          this.pageno = this.editor.currentpage;
 135          this.x = bounds.x;
 136          this.y = bounds.y;
 137          this.endx = bounds.x + bounds.width;
 138          this.endy = bounds.y + bounds.height;
 139          this.colour = edit.annotationcolour;
 140          this.path = edit.stamp;
 141  
 142          // Min width and height is always more than 40px.
 143          return true;
 144      },
 145  
 146      /**
 147       * Move an annotation to a new location.
 148       * @public
 149       * @param int newx
 150       * @param int newy
 151       * @method move_annotation
 152       */
 153      move : function(newx, newy) {
 154          var diffx = newx - this.x,
 155              diffy = newy - this.y;
 156  
 157          this.x += diffx;
 158          this.y += diffy;
 159          this.endx += diffx;
 160          this.endy += diffy;
 161  
 162          if (this.drawable) {
 163              this.drawable.erase();
 164          }
 165          this.editor.drawables.push(this.draw());
 166      }
 167  
 168  });

  96      public static function list_compatible_submission_files_for_attempt($assignment, $userid, $attemptnumber) {
  97          global $USER, $DB;
  98  
  99          $assignment = self::get_assignment_from_param($assignment);
 100  
 101          // Capability checks.
 102          if (!$assignment->can_view_submission($userid)) {
 103              \print_error('nopermission');
 104          }
 105  
 106          $files = array();
 107  
 108          if ($assignment->get_instance()->teamsubmission) {
 109              $submission = $assignment->get_group_submission($userid, 0, false);
 110          } else {
 111              $submission = $assignment->get_user_submission($userid, false);
 112          }
 113          $user = $DB->get_record('user', array('id' => $userid));
 114  
 115          // User has not submitted anything yet.
 116          if (!$submission) {
 117              return $files;
 118          }
 119          // Ask each plugin for it's list of files.
 120          foreach ($assignment->get_submission_plugins() as $plugin) {
 121              if ($plugin->is_enabled() && $plugin->is_visible()) {
 122                  $pluginfiles = $plugin->get_files($submission, $user);
 123                  foreach ($pluginfiles as $filename => $file) {
 124                      if (($file instanceof \stored_file) && ($file->get_mimetype() === 'application/pdf')) {
 125                          $files[$filename] = $file;
 126                      }
 127                  }
 128              }
 129          }
 130          return $files;
 131      }

 174      /**
 175       * This function will take all of the compatible files for a submission
 176       * and combine them into one PDF.
 177       * @param int|\assign $assignment
 178       * @param int $userid
 179       * @param int $attemptnumber (-1 means latest attempt)
 180       * @return stored_file
 181       */
 182      public static function generate_combined_pdf_for_attempt($assignment, $userid, $attemptnumber) {
 183          global $CFG;
 184  
 185          require_once($CFG->libdir . '/pdflib.php');
 186  
 187          $assignment = self::get_assignment_from_param($assignment);
 188  
 189          if (!$assignment->can_view_submission($userid)) {
 190              \print_error('nopermission');
 191          }
 192  
 193          $files = self::list_compatible_submission_files_for_attempt($assignment, $userid, $attemptnumber);
 194  
 195          $pdf = new pdf();
 196          if ($files) {
 197              // Create a mega joined PDF.
 198              $compatiblepdfs = array();
 199              foreach ($files as $file) {
 200                  $compatiblepdf = pdf::ensure_pdf_compatible($file);
 201                  if ($compatiblepdf) {
 202                      array_push($compatiblepdfs, $compatiblepdf);
 203                  }
 204              }
 205  
 206              $tmpdir = \make_temp_directory('assignfeedback_editpdf/combined/' . self::hash($assignment, $userid, $attemptnumber));
 207              $tmpfile = $tmpdir . '/' . self::COMBINED_PDF_FILENAME;
 208  
 209              @unlink($tmpfile);
 210              try {
 211                  $pagecount = $pdf->combine_pdfs($compatiblepdfs, $tmpfile);
 212              } catch (\Exception $e) {
 213                  debugging('TCPDF could not process the pdf files:' . $e->getMessage(), DEBUG_DEVELOPER);
 214                  // TCPDF does not recover from errors so we need to re-initialise the class.
 215                  $pagecount = 0;
 216              }
 217              if ($pagecount == 0) {
 218                  // We at least want a single blank page.
 219                  debugging('TCPDF did not produce a valid pdf:' . $tmpfile . '. Replacing with a blank pdf.', DEBUG_DEVELOPER);
 220                  @unlink($tmpfile);
 221                  $files = false;
 222              }
 223          }
 224          $pdf->Close(); // No real need to close this pdf, because it has been saved by combine_pdfs(), but for clarity.
 225  
 226          $grade = $assignment->get_user_grade($userid, true, $attemptnumber);
 227          $record = new \stdClass();
 228  
 229          $record->contextid = $assignment->get_context()->id;
 230          $record->component = 'assignfeedback_editpdf';
 231          $record->filearea = self::COMBINED_PDF_FILEAREA;
 232          $record->itemid = $grade->id;
 233          $record->filepath = '/';
 234          $record->filename = self::COMBINED_PDF_FILENAME;
 235          $fs = \get_file_storage();
 236  
 237          $fs->delete_area_files($record->contextid, $record->component, $record->filearea, $record->itemid);
 238  
 239          // Detect corrupt generated pdfs and replace with a blank one.
 240          if ($files) {
 241              $verifypdf = new pdf();
 242              $pagecount = $verifypdf->load_pdf($tmpfile);
 243              if ($pagecount <= 0) {
 244                  $files = false;
 245              }
 246              $verifypdf->Close(); // PDF loaded and never saved/outputted needs to be closed.
 247          }
 248  
 249          if (!$files) {
 250              // This was a blank pdf.
 251              $blankpdf = new pdf();
 252              $content = $blankpdf->Output(self::COMBINED_PDF_FILENAME, 'S');
 253              $file = $fs->create_file_from_string($record, $content);
 254              $blankpdf->Close(); // No real need to close this pdf, because it has been outputted, but for clarity.
 255          } else {
 256              // This was a combined pdf.
 257              $file = $fs->create_file_from_pathname($record, $tmpfile);
 258              @unlink($tmpfile);
 259  
 260              // Test the generated file for correctness.
 261              $compatiblepdf = pdf::ensure_pdf_compatible($file);
 262          }
 263  
 264          return $file;
 265      }

 275      public static function page_number_for_attempt($assignment, $userid, $attemptnumber, $readonly = false) {
 276          global $CFG;
 277  
 278          require_once($CFG->libdir . '/pdflib.php');
 279  
 280          $assignment = self::get_assignment_from_param($assignment);
 281  
 282          if (!$assignment->can_view_submission($userid)) {
 283              \print_error('nopermission');
 284          }
 285  
 286          // When in readonly we can return the number of images in the DB because they should already exist,
 287          // if for some reason they do not, then we proceed as for the normal version.
 288          if ($readonly) {
 289              $grade = $assignment->get_user_grade($userid, true, $attemptnumber);
 290              $fs = get_file_storage();
 291              $files = $fs->get_directory_files($assignment->get_context()->id, 'assignfeedback_editpdf',
 292                  self::PAGE_IMAGE_READONLY_FILEAREA, $grade->id, '/');
 293              $pagecount = count($files);
 294              if ($pagecount > 0) {
 295                  return $pagecount;
 296              }
 297          }
 298  
 299          // Get a combined pdf file from all submitted pdf files.
 300          $file = self::get_combined_pdf_for_attempt($assignment, $userid, $attemptnumber);
 301          if (!$file) {
 302              \print_error('Could not generate combined pdf.');
 303          }
 304  
 305          // Store the combined pdf file somewhere to be opened by tcpdf.
 306          $tmpdir = \make_temp_directory('assignfeedback_editpdf/pagetotal/'
 307              . self::hash($assignment, $userid, $attemptnumber));
 308          $combined = $tmpdir . '/' . self::COMBINED_PDF_FILENAME;
 309          $file->copy_content_to($combined); // Copy the file.
 310  
 311          // Get the total number of pages.
 312          $pdf = new pdf();
 313          $pagecount = $pdf->set_pdf($combined);
 314          $pdf->Close(); // PDF loaded and never saved/outputted needs to be closed.
 315  
 316          // Delete temporary folders and files.
 317          @unlink($combined);
 318          @rmdir($tmpdir);
 319  
 320          return $pagecount;
 321      }

 384      /**
 385       * This function returns a list of the page images from a pdf.
 386       *
 387       * The readonly version is different than the normal one. The readonly version contains a copy
 388       * of the pages in the state they were when the PDF was annotated, by doing so we prevent the
 389       * the pages that are displayed to change as soon as the submission changes.
 390       *
 391       * Though there is an edge case, if the PDF was annotated before MDL-45580, then it is possible
 392       * that we do not find any readonly version of the pages. In that case, we will get the normal
 393       * pages and copy them to the readonly area. This ensures that the pages will remain in that
 394       * state until the submission is updated. When the normal files do not exist, we throw an exception
 395       * because the readonly pages should only ever be displayed after a teacher has annotated the PDF,
 396       * they would not exist until they do.
 397       *
 398       * @param int|\assign $assignment
 399       * @param int $userid
 400       * @param int $attemptnumber (-1 means latest attempt)
 401       * @param bool $readonly If true, then we are requesting the readonly version.
 402       * @return array(stored_file)
 403       */
 404      public static function get_page_images_for_attempt($assignment, $userid, $attemptnumber, $readonly = false) {
 405  
 406          $assignment = self::get_assignment_from_param($assignment);
 407  
 408          if (!$assignment->can_view_submission($userid)) {
 409              \print_error('nopermission');
 410          }
 411  
 412          if ($assignment->get_instance()->teamsubmission) {
 413              $submission = $assignment->get_group_submission($userid, 0, false);
 414          } else {
 415              $submission = $assignment->get_user_submission($userid, false);
 416          }
 417          $grade = $assignment->get_user_grade($userid, true, $attemptnumber);
 418  
 419          $contextid = $assignment->get_context()->id;
 420          $component = 'assignfeedback_editpdf';
 421          $itemid = $grade->id;
 422          $filepath = '/';
 423          $filearea = self::PAGE_IMAGE_FILEAREA;
 424  
 425          $fs = \get_file_storage();
 426  
 427          // If we are after the readonly pages...
 428          if ($readonly) {
 429              $filearea = self::PAGE_IMAGE_READONLY_FILEAREA;
 430              if ($fs->is_area_empty($contextid, $component, $filearea, $itemid)) {
 431                  // We have a problem here, we were supposed to find the files.
 432                  // Attempt to re-generate the pages from the combined images.
 433                  self::generate_page_images_for_attempt($assignment, $userid, $attemptnumber);
 434                  self::copy_pages_to_readonly_area($assignment, $grade);
 435              }
 436          }
 437  
 438          $files = $fs->get_directory_files($contextid, $component, $filearea, $itemid, $filepath);
 439  
 440          $pages = array();
 441          if (!empty($files)) {
 442              $first = reset($files);
 443              if (!$readonly && $first->get_timemodified() < $submission->timemodified) {
 444                  // Image files are stale, we need to regenerate them, except in readonly mode.
 445                  // We also need to remove the draft annotations and comments associated with this attempt.
 446                  $fs->delete_area_files($contextid, $component, $filearea, $itemid);
 447                  page_editor::delete_draft_content($itemid);
 448                  $files = array();
 449              } else {
 450  
 451                  // Need to reorder the files following their name.
 452                  // because get_directory_files() return a different order than generate_page_images_for_attempt().
 453                  foreach($files as $file) {
 454                      // Extract the page number from the file name image_pageXXXX.png.
 455                      preg_match('/page([\d]+)\./', $file->get_filename(), $matches);
 456                      if (empty($matches) or !is_numeric($matches[1])) {
 457                          throw new \coding_exception("'" . $file->get_filename()
 458                              . "' file hasn't the expected format filename: image_pageXXXX.png.");
 459                      }
 460                      $pagenumber = (int)$matches[1];
 461  
 462                      // Save the page in the ordered array.
 463                      $pages[$pagenumber] = $file;
 464                  }
 465                  ksort($pages);
 466              }
 467          }
 468  
 469          if (empty($pages)) {
 470              if ($readonly) {
 471                  // This should never happen, there should be a version of the pages available
 472                  // whenever we are requesting the readonly version.
 473                  throw new \moodle_exception('Could not find readonly pages for grade ' . $grade->id);
 474              }
 475              $pages = self::generate_page_images_for_attempt($assignment, $userid, $attemptnumber);
 476          }
 477  
 478          return $pages;
 479      }

 531      public static function generate_feedback_document($assignment, $userid, $attemptnumber) {
 532  
 533          $assignment = self::get_assignment_from_param($assignment);
 534  
 535          if (!$assignment->can_view_submission($userid)) {
 536              \print_error('nopermission');
 537          }
 538          if (!$assignment->can_grade()) {
 539              \print_error('nopermission');
 540          }
 541  
 542          // Need to generate the page images - first get a combined pdf.
 543          $file = self::get_combined_pdf_for_attempt($assignment, $userid, $attemptnumber);
 544          if (!$file) {
 545              throw new \moodle_exception('Could not generate combined pdf.');
 546          }
 547  
 548          $tmpdir = \make_temp_directory('assignfeedback_editpdf/final/' . self::hash($assignment, $userid, $attemptnumber));
 549          $combined = $tmpdir . '/' . self::COMBINED_PDF_FILENAME;
 550          $file->copy_content_to($combined); // Copy the file.
 551  
 552          $pdf = new pdf();
 553  
 554          $fs = \get_file_storage();
 555          $stamptmpdir = \make_temp_directory('assignfeedback_editpdf/stamps/' . self::hash($assignment, $userid, $attemptnumber));
 556          $grade = $assignment->get_user_grade($userid, true, $attemptnumber);
 557          // Copy any new stamps to this instance.
 558          if ($files = $fs->get_area_files($assignment->get_context()->id,
 559                                           'assignfeedback_editpdf',
 560                                           'stamps',
 561                                           $grade->id,
 562                                           "filename",
 563                                           false)) {
 564              foreach ($files as $file) {
 565                  $filename = $stamptmpdir . '/' . $file->get_filename();
 566                  $file->copy_content_to($filename); // Copy the file.
 567              }
 568          }
 569  
 570          $pagecount = $pdf->set_pdf($combined);
 571          $grade = $assignment->get_user_grade($userid, true, $attemptnumber);
 572          page_editor::release_drafts($grade->id);
 573  
 574          for ($i = 0; $i < $pagecount; $i++) {
 575              $pdf->copy_page();
 576              $comments = page_editor::get_comments($grade->id, $i, false);
 577              $annotations = page_editor::get_annotations($grade->id, $i, false);
 578  
 579              foreach ($comments as $comment) {
 580                  $pdf->add_comment($comment->rawtext,
 581                                    $comment->x,
 582                                    $comment->y,
 583                                    $comment->width,
 584                                    $comment->colour);
 585              }
 586  
 587              foreach ($annotations as $annotation) {
 588                  $pdf->add_annotation($annotation->x,
 589                                       $annotation->y,
 590                                       $annotation->endx,
 591                                       $annotation->endy,
 592                                       $annotation->colour,
 593                                       $annotation->type,
 594                                       $annotation->path,
 595                                       $stamptmpdir);

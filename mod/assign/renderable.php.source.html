 808       * @param array $dir
 809       * @param string $filearea
 810       * @param string $component
 811       * @return void
 812       */
 813      public function preprocess($dir, $filearea, $component) {
 814          global $CFG;
 815          foreach ($dir['subdirs'] as $subdir) {
 816              $this->preprocess($subdir, $filearea, $component);
 817          }
 818          foreach ($dir['files'] as $file) {
 819              $file->portfoliobutton = '';
 820              if (!empty($CFG->enableportfolios)) {
 821                  $button = new portfolio_add_button();
 822                  if (has_capability('mod/assign:exportownsubmission', $this->context)) {
 823                      $portfolioparams = array('cmid' => $this->cm->id, 'fileid' => $file->get_id());

 825                                                    $portfolioparams,
 826                                                    'mod_assign');
 827                      $button->set_format_by_file($file);
 828                      $file->portfoliobutton = $button->to_html(PORTFOLIO_ADD_ICON_LINK);
 829                  }
 830              }
 831              $path = '/' .
 832                      $this->context->id .
 833                      '/' .
 834                      $component .
 835                      '/' .
 836                      $filearea .
 837                      '/' .
 838                      $file->get_itemid() .
 839                      $file->get_filepath() .
 840                      $file->get_filename();
 841              $url = file_encode_url("$CFG->wwwroot/pluginfile.php", $path, true);
 842              $filename = $file->get_filename();

 844          }

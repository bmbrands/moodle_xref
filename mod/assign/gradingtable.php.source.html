 484      /**
 485       * list current marking workflow state
 486       *
 487       * @param stdClass $row
 488       * @return string
 489       */
 490      public function col_workflowstatus(stdClass $row) {
 491          $o = '';
 492  
 493          $gradingdisabled = $this->assignment->grading_disabled($row->id);
 494          // The function in the assignment keeps a static cache of this list of states.
 495          $workflowstates = $this->assignment->get_marking_workflow_states_for_current_user();
 496          $workflowstate = $row->workflowstate;
 497          if (empty($workflowstate)) {
 498              $workflowstate = ASSIGN_MARKING_WORKFLOW_STATE_NOTMARKED;
 499          }
 500          if ($this->quickgrading && !$gradingdisabled) {
 501              $notmarked = get_string('markingworkflowstatenotmarked', 'assign');
 502              $name = 'quickgrade_' . $row->id . '_workflowstate';
 503              $o .= html_writer::select($workflowstates, $name, $workflowstate, array('' => $notmarked));
 504              // Check if this user is a marker that can't manage allocations and doesn't have the marker column added.
 505              if ($this->assignment->get_instance()->markingworkflow &&
 506                  $this->assignment->get_instance()->markingallocation &&
 507                  !has_capability('mod/assign:manageallocations', $this->assignment->get_context())) {
 508  
 509                  $name = 'quickgrade_' . $row->id . '_allocatedmarker';
 510                  $o .= html_writer::empty_tag('input', array('type' => 'hidden', 'name' => $name,
 511                          'value' => $row->allocatedmarker));
 512              }
 513          } else {
 514              $o .= $this->output->container(get_string('markingworkflowstate' . $workflowstate, 'assign'), $workflowstate);
 515          }
 516          return $o;
 517      }

 553              return '';
 554          }

 558              } else {
 559                  return '';
 560              }
 561          }
 562  
 563          if ($this->quickgrading && has_capability('mod/assign:manageallocations', $this->assignment->get_context()) &&
 564              (empty($row->workflowstate) ||
 565               $row->workflowstate == ASSIGN_MARKING_WORKFLOW_STATE_INMARKING ||
 566               $row->workflowstate == ASSIGN_MARKING_WORKFLOW_STATE_NOTMARKED)) {
 567  
 568              $name = 'quickgrade_' . $row->id . '_allocatedmarker';
 569              return  html_writer::select($markerlist, $name, $row->allocatedmarker, false);
 570          } else if (!empty($row->allocatedmarker)) {
 571              $output = '';

 646       *

 669              $submission = $this->groupsubmissions[$groupid . ':' . $attemptnumber];

 899       * Format a column of data for display
 900       *
 901       * @param stdClass $row
 902       * @return string
 903       */
 904      public function col_status(stdClass $row) {
 905          $o = '';
 906  
 907          $instance = $this->assignment->get_instance();
 908  
 909          $due = $instance->duedate;
 910          if ($row->extensionduedate) {
 911              $due = $row->extensionduedate;
 912          }
 913  
 914          $group = false;
 915          $submission = false;
 916          $this->get_group_and_submission($row->id, $group, $submission, -1);
 917  
 918          if ($instance->teamsubmission && !$group && !$instance->preventsubmissionnotingroup) {
 919              $group = true;
 920          }
 921  
 922          if ($group && $submission) {
 923              $timesubmitted = $submission->timemodified;
 924              $status = $submission->status;
 925          } else {
 926              $timesubmitted = $row->timesubmitted;
 927              $status = $row->status;
 928          }
 929  
 930          if ($this->assignment->is_any_submission_plugin_enabled()) {
 931  
 932              $o .= $this->output->container(get_string('submissionstatus_' . $status, 'assign'),
 933                                             array('class'=>'submissionstatus' .$status));
 934              if ($due && $timesubmitted > $due) {
 935                  $usertime = format_time($timesubmitted - $due);
 936                  $latemessage = get_string('submittedlateshort',
 937                                            'assign',
 938                                            $usertime);
 939                  $o .= $this->output->container($latemessage, 'latesubmission');
 940              }
 941              if ($row->locked) {
 942                  $lockedstr = get_string('submissionslockedshort', 'assign');
 943                  $o .= $this->output->container($lockedstr, 'lockedsubmission');
 944              }
 945  
 946              // Add status of "grading" if markflow is not enabled.
 947              if (!$instance->markingworkflow) {
 948                  if ($row->grade !== null && $row->grade >= 0) {
 949                      $o .= $this->output->container(get_string('graded', 'assign'), 'submissiongraded');
 950                  } else if (!$timesubmitted) {
 951                      $now = time();
 952                      if ($due && ($now > $due)) {
 953                          $overduestr = get_string('overdue', 'assign', format_time($now - $due));
 954                          $o .= $this->output->container($overduestr, 'overduesubmission');
 955                      }
 956                  }
 957              }
 958          }
 959  
 960          if ($instance->markingworkflow) {
 961              $o .= $this->col_workflowstatus($row);
 962          }
 963          if ($row->extensionduedate) {
 964              $userdate = userdate($row->extensionduedate);
 965              $extensionstr = get_string('userextensiondate', 'assign', $userdate);
 966              $o .= $this->output->container($extensionstr, 'extensiondate');
 967          }
 968  
 969          if ($this->is_downloading()) {
 970              $o = strip_tags(str_replace('</div>', "\n", $o));
 971          }
 972  
 973          return $o;
 974      }

 199  /**
 200   * Extracts scrom package, sets up all variables.
 201   * Called whenever scorm changes
 202   * @param object $scorm instance - fields are updated and changes saved into database
 203   * @param bool $full force full update if true
 204   * @return void
 205   */
 206  function scorm_parse($scorm, $full) {
 207      global $CFG, $DB;
 208      $cfgscorm = get_config('scorm');
 209  
 210      if (!isset($scorm->cmid)) {
 211          $cm = get_coursemodule_from_instance('scorm', $scorm->id);
 212          $scorm->cmid = $cm->id;
 213      }
 214      $context = context_module::instance($scorm->cmid);
 215      $newhash = $scorm->sha1hash;
 216  
 217      if ($scorm->scormtype === SCORM_TYPE_LOCAL or $scorm->scormtype === SCORM_TYPE_LOCALSYNC) {
 218  
 219          $fs = get_file_storage();
 220          $packagefile = false;
 221          $packagefileimsmanifest = false;
 222  
 223          if ($scorm->scormtype === SCORM_TYPE_LOCAL) {
 224              if ($packagefile = $fs->get_file($context->id, 'mod_scorm', 'package', 0, '/', $scorm->reference)) {
 225                  if ($packagefile->is_external_file()) { // Get zip file so we can check it is correct.
 226                      $packagefile->import_external_file_contents();
 227                  }
 228                  $newhash = $packagefile->get_contenthash();
 229                  if (strtolower($packagefile->get_filename()) == 'imsmanifest.xml') {
 230                      $packagefileimsmanifest = true;
 231                  }
 232              } else {
 233                  $newhash = null;
 234              }
 235          } else {
 236              if (!$cfgscorm->allowtypelocalsync) {
 237                  // Sorry - localsync disabled.
 238                  return;
 239              }
 240              if ($scorm->reference !== '' and (!$full or $scorm->sha1hash !== sha1($scorm->reference))) {
 241                  $fs->delete_area_files($context->id, 'mod_scorm', 'package');
 242                  $filerecord = array('contextid' => $context->id, 'component' => 'mod_scorm', 'filearea' => 'package',
 243                                      'itemid' => 0, 'filepath' => '/');
 244                  if ($packagefile = $fs->create_file_from_url($filerecord, $scorm->reference, array('calctimeout' => true))) {
 245                      $newhash = sha1($scorm->reference);
 246                  } else {
 247                      $newhash = null;
 248                  }
 249              }
 250          }
 251  
 252          if ($packagefile) {
 253              if (!$full and $packagefile and $scorm->sha1hash === $newhash) {
 254                  if (strpos($scorm->version, 'SCORM') !== false) {
 255                      if ($packagefileimsmanifest || $fs->get_file($context->id, 'mod_scorm', 'content', 0, '/', 'imsmanifest.xml')) {
 256                          // No need to update.
 257                          return;
 258                      }
 259                  } else if (strpos($scorm->version, 'AICC') !== false) {
 260                      // TODO: add more sanity checks - something really exists in scorm_content area.
 261                      return;
 262                  }
 263              }
 264              if (!$packagefileimsmanifest) {
 265                  // Now extract files.
 266                  $fs->delete_area_files($context->id, 'mod_scorm', 'content');
 267  
 268                  $packer = get_file_packer('application/zip');
 269                  $packagefile->extract_to_storage($packer, $context->id, 'mod_scorm', 'content', 0, '/');
 270              }
 271  
 272          } else if (!$full) {
 273              return;
 274          }
 275          if ($packagefileimsmanifest) {
 276              require_once("$CFG->dirroot/mod/scorm/datamodels/scormlib.php");
 277              // Direct link to imsmanifest.xml file.
 278              if (!scorm_parse_scorm($scorm, $packagefile)) {
 279                  $scorm->version = 'ERROR';
 280              }
 281  
 282          } else if ($manifest = $fs->get_file($context->id, 'mod_scorm', 'content', 0, '/', 'imsmanifest.xml')) {
 283              require_once("$CFG->dirroot/mod/scorm/datamodels/scormlib.php");
 284              // SCORM.
 285              if (!scorm_parse_scorm($scorm, $manifest)) {
 286                  $scorm->version = 'ERROR';
 287              }
 288          } else {
 289              require_once("$CFG->dirroot/mod/scorm/datamodels/aicclib.php");
 290              // AICC.
 291              $result = scorm_parse_aicc($scorm);
 292              if (!$result) {
 293                  $scorm->version = 'ERROR';
 294              } else {
 295                  $scorm->version = 'AICC';
 296              }
 297          }
 298  
 299      } else if ($scorm->scormtype === SCORM_TYPE_EXTERNAL and $cfgscorm->allowtypeexternal) {
 300          require_once("$CFG->dirroot/mod/scorm/datamodels/scormlib.php");
 301          // SCORM only, AICC can not be external.
 302          if (!scorm_parse_scorm($scorm, $scorm->reference)) {
 303              $scorm->version = 'ERROR';
 304          }
 305          $newhash = sha1($scorm->reference);
 306  
 307      } else if ($scorm->scormtype === SCORM_TYPE_AICCURL  and $cfgscorm->allowtypeexternalaicc) {
 308          require_once("$CFG->dirroot/mod/scorm/datamodels/aicclib.php");
 309          // AICC.
 310          $result = scorm_parse_aicc($scorm);
 311          if (!$result) {
 312              $scorm->version = 'ERROR';
 313          } else {
 314              $scorm->version = 'AICC';
 315          }
 316  
 317      } else {
 318          // Sorry, disabled type.
 319          return;
 320      }
 321  
 322      $scorm->revision++;
 323      $scorm->sha1hash = $newhash;
 324      $DB->update_record('scorm', $scorm);
 325  }

 328  function scorm_array_search($item, $needle, $haystacks, $strict=false) {
 329      if (!empty($haystacks)) {
 330          foreach ($haystacks as $key => $element) {
 331              if ($strict) {
 332                  if ($element->{$item} === $needle) {
 333                      return $key;
 334                  }
 335              } else {
 336                  if ($element->{$item} == $needle) {
 337                      return $key;
 338                  }
 339              }
 340          }
 341      }
 342      return false;
 343  }

 356  function scorm_external_link($link) {
 357      // Check if a link is external.
 358      $result = false;
 359      $link = strtolower($link);
 360      if (substr($link, 0, 7) == 'http://') {
 361          $result = true;
 362      } else if (substr($link, 0, 8) == 'https://') {
 363          $result = true;
 364      } else if (substr($link, 0, 4) == 'www.') {
 365          $result = true;
 366      }
 367      return $result;

 594      $usertrack->status = '';
 595      $usertrack->total_time = '00:00:00';
 596      $usertrack->session_time = '00:00:00';
 597      $usertrack->timemodified = 0;
 598  
 599      foreach ($trackdata as $track) {
 600          $element = $track->element;
 601          $usertrack->{$element} = $track->value;
 602          switch ($element) {
 603              case 'cmi.core.lesson_status':
 604              case 'cmi.completion_status':
 605                  if ($track->value == 'not attempted') {
 606                      $track->value = 'notattempted';
 607                  }
 608                  $usertrack->status = $track->value;
 609                  break;
 610              case 'cmi.core.score.raw':
 611              case 'cmi.score.raw':
 612                  $usertrack->score_raw = (float) sprintf('%2.2f', $track->value);
 613                  break;
 614              case 'cmi.core.session_time':
 615              case 'cmi.session_time':
 616                  $usertrack->session_time = $track->value;
 617                  break;
 618              case 'cmi.core.total_time':
 619              case 'cmi.total_time':
 620                  $usertrack->total_time = $track->value;
 621                  break;
 622          }
 623          if (isset($track->timemodified) && ($track->timemodified > $usertrack->timemodified)) {
 624              $usertrack->timemodified = $track->timemodified;
 625          }
 626      }
 627  
 628      return $usertrack;

 666  function scorm_grade_user_attempt($scorm, $userid, $attempt=1) {
 667      global $DB;
 668      $attemptscore = new stdClass();
 669      $attemptscore->scoes = 0;
 670      $attemptscore->values = 0;
 671      $attemptscore->max = 0;
 672      $attemptscore->sum = 0;
 673      $attemptscore->lastmodify = 0;
 674  
 675      if (!$scoes = $DB->get_records('scorm_scoes', array('scorm' => $scorm->id), 'sortorder, id')) {
 676          return null;
 677      }
 678  
 679      foreach ($scoes as $sco) {
 680          if ($userdata = scorm_get_tracks($sco->id, $userid, $attempt)) {
 681              if (($userdata->status == 'completed') || ($userdata->status == 'passed')) {
 682                  $attemptscore->scoes++;
 683              }
 684              if (!empty($userdata->score_raw) || (isset($scorm->type) && $scorm->type == 'sco' && isset($userdata->score_raw))) {
 685                  $attemptscore->values++;
 686                  $attemptscore->sum += $userdata->score_raw;
 687                  $attemptscore->max = ($userdata->score_raw > $attemptscore->max) ? $userdata->score_raw : $attemptscore->max;
 688                  if (isset($userdata->timemodified) && ($userdata->timemodified > $attemptscore->lastmodify)) {
 689                      $attemptscore->lastmodify = $userdata->timemodified;
 690                  } else {
 691                      $attemptscore->lastmodify = 0;
 692                  }
 693              }
 694          }
 695      }
 696      switch ($scorm->grademethod) {
 697          case GRADEHIGHEST:
 698              $score = (float) $attemptscore->max;
 699          break;
 700          case GRADEAVERAGE:
 701              if ($attemptscore->values > 0) {
 702                  $score = $attemptscore->sum / $attemptscore->values;
 703              } else {
 704                  $score = 0;
 705              }
 706          break;
 707          case GRADESUM:
 708              $score = $attemptscore->sum;
 709          break;
 710          case GRADESCOES:
 711              $score = $attemptscore->scoes;
 712          break;
 713          default:
 714              $score = $attemptscore->max;   // Remote Learner GRADEHIGHEST is default.
 715      }
 716  
 717      return $score;
 718  }

 720  function scorm_grade_user($scorm, $userid) {
 721  
 722      // Ensure we dont grade user beyond $scorm->maxattempt settings.
 723      $lastattempt = scorm_get_last_attempt($scorm->id, $userid);
 724      if ($scorm->maxattempt != 0 && $lastattempt >= $scorm->maxattempt) {
 725          $lastattempt = $scorm->maxattempt;
 726      }
 727  
 728      switch ($scorm->whatgrade) {
 729          case FIRSTATTEMPT:
 730              return scorm_grade_user_attempt($scorm, $userid, 1);
 731          break;
 732          case LASTATTEMPT:
 733              return scorm_grade_user_attempt($scorm, $userid, scorm_get_last_completed_attempt($scorm->id, $userid));
 734          break;
 735          case HIGHESTATTEMPT:
 736              $maxscore = 0;
 737              for ($attempt = 1; $attempt <= $lastattempt; $attempt++) {
 738                  $attemptscore = scorm_grade_user_attempt($scorm, $userid, $attempt);
 739                  $maxscore = $attemptscore > $maxscore ? $attemptscore : $maxscore;
 740              }
 741              return $maxscore;
 742  
 743          break;
 744          case AVERAGEATTEMPT:
 745              $attemptcount = scorm_get_attempt_count($userid, $scorm, true, true);
 746              if (empty($attemptcount)) {
 747                  return 0;
 748              } else {
 749                  $attemptcount = count($attemptcount);
 750              }
 751              $lastattempt = scorm_get_last_attempt($scorm->id, $userid);
 752              $sumscore = 0;
 753              for ($attempt = 1; $attempt <= $lastattempt; $attempt++) {
 754                  $attemptscore = scorm_grade_user_attempt($scorm, $userid, $attempt);
 755                  $sumscore += $attemptscore;
 756              }
 757  
 758              return round($sumscore / $attemptcount);
 759          break;
 760      }

1004  /**
1005   * Build up the JavaScript representation of an array element
1006   *
1007   * @param string $sversion SCORM API version
1008   * @param array $userdata User track data
1009   * @param string $elementname Name of array element to get values for
1010   * @param array $children list of sub elements of this array element that also need instantiating
1011   * @return Javascript array elements
1012   */
1013  function scorm_reconstitute_array_element($sversion, $userdata, $elementname, $children) {
1014      // Reconstitute comments_from_learner and comments_from_lms.
1015      $current = '';
1016      $currentsubelement = '';
1017      $currentsub = '';
1018      $count = 0;
1019      $countsub = 0;
1020      $scormseperator = '_';
1021      $return = '';
1022      if (scorm_version_check($sversion, SCORM_13)) { // Scorm 1.3 elements use a . instead of an _ .
1023          $scormseperator = '.';
1024      }
1025      // Filter out the ones we want.
1026      $elementlist = array();
1027      foreach ($userdata as $element => $value) {
1028          if (substr($element, 0, strlen($elementname)) == $elementname) {
1029              $elementlist[$element] = $value;
1030          }
1031      }
1032  
1033      // Sort elements in .n array order.
1034      uksort($elementlist, "scorm_element_cmp");
1035  
1036      // Generate JavaScript.
1037      foreach ($elementlist as $element => $value) {
1038          if (scorm_version_check($sversion, SCORM_13)) {
1039              $element = preg_replace('/\.(\d+)\./', ".N\$1.", $element);
1040              preg_match('/\.(N\d+)\./', $element, $matches);
1041          } else {
1042              $element = preg_replace('/\.(\d+)\./', "_\$1.", $element);
1043              preg_match('/\_(\d+)\./', $element, $matches);
1044          }
1045          if (count($matches) > 0 && $current != $matches[1]) {
1046              if ($countsub > 0) {
1047                  $return .= '    '.$elementname.$scormseperator.$current.'.'.$currentsubelement.'._count = '.$countsub.";\n";
1048              }
1049              $current = $matches[1];
1050              $count++;
1051              $currentsubelement = '';
1052              $currentsub = '';
1053              $countsub = 0;
1054              $end = strpos($element, $matches[1]) + strlen($matches[1]);
1055              $subelement = substr($element, 0, $end);
1056              $return .= '    '.$subelement." = new Object();\n";
1057              // Now add the children.
1058              foreach ($children as $child) {
1059                  $return .= '    '.$subelement.".".$child." = new Object();\n";
1060                  $return .= '    '.$subelement.".".$child."._children = ".$child."_children;\n";
1061              }
1062          }
1063  
1064          // Now - flesh out the second level elements if there are any.
1065          if (scorm_version_check($sversion, SCORM_13)) {
1066              $element = preg_replace('/(.*?\.N\d+\..*?)\.(\d+)\./', "\$1.N\$2.", $element);
1067              preg_match('/.*?\.N\d+\.(.*?)\.(N\d+)\./', $element, $matches);
1068          } else {
1069              $element = preg_replace('/(.*?\_\d+\..*?)\.(\d+)\./', "\$1_\$2.", $element);
1070              preg_match('/.*?\_\d+\.(.*?)\_(\d+)\./', $element, $matches);
1071          }
1072  
1073          // Check the sub element type.
1074          if (count($matches) > 0 && $currentsubelement != $matches[1]) {
1075              if ($countsub > 0) {
1076                  $return .= '    '.$elementname.$scormseperator.$current.'.'.$currentsubelement.'._count = '.$countsub.";\n";
1077              }
1078              $currentsubelement = $matches[1];
1079              $currentsub = '';
1080              $countsub = 0;
1081              $end = strpos($element, $matches[1]) + strlen($matches[1]);
1082              $subelement = substr($element, 0, $end);
1083              $return .= '    '.$subelement." = new Object();\n";
1084          }
1085  
1086          // Now check the subelement subscript.
1087          if (count($matches) > 0 && $currentsub != $matches[2]) {
1088              $currentsub = $matches[2];
1089              $countsub++;
1090              $end = strrpos($element, $matches[2]) + strlen($matches[2]);
1091              $subelement = substr($element, 0, $end);
1092              $return .= '    '.$subelement." = new Object();\n";
1093          }
1094  
1095          $return .= '    '.$element.' = \''.$value."';\n";
1096      }
1097      if ($countsub > 0) {
1098          $return .= '    '.$elementname.$scormseperator.$current.'.'.$currentsubelement.'._count = '.$countsub.";\n";
1099      }
1100      if ($count > 0) {
1101          $return .= '    '.$elementname.'._count = '.$count.";\n";
1102      }
1103      return $return;
1104  }

1106  /**
1107   * Build up the JavaScript representation of an array element
1108   *
1109   * @param string $a left array element
1110   * @param string $b right array element
1111   * @return comparator - 0,1,-1
1112   */
1113  function scorm_element_cmp($a, $b) {
1114      preg_match('/.*?(\d+)\./', $a, $matches);
1115      $left = intval($matches[1]);
1116      preg_match('/.?(\d+)\./', $b, $matches);
1117      $right = intval($matches[1]);
1118      if ($left < $right) {
1119          return -1; // Smaller.
1120      } else if ($left > $right) {
1121          return 1;  // Bigger.
1122      } else {
1123          // Look for a second level qualifier eg cmi.interactions_0.correct_responses_0.pattern.
1124          if (preg_match('/.*?(\d+)\.(.*?)\.(\d+)\./', $a, $matches)) {
1125              $leftterm = intval($matches[2]);
1126              $left = intval($matches[3]);
1127              if (preg_match('/.*?(\d+)\.(.*?)\.(\d+)\./', $b, $matches)) {
1128                  $rightterm = intval($matches[2]);
1129                  $right = intval($matches[3]);
1130                  if ($leftterm < $rightterm) {
1131                      return -1; // Smaller.
1132                  } else if ($leftterm > $rightterm) {
1133                      return 1;  // Bigger.
1134                  } else {

1379  /**
1380   * Converts SCORM duration notation to human-readable format
1381   * The function works with both SCORM 1.2 and SCORM 2004 time formats
1382   * @param $duration string SCORM duration
1383   * @return string human-readable date/time
1384   */
1385  function scorm_format_duration($duration) {
1386      // Fetch date/time strings.
1387      $stryears = get_string('years');
1388      $strmonths = get_string('nummonths');
1389      $strdays = get_string('days');
1390      $strhours = get_string('hours');
1391      $strminutes = get_string('minutes');
1392      $strseconds = get_string('seconds');
1393  
1394      if ($duration[0] == 'P') {
1395          // If timestamp starts with 'P' - it's a SCORM 2004 format
1396          // this regexp discards empty sections, takes Month/Minute ambiguity into consideration,
1397          // and outputs filled sections, discarding leading zeroes and any format literals
1398          // also saves the only zero before seconds decimals (if there are any) and discards decimals if they are zero.
1399          $pattern = array( '#([A-Z])0+Y#', '#([A-Z])0+M#', '#([A-Z])0+D#', '#P(|\d+Y)0*(\d+)M#',
1400                              '#0*(\d+)Y#', '#0*(\d+)D#', '#P#', '#([A-Z])0+H#', '#([A-Z])[0.]+S#',
1401                              '#\.0+S#', '#T(|\d+H)0*(\d+)M#', '#0*(\d+)H#', '#0+\.(\d+)S#',
1402                              '#0*([\d.]+)S#', '#T#' );
1403          $replace = array( '$1', '$1', '$1', '$1$2 '.$strmonths.' ', '$1 '.$stryears.' ', '$1 '.$strdays.' ',
1404                              '', '$1', '$1', 'S', '$1$2 '.$strminutes.' ', '$1 '.$strhours.' ',
1405                              '0.$1 '.$strseconds, '$1 '.$strseconds, '');
1406      } else {
1407          // Else we have SCORM 1.2 format there
1408          // first convert the timestamp to some SCORM 2004-like format for conveniency.
1409          $duration = preg_replace('#^(\d+):(\d+):([\d.]+)$#', 'T$1H$2M$3S', $duration);
1410          // Then convert in the same way as SCORM 2004.
1411          $pattern = array( '#T0+H#', '#([A-Z])0+M#', '#([A-Z])[0.]+S#', '#\.0+S#', '#0*(\d+)H#',
1412                              '#0*(\d+)M#', '#0+\.(\d+)S#', '#0*([\d.]+)S#', '#T#' );
1413          $replace = array( 'T', '$1', '$1', 'S', '$1 '.$strhours.' ', '$1 '.$strminutes.' ',
1414                              '0.$1 '.$strseconds, '$1 '.$strseconds, '' );
1415      }
1416  
1417      $result = preg_replace($pattern, $replace, $duration);
1418  
1419      return $result;
1420  }

1556  function scorm_get_toc_get_parent_child(&$result, $currentorg) {
1557      $final = array();
1558      $level = 0;
1559      // Organization is always the root, prevparent.
1560      if (!empty($currentorg)) {
1561          $prevparent = $currentorg;
1562      } else {
1563          $prevparent = '/';
1564      }
1565  
1566      foreach ($result as $sco) {
1567          if ($sco->parent == '/') {
1568              $final[$level][$sco->identifier] = $sco;
1569              $prevparent = $sco->identifier;
1570              unset($result[$sco->id]);
1571          } else {
1572              if ($sco->parent == $prevparent) {
1573                  $final[$level][$sco->identifier] = $sco;
1574                  $prevparent = $sco->identifier;
1575                  unset($result[$sco->id]);
1576              } else {
1577                  if (!empty($final[$level])) {
1578                      $found = false;
1579                      foreach ($final[$level] as $fin) {
1580                          if ($sco->parent == $fin->identifier) {
1581                              $found = true;
1582                          }
1583                      }
1584  
1585                      if ($found) {
1586                          $final[$level][$sco->identifier] = $sco;
1587                          unset($result[$sco->id]);
1588                          $found = false;
1589                      } else {
1590                          $level++;
1591                          $final[$level][$sco->identifier] = $sco;
1592                          unset($result[$sco->id]);
1593                      }
1594                  }
1595              }
1596          }
1597      }
1598  
1599      for ($i = 0; $i <= $level; $i++) {
1600          $prevparent = '';
1601          foreach ($final[$i] as $ident => $sco) {
1602              if (empty($prevparent)) {
1603                  $prevparent = $ident;
1604              }
1605              if (!isset($final[$i][$prevparent]->children)) {
1606                  $final[$i][$prevparent]->children = array();
1607              }
1608              if ($sco->parent == $prevparent) {
1609                  $final[$i][$prevparent]->children[] = $sco;
1610                  $prevparent = $ident;
1611              } else {
1612                  $parent = false;
1613                  foreach ($final[$i] as $identifier => $scoobj) {
1614                      if ($identifier == $sco->parent) {
1615                          $parent = $identifier;
1616                      }
1617                  }
1618  
1619                  if ($parent !== false) {
1620                      $final[$i][$parent]->children[] = $sco;
1621                  }
1622              }
1623          }
1624      }
1625  
1626      $results = array();
1627      for ($i = 0; $i <= $level; $i++) {
1628          $keys = array_keys($final[$i]);
1629          $results[] = $final[$i][$keys[0]];
1630      }
1631  
1632      return $results;
1633  }

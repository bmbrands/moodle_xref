 190      protected function after_execute() {
 191          global $DB;
 192  
 193          // Add scorm related files, no need to match by itemname (just internally handled context)
 194          $this->add_related_files('mod_scorm', 'intro', null);
 195          $this->add_related_files('mod_scorm', 'content', null);
 196          $this->add_related_files('mod_scorm', 'package', null);
 197  
 198          // Fix launch param in scorm table to use new sco id.
 199          $scormid = $this->get_new_parentid('scorm');
 200          $scorm = $DB->get_record('scorm', array('id' => $scormid));
 201          $scorm->launch = $this->get_mappingid('scorm_sco', $scorm->launch, '');
 202  
 203          if (!empty($scorm->launch)) {
 204              // Check that this sco has a valid launch value.
 205              $scolaunch = $DB->get_field('scorm_scoes', 'launch', array('id' => $scorm->launch));
 206              if (empty($scolaunch)) {
 207                  // This is not a valid sco - set to empty so we can find a valid launch sco.
 208                  $scorm->launch = '';
 209              }
 210          }
 211  
 212          if (empty($scorm->launch)) {
 213              // This scorm has an invalid launch param - we need to calculate it and get the first launchable sco.
 214              $sqlselect = 'scorm = ? AND '.$DB->sql_isnotempty('scorm_scoes', 'launch', false, true);
 215              // We use get_records here as we need to pass a limit in the query that works cross db.
 216              $scoes = $DB->get_records_select('scorm_scoes', $sqlselect, array($scormid), 'sortorder', 'id', 0, 1);
 217              if (!empty($scoes)) {
 218                  $sco = reset($scoes); // We only care about the first record - the above query only returns one.
 219                  $scorm->launch = $sco->id;
 220              }
 221          }
 222          if (!empty($scorm->launch)) {
 223              $DB->update_record('scorm', $scorm);
 224          }
 225      }

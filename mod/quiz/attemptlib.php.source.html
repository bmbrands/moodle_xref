 584       * Let each slot know which section it is part of.

 586      protected function link_sections_and_slots() {
 587          foreach ($this->sections as $i => $section) {
 588              if (isset($this->sections[$i + 1])) {

 590              } else {

 594                  $this->slots[$slot]->section = $section;

 599      /**
 600       * Parse attempt->layout to populate the other arrays the represent the layout.
 601       */
 602      protected function determine_layout() {
 603          $this->pagelayout = array();
 604  
 605          // Break up the layout string into pages.
 606          $pagelayouts = explode(',0', $this->attempt->layout);
 607  
 608          // Strip off any empty last page (normally there is one).
 609          if (end($pagelayouts) == '') {
 610              array_pop($pagelayouts);
 611          }
 612  
 613          // File the ids into the arrays.
 614          // Tracking which is the first slot in each section in this attempt is
 615          // trickier than you might guess, since the slots in this section
 616          // may be shuffled, so $section->firstslot (the lowest numbered slot in
 617          // the section) may not be the first one.
 618          $unseensections = $this->sections;
 619          $this->pagelayout = array();
 620          foreach ($pagelayouts as $page => $pagelayout) {
 621              $pagelayout = trim($pagelayout, ',');
 622              if ($pagelayout == '') {
 623                  continue;
 624              }
 625              $this->pagelayout[$page] = explode(',', $pagelayout);
 626              foreach ($this->pagelayout[$page] as $slot) {
 627                  $sectionkey = array_search($this->slots[$slot]->section, $unseensections);
 628                  if ($sectionkey !== false) {
 629                      $this->slots[$slot]->firstinsection = true;
 630                      unset($unseensections[$sectionkey]);
 631                  } else {
 632                      $this->slots[$slot]->firstinsection = false;
 633                  }
 634              }
 635          }
 636      }

 641      protected function number_questions() {
 642          $number = 1;
 643          foreach ($this->pagelayout as $page => $slots) {
 644              foreach ($slots as $slot) {
 645                  if ($length = $this->is_real_question($slot)) {

 898      /**
 899       * Check the appropriate capability to see whether this user may review their own attempt.
 900       * If not, prints an error.
 901       */
 902      public function check_review_capability() {
 903          if ($this->get_attempt_state() == mod_quiz_display_options::IMMEDIATELY_AFTER) {
 904              $capability = 'mod/quiz:attempt';
 905          } else {
 906              $capability = 'mod/quiz:reviewmyattempts';
 907          }
 908  
 909          // These next tests are in a slighly funny order. The point is that the
 910          // common and most performance-critical case is students attempting a quiz
 911          // so we want to check that permisison first.
 912  
 913          if ($this->has_capability($capability)) {
 914              // User has the permission that lets you do the quiz as a student. Fine.
 915              return;
 916          }
 917  
 918          if ($this->has_capability('mod/quiz:viewreports') ||
 919                  $this->has_capability('mod/quiz:preview')) {
 920              // User has the permission that lets teachers review. Fine.
 921              return;
 922          }
 923  
 924          // They should not be here. Trigger the standard no-permission error
 925          // but using the name of the student capability.
 926          // We know this will fail. We just want the stadard exception thown.
 927          $this->require_capability($capability);
 928      }

1274              $deadlines[] = $this->attempt->timestart + $this->quizobj->get_quiz()->timelimit;
1275          }
1276          if ($this->quizobj->get_quiz()->timeclose) {
1277              $deadlines[] = $this->quizobj->get_quiz()->timeclose;
1278          }
1279          if ($deadlines) {
1280              $duedate = min($deadlines);
1281          } else {
1282              return false;
1283          }
1284  
1285          switch ($this->attempt->state) {
1286              case self::IN_PROGRESS:
1287                  return $duedate;
1288  
1289              case self::OVERDUE:
1290                  return $duedate + $this->quizobj->get_quiz()->graceperiod;
1291  
1292              default:
1293                  throw new coding_exception('Unexpected state: ' . $this->attempt->state);
1294          }
1295      }

1669      /**
1670       * Check this attempt, to see if there are any state transitions that should
1671       * happen automatically.  This function will update the attempt checkstatetime.
1672       * @param int $timestamp the timestamp that should be stored as the modifed
1673       * @param bool $studentisonline is the student currently interacting with Moodle?
1674       */
1675      public function handle_if_time_expired($timestamp, $studentisonline) {
1676          global $DB;
1677  
1678          $timeclose = $this->get_access_manager($timestamp)->get_end_time($this->attempt);
1679  
1680          if ($timeclose === false || $this->is_preview()) {
1681              $this->update_timecheckstate(null);
1682              return; // No time limit
1683          }
1684          if ($timestamp < $timeclose) {
1685              $this->update_timecheckstate($timeclose);
1686              return; // Time has not yet expired.
1687          }
1688  
1689          // If the attempt is already overdue, look to see if it should be abandoned ...
1690          if ($this->attempt->state == self::OVERDUE) {
1691              $timeoverdue = $timestamp - $timeclose;
1692              $graceperiod = $this->quizobj->get_quiz()->graceperiod;
1693              if ($timeoverdue >= $graceperiod) {
1694                  $this->process_abandon($timestamp, $studentisonline);
1695              } else {
1696                  // Overdue time has not yet expired
1697                  $this->update_timecheckstate($timeclose + $graceperiod);
1698              }
1699              return; // ... and we are done.
1700          }
1701  
1702          if ($this->attempt->state != self::IN_PROGRESS) {
1703              $this->update_timecheckstate(null);
1704              return; // Attempt is already in a final state.
1705          }
1706  
1707          // Otherwise, we were in quiz_attempt::IN_PROGRESS, and time has now expired.
1708          // Transition to the appropriate state.
1709          switch ($this->quizobj->get_quiz()->overduehandling) {
1710              case 'autosubmit':
1711                  $this->process_finish($timestamp, false);
1712                  return;
1713  
1714              case 'graceperiod':
1715                  $this->process_going_overdue($timestamp, $studentisonline);
1716                  return;
1717  
1718              case 'autoabandon':
1719                  $this->process_abandon($timestamp, $studentisonline);
1720                  return;
1721          }
1722  
1723          // This is an overdue attempt with no overdue handling defined, so just abandon.
1724          $this->process_abandon($timestamp, $studentisonline);
1725          return;
1726      }

  74      protected function get_latest_steps($qubaids) {
  75          $summarks = array();
  76          $fakeusageid = 0;
  77          foreach ($this->lateststeps as $step) {
  78              // The same 'sumgrades' field is available in step data for every slot, we will ignore all slots but slot 1.
  79              // The step for slot 1 is always the first one in the csv file for each usage, we will use that to separate steps from
  80              // each usage.
  81              if ($step->slot == 1) {
  82                  $fakeusageid++;
  83                  $summarks[$fakeusageid] = $step->sumgrades;
  84              }
  85              unset($step->sumgrades);
  86              $step->questionusageid = $fakeusageid;
  87          }
  88  
  89          return array($this->lateststeps, $summarks);
  90      }

 143      public function qstats_q_fields($fieldname, $values, $multiplier=1) {
 144          foreach ($this->qstats->get_all_slots() as $slot) {

 158              if ($items[$key]!='') {

 164                          list($key) = each($items);

 166                      }

 168                  }

 175      public function get_records_from_csv($filename) {
 176          $filecontents = file($filename, FILE_IGNORE_NEW_LINES);
 177          $records = array();
 178          // Skip the first line containing field names.
 179          $keys = $this->get_fields_from_csv(array_shift($filecontents));
 180          while (null !== ($line = array_shift($filecontents))) {
 181              $data = $this->get_fields_from_csv($line);
 182              $arraykey = reset($data);
 183              $object = new stdClass();
 184              foreach ($keys as $key) {
 185                  $value = array_shift($data);
 186                  if ($value !== null) {
 187                      $object->{$key} = $value;
 188                  } else {
 189                      $object->{$key} = '';
 190                  }
 191              }
 192              $records[$arraykey] = $object;
 193          }
 194          return $records;
 195      }

  90      /**
  91       * Send a request using the REST API
  92       *
  93       * @method send_request
  94       * @param {Object} data The data to submit with the AJAX request
  95       * @param {Node} [statusspinner] A statusspinner which may contain a section loader
  96       * @param {Function} success_callback The callback to use on success
  97       * @param {Object} [optionalconfig] Any additional configuration to submit
  98       * @chainable
  99       */
 100      send_request: function(data, statusspinner, success_callback, optionalconfig) {
 101          // Default data structure
 102          if (!data) {
 103              data = {};
 104          }
 105          // Handle any variables which we must pass back through to
 106          var pageparams = this.get('config').pageparams,
 107              varname;
 108          for (varname in pageparams) {
 109              data[varname] = pageparams[varname];
 110          }
 111  
 112          data.sesskey = M.cfg.sesskey;
 113          data.courseid = this.get('courseid');
 114          data.quizid = this.get('quizid');
 115  
 116          var uri = M.cfg.wwwroot + this.get('ajaxurl');
 117  
 118          // Define the configuration to send with the request
 119          var responsetext = [];
 120          var config = {
 121              method: 'POST',
 122              data: data,
 123              on: {
 124                  success: function(tid, response) {
 125                      try {
 126                          responsetext = Y.JSON.parse(response.responseText);
 127                          if (responsetext.error) {
 128                              new M.core.ajaxException(responsetext);
 129                          }
 130                      } catch (e) {}
 131  
 132                      // Run the callback if we have one.
 133                      if (responsetext.hasOwnProperty('newsummarks')) {
 134                          Y.one(SELECTOR.SUMMARKS).setHTML(responsetext.newsummarks);
 135                      }
 136                      if (responsetext.hasOwnProperty('newnumquestions')) {
 137                          Y.one(SELECTOR.NUMQUESTIONS).setHTML(
 138                                  M.util.get_string('numquestionsx', 'quiz', responsetext.newnumquestions)
 139                              );
 140                      }
 141                      if (success_callback) {
 142                          Y.bind(success_callback, this, responsetext)();
 143                      }
 144  
 145                      if (statusspinner) {
 146                          window.setTimeout(function() {
 147                              statusspinner.hide();
 148                          }, 400);
 149                      }
 150                  },
 151                  failure: function(tid, response) {
 152                      if (statusspinner) {
 153                          statusspinner.hide();
 154                      }
 155                      new M.core.ajaxException(response);
 156                  }
 157              },
 158              context: this
 159          };
 160  
 161          // Apply optional config
 162          if (optionalconfig) {
 163              for (varname in optionalconfig) {
 164                  config[varname] = optionalconfig[varname];
 165              }
 166          }
 167  
 168          if (statusspinner) {
 169              statusspinner.show();
 170          }
 171  
 172          // Send the request
 173          Y.io(uri, config);
 174          return this;
 175      }

 289      /**
 290       * Handles the delegation event. When this is fired someone has triggered an action.
 291       *
 292       * Note not all actions will result in an AJAX enhancement.
 293       *
 294       * @protected
 295       * @method handle_data_action
 296       * @param {EventFacade} ev The event that was triggered.
 297       * @returns {boolean}
 298       */
 299      handle_data_action: function(ev) {
 300          // We need to get the anchor element that triggered this event.
 301          var node = ev.target;
 302          if (!node.test('a')) {
 303              node = node.ancestor(SELECTOR.ACTIVITYACTION);
 304          }
 305  
 306          // From the anchor we can get both the activity (added during initialisation) and the action being
 307          // performed (added by the UI as a data attribute).
 308          var action = node.getData('action'),
 309              activity = node.ancestor(SELECTOR.ACTIVITYLI);
 310  
 311          if (!node.test('a') || !action || !activity) {
 312              // It wasn't a valid action node.
 313              return;
 314          }
 315  
 316          // Switch based upon the action and do the desired thing.
 317          switch (action) {
 318              case 'editmaxmark':
 319                  // The user wishes to edit the maxmark of the resource.
 320                  this.edit_maxmark(ev, node, activity, action);
 321                  break;
 322              case 'delete':
 323                  // The user is deleting the activity.
 324                  this.delete_with_confirmation(ev, node, activity, action);
 325                  break;
 326              case 'addpagebreak':
 327              case 'removepagebreak':
 328                  // The user is adding or removing a page break.
 329                  this.update_page_break(ev, node, activity, action);
 330                  break;
 331              case 'adddependency':
 332              case 'removedependency':
 333                  // The user is adding or removing a dependency between questions.
 334                  this.update_dependency(ev, node, activity, action);
 335                  break;
 336              default:
 337                  // Nothing to do here!
 338                  break;
 339          }

 734      /**
 735       * Handles the delegation event. When this is fired someone has triggered an action.
 736       *
 737       * Note not all actions will result in an AJAX enhancement.
 738       *
 739       * @protected
 740       * @method handle_data_action
 741       * @param {EventFacade} ev The event that was triggered.
 742       * @returns {boolean}
 743       */
 744      handle_data_action: function(ev) {
 745          // We need to get the anchor element that triggered this event.
 746          var node = ev.target;
 747          if (!node.test('a') && !node.test('input[data-action]')) {
 748              node = node.ancestor(SELECTOR.ACTIVITYACTION);
 749          }
 750  
 751          // From the anchor we can get both the activity (added during initialisation) and the action being
 752          // performed (added by the UI as a data attribute).
 753          var action = node.getData('action'),
 754              activity = node.ancestor(SELECTOR.ACTIVITYLI);
 755  
 756          if ((!node.test('a') && !node.test('input[data-action]')) || !action || !activity) {
 757              // It wasn't a valid action node.
 758              return;
 759          }
 760  
 761          // Switch based upon the action and do the desired thing.
 762          switch (action) {
 763              case 'edit_section_title':
 764                  // The user wishes to edit the section headings.
 765                  this.edit_section_title(ev, node, activity, action);
 766                  break;
 767              case 'shuffle_questions':
 768                  // The user wishes to edit the shuffle questions of the section (resource).
 769                  this.edit_shuffle_questions(ev, node, activity, action);
 770                  break;
 771              case 'deletesection':
 772                  // The user is deleting the activity.
 773                  this.delete_section_with_confirmation(ev, node, activity, action);
 774                  break;
 775              default:
 776                  // Nothing to do here!
 777                  break;
 778          }
 779      },

 830      edit_section_title: function(ev, button, activity) {
 831          // Get the element we're working on

 833              instancesection  = activity.one(SELECTOR.INSTANCESECTION),

 852                  .set('innerHTML', M.util.get_string('edittitleinstructions', 'moodle'));
 853              var editor = Y.Node.create('<input name="section" type="text" />').setAttrs({

 857                  'maxLength': '255' // This is the maxlength in DB.

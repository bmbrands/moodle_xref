 162      getPreviousNumbered: function(slot) {

 164          if (previous) {
 165              return previous;
 166          }

 168          var section = slot.ancestor('li.section').previous('li.section');
 169          while (section) {
 170              var questions = section.all(this.SELECTORS.SLOT + ':not(' + this.SELECTORS.QUESTIONTYPEDESCRIPTION + ')');

 300      /**
 301       * Reorder each pagebreak by iterating through each related slot.
 302       *
 303       * @method reorderPageBreaks
 304       * @return void
 305       */
 306      reorderPageBreaks: function() {
 307          // Get list of slot nodes.
 308          var slots = this.getSlots(), slotnumber = 0;
 309          // Loop through slots incrementing the number each time.
 310          slots.each(function(slot, key) {
 311              slotnumber++;
 312              var pagebreak = this.getPageBreak(slot);
 313              var nextitem = slot.next('li.activity');
 314              if (!nextitem) {
 315                  // Last slot in a section. Should not have an icon.
 316                  return;
 317              }
 318  
 319              // No pagebreak and not last slot. Add one.
 320              if (!pagebreak) {
 321                  pagebreak = this.addPageBreak(slot);
 322              }
 323  
 324              // Remove last page break if there is one.
 325              if (pagebreak && key === slots.size() - 1) {
 326                  this.removePageBreak(slot);
 327              }
 328  
 329              // Get page break anchor element.
 330              var pagebreaklink = pagebreak.get('childNodes').item(0);
 331  
 332              // Get the correct title.
 333              var action = '', iconname = '';
 334              if (Y.Moodle.mod_quiz.util.page.isPage(nextitem)) {
 335                  action = 'removepagebreak';
 336                  iconname = 'e/remove_page_break';
 337              } else {
 338                  action = 'addpagebreak';
 339                  iconname = 'e/insert_page_break';
 340              }
 341  
 342              // Update the link and image titles
 343              pagebreaklink.set('title', M.util.get_string(action, 'quiz'));
 344              pagebreaklink.setData('action', action);
 345              // Update the image title.
 346              var icon = pagebreaklink.one(this.SELECTORS.ICON);
 347              icon.set('title', M.util.get_string(action, 'quiz'));
 348              icon.set('alt', M.util.get_string(action, 'quiz'));
 349  
 350              // Update the image src.
 351              icon.set('src', M.util.image_url(iconname));
 352  
 353              // Get anchor url parameters as an associative array.
 354              var params = Y.QueryString.parse(pagebreaklink.get('href'));
 355              // Update slot number.
 356              params.slot = slotnumber;
 357              // Create the new url.
 358              var newurl = '';
 359              for (var index in params) {
 360                  if (newurl.length) {
 361                      newurl += "&";
 362                  }
 363                  newurl += index + "=" + params[index];
 364              }
 365              // Update the anchor.
 366              pagebreaklink.set('href', newurl);
 367          }, this);
 368      },

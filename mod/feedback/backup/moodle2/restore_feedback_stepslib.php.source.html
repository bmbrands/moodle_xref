  49      protected function process_feedback($data) {
  50          global $DB;
  51  
  52          $data = (object)$data;
  53          $oldid = $data->id;
  54          $data->course = $this->get_courseid();
  55  
  56          $data->timeopen = $this->apply_date_offset($data->timeopen);
  57          $data->timeclose = $this->apply_date_offset($data->timeclose);

  44          $action = new action_link($url1, get_string('printbook', 'booktool_print'), new popup_action('click', $url1));
  45          $node->add(get_string('printbook', 'booktool_print'), $action, navigation_node::TYPE_SETTING, null, null,
  46                  new pix_icon('book', '', 'booktool_print', array('class'=>'icon')));
  47          $action = new action_link($url2, get_string('printchapter', 'booktool_print'), new popup_action('click', $url2));
  48          $node->add(get_string('printchapter', 'booktool_print'), $action, navigation_node::TYPE_SETTING, null, null,
  49                  new pix_icon('chapter', '', 'booktool_print', array('class'=>'icon')));
  50      }
  51  }

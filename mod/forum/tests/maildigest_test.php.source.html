  42       * Set up message and mail sinks, and set up other requirements for the
  43       * cron to be tested here.
  44       */
  45      public function setUp() {
  46          global $CFG;
  47  
  48          $this->helper = new stdClass();
  49  
  50          // Messaging is not compatible with transactions...
  51          $this->preventResetByRollback();
  52  
  53          // Catch all messages
  54          $this->helper->messagesink = $this->redirectMessages();
  55          $this->helper->mailsink = $this->redirectEmails();
  56  
  57          // Confirm that we have an empty message sink so far.
  58          $messages = $this->helper->messagesink->get_messages();
  59          $this->assertEquals(0, count($messages));
  60  
  61          $messages = $this->helper->mailsink->get_messages();
  62          $this->assertEquals(0, count($messages));
  63  
  64          // Tell Moodle that we've not sent any digest messages out recently.
  65          $CFG->digestmailtimelast = 0;
  66  
  67          // And set the digest sending time to a negative number - this has
  68          // the effect of making it 11pm the previous day.
  69          $CFG->digestmailtime = -1;
  70  
  71          // Forcibly reduce the maxeditingtime to a one second to ensure that
  72          // messages are sent out.
  73          $CFG->maxeditingtime = 1;
  74  
  75          // We must clear the subscription caches. This has to be done both before each test, and after in case of other
  76          // tests using these functions.
  77          \mod_forum\subscriptions::reset_forum_cache();
  78          \mod_forum\subscriptions::reset_discussion_cache();
  79      }

 135       * Helper to falsify all forum post records for a digest run.

 137      protected function helper_force_digest_mail_times() {
 138          global $CFG, $DB;
 139          // Fake all of the post editing times because digests aren't sent until
 140          // the start of an hour where the modification time on the message is before
 141          // the start of that hour
 142          $sitetimezone = core_date::get_server_timezone();
 143          $digesttime = usergetmidnight(time(), $sitetimezone) + ($CFG->digestmailtime * 3600) - (60 * 60);
 144          $DB->set_field('forum_posts', 'modified', $digesttime, array('mailed' => 0));
 145          $DB->set_field('forum_posts', 'created', $digesttime, array('mailed' => 0));
 146      }

 148      /**
 149       * Run the forum cron, and check that the specified post was sent the
 150       * specified number of times.
 151       *
 152       * @param integer $expected The number of times that the post should have been sent
 153       * @return array An array of the messages caught by the message sink
 154       */
 155      protected function helper_run_cron_check_count($expected, $messagecount, $mailcount) {
 156          if ($expected === 0) {
 157              $this->expectOutputRegex('/(Email digests successfully sent to .* users.){0}/');
 158          } else {
 159              $this->expectOutputRegex("/Email digests successfully sent to {$expected} users/");
 160          }
 161          forum_cron();
 162  
 163          // Now check the results in the message sink.
 164          $messages = $this->helper->messagesink->get_messages();
 165          // There should be the expected number of messages.
 166          $this->assertEquals($messagecount, count($messages));
 167  
 168          // Now check the results in the mail sink.
 169          $messages = $this->helper->mailsink->get_messages();
 170          // There should be the expected number of messages.
 171          $this->assertEquals($mailcount, count($messages));
 172  
 173          return $messages;
 174      }

 176      public function test_set_maildigest() {
 177          global $DB;
 178  
 179          $this->resetAfterTest(true);
 180  
 181          $helper = $this->helper_setup_user_in_course();
 182          $user = $helper->user;
 183          $course1 = $helper->courses->course1;
 184          $forum1 = $helper->forums->forum1;
 185  
 186          // Set to the user.
 187          self::setUser($helper->user);
 188  
 189          // Confirm that there is no current value.
 190          $currentsetting = $DB->get_record('forum_digests', array(
 191              'forum' => $forum1->id,
 192              'userid' => $user->id,
 193          ));
 194          $this->assertFalse($currentsetting);
 195  
 196          // Test with each of the valid values:
 197          // 0, 1, and 2 are valid values.
 198          forum_set_user_maildigest($forum1, 0, $user);
 199          $currentsetting = $DB->get_record('forum_digests', array(
 200              'forum' => $forum1->id,
 201              'userid' => $user->id,
 202          ));
 203          $this->assertEquals($currentsetting->maildigest, 0);
 204  
 205          forum_set_user_maildigest($forum1, 1, $user);
 206          $currentsetting = $DB->get_record('forum_digests', array(
 207              'forum' => $forum1->id,
 208              'userid' => $user->id,
 209          ));
 210          $this->assertEquals($currentsetting->maildigest, 1);
 211  
 212          forum_set_user_maildigest($forum1, 2, $user);
 213          $currentsetting = $DB->get_record('forum_digests', array(
 214              'forum' => $forum1->id,
 215              'userid' => $user->id,
 216          ));
 217          $this->assertEquals($currentsetting->maildigest, 2);
 218  
 219          // And the default value - this should delete the record again
 220          forum_set_user_maildigest($forum1, -1, $user);
 221          $currentsetting = $DB->get_record('forum_digests', array(
 222              'forum' => $forum1->id,
 223              'userid' => $user->id,
 224          ));
 225          $this->assertFalse($currentsetting);
 226  
 227          // Try with an invalid value.
 228          $this->setExpectedException('moodle_exception');
 229          forum_set_user_maildigest($forum1, 42, $user);
 230      }

 233          global $USER, $DB;

 245  
 246          // We test against these options.
 247          $digestoptions = array(
 248              '0' => get_string('emaildigestoffshort', 'mod_forum'),
 249              '1' => get_string('emaildigestcompleteshort', 'mod_forum'),
 250              '2' => get_string('emaildigestsubjectsshort', 'mod_forum'),
 251          );
 252  
 253          // The default settings is 0.
 254          $this->assertEquals(0, $user->maildigest);
 255          $options = forum_get_user_digest_options();
 256          $this->assertEquals($options[-1], get_string('emaildigestdefault', 'mod_forum', $digestoptions[0]));
 257  
 258          // Update the setting to 1.
 259          $USER->maildigest = 1;
 260          $this->assertEquals(1, $USER->maildigest);
 261          $options = forum_get_user_digest_options();
 262          $this->assertEquals($options[-1], get_string('emaildigestdefault', 'mod_forum', $digestoptions[1]));
 263  
 264          // Update the setting to 2.
 265          $USER->maildigest = 2;
 266          $this->assertEquals(2, $USER->maildigest);
 267          $options = forum_get_user_digest_options();
 268          $this->assertEquals($options[-1], get_string('emaildigestdefault', 'mod_forum', $digestoptions[2]));

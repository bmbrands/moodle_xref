  49      public function create_instance($record = null, array $options = null) {
  50          global $DB;
  51  
  52          if ($this->templates === null) {
  53              $this->templates = $DB->get_records_menu('survey', array('template' => 0), 'name', 'id, name');
  54          }
  55          if (empty($this->templates)) {
  56              throw new moodle_exception('cannotfindsurveytmpt', 'survey');
  57          }
  58          $record = (array)$record;
  59          if (isset($record['template']) && !is_number($record['template'])) {
  60              // Substitute template name with template id.
  61              $record['template'] = array_search($record['template'], $this->templates);
  62          }
  63          if (isset($record['template']) && !array_key_exists($record['template'], $this->templates)) {
  64              throw new moodle_exception('cannotfindsurveytmpt', 'survey');
  65          }
  66  
  67          // Add default values for survey.
  68          if (!isset($record['template'])) {
  69              reset($this->templates);
  70              $record['template'] = key($this->templates);
  71          }
  72  
  73          return parent::create_instance($record, (array)$options);
  74      }

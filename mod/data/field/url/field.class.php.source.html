 112      function display_browse_field($recordid, $template) {
 113          global $DB;
 114  
 115          if ($content = $DB->get_record('data_content', array('fieldid'=>$this->field->id, 'recordid'=>$recordid))) {
 116              $url = empty($content->content)? '':$content->content;
 117              $text = empty($content->content1)? '':$content->content1;
 118              if (empty($url) or ($url == 'http://')) {
 119                  return '';
 120              }
 121              if (!empty($this->field->param2)) {
 122                  // param2 forces the text to something
 123                  $text = $this->field->param2;
 124              }
 125              if ($this->field->param1) {
 126                  // param1 defines whether we want to autolink the url.
 127                  $attributes = array();
 128                  if ($this->field->param3) {
 129                      // param3 defines whether this URL should open in a new window.
 130                      $attributes['target'] = '_blank';
 131                  }
 132  
 133                  if (empty($text)) {
 134                      $text = $url;
 135                  }
 136  
 137                  $str = html_writer::link($url, $text, $attributes);
 138              } else {
 139                  $str = $url;
 140              }
 141              return $str;
 142          }
 143          return false;
 144      }

 146      function update_content($recordid, $value, $name='') {
 147          global $DB;
 148  
 149          $content = new stdClass();
 150          $content->fieldid = $this->field->id;
 151          $content->recordid = $recordid;
 152          $names = explode('_', $name);
 153  
 154          switch ($names[2]) {
 155              case 0:
 156                  // update link
 157                  $content->content = clean_param($value, PARAM_URL);
 158                  break;
 159              case 1:
 160                  // add text
 161                  $content->content1 = clean_param($value, PARAM_NOTAGS);
 162                  break;
 163              default:
 164                  break;
 165          }
 166  
 167          if (!empty($content->content) && (strpos($content->content, '://') === false)
 168                  && (strpos($content->content, '/') !== 0)) {
 169              $content->content = 'http://' . $content->content;
 170          }
 171  
 172          if ($oldcontent = $DB->get_record('data_content', array('fieldid'=>$this->field->id, 'recordid'=>$recordid))) {
 173              $content->id = $oldcontent->id;
 174              return $DB->update_record('data_content', $content);
 175          } else {
 176              return $DB->insert_record('data_content', $content);
 177          }
 178      }

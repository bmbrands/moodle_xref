  89          $varcharlat = $DB->sql_compare_text('content');

  91          $latlongsrs = $DB->get_recordset_sql(
  92              "SELECT DISTINCT $varcharlat AS la, $varcharlong AS lo

  95               ORDER BY $varcharlat, $varcharlong", array($this->field->id));
  96  
  97          $options = array();
  98          foreach ($latlongsrs as $latlong) {
  99              $latitude = format_float($latlong->la, 4);
 100              $longitude = format_float($latlong->lo, 4);
 101              $options[$latlong->la . ',' . $latlong->lo] = $latitude . ' ' . $longitude;

 132      function display_browse_field($recordid, $template) {
 133          global $CFG, $DB;
 134          if ($content = $DB->get_record('data_content', array('fieldid'=>$this->field->id, 'recordid'=>$recordid))) {
 135              $lat = $content->content;
 136              if (strlen($lat) < 1) {
 137                  return false;
 138              }
 139              $long = $content->content1;
 140              if (strlen($long) < 1) {
 141                  return false;
 142              }
 143              // We use format_float to display in the regional format.
 144              if($lat < 0) {
 145                  $compasslat = format_float(-$lat, 4) . '°S';
 146              } else {
 147                  $compasslat = format_float($lat, 4) . '°N';
 148              }
 149              if($long < 0) {
 150                  $compasslong = format_float(-$long, 4) . '°W';
 151              } else {
 152                  $compasslong = format_float($long, 4) . '°E';
 153              }
 154  
 155              // Now let's create the jump-to-services link
 156              $servicesshown = explode(',', $this->field->param1);
 157  
 158              // These are the different things that can be magically inserted into URL schemes
 159              $urlreplacements = array(
 160                  '@lat@'=> $lat,
 161                  '@long@'=> $long,
 162                  '@wwwroot@'=> $CFG->wwwroot,
 163                  '@contentid@'=> $content->id,
 164                  '@dataid@'=> $this->data->id,
 165                  '@courseid@'=> $this->data->course,
 166                  '@fieldid@'=> $content->fieldid,
 167                  '@recordid@'=> $content->recordid,
 168              );
 169  
 170              if(sizeof($servicesshown)==1 && $servicesshown[0]) {
 171                  $str = " <a href='"
 172                            . str_replace(array_keys($urlreplacements), array_values($urlreplacements), $this->linkoutservices[$servicesshown[0]])
 173                            ."' title='$servicesshown[0]'>$compasslat $compasslong</a>";
 174              } elseif (sizeof($servicesshown)>1) {
 175                  $str = '<form id="latlongfieldbrowse">';
 176                  $str .= "$compasslat, $compasslong\n";
 177                  $str .= "<label class='accesshide' for='jumpto'>". get_string('jumpto') ."</label>";
 178                  $str .= "<select id='jumpto' name='jumpto'>";
 179                  foreach($servicesshown as $servicename){
 180                      // Add a link to a service
 181                      $str .= "\n  <option value='"
 182                                 . str_replace(array_keys($urlreplacements), array_values($urlreplacements), $this->linkoutservices[$servicename])
 183                                 . "'>".htmlspecialchars($servicename)."</option>";
 184                  }
 185                  // NB! If you are editing this, make sure you don't break the javascript reference "previousSibling"
 186                  //   which allows the "Go" button to refer to the drop-down selector.
 187                  $str .= "\n</select><input type='button' value='" . get_string('go') . "' onclick='if(previousSibling.value){self.location=previousSibling.value}'/>";
 188                  $str .= '</form>';
 189              } else {
 190                  $str = "$compasslat, $compasslong";
 191              }
 192  
 193              return $str;
 194          }
 195          return false;
 196      }

 198      function update_content($recordid, $value, $name='') {
 199          global $DB;
 200  
 201          $content = new stdClass();
 202          $content->fieldid = $this->field->id;
 203          $content->recordid = $recordid;
 204          // When updating these values (which might be region formatted) we should format
 205          // the float to allow for a consistent float format in the database.
 206          $value = unformat_float($value);
 207          $value = trim($value);
 208          if (strlen($value) > 0) {
 209              $value = floatval($value);
 210          } else {
 211              $value = null;
 212          }
 213          $names = explode('_', $name);
 214          switch ($names[2]) {
 215              case 0:
 216                  // update lat
 217                  $content->content = $value;
 218                  break;
 219              case 1:
 220                  // update long
 221                  $content->content1 = $value;
 222                  break;
 223              default:
 224                  break;
 225          }
 226          if ($oldcontent = $DB->get_record('data_content', array('fieldid'=>$this->field->id, 'recordid'=>$recordid))) {
 227              $content->id = $oldcontent->id;
 228              return $DB->update_record('data_content', $content);
 229          } else {
 230              return $DB->insert_record('data_content', $content);
 231          }
 232      }

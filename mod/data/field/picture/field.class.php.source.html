 186      function update_field() {
 187          global $DB, $OUTPUT;
 188  
 189          // Get the old field data so that we can check whether the thumbnail dimensions have changed
 190          $oldfield = $DB->get_record('data_fields', array('id'=>$this->field->id));
 191          $DB->update_record('data_fields', $this->field);
 192  
 193          // Have the thumbnail dimensions changed?
 194          if ($oldfield && ($oldfield->param4 != $this->field->param4 || $oldfield->param5 != $this->field->param5)) {
 195              // Check through all existing records and update the thumbnail
 196              if ($contents = $DB->get_records('data_content', array('fieldid'=>$this->field->id))) {
 197                  $fs = get_file_storage();
 198                  if (count($contents) > 20) {
 199                      echo $OUTPUT->notification(get_string('resizingimages', 'data'), 'notifysuccess');
 200                      echo "\n\n";
 201                      // To make sure that ob_flush() has the desired effect
 202                      ob_flush();
 203                  }
 204                  foreach ($contents as $content) {
 205                      if (!$file = $fs->get_file($this->context->id, 'mod_data', 'content', $content->id, '/', $content->content)) {
 206                          continue;
 207                      }
 208                      if ($thumbfile = $fs->get_file($this->context->id, 'mod_data', 'content', $content->id, '/', 'thumb_'.$content->content)) {
 209                          $thumbfile->delete();
 210                      }
 211                      core_php_time_limit::raise(300);
 212                      // Might be slow!
 213                      $this->update_thumbnail($content, $file);
 214                  }
 215              }
 216          }
 217          return true;

 220      function update_content($recordid, $value, $name='') {
 221          global $CFG, $DB, $USER;
 222  
 223          if (!$content = $DB->get_record('data_content', array('fieldid'=>$this->field->id, 'recordid'=>$recordid))) {
 224          // Quickly make one now!
 225              $content = new stdClass();
 226              $content->fieldid  = $this->field->id;
 227              $content->recordid = $recordid;
 228              $id = $DB->insert_record('data_content', $content);
 229              $content = $DB->get_record('data_content', array('id'=>$id));
 230          }
 231  
 232          $names = explode('_', $name);
 233          switch ($names[2]) {
 234              case 'file':
 235                  $fs = get_file_storage();
 236                  $fs->delete_area_files($this->context->id, 'mod_data', 'content', $content->id);
 237                  $usercontext = context_user::instance($USER->id);
 238                  $files = $fs->get_area_files($usercontext->id, 'user', 'draft', $value);
 239                  if (count($files)<2) {
 240                      // no file
 241                  } else {
 242                      $count = 0;
 243                      foreach ($files as $draftfile) {
 244                          $file_record = array('contextid'=>$this->context->id, 'component'=>'mod_data', 'filearea'=>'content', 'itemid'=>$content->id, 'filepath'=>'/');
 245                          if (!$draftfile->is_directory()) {
 246                              $file_record['filename'] = $draftfile->get_filename();
 247  
 248                              $content->content = $draftfile->get_filename();
 249  
 250                              $file = $fs->create_file_from_storedfile($file_record, $draftfile);
 251  
 252                              // If the file is not a valid image, redirect back to the upload form.
 253                              if ($file->get_imageinfo() === false) {
 254                                  $url = new moodle_url('/mod/data/edit.php', array('d' => $this->field->dataid));
 255                                  redirect($url, get_string('invalidfiletype', 'error', $file->get_filename()));
 256                              }
 257  
 258                              $DB->update_record('data_content', $content);
 259                              $this->update_thumbnail($content, $file);
 260  
 261                              if ($count > 0) {
 262                                  break;
 263                              } else {
 264                                  $count++;
 265                              }
 266                          }
 267                      }
 268                  }
 269  
 270                  break;
 271  
 272              case 'alttext':
 273                  // only changing alt tag
 274                  $content->content1 = clean_param($value, PARAM_NOTAGS);
 275                  $DB->update_record('data_content', $content);
 276                  break;
 277  
 278              default:
 279                  break;
 280          }
 281      }

 283      function update_thumbnail($content, $file) {
 284          // (Re)generate thumbnail image according to the dimensions specified in the field settings.
 285          // If thumbnail width and height are BOTH not specified then no thumbnail is generated, and
 286          // additionally an attempted delete of the existing thumbnail takes place.
 287          $fs = get_file_storage();
 288          $file_record = array('contextid'=>$file->get_contextid(), 'component'=>$file->get_component(), 'filearea'=>$file->get_filearea(),
 289                               'itemid'=>$file->get_itemid(), 'filepath'=>$file->get_filepath(),
 290                               'filename'=>'thumb_'.$file->get_filename(), 'userid'=>$file->get_userid());
 291          try {
 292              // this may fail for various reasons
 293              $fs->convert_image($file_record, $file, $this->field->param4, $this->field->param5, true);
 294              return true;
 295          } catch (Exception $e) {
 296              debugging($e->getMessage());
 297              return false;
 298          }
 299      }

 316      function notemptyfield($value, $name) {
 317          global $USER;
 318  
 319          $names = explode('_', $name);
 320          if ($names[2] == 'file') {
 321              $usercontext = context_user::instance($USER->id);
 322              $fs = get_file_storage();
 323              $files = $fs->get_area_files($usercontext->id, 'user', 'draft', $value);
 324              return count($files) >= 2;
 325          }
 326          return false;

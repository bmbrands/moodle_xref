 185      function update_content($recordid, $value, $name='') {
 186          global $DB;
 187  
 188          $content = new stdClass();
 189          $content->fieldid = $this->field->id;
 190          $content->recordid = $recordid;
 191  
 192          $names = explode('_', $name);
 193          if (!empty($names[2])) {
 194              if ($names[2] == 'itemid') {
 195                  // the value will be retrieved by file_get_submitted_draft_itemid, do not need to save in DB
 196                  return true;
 197              } else {
 198                  $content->$names[2] = clean_param($value, PARAM_NOTAGS);  // content[1-4]
 199              }
 200          } else {
 201              $content->content = clean_param($value, PARAM_CLEAN);
 202          }
 203  
 204          if ($oldcontent = $DB->get_record('data_content', array('fieldid'=>$this->field->id, 'recordid'=>$recordid))) {
 205              $content->id = $oldcontent->id;
 206          } else {
 207              $content->id = $DB->insert_record('data_content', $content);
 208              if (!$content->id) {
 209                  return false;
 210              }
 211          }
 212          if (!empty($content->content)) {
 213              $draftitemid = file_get_submitted_draft_itemid('field_'. $this->field->id. '_itemid');
 214              $options = $this->get_options();
 215              $content->content = file_save_draft_area_files($draftitemid, $this->context->id, 'mod_data', 'content', $content->id, $options, $content->content);
 216          }
 217          $rv = $DB->update_record('data_content', $content);
 218          return $rv;
 219      }

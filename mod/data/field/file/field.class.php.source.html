 163      function update_content($recordid, $value, $name='') {
 164          global $CFG, $DB, $USER;
 165          $fs = get_file_storage();
 166  
 167          if (!$content = $DB->get_record('data_content', array('fieldid'=>$this->field->id, 'recordid'=>$recordid))) {
 168  
 169          // Quickly make one now!
 170              $content = new stdClass();
 171              $content->fieldid  = $this->field->id;
 172              $content->recordid = $recordid;
 173              $id = $DB->insert_record('data_content', $content);
 174              $content = $DB->get_record('data_content', array('id'=>$id));
 175          }
 176  
 177          // delete existing files
 178          $fs->delete_area_files($this->context->id, 'mod_data', 'content', $content->id);
 179  
 180          $usercontext = context_user::instance($USER->id);
 181          $files = $fs->get_area_files($usercontext->id, 'user', 'draft', $value, 'timecreated DESC');
 182  
 183          if (count($files)<2) {
 184              // no file
 185          } else {
 186              foreach ($files as $draftfile) {
 187                  if (!$draftfile->is_directory()) {
 188                      $file_record = array(
 189                          'contextid' => $this->context->id,
 190                          'component' => 'mod_data',
 191                          'filearea' => 'content',
 192                          'itemid' => $content->id,
 193                          'filepath' => '/',
 194                          'filename' => $draftfile->get_filename(),
 195                      );
 196  
 197                      $content->content = $file_record['filename'];
 198  
 199                      $fs->create_file_from_storedfile($file_record, $draftfile);
 200                      $DB->update_record('data_content', $content);
 201  
 202                      // Break from the loop now to avoid overwriting the uploaded file record
 203                      break;
 204                  }
 205              }
 206          }

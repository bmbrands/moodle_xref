 180      /**
 181       * Test 1: The function data_get_all_recordids.
 182       *
 183       * Test 2: This tests the data_get_advance_search_ids() function. The function takes a set
 184       * of all the record IDs in the database and then with the search details ($this->recordsearcharray)
 185       * returns a comma seperated string of record IDs that match the search criteria.
 186       *
 187       * Test 3: This function tests data_get_recordids(). This is the function that is nested in the last
 188       * function (@see data_get_advance_search_ids). This function takes a couple of
 189       * extra parameters. $alias is the field alias used in the sql query and $commaid
 190       * is a comma seperated string of record IDs.
 191       *
 192       * Test 4: data_get_advanced_search_sql provides an array which contains an sql string to be used for displaying records
 193       * to the user when they use the advanced search criteria and the parameters that go with the sql statement. This test
 194       * takes that information and does a search on the database, returning a record.
 195       *
 196       * Test 5: Returning to data_get_all_recordids(). Here we are ensuring that the total amount of record ids is reduced to
 197       * match the group conditions that are provided. There are 25 entries which relate to group 2. They are removed
 198       * from the total so we should only have 75 records total.
 199       *
 200       * Test 6: data_get_all_recordids() again. This time we are testing approved database records. We only want to
 201       * display the records that have been approved. In this record set we have 89 approved records.
 202       */
 203      function test_advanced_search_sql_section() {
 204          global $DB;
 205  
 206          // Test 1
 207          $recordids = data_get_all_recordids($this->recorddata->id);
 208          $this->assertEquals(count($recordids), $this->datarecordcount);
 209  
 210          // Test 2
 211          $key = array_keys($this->recordsearcharray);
 212          $alias = $key[0];
 213          $newrecordids = data_get_recordids($alias, $this->recordsearcharray, $this->recorddata->id, $recordids);
 214          $this->assertEquals($this->datarecordset, $newrecordids);
 215  
 216          // Test 3
 217          $newrecordids = data_get_advance_search_ids($recordids, $this->recordsearcharray, $this->recorddata->id);
 218          $this->assertEquals($this->datarecordset, $newrecordids);
 219  
 220          // Test 4
 221          $sortorder = 'ORDER BY r.timecreated ASC , r.id ASC';
 222          $html = data_get_advanced_search_sql('0', $this->recorddata, $newrecordids, '', $sortorder);
 223          $allparams = array_merge($html['params'], array('dataid' => $this->recorddata->id));
 224          $records = $DB->get_records_sql($html['sql'], $allparams);
 225          $this->assertEquals($records, $this->finalrecord);
 226  
 227          // Test 5
 228          $groupsql = " AND (r.groupid = :currentgroup OR r.groupid = 0)";
 229          $params = array('currentgroup' => 1);
 230          $recordids = data_get_all_recordids($this->recorddata->id, $groupsql, $params);
 231          $this->assertEquals($this->groupdatarecordcount, count($recordids));
 232  
 233          // Test 6
 234          $approvesql = ' AND r.approved=1 ';
 235          $recordids = data_get_all_recordids($this->recorddata->id, $approvesql, $params);
 236          $this->assertEquals($this->approvedatarecordcount, count($recordids));
 237      }

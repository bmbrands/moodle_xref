 138      private function getTokens($format, $token) {
 139          $tokens = wiki_parser_get_token($format, $token);
 140  
 141          if (is_array($tokens)) {
 142              foreach ($tokens as & $t) {
 143                  $this->escapeToken($t);
 144              }
 145          } else {
 146              $this->escapeToken($tokens);
 147          }
 148  
 149          return $tokens;
 150      }
 151  
 152      private function escapeToken(&$token) {
 153          $token = urlencode(str_replace("'", "\'", $token));
 154      }

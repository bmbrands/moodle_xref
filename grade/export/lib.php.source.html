 147      function process_form($formdata) {
 148          global $USER;
 149  
 150          $this->columns = array();
 151          if (!empty($formdata->itemids)) {
 152              if ($formdata->itemids=='-1') {
 153                  //user deselected all items
 154              } else {
 155                  foreach ($formdata->itemids as $itemid=>$selected) {
 156                      if ($selected and array_key_exists($itemid, $this->grade_items)) {
 157                          $this->columns[$itemid] =& $this->grade_items[$itemid];
 158                      }
 159                  }
 160              }
 161          } else {
 162              foreach ($this->grade_items as $itemid=>$unused) {
 163                  $this->columns[$itemid] =& $this->grade_items[$itemid];
 164              }
 165          }
 166  
 167          if (isset($formdata->key)) {
 168              if ($formdata->key == 1 && isset($formdata->iprestriction) && isset($formdata->validuntil)) {
 169                  // Create a new key
 170                  $formdata->key = create_user_key('grade/export', $USER->id, $this->course->id, $formdata->iprestriction, $formdata->validuntil);
 171              }
 172              $this->userkey = $formdata->key;
 173          }
 174  
 175          if (isset($formdata->decimals)) {
 176              $this->decimalpoints = $formdata->decimals;
 177          }
 178  
 179          if (isset($formdata->export_letters)) {
 180              $this->export_letters = $formdata->export_letters;
 181          }
 182  
 183          if (isset($formdata->export_feedback)) {
 184              $this->export_feedback = $formdata->export_feedback;
 185          }
 186  
 187          if (isset($formdata->export_onlyactive)) {
 188              $this->onlyactive = $formdata->export_onlyactive;
 189          }
 190  
 191          if (isset($formdata->previewrows)) {
 192              $this->previewrows = $formdata->previewrows;
 193          }
 194  
 195          if (isset($formdata->display)) {
 196              $this->displaytype = $formdata->display;
 197  
 198              // Used by grade exports which accept multiple display types.
 199              // If the checkbox value is 0 (unchecked) then remove it.
 200              if (is_array($formdata->display)) {
 201                  $this->displaytype = array_filter($formdata->display);
 202              }
 203          }
 204  
 205          if (isset($formdata->updatedgradesonly)) {

 288

 315          $gui->require_active_enrolment($this->onlyactive);
 316          $gui->allow_user_custom_fields($this->usercustomfields);

 318          while ($userdata = $gui->next_user()) {
 319              // number of preview rows
 320              if ($this->previewrows and $this->previewrows <= $i) {
 321                  break;

 360                  echo '<td>' . format_text($fieldvalue, FORMAT_MOODLE, $formatoptions) . '</td>';

 363                  $issuspended = ($user->suspendedenrolment) ? get_string('yes') : '';
 364                  echo "<td>$issuspended</td>";

 465      /**
 466       * Convert the grade display types parameter into the required array to grade exporting class.
 467       *
 468       * In order to export, the array key must be the display type name and the value must be the grade display type
 469       * constant.
 470       *
 471       * Note: Added support for combined display types constants like the (GRADE_DISPLAY_TYPE_PERCENTAGE_REAL) as
 472       *       the $CFG->grade_export_displaytype config is still used on 2.7 in case of missing displaytype url param.
 473       *       In these cases, the file will be exported with a column for each display type.
 474       *
 475       * @param string $displaytypes can be a single or multiple display type constants comma separated.
 476       * @return array $types
 477       */
 478      public static function convert_flat_displaytypes_to_array($displaytypes) {
 479          $types = array();
 480  
 481          // We have a single grade display type constant.
 482          if (is_int($displaytypes)) {
 483              $displaytype = clean_param($displaytypes, PARAM_INT);
 484  
 485              // Let's set a default value, will be replaced below by the grade display type constant.
 486              $display[$displaytype] = 1;
 487          } else {
 488              // Multiple grade display types constants.
 489              $display = array_flip(explode(',', $displaytypes));
 490          }
 491  
 492          // Now, create the array in the required format by grade exporting class.
 493          foreach ($display as $type => $value) {
 494              $type = clean_param($type, PARAM_INT);
 495              if ($type == GRADE_DISPLAY_TYPE_LETTER) {
 496                  $types['letter'] = GRADE_DISPLAY_TYPE_LETTER;
 497              } else if ($type == GRADE_DISPLAY_TYPE_PERCENTAGE) {
 498                  $types['percentage'] = GRADE_DISPLAY_TYPE_PERCENTAGE;
 499              } else if ($type == GRADE_DISPLAY_TYPE_REAL) {
 500                  $types['real'] = GRADE_DISPLAY_TYPE_REAL;
 501              } else if ($type == GRADE_DISPLAY_TYPE_REAL_PERCENTAGE) {
 502                  $types['real'] = GRADE_DISPLAY_TYPE_REAL;
 503                  $types['percentage'] = GRADE_DISPLAY_TYPE_PERCENTAGE;
 504              } else if ($type == GRADE_DISPLAY_TYPE_REAL_LETTER) {
 505                  $types['real'] = GRADE_DISPLAY_TYPE_REAL;
 506                  $types['letter'] = GRADE_DISPLAY_TYPE_LETTER;
 507              } else if ($type == GRADE_DISPLAY_TYPE_LETTER_REAL) {
 508                  $types['letter'] = GRADE_DISPLAY_TYPE_LETTER;
 509                  $types['real'] = GRADE_DISPLAY_TYPE_REAL;
 510              } else if ($type == GRADE_DISPLAY_TYPE_LETTER_PERCENTAGE) {
 511                  $types['letter'] = GRADE_DISPLAY_TYPE_LETTER;
 512                  $types['percentage'] = GRADE_DISPLAY_TYPE_PERCENTAGE;
 513              } else if ($type == GRADE_DISPLAY_TYPE_PERCENTAGE_LETTER) {
 514                  $types['percentage'] = GRADE_DISPLAY_TYPE_PERCENTAGE;
 515                  $types['letter'] = GRADE_DISPLAY_TYPE_LETTER;
 516              } else if ($type == GRADE_DISPLAY_TYPE_PERCENTAGE_REAL) {
 517                  $types['percentage'] = GRADE_DISPLAY_TYPE_PERCENTAGE;
 518                  $types['real'] = GRADE_DISPLAY_TYPE_REAL;
 519              }
 520          }
 521          return $types;
 522      }

 616      public function flush($buffersize) {
 617          global $CFG, $DB;
 618  
 619          if (count($this->update_list) > $buffersize) {

 638                  // grade does not exist yet
 639                  $status = 'unknown';

 642                  $this->update_list[] = $grade_grade->id;

 644

 646              $status = 'regrade';
 647              $this->update_list[] = $grade_grade->id;
 648  
 649          } else if ($grade_grade->exported >= $grade_grade->timemodified) {
 650              $status = 'nochange';
 651  
 652          } else {
 653              // something is wrong?
 654              $status = 'unknown';

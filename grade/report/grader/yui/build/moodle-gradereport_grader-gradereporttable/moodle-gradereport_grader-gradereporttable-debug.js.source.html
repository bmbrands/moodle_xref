 390      /**
 391       * Setup the grader report table.
 392       *
 393       * @method setupFloatingHeaders
 394       * @chainable

 396      setupFloatingHeaders: function() {
 397          // Grab references to commonly used Nodes.
 398          this.firstUserCell = Y.one(SELECTORS.USERCELL);
 399          this.container = Y.one(SELECTORS.GRADEPARENT);
 400          this.firstNonUserCell = Y.one(SELECTORS.GRADECELL);
 401  
 402          if (!this.firstUserCell) {
 403              // No need for floating elements, there are no users.
 404              return this;
 405          }
 406  
 407          if (M.cfg.behatsiterunning) {
 408              // If the behat site is running we don't want floating elements.
 409              return;
 410          }
 411  
 412          // Generate floating elements.
 413          this._setupFloatingUserColumn();
 414          this._setupFloatingUserHeader();
 415          this._setupFloatingAssignmentHeaders();
 416          this._setupFloatingAssignmentFooter();
 417  
 418          // Setup generic floating left-aligned headers.
 419          this.floatingHeaderRow = {};
 420  
 421          // The 'Controls' row (shown in editing mode when certain options are set).
 422          this._setupFloatingLeftHeaders('.controls .controls');
 423  
 424          // The 'Range' row (shown in editing mode when certain options are set).
 425          this._setupFloatingLeftHeaders('.range .range');
 426  
 427          // The 'Overall Average' field.
 428          this._setupFloatingLeftHeaders(SELECTORS.FOOTERTITLE);
 429  
 430          // Additional setup for the footertitle.
 431          this._setupFloatingAssignmentFooterTitle();
 432  
 433          // Calculate the positions of edge cells. These are used for positioning of the floating headers.
 434          // This must be called after the floating headers are setup, but before the scroll event handler is invoked.
 435          this._calculateCellPositions();
 436  
 437          // Setup the floating element initial positions by simulating scroll.
 438          this._handleScrollEvent();
 439  
 440          // Setup the event handlers.
 441          this._setupEventHandlers();
 442  
 443          // Listen for a resize event globally - other parts of the code not in this YUI wrapper may make changes to the
 444          // fields which result in size changes.
 445          Y.Global.on('moodle-gradereport_grader:resized', this._handleResizeEvent, this);
 446  
 447          return this;
 448      },

 450      /**
 451       * Calculate the positions of some cells. These values are used heavily
 452       * in scroll event handling.
 453       *
 454       * @method _calculateCellPositions
 455       * @protected
 456       */
 457      _calculateCellPositions: function() {
 458          // The header row shows the grade item headers and is floated to the top of the window.
 459          this.headerRowTop = this.headerRow.getY();
 460  
 461          // The footer row shows the grade averages and will be floated to the page bottom.
 462          if (this.tableFooterRow) {
 463              this.footerRowPosition = this.tableFooterRow.getY();
 464          }
 465  
 466          // Add the width of the dock if it is visible.
 467          this.dockWidth = 0;
 468          var dock = Y.one('.has_dock #dock');
 469          if (dock) {
 470              this.dockWidth = dock.get(OFFSETWIDTH);
 471          }
 472  
 473          var userCellList = Y.all(SELECTORS.USERCELL);
 474  
 475          // The left of the user cells matches the left of the headerRow.
 476          this.firstUserCellLeft = this.firstUserCell.getX();
 477          this.firstUserCellWidth = this.firstUserCell.get(OFFSETWIDTH);
 478  
 479          // The left of the user cells matches the left of the footer title.
 480          this.firstNonUserCellLeft = this.firstNonUserCell.getX();
 481          this.firstNonUserCellWidth = this.firstNonUserCell.get(OFFSETWIDTH);
 482  
 483          if (userCellList.size() > 1) {
 484              // Use the top of the second cell for the bottom of the first cell.
 485              // This is used when scrolling to fix the footer to the top edge of the window.
 486              var firstUserCell = userCellList.item(1);
 487              this.firstUserCellBottom = firstUserCell.getY() + parseInt(firstUserCell.getComputedStyle(HEIGHT), 10);
 488  
 489              // Use the top of the penultimate cell when scrolling the header.
 490              // The header is the same size as the cells.
 491              this.lastUserCellTop = userCellList.item(userCellList.size() - 2).getY();
 492          } else {
 493              var firstItem = userCellList.item(0);
 494              // We can't use the top of the second row as there is only one row.
 495              this.lastUserCellTop = firstItem.getY();
 496  
 497              if (this.tableFooterRow) {
 498                  // The footer is present so we can use that.
 499                  this.firstUserCellBottom = this.footerRowPosition + parseInt(this.tableFooterRow.getComputedStyle(HEIGHT), 10);
 500              } else {
 501                  // No other clues - calculate the top instead.
 502                  this.firstUserCellBottom = firstItem.getY() + firstItem.get('offsetHeight');
 503              }
 504          }
 505  
 506          // Check whether a header is present and whether it is floating.
 507          var header = Y.one('header');
 508          this.pageHeaderHeight = 0;
 509          if (header) {
 510              if (header.getComputedStyle('position') === 'fixed') {
 511                  this.pageHeaderHeight = header.get(OFFSETHEIGHT);
 512              } else {
 513                  var navbar = Y.one('.navbar');
 514  
 515                  if (navbar && navbar.getComputedStyle('position') === 'fixed') {
 516                      // If the navbar exists and isn't fixed, we need to offset the page header to accommodate for it.
 517                      this.pageHeaderHeight = navbar.get(OFFSETHEIGHT);
 518                  }
 519              }
 520          }
 521      },

 634                  var marginHeight = parseInt(node.getComputedStyle('marginTop'), 10) +

 638                  var borderHeight = parseInt(node.getComputedStyle('borderTopWidth'), 10) +
 639                      parseInt(node.getComputedStyle('borderBottomWidth'), 10);
 640                  height = allHeight - marginHeight - paddingHeight - borderHeight;
 641              }
 642              // Create and configure the new container.

 645                      .setAttribute('class', node.getAttribute('class'))

 653              floatingUserColumn.appendChild(containerNode);
 654          }, this);

 676      _setupFloatingUserHeader: function() {
 677          // We make various references to the header cells. Store it for later.
 678          this.headerRow = Y.one(SELECTORS.HEADERROW);
 679          this.headerCell = Y.one(SELECTORS.STUDENTHEADER);

 685              nodepos = this._getRelativeXY(this.headerCell)[0],
 686              coordinates = this._getRelativeXY(this.headerRow),
 687              gradeHeadersOffset = coordinates[0];
 688  
 689          // Append the content and style to the floating cell.
 690          floatingUserHeaderCell
 691              .set('innerHTML', this.headerCell.getHTML())
 692              .setAttribute('class', this.headerCell.getAttribute('class'))
 693              .setStyles({
 694                  // The header is larger than the user cells, so we take the user cell.
 695                  width:      this.firstUserCell.getComputedStyle(WIDTH),
 696                  left:       (nodepos - gradeHeadersOffset) + 'px'
 697              });
 698  
 699          // Style the floating row.

 701              .setStyles({
 702                  left:       coordinates[0] + 'px',
 703                  position:   'absolute',
 704                  top:        coordinates[1] + 'px'
 705              });
 706  
 707          // Append the cell to the row, and finally to the region.
 708          floatingUserHeaderRow.append(floatingUserHeaderCell);
 709          this.graderRegion.append(floatingUserHeaderRow);
 710

 901       * @protected
 902       */
 903      _handleScrollEvent: function() {
 904          // Performance is important in this function as it is called frequently and in quick succesion.
 905          // To prevent layout thrashing when the DOM is repeatedly updated and queried, updated and queried,
 906          // updates must be batched.
 907  
 908          // Next do all the calculations.
 909          var gradeItemHeadingContainerStyles = {},
 910              userColumnHeaderStyles = {},
 911              userColumnStyles = {},
 912              footerStyles = {},
 913              coord = 0,
 914              floatingUserTriggerPoint = 0,       // The X position at which the floating should start.
 915              floatingUserRelativePoint = 0,      // The point to use when calculating the new position.
 916              headerFloats = false,
 917              userFloats = false,
 918              footerFloats = false,
 919              leftTitleFloats = false,
 920              floatingHeaderStyles = {},
 921              floatingFooterTitleStyles = {},
 922              floatingFooterTitleRow = false;
 923  
 924          // Header position.
 925          gradeItemHeadingContainerStyles.left = this._getRelativeXFromX(this.headerRow.getX());
 926          if (Y.config.win.pageYOffset + this.pageHeaderHeight > this.headerRowTop) {
 927              headerFloats = true;
 928              if (Y.config.win.pageYOffset + this.pageHeaderHeight < this.lastUserCellTop) {
 929                  coord = this._getRelativeYFromY(Y.config.win.pageYOffset + this.pageHeaderHeight);
 930                  gradeItemHeadingContainerStyles.top = coord + 'px';
 931                  userColumnHeaderStyles.top = coord + 'px';
 932              } else {
 933                  coord = this._getRelativeYFromY(this.lastUserCellTop);
 934                  gradeItemHeadingContainerStyles.top = coord + 'px';
 935                  userColumnHeaderStyles.top = coord + 'px';
 936              }
 937          } else {
 938              headerFloats = false;
 939              coord = this._getRelativeYFromY(this.headerRowTop);
 940              gradeItemHeadingContainerStyles.top = coord + 'px';
 941              userColumnHeaderStyles.top = coord + 'px';
 942          }
 943  
 944          // User column position.
 945          if (window.right_to_left()) {
 946              floatingUserTriggerPoint = Y.config.win.innerWidth + Y.config.win.pageXOffset - this.dockWidth;
 947              floatingUserRelativePoint = floatingUserTriggerPoint - this.firstUserCellWidth;
 948              userFloats = floatingUserTriggerPoint < (this.firstUserCellLeft + this.firstUserCellWidth);
 949              leftTitleFloats = (floatingUserTriggerPoint - this.firstNonUserCellWidth) <
 950                                (this.firstNonUserCellLeft + this.firstUserCellWidth);
 951          } else {
 952              floatingUserRelativePoint = Y.config.win.pageXOffset;
 953              floatingUserTriggerPoint = floatingUserRelativePoint + this.dockWidth;
 954              userFloats = floatingUserTriggerPoint > this.firstUserCellLeft;
 955              leftTitleFloats = floatingUserTriggerPoint > (this.firstNonUserCellLeft - this.firstUserCellWidth);
 956          }
 957  
 958          if (userFloats) {
 959              coord = this._getRelativeXFromX(floatingUserRelativePoint);
 960              userColumnStyles.left = coord + 'px';
 961              userColumnHeaderStyles.left = coord + 'px';
 962          } else {
 963              coord = this._getRelativeXFromX(this.firstUserCellLeft);
 964              userColumnStyles.left = coord + 'px';
 965              userColumnHeaderStyles.left = coord + 'px';
 966          }
 967  
 968          // Update the miscellaneous left-only floats.
 969          Y.Object.each(this.floatingHeaderRow, function(origin, key) {
 970              floatingHeaderStyles[key] = {
 971                  left: userColumnStyles.left
 972              };
 973          }, this);
 974  
 975          // Update footer.
 976          if (this.footerRow) {
 977              footerStyles.left = this._getRelativeXFromX(this.headerRow.getX());
 978  
 979              // Determine whether the footer should now be shown as sticky.
 980              var pageHeight = Y.config.win.innerHeight,
 981                  pageOffset = Y.config.win.pageYOffset,
 982                  bottomScrollPosition = pageHeight - this._getScrollBarHeight() + pageOffset,
 983                  footerRowHeight = parseInt(this.footerRow.getComputedStyle(HEIGHT), 10),
 984                  footerBottomPosition = footerRowHeight + this.footerRowPosition;
 985

 987              floatingFooterTitleRow = this.floatingHeaderRow[SELECTORS.FOOTERTITLE];
 988              if (bottomScrollPosition < footerBottomPosition && bottomScrollPosition > this.firstUserCellBottom) {
 989                  // We have not scrolled below the footer, nor above the first row.
 990                  footerStyles.bottom = Math.ceil(footerBottomPosition - bottomScrollPosition) + 'px';
 991                  footerFloats = true;
 992              } else {
 993                  // The footer should not float any more.
 994                  footerStyles.bottom = '1px';
 995                  footerFloats = false;
 996              }
 997              if (floatingFooterTitleStyles) {
 998                  floatingFooterTitleStyles.bottom = footerStyles.bottom;
 999                  floatingFooterTitleStyles.top = null;
1000              }
1001              floatingHeaderStyles[SELECTORS.FOOTERTITLE] = floatingFooterTitleStyles;

1003

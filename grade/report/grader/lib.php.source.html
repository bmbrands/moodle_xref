 163      /**
 164       * Processes the data sent by the form (grades and feedbacks).
 165       * Caller is responsible for all access control checks
 166       * @param array $data form submission (with magic quotes)
 167       * @return array empty array if success, array of warnings if something fails.
 168       */
 169      public function process_data($data) {
 170          global $DB;
 171          $warnings = array();
 172  
 173          $separategroups = false;
 174          $mygroups       = array();
 175          if ($this->groupmode == SEPARATEGROUPS and !has_capability('moodle/site:accessallgroups', $this->context)) {
 176              $separategroups = true;
 177              $mygroups = groups_get_user_groups($this->course->id);
 178              $mygroups = $mygroups[0]; // ignore groupings
 179              // reorder the groups fro better perf below
 180              $current = array_search($this->currentgroup, $mygroups);
 181              if ($current !== false) {
 182                  unset($mygroups[$current]);
 183                  array_unshift($mygroups, $this->currentgroup);
 184              }
 185          }
 186  
 187          // always initialize all arrays
 188          $queue = array();
 189  
 190          $this->load_users();
 191          $this->load_final_grades();
 192  
 193          // Were any changes made?
 194          $changedgrades = false;
 195          $timepageload = clean_param($data->timepageload, PARAM_INT);
 196  
 197          foreach ($data as $varname => $students) {
 198  
 199              $needsupdate = false;
 200  
 201              // skip, not a grade nor feedback
 202              if (strpos($varname, 'grade') === 0) {
 203                  $datatype = 'grade';
 204              } else if (strpos($varname, 'feedback') === 0) {
 205                  $datatype = 'feedback';
 206              } else {
 207                  continue;
 208              }
 209  
 210              foreach ($students as $userid => $items) {
 211                  $userid = clean_param($userid, PARAM_INT);
 212                  foreach ($items as $itemid => $postedvalue) {
 213                      $itemid = clean_param($itemid, PARAM_INT);
 214  
 215                      // Was change requested?
 216                      $oldvalue = $this->grades[$userid][$itemid];
 217                      if ($datatype === 'grade') {
 218                          // If there was no grade and there still isn't
 219                          if (is_null($oldvalue->finalgrade) && $postedvalue == -1) {
 220                              // -1 means no grade
 221                              continue;
 222                          }
 223  
 224                          // If the grade item uses a custom scale
 225                          if (!empty($oldvalue->grade_item->scaleid)) {
 226  
 227                              if ((int)$oldvalue->finalgrade === (int)$postedvalue) {
 228                                  continue;
 229                              }
 230                          } else {
 231                              // The grade item uses a numeric scale
 232  
 233                              // Format the finalgrade from the DB so that it matches the grade from the client
 234                              if ($postedvalue === format_float($oldvalue->finalgrade, $oldvalue->grade_item->get_decimals())) {
 235                                  continue;
 236                              }
 237                          }
 238  
 239                          $changedgrades = true;
 240  
 241                      } else if ($datatype === 'feedback') {
 242                          // If quick grading is on, feedback needs to be compared without line breaks.
 243                          if ($this->get_pref('quickgrading')) {
 244                              $oldvalue->feedback = preg_replace("/\r\n|\r|\n/", "", $oldvalue->feedback);
 245                          }
 246                          if (($oldvalue->feedback === $postedvalue) or ($oldvalue->feedback === null and empty($postedvalue))) {
 247                              continue;
 248                          }
 249                      }
 250  
 251                      if (!$gradeitem = grade_item::fetch(array('id'=>$itemid, 'courseid'=>$this->courseid))) {
 252                          print_error('invalidgradeitemid');
 253                      }
 254  
 255                      // Pre-process grade
 256                      if ($datatype == 'grade') {
 257                          $feedback = false;
 258                          $feedbackformat = false;
 259                          if ($gradeitem->gradetype == GRADE_TYPE_SCALE) {
 260                              if ($postedvalue == -1) { // -1 means no grade
 261                                  $finalgrade = null;
 262                              } else {
 263                                  $finalgrade = $postedvalue;
 264                              }
 265                          } else {
 266                              $finalgrade = unformat_float($postedvalue);
 267                          }
 268  
 269                          $errorstr = '';
 270                          $skip = false;
 271  
 272                          $dategraded = $oldvalue->get_dategraded();
 273                          if (!empty($dategraded) && $timepageload < $dategraded) {
 274                              // Warn if the grade was updated while we were editing this form.
 275                              $errorstr = 'gradewasmodifiedduringediting';
 276                              $skip = true;
 277                          } else if (!is_null($finalgrade)) {
 278                              // Warn if the grade is out of bounds.
 279                              $bounded = $gradeitem->bounded_grade($finalgrade);
 280                              if ($bounded > $finalgrade) {
 281                                  $errorstr = 'lessthanmin';
 282                              } else if ($bounded < $finalgrade) {
 283                                  $errorstr = 'morethanmax';
 284                              }
 285                          }
 286  
 287                          if ($errorstr) {
 288                              $userfields = 'id, ' . get_all_user_name_fields(true);
 289                              $user = $DB->get_record('user', array('id' => $userid), $userfields);
 290                              $gradestr = new stdClass();
 291                              $gradestr->username = fullname($user);
 292                              $gradestr->itemname = $gradeitem->get_name();
 293                              $warnings[] = get_string($errorstr, 'grades', $gradestr);
 294                              if ($skip) {
 295                                  // Skipping the update of this grade it failed the tests above.
 296                                  continue;
 297                              }
 298                          }
 299  
 300                      } else if ($datatype == 'feedback') {
 301                          $finalgrade = false;
 302                          $trimmed = trim($postedvalue);
 303                          if (empty($trimmed)) {
 304                               $feedback = null;
 305                          } else {
 306                               $feedback = $postedvalue;
 307                          }
 308                      }
 309  
 310                      // group access control
 311                      if ($separategroups) {
 312                          // note: we can not use $this->currentgroup because it would fail badly
 313                          //       when having two browser windows each with different group
 314                          $sharinggroup = false;
 315                          foreach ($mygroups as $groupid) {
 316                              if (groups_is_member($groupid, $userid)) {
 317                                  $sharinggroup = true;
 318                                  break;
 319                              }
 320                          }
 321                          if (!$sharinggroup) {
 322                              // either group membership changed or somebody is hacking grades of other group
 323                              $warnings[] = get_string('errorsavegrade', 'grades');
 324                              continue;
 325                          }
 326                      }
 327  
 328                      $gradeitem->update_final_grade($userid, $finalgrade, 'gradebook', $feedback, FORMAT_MOODLE);
 329  
 330                      // We can update feedback without reloading the grade item as it doesn't affect grade calculations
 331                      if ($datatype === 'feedback') {
 332                          $this->grades[$userid][$itemid]->feedback = $feedback;
 333                      }
 334                  }
 335              }
 336          }
 337  
 338          if ($changedgrades) {
 339              // If a final grade was overriden reload grades so dependent grades like course total will be correct
 340              $this->grades = null;
 341          }
 342  
 343          return $warnings;
 344      }

 369                  if (!isset($SESSION->gradeuserreport->sortitemid)) {

 371                          $this->sortorder = $SESSION->gradeuserreport->sort = 'ASC';
 372                      } else {
 373                          $this->sortorder = $SESSION->gradeuserreport->sort = 'DESC';
 374                      }
 375                  } else if ($SESSION->gradeuserreport->sortitemid == $this->sortitemid) {
 376                      // same as last sort
 377                      if ($SESSION->gradeuserreport->sort == 'ASC') {
 378                          $this->sortorder = $SESSION->gradeuserreport->sort = 'DESC';
 379                      } else {
 380                          $this->sortorder = $SESSION->gradeuserreport->sort = 'ASC';
 381                      }
 382                  } else {
 383                      if ($this->sortitemid == 'firstname' || $this->sortitemid == 'lastname') {
 384                          $this->sortorder = $SESSION->gradeuserreport->sort = 'ASC';
 385                      } else {
 386                          $this->sortorder = $SESSION->gradeuserreport->sort = 'DESC';
 387                      }
 388                  }
 389              }
 390              $SESSION->gradeuserreport->sortitemid = $this->sortitemid;
 391          } else {
 392              // not requesting sort, use last setting (for paging)
 393  
 394              if (isset($SESSION->gradeuserreport->sortitemid)) {
 395                  $this->sortitemid = $SESSION->gradeuserreport->sortitemid;

 411      public function load_users() {
 412          global $CFG, $DB;
 413  
 414          if (!empty($this->users)) {
 415              return;
 416          }
 417          $this->setup_users();
 418  
 419          // Limit to users with a gradeable role.
 420          list($gradebookrolessql, $gradebookrolesparams) = $DB->get_in_or_equal(explode(',', $this->gradebookroles), SQL_PARAMS_NAMED, 'grbr0');
 421  
 422          // Check the status of showing only active enrolments.
 423          $coursecontext = $this->context->get_course_context(true);
 424          $defaultgradeshowactiveenrol = !empty($CFG->grade_report_showonlyactiveenrol);
 425          $showonlyactiveenrol = get_user_preferences('grade_report_showonlyactiveenrol', $defaultgradeshowactiveenrol);
 426          $showonlyactiveenrol = $showonlyactiveenrol || !has_capability('moodle/course:viewsuspendedusers', $coursecontext);
 427  
 428          // Limit to users with an active enrollment.
 429          list($enrolledsql, $enrolledparams) = get_enrolled_sql($this->context, '', 0, $showonlyactiveenrol);
 430  
 431          // Fields we need from the user table.
 432          $userfields = user_picture::fields('u', get_extra_user_fields($this->context));
 433  
 434          // We want to query both the current context and parent contexts.
 435          list($relatedctxsql, $relatedctxparams) = $DB->get_in_or_equal($this->context->get_parent_context_ids(true), SQL_PARAMS_NAMED, 'relatedctx');
 436  
 437          // If the user has clicked one of the sort asc/desc arrows.
 438          if (is_numeric($this->sortitemid)) {
 439              $params = array_merge(array('gitemid' => $this->sortitemid), $gradebookrolesparams, $this->userwheresql_params,
 440                      $this->groupwheresql_params, $enrolledparams, $relatedctxparams);
 441  
 442              $sortjoin = "LEFT JOIN {grade_grades} g ON g.userid = u.id AND g.itemid = $this->sortitemid";
 443              $sort = "g.finalgrade $this->sortorder";
 444          } else {
 445              $sortjoin = '';
 446              switch($this->sortitemid) {
 447                  case 'lastname':
 448                      $sort = "u.lastname $this->sortorder, u.firstname $this->sortorder";
 449                      break;
 450                  case 'firstname':
 451                      $sort = "u.firstname $this->sortorder, u.lastname $this->sortorder";
 452                      break;
 453                  case 'email':
 454                      $sort = "u.email $this->sortorder";
 455                      break;
 456                  case 'idnumber':
 457                  default:
 458                      $sort = "u.idnumber $this->sortorder";
 459                      break;
 460              }
 461  
 462              $params = array_merge($gradebookrolesparams, $this->userwheresql_params, $this->groupwheresql_params, $enrolledparams, $relatedctxparams);
 463          }
 464  
 465          $sql = "SELECT $userfields
 466                    FROM {user} u
 467                    JOIN ($enrolledsql) je ON je.id = u.id
 468                         $this->groupsql
 469                         $sortjoin
 470                    JOIN (
 471                             SELECT DISTINCT ra.userid
 472                               FROM {role_assignments} ra
 473                              WHERE ra.roleid IN ($this->gradebookroles)
 474                                AND ra.contextid $relatedctxsql
 475                         ) rainner ON rainner.userid = u.id
 476                     AND u.deleted = 0
 477                     $this->userwheresql
 478                     $this->groupwheresql
 479                ORDER BY $sort";
 480          $studentsperpage = $this->get_students_per_page();
 481          $this->users = $DB->get_records_sql($sql, $params, $studentsperpage * $this->page, $studentsperpage);
 482  
 483          if (empty($this->users)) {
 484              $this->userselect = '';
 485              $this->users = array();
 486              $this->userselect_params = array();
 487          } else {
 488              list($usql, $uparams) = $DB->get_in_or_equal(array_keys($this->users), SQL_PARAMS_NAMED, 'usid0');
 489              $this->userselect = "AND g.userid $usql";
 490              $this->userselect_params = $uparams;
 491  
 492              // First flag everyone as not suspended.
 493              foreach ($this->users as $user) {
 494                  $this->users[$user->id]->suspendedenrolment = false;
 495              }
 496  
 497              // If we want to mix both suspended and not suspended users, let's find out who is suspended.
 498              if (!$showonlyactiveenrol) {
 499                  $sql = "SELECT ue.userid
 500                            FROM {user_enrolments} ue
 501                            JOIN {enrol} e ON e.id = ue.enrolid
 502                           WHERE ue.userid $usql
 503                                 AND ue.status = :uestatus
 504                                 AND e.status = :estatus
 505                                 AND e.courseid = :courseid
 506                                 AND ue.timestart < :now1 AND (ue.timeend = 0 OR ue.timeend > :now2)
 507                        GROUP BY ue.userid";
 508  
 509                  $time = time();
 510                  $params = array_merge($uparams, array('estatus' => ENROL_INSTANCE_ENABLED, 'uestatus' => ENROL_USER_ACTIVE,
 511                          'courseid' => $coursecontext->instanceid, 'now1' => $time, 'now2' => $time));
 512                  $useractiveenrolments = $DB->get_records_sql($sql, $params);
 513  
 514                  foreach ($this->users as $user) {
 515                      $this->users[$user->id]->suspendedenrolment = !array_key_exists($user->id, $useractiveenrolments);
 516                  }
 517              }
 518          }
 519          return $this->users;
 520      }

 526      public function load_final_grades() {
 527          global $CFG, $DB;
 528  
 529          if (!empty($this->grades)) {
 530              return;
 531          }
 532  
 533          if (empty($this->users)) {
 534              return;
 535          }
 536  
 537          // please note that we must fetch all grade_grades fields if we want to construct grade_grade object from it!
 538          $params = array_merge(array('courseid'=>$this->courseid), $this->userselect_params);
 539          $sql = "SELECT g.*
 540                    FROM {grade_items} gi,
 541                         {grade_grades} g
 542                   WHERE g.itemid = gi.id AND gi.courseid = :courseid {$this->userselect}";
 543  
 544          $userids = array_keys($this->users);
 545  
 546          if ($grades = $DB->get_records_sql($sql, $params)) {
 547              foreach ($grades as $graderec) {
 548                  $grade = new grade_grade($graderec, false);
 549                  $this->allgrades[$graderec->userid][$graderec->itemid] = $grade;
 550                  if (in_array($graderec->userid, $userids) and array_key_exists($graderec->itemid, $this->gtree->get_items())) { // some items may not be present!!
 551                      $this->grades[$graderec->userid][$graderec->itemid] = $grade;
 552                      $this->grades[$graderec->userid][$graderec->itemid]->grade_item = $this->gtree->get_item($graderec->itemid); // db caching
 553                  }
 554              }
 555          }
 556  
 557          // prefil grades that do not exist yet
 558          foreach ($userids as $userid) {
 559              foreach ($this->gtree->get_items() as $itemid => $unused) {
 560                  if (!isset($this->grades[$userid][$itemid])) {
 561                      $this->grades[$userid][$itemid] = new grade_grade();
 562                      $this->grades[$userid][$itemid]->itemid = $itemid;
 563                      $this->grades[$userid][$itemid]->userid = $userid;
 564                      $this->grades[$userid][$itemid]->grade_item = $this->gtree->get_item($itemid); // db caching
 565  
 566                      $this->allgrades[$userid][$itemid] = $this->grades[$userid][$itemid];
 567                  }
 568              }
 569          }

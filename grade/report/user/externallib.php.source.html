  55      /**
  56       * Returns a list of grades tables for users in a course.
  57       *
  58       * @param int $courseid Course Id
  59       * @param int $userid   Only this user (optional)
  60       *
  61       * @return array the grades tables
  62       * @since Moodle 2.9
  63       */
  64      public static function get_grades_table($courseid, $userid = 0) {
  65          global $CFG, $USER;
  66  
  67          $warnings = array();
  68  
  69          // Validate the parameter.
  70          $params = self::validate_parameters(self::get_grades_table_parameters(),
  71              array(
  72                  'courseid' => $courseid,
  73                  'userid' => $userid)
  74              );
  75  
  76          // Compact/extract functions are not recommended.
  77          $courseid = $params['courseid'];
  78          $userid   = $params['userid'];
  79  
  80          // Function get_course internally throws an exception if the course doesn't exist.
  81          $course = get_course($courseid);
  82  
  83          $context = context_course::instance($courseid);
  84          self::validate_context($context);
  85  
  86          // Specific capabilities.
  87          require_capability('gradereport/user:view', $context);
  88  
  89          $user = null;
  90  
  91          if (empty($userid)) {
  92              require_capability('moodle/grade:viewall', $context);
  93          } else {
  94              $user = core_user::get_user($userid, '*', MUST_EXIST);
  95          }
  96  
  97          $access = false;
  98  
  99          if (has_capability('moodle/grade:viewall', $context)) {
 100              // Can view all course grades.
 101              $access = true;
 102          } else if ($userid == $USER->id and has_capability('moodle/grade:view', $context) and $course->showgrades) {
 103              // View own grades.
 104              $access = true;
 105          }
 106  
 107          if (!$access) {
 108              throw new moodle_exception('nopermissiontoviewgrades', 'error');
 109          }
 110  
 111          // Require files here to save some memory in case validation fails.
 112          require_once($CFG->dirroot . '/group/lib.php');
 113          require_once($CFG->libdir  . '/gradelib.php');
 114          require_once($CFG->dirroot . '/grade/lib.php');
 115          require_once($CFG->dirroot . '/grade/report/user/lib.php');
 116  
 117          $gpr = new grade_plugin_return(
 118              array(
 119                  'type' => 'report',
 120                  'plugin' => 'user',
 121                  'courseid' => $courseid,
 122                  'userid' => $userid)
 123              );
 124  
 125          $tables = array();
 126  
 127          // Just one user.
 128          if ($user) {
 129              $report = new grade_report_user($courseid, $gpr, $context, $userid);
 130              $report->fill_table();
 131  
 132              $tables[] = array(
 133                  'courseid'      => $courseid,
 134                  'userid'        => $user->id,
 135                  'userfullname'  => fullname($user),
 136                  'maxdepth'      => $report->maxdepth,
 137                  'tabledata'     => $report->tabledata
 138              );
 139  
 140          } else {
 141              $defaultgradeshowactiveenrol = !empty($CFG->grade_report_showonlyactiveenrol);
 142              $showonlyactiveenrol = get_user_preferences('grade_report_showonlyactiveenrol', $defaultgradeshowactiveenrol);
 143              $showonlyactiveenrol = $showonlyactiveenrol || !has_capability('moodle/course:viewsuspendedusers', $context);
 144  
 145              $gui = new graded_users_iterator($course);
 146              $gui->require_active_enrolment($showonlyactiveenrol);
 147              $gui->init();
 148  
 149              while ($userdata = $gui->next_user()) {
 150                  $currentuser = $userdata->user;
 151                  $report = new grade_report_user($courseid, $gpr, $context, $currentuser->id);
 152                  $report->fill_table();
 153  
 154                  $tables[] = array(
 155                      'courseid'      => $courseid,
 156                      'userid'        => $currentuser->id,
 157                      'userfullname'  => fullname($currentuser),
 158                      'maxdepth'      => $report->maxdepth,
 159                      'tabledata'     => $report->tabledata
 160                  );
 161              }
 162              $gui->close();
 163          }
 164  
 165          $result = array();
 166          $result['tables'] = $tables;
 167          $result['warnings'] = $warnings;
 168          return $result;
 169      }

 272      /**
 273       * Trigger the user report events, do the same that the web interface view of the report
 274       *
 275       * @param int $courseid id of course
 276       * @param int $userid id of the user the report belongs to
 277       * @return array of warnings and status result
 278       * @since Moodle 2.9
 279       * @throws moodle_exception
 280       */
 281      public static function view_grade_report($courseid, $userid = 0) {
 282          global $CFG, $USER;
 283          require_once($CFG->dirroot . "/grade/lib.php");
 284          require_once($CFG->dirroot . "/grade/report/user/lib.php");
 285  
 286          $params = self::validate_parameters(self::view_grade_report_parameters(),
 287                                              array(
 288                                                  'courseid' => $courseid,
 289                                                  'userid' => $userid
 290                                              ));
 291  
 292          $warnings = array();
 293  
 294          $course = get_course($params['courseid']);
 295  
 296          $context = context_course::instance($course->id);
 297          self::validate_context($context);
 298  
 299          $userid = $params['userid'];
 300          if (empty($userid)) {
 301              $userid = $USER->id;
 302          } else {
 303              $user = core_user::get_user($userid, '*', MUST_EXIST);
 304              if ($user->deleted) {
 305                  throw new moodle_exception('userdeleted');
 306              }
 307              if (isguestuser($user)) {
 308                  // Can not view profile of guest - thre is nothing to see there.
 309                  throw new moodle_exception('invaliduserid');
 310              }
 311          }
 312  
 313          $access = false;
 314  
 315          if (has_capability('moodle/grade:viewall', $context)) {
 316              // Can view all course grades (any user).
 317              $access = true;
 318          } else if ($userid == $USER->id and has_capability('moodle/grade:view', $context) and $course->showgrades) {
 319              // View own grades.
 320              $access = true;
 321          }
 322  
 323          if (!$access) {
 324              throw new moodle_exception('nopermissiontoviewgrades', 'error');
 325          }
 326  
 327          // Create a report instance. We don't need the gpr second parameter.
 328          $report = new grade_report_user($course->id, null, $context, $userid);
 329          $report->viewed();
 330  
 331          $result = array();
 332          $result['status'] = true;
 333          $result['warnings'] = $warnings;
 334          return $result;
 335      }

  99      public function init($selfitemisempty = false) {
 100          global $DB;
 101  
 102          if (!$selfitemisempty) {
 103              $validusers = $this->load_users();
 104              if (!isset($validusers[$this->itemid])) {
 105                  // If the passed user id is not valid, show the first user from the list instead.
 106                  $this->item = reset($validusers);
 107                  $this->itemid = $this->item->id;
 108              } else {
 109                  $this->item = $validusers[$this->itemid];
 110              }
 111          }
 112  
 113          $params = array('courseid' => $this->courseid);
 114  
 115          $seq = new grade_seq($this->courseid, true);
 116          foreach ($seq->items as $key => $item) {
 117              if (isset($item->itemmodule)) {
 118                  list($courseid, $cmid) = get_course_and_cm_from_instance($item->iteminstance, $item->itemmodule);
 119                  $seq->items[$key]->cmid = $cmid->id;
 120              }
 121          }
 122  
 123          $this->items = array();
 124          foreach ($seq->items as $itemid => $item) {
 125              if (grade::filter($item)) {
 126                  $this->items[$itemid] = $item;
 127              }
 128          }
 129  
 130          $this->requirespaging = count($this->items) > $this->perpage;
 131  
 132          $this->setup_structure();
 133  
 134          $this->definition = array(
 135              'finalgrade', 'feedback', 'override', 'exclude'
 136          );
 137          $this->set_headers($this->original_headers());
 138      }

 164      public function format_line($item) {
 165          global $OUTPUT;
 166  
 167          $grade = $this->fetch_grade_or_default($item, $this->item->id);
 168          $lockicon = '';
 169  
 170          $lockeditem = $lockeditemgrade = 0;
 171          if (!empty($grade->locked)) {
 172              $lockeditem = 1;
 173          }
 174          if (!empty($grade->grade_item->locked)) {
 175              $lockeditemgrade = 1;
 176          }
 177          // Check both grade and grade item.
 178          if ($lockeditem || $lockeditemgrade) {
 179               $lockicon = $OUTPUT->pix_icon('t/locked', 'grade is locked');
 180          }
 181  
 182          $realmodid = '';
 183          if (isset($item->cmid)) {
 184              $realmodid = $item->cmid;
 185          }
 186  
 187          $iconstring = get_string('filtergrades', 'gradereport_singleview', $item->get_name());
 188  
 189          // Create a fake gradetreeitem so we can call get_element_header().
 190          // The type logic below is from grade_category->_get_children_recursion().
 191          $gradetreeitem = array();
 192          if (in_array($item->itemtype, array('course', 'category'))) {
 193              $gradetreeitem['type'] = $item->itemtype.'item';
 194          } else {
 195              $gradetreeitem['type'] = 'item';
 196          }
 197          $gradetreeitem['object'] = $item;
 198          $gradetreeitem['userid'] = $this->item->id;
 199  
 200          $itemlabel = $this->structure->get_element_header($gradetreeitem, true, false, false, false, true);
 201          $grade->label = $item->get_name();
 202  
 203          $itemlabel = $item->get_name();
 204          if (!empty($realmodid)) {
 205              $url = new moodle_url('/mod/' . $item->itemmodule . '/view.php', array('id' => $realmodid));
 206              $itemlabel = html_writer::link($url, $item->get_name());
 207          }
 208  
 209          $line = array(
 210              $OUTPUT->action_icon($this->format_link('grade', $item->id), new pix_icon('t/editstring', $iconstring)),
 211              $this->format_icon($item) . $lockicon . $itemlabel,
 212              $this->category($item),
 213              new range($item)
 214          );
 215          $lineclasses = array(
 216              "action",
 217              "gradeitem",
 218              "category",
 219              "range"
 220          );
 221  
 222          $outputline = array();
 223          $i = 0;
 224          foreach ($line as $key => $value) {
 225              $cell = new \html_table_cell($value);
 226              if ($isheader = $i == 1) {
 227                  $cell->header = $isheader;
 228                  $cell->scope = "row";
 229              }
 230              if (array_key_exists($key, $lineclasses)) {
 231                  $cell->attributes['class'] = $lineclasses[$key];
 232              }
 233              $outputline[] = $cell;
 234              $i++;
 235          }

 263                  return $this->course->fullname;
 264              }
 265  
 266              $params = array('id' => $item->iteminstance);
 267              $elem = $DB->get_record('grade_categories', $params);
 268  
 269              return $elem->fullname;
 270          }
 271  
 272          if (!isset($this->categories[$item->categoryid])) {
 273              $category = $item->get_parent_category();
 274  
 275              $this->categories[$category->id] = $category;
 276          }

 333      /**
 334       * Process the data from the form.
 335       *
 336       * @param array $data
 337       * @return array of warnings
 338       */
 339      public function process($data) {
 340          $bulk = new bulk_insert($this->item);
 341          // Bulk insert messages the data to be passed in
 342          // ie: for all grades of empty grades apply the specified value.
 343          if ($bulk->is_applied($data)) {
 344              $filter = $bulk->get_type($data);
 345              $insertvalue = $bulk->get_insert_value($data);
 346  
 347              $userid = $this->item->id;
 348              foreach ($this->items as $gradeitemid => $gradeitem) {
 349                  $null = $gradeitem->gradetype == GRADE_TYPE_SCALE ? -1 : '';
 350                  $field = "finalgrade_{$gradeitem->id}_{$this->itemid}";
 351                  if (isset($data->$field)) {
 352                      continue;
 353                  }
 354  
 355                  $oldfinalgradefield = "oldfinalgrade_{$gradeitem->id}_{$this->itemid}";
 356                  if (!empty($data->$oldfinalgradefield)) {
 357                      continue;
 358                  }
 359                  $grade = grade_grade::fetch(array(
 360                      'itemid' => $this->itemid,
 361                      'userid' => $userid
 362                  ));
 363  
 364                  $data->$field = empty($grade) ? $null : $grade->finalgrade;
 365                  $data->{"old$field"} = $data->$field;
 366  
 367                  preg_match('/_(\d+)_(\d+)/', $field, $oldoverride);
 368                  $oldoverride = 'oldoverride' . $oldoverride[0];
 369                  if (empty($data->$oldoverride)) {
 370                      $data->$field = (!isset($grade->rawgrade)) ? $null : $grade->rawgrade;
 371                  }
 372  
 373              }
 374  
 375              foreach ($data as $varname => $value) {
 376                  if (preg_match('/^oldoverride_(\d+)_(\d+)/', $varname, $matches)) {
 377                      // If we've selected override or overriding all grades.
 378                      if (!empty($data->$matches[0]) || $filter == 'all') {
 379                          $override = "override_{$matches[1]}_{$matches[2]}";
 380                          $data->$override = '1';
 381                      }
 382                  }
 383                  if (!preg_match('/^finalgrade_(\d+)_(\d+)/', $varname, $matches)) {
 384                      continue;
 385                  }
 386  
 387                  $gradeitem = grade_item::fetch(array(
 388                      'courseid' => $this->courseid,
 389                      'id' => $matches[1]
 390                  ));
 391  
 392                  $isscale = ($gradeitem->gradetype == GRADE_TYPE_SCALE);
 393  
 394                  $empties = (trim($value) === '' or ($isscale and $value == -1));
 395  
 396                  if ($filter == 'all' or $empties) {
 397                      $data->$varname = ($isscale and empty($insertvalue)) ?
 398                          -1 : $insertvalue;
 399                  }
 400              }
 401          }
 402          return parent::process($data);
 403      }

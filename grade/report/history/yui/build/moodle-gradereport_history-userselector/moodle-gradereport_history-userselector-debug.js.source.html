 605      /**
 606       * Find the next or previous focusable node.
 607       *
 608       * @param {NodeList} users The list of users.
 609       * @param {Node} user The user to start with.
 610       * @param {Number} direction The direction in which to go.
 611       * @return {Node|null} A user node, or null if not found.
 612       * @method findFocusableUser
 613       */
 614      findFocusableUser: function(users, user, direction) {
 615          var index = users.indexOf(user);
 616  
 617          if (users.size() < 1) {
 618              Y.log('The users list is empty', 'debug', COMPONENT);
 619              return null;
 620          }
 621  
 622          if (index < 0) {
 623              Y.log('Unable to find the user in the list of users', 'debug', COMPONENT);
 624              return users.item(0);
 625          }
 626  
 627          index += direction;
 628  
 629          // Wrap the navigation when reaching the top of the bottom.
 630          if (index < 0) {
 631              index = users.size() - 1;
 632          } else if (index >= users.size()) {
 633              index = 0;
 634          }
 635  
 636          return users.item(index);
 637      },

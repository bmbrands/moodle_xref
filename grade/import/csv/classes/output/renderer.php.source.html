  43      public function standard_upload_file_form($course, $mform) {
  44  
  45          $output = groups_print_course_menu($course, 'index.php?id=' . $course->id, true);
  46          $output .= html_writer::start_tag('div', array('class' => 'clearer'));

  74          return $html;

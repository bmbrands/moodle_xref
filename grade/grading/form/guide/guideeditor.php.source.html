 115          } else {

 117              if ($this->_persistantFreeze) {

 135      /**
 136       * Prepares the data passed in $_POST:
 137       * - processes the pressed buttons 'addlevel', 'addcriterion', 'moveup', 'movedown', 'delete' (when JavaScript is disabled)
 138       *   sets $this->nonjsbuttonpressed to true/false if such button was pressed
 139       * - if options not passed (i.e. we create a new guide) fills the options array with the default values
 140       * - if options are passed completes the options array with unchecked checkboxes
 141       * - if $withvalidation is set, adds 'error_xxx' attributes to elements that contain errors and creates an error string
 142       *   and stores it in $this->validationerrors
 143       *
 144       * @param array $value
 145       * @param boolean $withvalidation whether to enable data validation
 146       * @return array
 147       */
 148      protected function prepare_data($value = null, $withvalidation = false) {
 149          if (null === $value) {
 150              $value = $this->getValue();
 151          }
 152          if ($this->nonjsbuttonpressed === null) {
 153              $this->nonjsbuttonpressed = false;
 154          }
 155  
 156          $errors = array();
 157          $return = array('criteria' => array(), 'options' => gradingform_guide_controller::get_default_options(),
 158              'comments' => array());
 159          if (!isset($value['criteria'])) {
 160              $value['criteria'] = array();
 161              $errors['err_nocriteria'] = 1;
 162          }
 163          // If options are present in $value, replace default values with submitted values.
 164          if (!empty($value['options'])) {
 165              foreach (array_keys($return['options']) as $option) {
 166                  // Special treatment for checkboxes.
 167                  if (!empty($value['options'][$option])) {
 168                      $return['options'][$option] = $value['options'][$option];
 169                  } else {
 170                      $return['options'][$option] = null;
 171                  }
 172  
 173              }
 174          }
 175  
 176          if (is_array($value)) {
 177              // For other array keys of $value no special treatmeant neeeded, copy them to return value as is.
 178              foreach (array_keys($value) as $key) {
 179                  if ($key != 'options' && $key != 'criteria' && $key != 'comments') {
 180                      $return[$key] = $value[$key];
 181                  }
 182              }
 183          }
 184  
 185          // Iterate through criteria.
 186          $lastaction = null;
 187          $lastid = null;
 188          foreach ($value['criteria'] as $id => $criterion) {
 189              if ($id == 'addcriterion') {
 190                  $id = $this->get_next_id(array_keys($value['criteria']));
 191                  $criterion = array('description' => '');
 192                  $this->nonjsbuttonpressed = true;
 193              }
 194  
 195              if ($withvalidation && !array_key_exists('delete', $criterion)) {
 196                  if (!strlen(trim($criterion['shortname']))) {
 197                      $errors['err_noshortname'] = 1;
 198                      $criterion['error_description'] = true;
 199                  }
 200                  if (strlen(trim($criterion['shortname'])) > 255) {
 201                      $errors['err_shortnametoolong'] = 1;
 202                      $criterion['error_description'] = true;
 203                  }
 204                  if (!strlen(trim($criterion['maxscore']))) {
 205                      $errors['err_nomaxscore'] = 1;
 206                      $criterion['error_description'] = true;
 207                  } else if (!is_numeric($criterion['maxscore']) || $criterion['maxscore'] < 0) {
 208                      $errors['err_maxscorenotnumeric'] = 1;
 209                      $criterion['error_description'] = true;
 210                  }
 211              }
 212              if (array_key_exists('moveup', $criterion) || $lastaction == 'movedown') {
 213                  unset($criterion['moveup']);
 214                  if ($lastid !== null) {
 215                      $lastcriterion = $return['criteria'][$lastid];
 216                      unset($return['criteria'][$lastid]);
 217                      $return['criteria'][$id] = $criterion;
 218                      $return['criteria'][$lastid] = $lastcriterion;
 219                  } else {
 220                      $return['criteria'][$id] = $criterion;
 221                  }
 222                  $lastaction = null;
 223                  $lastid = $id;
 224                  $this->nonjsbuttonpressed = true;
 225              } else if (array_key_exists('delete', $criterion)) {
 226                  $this->nonjsbuttonpressed = true;
 227              } else {
 228                  if (array_key_exists('movedown', $criterion)) {
 229                      unset($criterion['movedown']);
 230                      $lastaction = 'movedown';
 231                      $this->nonjsbuttonpressed = true;
 232                  }
 233                  $return['criteria'][$id] = $criterion;
 234                  $lastid = $id;
 235              }
 236          }
 237  
 238          // Add sort order field to criteria.
 239          $csortorder = 1;
 240          foreach (array_keys($return['criteria']) as $id) {
 241              $return['criteria'][$id]['sortorder'] = $csortorder++;
 242          }
 243  
 244          // Iterate through comments.
 245          $lastaction = null;
 246          $lastid = null;
 247          if (!empty($value['comments'])) {
 248              foreach ($value['comments'] as $id => $comment) {
 249                  if ($id == 'addcomment') {
 250                      $id = $this->get_next_id(array_keys($value['comments']));
 251                      $comment = array('description' => '');
 252                      $this->nonjsbuttonpressed = true;
 253                  }
 254  
 255                  if (array_key_exists('moveup', $comment) || $lastaction == 'movedown') {
 256                      unset($comment['moveup']);
 257                      if ($lastid !== null) {
 258                          $lastcomment = $return['comments'][$lastid];
 259                          unset($return['comments'][$lastid]);
 260                          $return['comments'][$id] = $comment;
 261                          $return['comments'][$lastid] = $lastcomment;
 262                      } else {
 263                          $return['comments'][$id] = $comment;
 264                      }
 265                      $lastaction = null;
 266                      $lastid = $id;
 267                      $this->nonjsbuttonpressed = true;
 268                  } else if (array_key_exists('delete', $comment)) {
 269                      $this->nonjsbuttonpressed = true;
 270                  } else {
 271                      if (array_key_exists('movedown', $comment)) {
 272                          unset($comment['movedown']);
 273                          $lastaction = 'movedown';
 274                          $this->nonjsbuttonpressed = true;
 275                      }
 276                      $return['comments'][$id] = $comment;
 277                      $lastid = $id;
 278                  }
 279              }
 280              // Add sort order field to comments.
 281              $csortorder = 1;
 282              foreach (array_keys($return['comments']) as $id) {
 283                  $return['comments'][$id]['sortorder'] = $csortorder++;
 284              }
 285          }
 286          // Create validation error string (if needed).
 287          if ($withvalidation) {
 288              if (count($errors)) {
 289                  $rv = array();
 290                  foreach ($errors as $error => $v) {
 291                      $rv[] = get_string($error, 'gradingform_guide');
 292                  }
 293                  $this->validationerrors = join('<br/ >', $rv);
 294              } else {
 295                  $this->validationerrors = false;
 296              }
 297              $this->wasvalidated = true;
 298          }
 299          return $return;

 301      }

 303      /**
 304       * Scans array $ids to find the biggest element ! NEWID*, increments it by 1 and returns
 305       *
 306       * @param array $ids
 307       * @return string
 308       */
 309      protected function get_next_id($ids) {
 310          $maxid = 0;
 311          foreach ($ids as $id) {
 312              if (preg_match('/^NEWID(\d+)$/', $id, $matches) && ((int)$matches[1]) > $maxid) {
 313                  $maxid = (int)$matches[1];
 314              }
 315          }
 316          return 'NEWID'.($maxid+1);
 317      }

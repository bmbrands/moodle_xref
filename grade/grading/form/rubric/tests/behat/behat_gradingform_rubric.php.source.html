  52      /**
  53       * Defines the rubric with the provided data, following rubric's definition grid cells.
  54       *
  55       * This method fills the rubric of the rubric definition
  56       * form; the provided TableNode should contain one row for
  57       * each criterion and each cell of the row should contain:
  58       * # Criterion description
  59       * # Criterion level 1 name
  60       * # Criterion level 1 points
  61       * # Criterion level 2 name
  62       * # Criterion level 2 points
  63       * # Criterion level 3 .....
  64       *
  65       * Works with both JS and non-JS.
  66       *
  67       * @When /^I define the following rubric:$/
  68       * @throws ExpectationException
  69       * @param TableNode $rubric
  70       */
  71      public function i_define_the_following_rubric(TableNode $rubric) {
  72  
  73          // Being a smart method is nothing good when we talk about step definitions, in
  74          // this case we didn't have any other options as there are no labels no elements
  75          // id we can point to without having to "calculate" them.
  76  
  77          $steptableinfo = '| criterion description | level1 name  | level1 points | level2 name | level2 points | ...';
  78  
  79          $criteria = $rubric->getRows();
  80  
  81          $addcriterionbutton = $this->find_button(get_string('addcriterion', 'gradingform_rubric'));
  82  
  83          // Cleaning the current ones.
  84          $deletebuttons = $this->find_all('css', "input[title='" . get_string('criteriondelete', 'gradingform_rubric') . "']");
  85          if ($deletebuttons) {
  86  
  87              // We should reverse the deletebuttons because otherwise once we delete
  88              // the first one the DOM will change and the [X] one will not exist anymore.
  89              $deletebuttons = array_reverse($deletebuttons, true);
  90              foreach ($deletebuttons as $button) {
  91                  $this->click_and_confirm($button);
  92              }
  93          }
  94  
  95          // The level number (NEWID$N) is not reset after each criterion.
  96          $levelnumber = 1;
  97  
  98          // The next criterion is created with the same number of levels than the last criterion.
  99          $defaultnumberoflevels = self::DEFAULT_RUBRIC_LEVELS;
 100  
 101          if ($criteria) {
 102              foreach ($criteria as $criterionit => $criterion) {
 103  
 104                  // Checking the number of cells.
 105                  if (count($criterion) % 2 === 0) {
 106                      throw new ExpectationException(
 107                          'The criterion levels should contain both definition and points, follow this format:' . $steptableinfo,
 108                          $this->getSession()
 109                      );
 110                  }
 111  
 112                  // Minimum 2 levels per criterion.
 113                  // description + definition1 + score1 + definition2 + score2 = 5.
 114                  if (count($criterion) < 5) {
 115                      throw new ExpectationException(
 116                          get_string('err_mintwolevels', 'gradingform_rubric'),
 117                          $this->getSession()
 118                      );
 119  
 120                  }
 121  
 122                  // Add new criterion.
 123                  $addcriterionbutton->click();
 124  
 125                  $criterionroot = 'rubric[criteria][NEWID' . ($criterionit + 1) . ']';
 126  
 127                  // Getting the criterion description, this one is visible by default.
 128                  $this->set_rubric_field_value($criterionroot . '[description]', array_shift($criterion), true);
 129  
 130                  // When JS is disabled each criterion's levels name numbers starts from 0.
 131                  if (!$this->running_javascript()) {
 132                      $levelnumber = 0;
 133                  }
 134  
 135                  // Setting the correct number of levels.
 136                  $nlevels = count($criterion) / 2;
 137                  if ($nlevels < $defaultnumberoflevels) {
 138  
 139                      // Removing levels if there are too much levels.
 140                      // When we add a new level the NEWID$N is increased from the last criterion.
 141                      $lastcriteriondefaultlevel = $defaultnumberoflevels + $levelnumber - 1;
 142                      $lastcriterionlevel = $nlevels + $levelnumber - 1;
 143                      for ($i = $lastcriteriondefaultlevel; $i > $lastcriterionlevel; $i--) {
 144  
 145                          // If JS is disabled seems that new levels are not added.
 146                          if ($this->running_javascript()) {
 147                              $deletelevel = $this->find_button($criterionroot . '[levels][NEWID' . $i . '][delete]');
 148                              $this->click_and_confirm($deletelevel);
 149  
 150                          } else {
 151                              // Only if the level exists.
 152                              $buttonname = $criterionroot . '[levels][NEWID' . $i . '][delete]';
 153                              if ($deletelevel = $this->getSession()->getPage()->findButton($buttonname)) {
 154                                  $deletelevel->click();
 155                              }
 156                          }
 157                      }
 158                  } else if ($nlevels > $defaultnumberoflevels) {
 159                      // Adding levels if we don't have enough.
 160                      $addlevel = $this->find_button($criterionroot . '[levels][addlevel]');
 161                      for ($i = ($defaultnumberoflevels + 1); $i <= $nlevels; $i++) {
 162                          $addlevel->click();
 163                      }
 164                  }
 165  
 166                  // Updating it.
 167                  if ($nlevels > self::DEFAULT_RUBRIC_LEVELS) {
 168                      $defaultnumberoflevels = $nlevels;
 169                  } else {
 170                      // If it is less than the default value it sets it to
 171                      // the default value.
 172                      $defaultnumberoflevels = self::DEFAULT_RUBRIC_LEVELS;
 173                  }
 174  
 175                  foreach ($criterion as $i => $value) {
 176  
 177                      $levelroot = $criterionroot . '[levels][NEWID' . $levelnumber . ']';
 178  
 179                      if ($i % 2 === 0) {
 180                          // Pairs are the definitions.
 181                          $fieldname = $levelroot . '[definition]';
 182                          $this->set_rubric_field_value($fieldname, $value);
 183  
 184                      } else {
 185                          // Odds are the points.
 186  
 187                          // Checking it now, we would need to remove it if we are testing the form validations...
 188                          if (!is_numeric($value)) {
 189                              throw new ExpectationException(
 190                                  'The points cells should contain numeric values, follow this format: ' . $steptableinfo,
 191                                  $this->getSession()
 192                              );
 193                          }
 194  
 195                          $fieldname = $levelroot . '[score]';
 196                          $this->set_rubric_field_value($fieldname, $value, true);
 197  
 198                          // Increase the level by one every 2 cells.
 199                          $levelnumber++;
 200                      }
 201  
 202                  }
 203              }
 204          }
 205      }

 258      /**
 259       * Grades filling the current page rubric. Set one line per criterion and for each criterion set "| Criterion name | Points | Remark |".
 260       *
 261       * @When /^I grade by filling the rubric with:$/
 262       *
 263       * @throws ExpectationException
 264       * @param TableNode $rubric
 265       * @return void
 266       */
 267      public function i_grade_by_filling_the_rubric_with(TableNode $rubric) {
 268  
 269          $criteria = $rubric->getRowsHash();
 270  
 271          $stepusage = '"I grade by filling the rubric with:" step needs you to provide a table where each row is a criterion' .
 272              ' and each criterion has 3 different values: | Criterion name | Number of points | Remark text |';
 273  
 274          // To fill with the steps to execute.
 275          $steps = array();
 276  
 277          // First element -> name, second -> points, third -> Remark.
 278          foreach ($criteria as $name => $criterion) {
 279  
 280              // We only expect the points and the remark, as the criterion name is $name.
 281              if (count($criterion) !== 2) {
 282                  throw new ExpectationException($stepusage, $this->getSession());
 283              }
 284  
 285              // Numeric value here.
 286              $points = $criterion[0];
 287              if (!is_numeric($points)) {
 288                  throw new ExpectationException($stepusage, $this->getSession());
 289              }
 290  
 291              // Selecting a value.
 292              // When JS is disabled there are radio options, with JS enabled divs.
 293              $selectedlevelxpath = $this->get_level_xpath($points);
 294              if ($this->running_javascript()) {
 295  
 296                  // Only clicking on the selected level if it was not already selected.
 297                  $levelnode = $this->find('xpath', $selectedlevelxpath);
 298  
 299                  // Using in_array() as there are only a few elements.
 300                  if (!in_array('checked', explode(' ', $levelnode->getAttribute('class')))) {
 301                      $steps[] = new Given('I click on "' . $selectedlevelxpath . '" "xpath_element" in the "' .
 302                          $this->escape($name) . '" "table_row"');
 303                  }
 304  
 305              } else {
 306  
 307                  // Getting the name of the field.
 308                  $radioxpath = $this->get_criterion_xpath($name) .
 309                      $selectedlevelxpath . "/descendant::input[@type='radio']";
 310                  $radionode = $this->find('xpath', $radioxpath);
 311                  // TODO MDL-43738: Change setValue() to use the generic set_value()
 312                  // which will delegate the process to the field type.
 313                  $radionode->setValue($radionode->getAttribute('value'));
 314              }
 315  
 316              // Setting the remark.
 317  
 318              // First we need to get the textarea name, then we can set the value.
 319              $textarea = $this->get_node_in_container('css_element', 'textarea', 'table_row', $name);
 320              $steps[] = new Given('I set the field "' . $textarea->getAttribute('name') . '" to "' . $criterion[1] . '"');
 321          }
 322  
 323          return $steps;
 324      }

 326      /**
 327       * Checks that the level was previously selected and the user changed to another level.
 328       *
 329       * @Then /^the level with "(?P<points_number>\d+)" points was previously selected for the rubric criterion "(?P<criterion_name_string>(?:[^"]|\\")*)"$/
 330       * @throws ExpectationException
 331       * @param string $criterionname
 332       * @param int $points
 333       * @return void
 334       */
 335      public function the_level_with_points_was_previously_selected_for_the_rubric_criterion($points, $criterionname) {
 336  
 337          $levelxpath = $this->get_criterion_xpath($criterionname) .
 338              $this->get_level_xpath($points) .
 339              "[contains(concat(' ', normalize-space(@class), ' '), ' currentchecked ')]";
 340  
 341          // Works both for JS and non-JS.
 342          // - JS: Class -> checked is there when is marked as green.
 343          // - Non-JS: When editing a rubric definition, there are radio inputs and when viewing a
 344          //   grade @class contains checked.
 345          $levelxpath .= "[not(contains(concat(' ', normalize-space(@class), ' '), ' checked '))]" .
 346              "[not(/descendant::input[@type='radio'][@checked!='checked'])]";
 347  
 348          try {
 349              $this->find('xpath', $levelxpath);
 350          } catch (ElementNotFoundException $e) {
 351              throw new ExpectationException('"' . $points . '" points level was not previously selected', $this->getSession());
 352          }

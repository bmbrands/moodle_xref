  34      public function definition() {
  35  
  36          $mform = $this->_form;

  40          $mform->addElement('text', 'name', get_string('name', 'cohort'), 'maxlength="254" size="50"');

  42          $mform->setType('name', PARAM_TEXT);
  43  
  44          $options = $this->get_category_options($cohort->contextid);
  45          $mform->addElement('select', 'contextid', get_string('context', 'role'), $options);

  47          $mform->addElement('text', 'idnumber', get_string('idnumber', 'cohort'), 'maxlength="254" size="50"');
  48          $mform->setType('idnumber', PARAM_RAW); // Idnumbers are plain text, must not be changed.
  49  
  50          $mform->addElement('advcheckbox', 'visible', get_string('visible', 'cohort'));
  51          $mform->setDefault('visible', 1);

  54          $mform->addElement('editor', 'description_editor', get_string('description', 'cohort'), null, $editoroptions);
  55          $mform->setType('description_editor', PARAM_RAW);

  70      public function validation($data, $files) {
  71          global $DB;
  72  
  73          $errors = parent::validation($data, $files);
  74  
  75          $idnumber = trim($data['idnumber']);
  76          if ($idnumber === '') {
  77              // Fine, empty is ok.
  78  
  79          } else if ($data['id']) {
  80              $current = $DB->get_record('cohort', array('id'=>$data['id']), '*', MUST_EXIST);
  81              if ($current->idnumber !== $idnumber) {
  82                  if ($DB->record_exists('cohort', array('idnumber'=>$idnumber))) {
  83                      $errors['idnumber'] = get_string('duplicateidnumber', 'cohort');
  84                  }
  85              }
  86  
  87          } else {
  88              if ($DB->record_exists('cohort', array('idnumber'=>$idnumber))) {
  89                  $errors['idnumber'] = get_string('duplicateidnumber', 'cohort');
  90              }
  91          }
  92  
  93          return $errors;
  94      }

  96      protected function get_category_options($currentcontextid) {
  97          global $CFG;
  98          require_once($CFG->libdir. '/coursecatlib.php');
  99          $displaylist = coursecat::make_categories_list('moodle/cohort:manage');
 100          $options = array();
 101          $syscontext = context_system::instance();
 102          if (has_capability('moodle/cohort:manage', $syscontext)) {
 103              $options[$syscontext->id] = $syscontext->get_context_name();
 104          }
 105          foreach ($displaylist as $cid=>$name) {
 106              $context = context_coursecat::instance($cid);
 107              $options[$context->id] = $name;
 108          }
 109          // Always add current - this is not likely, but if the logic gets changed it might be a problem.
 110          if (!isset($options[$currentcontextid])) {
 111              $context = context::instance_by_id($currentcontextid, MUST_EXIST);
 112              $options[$context->id] = $syscontext->get_context_name();
 113          }
 114          return $options;
 115      }

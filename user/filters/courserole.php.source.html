  64      /**
  65       * Adds controls specific to this filter in the form.
  66       * @param moodleform $mform a MoodleForm object to setup
  67       */
  68      public function setupForm(&$mform) {
  69          $objs = array();
  70          $objs['role'] = $mform->createElement('select', $this->_name .'_rl', null, $this->get_roles());
  71          $objs['role']->setLabel(get_string('courserole', 'filters'));
  72          $objs['category'] = $mform->createElement('select', $this->_name .'_ct', null, $this->get_course_categories());
  73          $objs['category']->setLabel(get_string('coursecategory', 'filters'));
  74          $objs['value'] = $mform->createElement('text', $this->_name, null);
  75          $objs['value']->setLabel(get_string('coursevalue', 'filters'));
  76          $grp =& $mform->addElement('group', $this->_name.'_grp', $this->_label, $objs, '', false);
  77          $mform->setType($this->_name, PARAM_TEXT);
  78          if ($this->_advanced) {
  79              $mform->setAdvanced($this->_name.'_grp');
  80          }
  81      }

  83      /**
  84       * Retrieves data from the form data
  85       * @param stdClass $formdata data submited with the form
  86       * @return mixed array filter data or false when filter not set
  87       */
  88      public function check_data($formdata) {
  89          $field    = $this->_name;
  90          $role     = $field .'_rl';
  91          $category = $field .'_ct';
  92  
  93          if (array_key_exists($field, $formdata)) {
  94              if (empty($formdata->$field) and empty($formdata->$role) and empty($formdata->$category)) {
  95                  // Nothing selected.
  96                  return false;
  97              }
  98              return array('value'      => (string)$formdata->$field,
  99                           'roleid'     => (int)$formdata->$role,
 100                           'categoryid' => (int)$formdata->$category);
 101          }
 102          return false;
 103      }

 105      /**
 106       * Returns the condition to be used with SQL where
 107       * @param array $data filter settings
 108       * @return array sql string and $params
 109       */
 110      public function get_sql_filter($data) {
 111          global $CFG, $DB;
 112          static $counter = 0;
 113          $pref = 'ex_courserole'.($counter++).'_';
 114  
 115          $value      = $data['value'];
 116          $roleid     = $data['roleid'];
 117          $categoryid = $data['categoryid'];
 118  
 119          $params = array();
 120  
 121          if (empty($value) and empty($roleid) and empty($categoryid)) {
 122              return array('', $params);
 123          }
 124  
 125          $where = "b.contextlevel=50";
 126          if ($roleid) {
 127              $where .= " AND a.roleid = :{$pref}roleid";
 128              $params[$pref.'roleid'] = $roleid;
 129          }
 130          if ($categoryid) {
 131              $where .= " AND c.category = :{$pref}categoryid";
 132              $params[$pref.'categoryid'] = $categoryid;
 133          }
 134          if ($value) {
 135              $where .= " AND c.shortname = :{$pref}course";
 136              $params[$pref.'course'] = $value;
 137          }
 138          return array("id IN (SELECT userid
 139                                 FROM {role_assignments} a
 140                           INNER JOIN {context} b ON a.contextid=b.id
 141                           INNER JOIN {course} c ON b.instanceid=c.id
 142                                WHERE $where)", $params);
 143      }

 145      /**
 146       * Returns a human friendly description of the filter used as label.
 147       * @param array $data filter settings
 148       * @return string active filter label
 149       */
 150      public function get_label($data) {
 151          global $DB;
 152  
 153          $value      = $data['value'];
 154          $roleid     = $data['roleid'];
 155          $categoryid = $data['categoryid'];
 156  
 157          $a = new stdClass();
 158          $a->label = $this->_label;
 159  
 160          if ($roleid) {
 161              $role = $DB->get_record('role', array('id' => $roleid));
 162              $a->rolename = '"'.role_get_name($role).'"';
 163          } else {
 164              $a->rolename = get_string('anyrole', 'filters');
 165          }
 166  
 167          if ($categoryid) {
 168              $catname = $DB->get_field('course_categories', 'name', array('id' => $categoryid));
 169              $a->categoryname = '"'.format_string($catname).'"';
 170          } else {
 171              $a->categoryname = get_string('anycategory', 'filters');
 172          }
 173  
 174          if ($value) {
 175              $a->coursename = '"'.s($value).'"';
 176              if (!$DB->record_exists('course', array('shortname' => $value))) {
 177                  return '<span class="notifyproblem">'.get_string('courserolelabelerror', 'filters', $a).'</span>';
 178              }
 179          } else {
 180              $a->coursename = get_string('anycourse', 'filters');
 181          }
 182  
 183          return get_string('courserolelabel', 'filters', $a);
 184      }

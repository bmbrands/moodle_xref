  39      /**
  40       * Constructor method.
  41       *
  42       * Pulls out the options for the menu from the database and sets the the corresponding key for the data if it exists.
  43       *
  44       * @param int $fieldid
  45       * @param int $userid
  46       */
  47      public function profile_field_menu($fieldid = 0, $userid = 0) {
  48          // First call parent constructor.
  49          $this->profile_field_base($fieldid, $userid);
  50  
  51          // Param 1 for menu type is the options.
  52          if (isset($this->field->param1)) {
  53              $options = explode("\n", $this->field->param1);
  54          } else {
  55              $options = array();
  56          }
  57          $this->options = array();
  58          if (!empty($this->field->required)) {
  59              $this->options[''] = get_string('choose').'...';
  60          }
  61          foreach ($options as $key => $option) {
  62              $this->options[$option] = format_string($option); // Multilang formatting with filters.
  63          }
  64  
  65          // Set the data key.
  66          if ($this->data !== null) {
  67              $key = $this->data;
  68              if (isset($this->options[$key]) || ($key = array_search($key, $this->options)) !== false) {
  69                  $this->data = $key;
  70                  $this->datakey = $key;
  71              }
  72          }
  73      }

 126       * HardFreeze the field if locked.

 129      public function edit_field_set_locked($mform) {
 130          if (!$mform->elementExists($this->inputname)) {
 131              return;
 132          }
 133          if ($this->is_locked() and !has_capability('moodle/user:update', context_system::instance())) {
 134              $mform->hardFreeze($this->inputname);
 135              $mform->setConstant($this->inputname, format_string($this->datakey));
 136          }
 137      }
 138      /**
 139       * Convert external data (csv file) from value to key for processing later by edit_save_data_preprocess
 140       *
 141       * @param string $value one of the values in menu options.
 142       * @return int options key for the menu
 143       */
 144      public function convert_external_data($value) {
 145          if (isset($this->options[$value])) {
 146              $retval = $value;
 147          } else {
 148              $retval = array_search($value, $this->options);
 149          }
 150  
 151          // If value is not found in options then return null, so that it can be handled
 152          // later by edit_save_data_preprocess.
 153          if ($retval === false) {
 154              $retval = null;
 155          }
 156          return $retval;
 157      }

  58      /**
  59       * Update completion status for the current user in an activity, only for activities with manual tracking.
  60       * @param  int $cmid      Course module id
  61       * @param  bool $completed Activity completed or not
  62       * @return array            Result and possible warnings
  63       * @since Moodle 2.9
  64       * @throws moodle_exception
  65       */
  66      public static function update_activity_completion_status_manually($cmid,  $completed) {
  67  
  68          // Validate and normalize parameters.
  69          $params = self::validate_parameters(self::update_activity_completion_status_manually_parameters(),
  70              array('cmid' => $cmid, 'completed' => $completed));
  71          $cmid = $params['cmid'];
  72          $completed = $params['completed'];
  73  
  74          $warnings = array();
  75  
  76          $context = context_module::instance($cmid);
  77          self::validate_context($context);
  78  
  79          list($course, $cm) = get_course_and_cm_from_cmid($cmid);
  80  
  81          // Set up completion object and check it is enabled.
  82          $completion = new completion_info($course);
  83          if (!$completion->is_enabled()) {
  84              throw new moodle_exception('completionnotenabled', 'completion');
  85          }
  86  
  87          // Check completion state is manual.
  88          if ($cm->completion != COMPLETION_TRACKING_MANUAL) {
  89              throw new moodle_exception('cannotmanualctrack', 'error');
  90          }
  91  
  92          $targetstate = ($completed) ? COMPLETION_COMPLETE : COMPLETION_INCOMPLETE;
  93          $completion->update_state($cm, $targetstate);
  94  
  95          $result = array();
  96          $result['status'] = true;
  97          $result['warnings'] = $warnings;
  98          return $result;
  99      }

 132      /**
 133       * Get Activities completion status
 134       *
 135       * @param int $courseid ID of the Course
 136       * @param int $userid ID of the User
 137       * @return array of activities progress and warnings
 138       * @throws moodle_exception
 139       * @since Moodle 2.9
 140       * @throws moodle_exception
 141       */
 142      public static function get_activities_completion_status($courseid, $userid) {
 143          global $CFG, $USER;
 144          require_once($CFG->libdir . '/grouplib.php');
 145  
 146          $warnings = array();
 147          $arrayparams = array(
 148              'courseid' => $courseid,
 149              'userid'   => $userid,
 150          );
 151  
 152          $params = self::validate_parameters(self::get_activities_completion_status_parameters(), $arrayparams);
 153  
 154          $course = get_course($params['courseid']);
 155          $user = core_user::get_user($params['userid'], 'id', MUST_EXIST);
 156  
 157          $context = context_course::instance($course->id);
 158          self::validate_context($context);
 159  
 160          // Check that current user have permissions to see this user's activities.
 161          if ($user->id != $USER->id) {
 162              require_capability('report/progress:view', $context);
 163              if (!groups_user_groups_visible($course, $user->id)) {
 164                  // We are not in the same group!
 165                  throw new moodle_exception('accessdenied', 'admin');
 166              }
 167          }
 168  
 169          $completion = new completion_info($course);
 170          $activities = $completion->get_activities();
 171          $progresses = $completion->get_progress_all();
 172          $userprogress = $progresses[$user->id];
 173  
 174          $results = array();
 175          foreach ($activities as $activity) {
 176  
 177              // Check if current user has visibility on this activity.
 178              if (!$activity->uservisible) {
 179                  continue;
 180              }
 181  
 182              // Get progress information and state.
 183              if (array_key_exists($activity->id, $userprogress->progress)) {
 184                  $thisprogress  = $userprogress->progress[$activity->id];
 185                  $state         = $thisprogress->completionstate;
 186                  $timecompleted = $thisprogress->timemodified;
 187              } else {
 188                  $state = COMPLETION_INCOMPLETE;
 189                  $timecompleted = 0;
 190              }
 191  
 192              $results[] = array(
 193                         'cmid'          => $activity->id,
 194                         'modname'       => $activity->modname,
 195                         'instance'      => $activity->instance,
 196                         'state'         => $state,
 197                         'timecompleted' => $timecompleted,
 198                         'tracking'      => $activity->completion
 199              );
 200          }
 201  
 202          $results = array(
 203              'statuses' => $results,
 204              'warnings' => $warnings
 205          );
 206          return $results;
 207      }

 252      /**
 253       * Get Course completion status
 254       *
 255       * @param int $courseid ID of the Course
 256       * @param int $userid ID of the User
 257       * @return array of course completion status and warnings
 258       * @since Moodle 2.9
 259       * @throws moodle_exception
 260       */
 261      public static function get_course_completion_status($courseid, $userid) {
 262          global $CFG, $USER;
 263          require_once($CFG->libdir . '/grouplib.php');
 264  
 265          $warnings = array();
 266          $arrayparams = array(
 267              'courseid' => $courseid,
 268              'userid'   => $userid,
 269          );
 270          $params = self::validate_parameters(self::get_course_completion_status_parameters(), $arrayparams);
 271  
 272          $course = get_course($params['courseid']);
 273          $user = core_user::get_user($params['userid'], 'id', MUST_EXIST);
 274          $context = context_course::instance($course->id);
 275          self::validate_context($context);
 276  
 277          // Can current user see user's course completion status?
 278          // This check verifies if completion is enabled because $course is mandatory.
 279          if (!completion_can_view_data($user->id, $course)) {
 280              throw new moodle_exception('cannotviewreport');
 281          }
 282  
 283          // The previous function doesn't check groups.
 284          if ($user->id != $USER->id) {
 285              if (!groups_user_groups_visible($course, $user->id)) {
 286                  // We are not in the same group!
 287                  throw new moodle_exception('accessdenied', 'admin');
 288              }
 289          }
 290  
 291          $info = new completion_info($course);
 292  
 293          // Check this user is enroled.
 294          if (!$info->is_tracked_user($user->id)) {
 295              if ($USER->id == $user->id) {
 296                  throw new moodle_exception('notenroled', 'completion');
 297              } else {
 298                  throw new moodle_exception('usernotenroled', 'completion');
 299              }
 300          }
 301  
 302          $completions = $info->get_completions($user->id);
 303          if (empty($completions)) {
 304              throw new moodle_exception('nocriteriaset', 'completion');
 305          }
 306  
 307          // Load course completion.
 308          $completionparams = array(
 309              'userid' => $user->id,
 310              'course' => $course->id,
 311          );
 312          $ccompletion = new completion_completion($completionparams);
 313  
 314          $completionrows = array();
 315          // Loop through course criteria.
 316          foreach ($completions as $completion) {
 317              $criteria = $completion->get_criteria();
 318  
 319              $completionrow = array();
 320              $completionrow['type'] = $criteria->criteriatype;
 321              $completionrow['title'] = $criteria->get_title();
 322              $completionrow['status'] = $completion->get_status();
 323              $completionrow['complete'] = $completion->is_complete();
 324              $completionrow['timecompleted'] = $completion->timecompleted;
 325              $completionrow['details'] = $criteria->get_details($completion);
 326              $completionrows[] = $completionrow;
 327          }
 328  
 329          $result = array(
 330                    'completed'   => $info->is_course_complete($user->id),
 331                    'aggregation' => $info->get_aggregation_method(),
 332                    'completions' => $completionrows
 333          );
 334  
 335          $results = array(
 336              'completionstatus' => $result,
 337              'warnings' => $warnings
 338          );
 339          return $results;
 340  
 341      }

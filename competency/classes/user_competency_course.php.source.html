 155       * Validate the proficiency.
 156       *
 157       * @param int $value The value.
 158       * @return true|lang_string
 159       */
 160      protected function validate_proficiency($value) {
 161          $grade = $this->get('grade');
 162  
 163          if ($grade !== null && $value === null) {
 164              // We must set a proficiency when we set a grade.
 165              return new lang_string('invaliddata', 'error');
 166  
 167          } else if ($grade === null && $value !== null) {
 168              // We must not set a proficiency when we don't set a grade.
 169              return new lang_string('invaliddata', 'error');
 170          }
 171  
 172          return true;
 173      }

 181      protected function validate_grade($value) {

 184                  return new lang_string('invalidgrade', 'core_competency');
 185              }
 186  
 187              // TODO MDL-52243 Use a core method to validate the grade_scale item.
 188              // Check if grade exist in the scale item values.
 189              $competency = $this->get_competency();
 190              if (!array_key_exists($value - 1 , $competency->get_scale()->scale_items)) {
 191                  return new lang_string('invalidgrade', 'core_competency');

 264      public static function get_least_proficient_competencies_for_course($courseid, $skip = 0, $limit = 0) {
 265          global $DB;
 266  
 267          $fields = competency::get_sql_fields('c', 'c_');
 268          $params = array('courseid' => $courseid);
 269          $sql = 'SELECT ' . $fields . '
 270                    FROM (SELECT cc.competencyid, SUM(COALESCE(ucc.proficiency, 0)) AS timesproficient
 271                            FROM {' . course_competency::TABLE . '} cc
 272                       LEFT JOIN {' . self::TABLE . '} ucc
 273                                  ON ucc.competencyid = cc.competencyid
 274                                 AND ucc.courseid = cc.courseid
 275                           WHERE cc.courseid = :courseid
 276                        GROUP BY cc.competencyid
 277                       ) p
 278                    JOIN {' . competency::TABLE . '} c
 279                      ON c.id = p.competencyid
 280                ORDER BY p.timesproficient ASC, c.id DESC';
 281  
 282          $results = $DB->get_records_sql($sql, $params, $skip, $limit);
 283          $a = $DB->get_records_sql('SELECT * from {' . self::TABLE . '}');
 284  
 285          $comps = array();
 286          foreach ($results as $r) {
 287              $c = competency::extract_record($r, 'c_');
 288              $comps[] = new competency(0, $c);
 289          }
 290          return $comps;
 291      }

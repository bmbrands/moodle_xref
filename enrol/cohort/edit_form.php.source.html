 112      protected function add_add_buttons() {
 113          $mform = $this->_form;
 114          $buttonarray = array();
 115          $buttonarray[0] = $mform->createElement('submit', 'submitbutton', get_string('addinstance', 'enrol'));
 116          $buttonarray[1] = $mform->createElement('submit', 'submitbuttonnext', get_string('addinstanceanother', 'enrol'));
 117          $buttonarray[2] = $mform->createElement('cancel');
 118          $mform->addGroup($buttonarray, 'buttonar', '', array(' '), false);
 119          $mform->closeHeaderBefore('buttonar');
 120      }

 122      function validation($data, $files) {
 123          global $DB;
 124  
 125          $errors = parent::validation($data, $files);
 126  
 127          $params = array('roleid'=>$data['roleid'], 'customint1'=>$data['customint1'], 'courseid'=>$data['courseid'], 'id'=>$data['id']);
 128          if ($DB->record_exists_select('enrol', "roleid = :roleid AND customint1 = :customint1 AND courseid = :courseid AND enrol = 'cohort' AND id <> :id", $params)) {
 129              $errors['roleid'] = get_string('instanceexists', 'enrol_cohort');

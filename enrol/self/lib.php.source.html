  35      /**
  36       * Returns optional enrolment information icons.
  37       *
  38       * This is used in course list for quick overview of enrolment options.
  39       *
  40       * We are not using single instance parameter because sometimes
  41       * we might want to prevent icon repetition when multiple instances
  42       * of one type exist. One instance may also produce several icons.
  43       *
  44       * @param array $instances all enrol instances of this type in one course
  45       * @return array of pix_icon
  46       */
  47      public function get_info_icons(array $instances) {
  48          $key = false;
  49          $nokey = false;
  50          foreach ($instances as $instance) {
  51              if ($this->can_self_enrol($instance, false) !== true) {
  52                  // User can not enrol himself.
  53                  // Note that we do not check here if user is already enrolled for performance reasons -
  54                  // such check would execute extra queries for each course in the list of courses and
  55                  // would hide self-enrolment icons from guests.
  56                  continue;
  57              }
  58              if ($instance->password or $instance->customint1) {
  59                  $key = true;
  60              } else {
  61                  $nokey = true;
  62              }
  63          }
  64          $icons = array();
  65          if ($nokey) {
  66              $icons[] = new pix_icon('withoutkey', get_string('pluginname', 'enrol_self'), 'enrol_self');
  67          }
  68          if ($key) {
  69              $icons[] = new pix_icon('withkey', get_string('pluginname', 'enrol_self'), 'enrol_self');
  70          }
  71          return $icons;
  72      }

 178      /**
 179       * Self enrol user to course
 180       *
 181       * @param stdClass $instance enrolment instance
 182       * @param stdClass $data data needed for enrolment.
 183       * @return bool|array true if enroled else eddor code and messege
 184       */
 185      public function enrol_self(stdClass $instance, $data = null) {
 186          global $DB, $USER, $CFG;
 187  
 188          // Don't enrol user if password is not passed when required.
 189          if ($instance->password && !isset($data->enrolpassword)) {
 190              return;
 191          }
 192  
 193          $timestart = time();
 194          if ($instance->enrolperiod) {
 195              $timeend = $timestart + $instance->enrolperiod;
 196          } else {
 197              $timeend = 0;
 198          }
 199  
 200          $this->enrol_user($instance, $USER->id, $instance->roleid, $timestart, $timeend);
 201  
 202          if ($instance->password and $instance->customint1 and $data->enrolpassword !== $instance->password) {
 203              // It must be a group enrolment, let's assign group too.
 204              $groups = $DB->get_records('groups', array('courseid'=>$instance->courseid), 'id', 'id, enrolmentkey');
 205              foreach ($groups as $group) {
 206                  if (empty($group->enrolmentkey)) {
 207                      continue;
 208                  }
 209                  if ($group->enrolmentkey === $data->enrolpassword) {
 210                      // Add user to group.
 211                      require_once($CFG->dirroot.'/group/lib.php');
 212                      groups_add_member($group->id, $USER->id);
 213                      break;
 214                  }
 215              }
 216          }
 217          // Send welcome message.
 218          if ($instance->customint4) {
 219              $this->email_welcome_message($instance, $USER);
 220          }
 221      }

 256      /**
 257       * Checks if user can self enrol.
 258       *
 259       * @param stdClass $instance enrolment instance
 260       * @param bool $checkuserenrolment if true will check if user enrolment is inactive.
 261       *             used by navigation to improve performance.
 262       * @return bool|string true if successful, else error message or false.
 263       */
 264      public function can_self_enrol(stdClass $instance, $checkuserenrolment = true) {
 265          global $CFG, $DB, $OUTPUT, $USER;
 266  
 267          if ($checkuserenrolment) {
 268              if (isguestuser()) {
 269                  // Can not enrol guest.
 270                  return get_string('noguestaccess', 'enrol') . $OUTPUT->continue_button(get_login_url());
 271              }
 272              // Check if user is already enroled.
 273              if ($DB->get_record('user_enrolments', array('userid' => $USER->id, 'enrolid' => $instance->id))) {
 274                  return get_string('canntenrol', 'enrol_self');
 275              }
 276          }
 277  
 278          if ($instance->status != ENROL_INSTANCE_ENABLED) {
 279              return get_string('canntenrol', 'enrol_self');
 280          }
 281  
 282          if ($instance->enrolstartdate != 0 and $instance->enrolstartdate > time()) {
 283              return get_string('canntenrol', 'enrol_self');
 284          }
 285  
 286          if ($instance->enrolenddate != 0 and $instance->enrolenddate < time()) {
 287              return get_string('canntenrol', 'enrol_self');
 288          }
 289  
 290          if (!$instance->customint6) {
 291              // New enrols not allowed.
 292              return get_string('canntenrol', 'enrol_self');
 293          }
 294  
 295          if ($DB->record_exists('user_enrolments', array('userid' => $USER->id, 'enrolid' => $instance->id))) {
 296              return get_string('canntenrol', 'enrol_self');
 297          }
 298  
 299          if ($instance->customint3 > 0) {
 300              // Max enrol limit specified.
 301              $count = $DB->count_records('user_enrolments', array('enrolid' => $instance->id));
 302              if ($count >= $instance->customint3) {
 303                  // Bad luck, no more self enrolments here.
 304                  return get_string('maxenrolledreached', 'enrol_self');
 305              }
 306          }
 307  
 308          if ($instance->customint5) {
 309              require_once("$CFG->dirroot/cohort/lib.php");
 310              if (!cohort_is_member($instance->customint5, $USER->id)) {
 311                  $cohort = $DB->get_record('cohort', array('id' => $instance->customint5));
 312                  if (!$cohort) {
 313                      return null;
 314                  }
 315                  $a = format_string($cohort->name, true, array('context' => context::instance_by_id($cohort->contextid)));
 316                  return markdown_to_html(get_string('cohortnonmemberinfo', 'enrol_self', $a));
 317              }
 318          }
 319  
 320          return true;
 321      }

 403      protected function email_welcome_message($instance, $user) {
 404          global $CFG, $DB;
 405  
 406          $course = $DB->get_record('course', array('id'=>$instance->courseid), '*', MUST_EXIST);
 407          $context = context_course::instance($course->id);
 408  
 409          $a = new stdClass();
 410          $a->coursename = format_string($course->fullname, true, array('context'=>$context));
 411          $a->profileurl = "$CFG->wwwroot/user/view.php?id=$user->id&course=$course->id";
 412  
 413          if (trim($instance->customtext1) !== '') {
 414              $message = $instance->customtext1;
 415              $key = array('{$a->coursename}', '{$a->profileurl}', '{$a->fullname}', '{$a->email}');
 416              $value = array($a->coursename, $a->profileurl, fullname($user), $user->email);
 417              $message = str_replace($key, $value, $message);
 418              if (strpos($message, '<') === false) {
 419                  // Plain text only.
 420                  $messagetext = $message;
 421                  $messagehtml = text_to_html($messagetext, null, false, true);
 422              } else {
 423                  // This is most probably the tag/newline soup known as FORMAT_MOODLE.
 424                  $messagehtml = format_text($message, FORMAT_MOODLE, array('context'=>$context, 'para'=>false, 'newlines'=>true, 'filter'=>true));
 425                  $messagetext = html_to_text($messagehtml);
 426              }
 427          } else {
 428              $messagetext = get_string('welcometocoursetext', 'enrol_self', $a);
 429              $messagehtml = text_to_html($messagetext, null, false, true);
 430          }
 431  
 432          $subject = get_string('welcometocourse', 'enrol_self', format_string($course->fullname, true, array('context'=>$context)));
 433  
 434          $rusers = array();
 435          if (!empty($CFG->coursecontact)) {
 436              $croles = explode(',', $CFG->coursecontact);
 437              list($sort, $sortparams) = users_order_by_sql('u');
 438              // We only use the first user.
 439              $i = 0;
 440              do {
 441                  $rusers = get_role_users($croles[$i], $context, true, '',
 442                      'r.sortorder ASC, ' . $sort, null, '', '', '', '', $sortparams);
 443                  $i++;
 444              } while (empty($rusers) && !empty($croles[$i]));
 445          }
 446          if ($rusers) {
 447              $contact = reset($rusers);
 448          } else {
 449              $contact = core_user::get_support_user();
 450          }
 451  
 452          // Directly emailing welcome message rather than using messaging.
 453          email_to_user($user, $contact, $subject, $messagetext, $messagehtml);
 454      }

 466      /**
 467       * Sync all meta course links.
 468       *
 469       * @param progress_trace $trace
 470       * @param int $courseid one course, empty mean all
 471       * @return int 0 means ok, 1 means error, 2 means plugin disabled
 472       */
 473      public function sync(progress_trace $trace, $courseid = null) {
 474          global $DB;
 475  
 476          if (!enrol_is_enabled('self')) {
 477              $trace->finished();
 478              return 2;
 479          }
 480  
 481          // Unfortunately this may take a long time, execution can be interrupted safely here.
 482          core_php_time_limit::raise();
 483          raise_memory_limit(MEMORY_HUGE);
 484  
 485          $trace->output('Verifying self-enrolments...');
 486  
 487          $params = array('now'=>time(), 'useractive'=>ENROL_USER_ACTIVE, 'courselevel'=>CONTEXT_COURSE);
 488          $coursesql = "";
 489          if ($courseid) {
 490              $coursesql = "AND e.courseid = :courseid";
 491              $params['courseid'] = $courseid;
 492          }
 493  
 494          // Note: the logic of self enrolment guarantees that user logged in at least once (=== u.lastaccess set)
 495          //       and that user accessed course at least once too (=== user_lastaccess record exists).
 496  
 497          // First deal with users that did not log in for a really long time - they do not have user_lastaccess records.
 498          $sql = "SELECT e.*, ue.userid
 499                    FROM {user_enrolments} ue
 500                    JOIN {enrol} e ON (e.id = ue.enrolid AND e.enrol = 'self' AND e.customint2 > 0)
 501                    JOIN {user} u ON u.id = ue.userid
 502                   WHERE :now - u.lastaccess > e.customint2
 503                         $coursesql";
 504          $rs = $DB->get_recordset_sql($sql, $params);
 505          foreach ($rs as $instance) {
 506              $userid = $instance->userid;
 507              unset($instance->userid);
 508              $this->unenrol_user($instance, $userid);
 509              $days = $instance->customint2 / 60*60*24;
 510              $trace->output("unenrolling user $userid from course $instance->courseid as they have did not log in for at least $days days", 1);
 511          }
 512          $rs->close();
 513  
 514          // Now unenrol from course user did not visit for a long time.
 515          $sql = "SELECT e.*, ue.userid
 516                    FROM {user_enrolments} ue
 517                    JOIN {enrol} e ON (e.id = ue.enrolid AND e.enrol = 'self' AND e.customint2 > 0)
 518                    JOIN {user_lastaccess} ul ON (ul.userid = ue.userid AND ul.courseid = e.courseid)
 519                   WHERE :now - ul.timeaccess > e.customint2
 520                         $coursesql";
 521          $rs = $DB->get_recordset_sql($sql, $params);
 522          foreach ($rs as $instance) {
 523              $userid = $instance->userid;
 524              unset($instance->userid);
 525              $this->unenrol_user($instance, $userid);
 526                  $days = $instance->customint2 / 60*60*24;
 527              $trace->output("unenrolling user $userid from course $instance->courseid as they have did not access course for at least $days days", 1);
 528          }
 529          $rs->close();
 530  
 531          $trace->output('...user self-enrolment updates finished.');
 532          $trace->finished();
 533  
 534          $this->process_expirations($trace, $courseid);
 535  
 536          return 0;
 537      }

 539      /**
 540       * Returns the user who is responsible for self enrolments in given instance.
 541       *
 542       * Usually it is the first editing teacher - the person with "highest authority"
 543       * as defined by sort_by_roleassignment_authority() having 'enrol/self:manage'
 544       * capability.
 545       *
 546       * @param int $instanceid enrolment instance id
 547       * @return stdClass user record
 548       */
 549      protected function get_enroller($instanceid) {
 550          global $DB;
 551  
 552          if ($this->lasternollerinstanceid == $instanceid and $this->lasternoller) {
 553              return $this->lasternoller;
 554          }
 555  
 556          $instance = $DB->get_record('enrol', array('id'=>$instanceid, 'enrol'=>$this->get_name()), '*', MUST_EXIST);
 557          $context = context_course::instance($instance->courseid);
 558  
 559          if ($users = get_enrolled_users($context, 'enrol/self:manage')) {
 560              $users = sort_by_roleassignment_authority($users, $context);
 561              $this->lasternoller = reset($users);
 562              unset($users);
 563          } else {
 564              $this->lasternoller = parent::get_enroller($instanceid);
 565          }
 566  
 567          $this->lasternollerinstanceid = $instanceid;
 568  
 569          return $this->lasternoller;
 570      }

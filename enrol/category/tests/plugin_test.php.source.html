  44      protected function enable_role_sync($roleid) {
  45          global $DB;
  46  
  47          $syscontext = context_system::instance();
  48  
  49          if ($rc = $DB->record_exists('role_capabilities', array('capability'=>'enrol/category:synchronised', 'roleid'=>$roleid, 'contextid'=>$syscontext->id))) {
  50              if ($rc->permission != CAP_ALLOW) {
  51                  $rc->permission = CAP_ALLOW;
  52                  $DB->update_record('role_capabilities', $rc);
  53              }
  54          } else {
  55              $rc = new stdClass();
  56              $rc->capability = 'enrol/category:synchronised';
  57              $rc->roleid = $roleid;
  58              $rc->contextid = $syscontext->id;
  59              $rc->permission = CAP_ALLOW;
  60              $rc->timemodified = time();
  61              $rc->modifierid = 0;
  62              $DB->insert_record('role_capabilities', $rc);
  63          }
  64      }

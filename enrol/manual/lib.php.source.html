 295      /**
 296       * Sync all meta course links.
 297       *
 298       * @param progress_trace $trace
 299       * @param int $courseid one course, empty mean all
 300       * @return int 0 means ok, 1 means error, 2 means plugin disabled
 301       */
 302      public function sync(progress_trace $trace, $courseid = null) {
 303          global $DB;
 304  
 305          if (!enrol_is_enabled('manual')) {
 306              $trace->finished();
 307              return 2;
 308          }
 309  
 310          // Unfortunately this may take a long time, execution can be interrupted safely here.
 311          core_php_time_limit::raise();
 312          raise_memory_limit(MEMORY_HUGE);
 313  
 314          $trace->output('Verifying manual enrolment expiration...');
 315  
 316          $params = array('now'=>time(), 'useractive'=>ENROL_USER_ACTIVE, 'courselevel'=>CONTEXT_COURSE);
 317          $coursesql = "";
 318          if ($courseid) {
 319              $coursesql = "AND e.courseid = :courseid";
 320              $params['courseid'] = $courseid;
 321          }
 322  
 323          // Deal with expired accounts.
 324          $action = $this->get_config('expiredaction', ENROL_EXT_REMOVED_KEEP);
 325  
 326          if ($action == ENROL_EXT_REMOVED_UNENROL) {
 327              $instances = array();
 328              $sql = "SELECT ue.*, e.courseid, c.id AS contextid
 329                        FROM {user_enrolments} ue
 330                        JOIN {enrol} e ON (e.id = ue.enrolid AND e.enrol = 'manual')
 331                        JOIN {context} c ON (c.instanceid = e.courseid AND c.contextlevel = :courselevel)
 332                       WHERE ue.timeend > 0 AND ue.timeend < :now
 333                             $coursesql";
 334              $rs = $DB->get_recordset_sql($sql, $params);
 335              foreach ($rs as $ue) {
 336                  if (empty($instances[$ue->enrolid])) {
 337                      $instances[$ue->enrolid] = $DB->get_record('enrol', array('id'=>$ue->enrolid));
 338                  }
 339                  $instance = $instances[$ue->enrolid];
 340                  // Always remove all manually assigned roles here, this may break enrol_self roles but we do not want hardcoded hacks here.
 341                  role_unassign_all(array('userid'=>$ue->userid, 'contextid'=>$ue->contextid, 'component'=>'', 'itemid'=>0), true);
 342                  $this->unenrol_user($instance, $ue->userid);
 343                  $trace->output("unenrolling expired user $ue->userid from course $instance->courseid", 1);
 344              }
 345              $rs->close();
 346              unset($instances);
 347  
 348          } else if ($action == ENROL_EXT_REMOVED_SUSPENDNOROLES or $action == ENROL_EXT_REMOVED_SUSPEND) {
 349              $instances = array();
 350              $sql = "SELECT ue.*, e.courseid, c.id AS contextid
 351                        FROM {user_enrolments} ue
 352                        JOIN {enrol} e ON (e.id = ue.enrolid AND e.enrol = 'manual')
 353                        JOIN {context} c ON (c.instanceid = e.courseid AND c.contextlevel = :courselevel)
 354                       WHERE ue.timeend > 0 AND ue.timeend < :now
 355                             AND ue.status = :useractive
 356                             $coursesql";
 357              $rs = $DB->get_recordset_sql($sql, $params);
 358              foreach ($rs as $ue) {
 359                  if (empty($instances[$ue->enrolid])) {
 360                      $instances[$ue->enrolid] = $DB->get_record('enrol', array('id'=>$ue->enrolid));
 361                  }
 362                  $instance = $instances[$ue->enrolid];
 363                  if ($action == ENROL_EXT_REMOVED_SUSPENDNOROLES) {
 364                      // Remove all manually assigned roles here, this may break enrol_self roles but we do not want hardcoded hacks here.
 365                      role_unassign_all(array('userid'=>$ue->userid, 'contextid'=>$ue->contextid, 'component'=>'', 'itemid'=>0), true);
 366                      $this->update_user_enrol($instance, $ue->userid, ENROL_USER_SUSPENDED);
 367                      $trace->output("suspending expired user $ue->userid in course $instance->courseid, roles unassigned", 1);
 368                  } else {
 369                      $this->update_user_enrol($instance, $ue->userid, ENROL_USER_SUSPENDED);
 370                      $trace->output("suspending expired user $ue->userid in course $instance->courseid, roles kept", 1);
 371                  }
 372              }
 373              $rs->close();
 374              unset($instances);
 375  
 376          } else {
 377              // ENROL_EXT_REMOVED_KEEP means no changes.
 378          }
 379  
 380          $trace->output('...manual enrolment updates finished.');
 381          $trace->finished();
 382  
 383          return 0;
 384      }

 386      /**
 387       * Returns the user who is responsible for manual enrolments in given instance.
 388       *
 389       * Usually it is the first editing teacher - the person with "highest authority"
 390       * as defined by sort_by_roleassignment_authority() having 'enrol/manual:manage'
 391       * capability.
 392       *
 393       * @param int $instanceid enrolment instance id
 394       * @return stdClass user record
 395       */
 396      protected function get_enroller($instanceid) {
 397          global $DB;
 398  
 399          if ($this->lasternollerinstanceid == $instanceid and $this->lasternoller) {
 400              return $this->lasternoller;
 401          }
 402  
 403          $instance = $DB->get_record('enrol', array('id'=>$instanceid, 'enrol'=>$this->get_name()), '*', MUST_EXIST);
 404          $context = context_course::instance($instance->courseid);
 405  
 406          if ($users = get_enrolled_users($context, 'enrol/manual:manage')) {
 407              $users = sort_by_roleassignment_authority($users, $context);
 408              $this->lasternoller = reset($users);
 409              unset($users);
 410          } else {
 411              $this->lasternoller = parent::get_enroller($instanceid);
 412          }
 413  
 414          $this->lasternollerinstanceid = $instanceid;
 415  
 416          return $this->lasternoller;
 417      }

 482      /**
 483       * Restore user enrolment.
 484       *
 485       * @param restore_enrolments_structure_step $step
 486       * @param stdClass $data
 487       * @param stdClass $instance
 488       * @param int $oldinstancestatus
 489       * @param int $userid
 490       */
 491      public function restore_user_enrolment(restore_enrolments_structure_step $step, $data, $instance, $userid, $oldinstancestatus) {
 492          global $DB;
 493  
 494          // Note: this is a bit tricky because other types may be converted to manual enrolments,
 495          //       and manual is restricted to one enrolment per user.
 496  
 497          $ue = $DB->get_record('user_enrolments', array('enrolid'=>$instance->id, 'userid'=>$userid));
 498          $enrol = false;
 499          if ($ue and $ue->status == ENROL_USER_ACTIVE) {
 500              // We do not want to restrict current active enrolments, let's kind of merge the times only.
 501              // This prevents some teacher lockouts too.
 502              if ($data->status == ENROL_USER_ACTIVE) {
 503                  if ($data->timestart > $ue->timestart) {
 504                      $data->timestart = $ue->timestart;
 505                      $enrol = true;
 506                  }
 507  
 508                  if ($data->timeend == 0) {
 509                      if ($ue->timeend != 0) {
 510                          $enrol = true;
 511                      }
 512                  } else if ($ue->timeend == 0) {
 513                      $data->timeend = 0;
 514                  } else if ($data->timeend < $ue->timeend) {
 515                      $data->timeend = $ue->timeend;
 516                      $enrol = true;
 517                  }
 518              }
 519          } else {
 520              if ($instance->status == ENROL_INSTANCE_ENABLED and $oldinstancestatus != ENROL_INSTANCE_ENABLED) {
 521                  // Make sure that user enrolments are not activated accidentally,
 522                  // we do it only here because it is not expected that enrolments are migrated to other plugins.
 523                  $data->status = ENROL_USER_SUSPENDED;
 524              }
 525              $enrol = true;
 526          }
 527  
 528          if ($enrol) {
 529              $this->enrol_user($instance, $userid, null, $data->timestart, $data->timeend, $data->status);
 530          }
 531      }

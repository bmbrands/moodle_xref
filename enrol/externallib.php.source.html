  79      /**
  80       * Return users that have the capabilities for each course specified. For each course and capability specified,
  81       * a list of the users that are enrolled in the course and have that capability are returned.
  82       *
  83       * @param array $coursecapabilities array of course ids and associated capability names {courseid, {capabilities}}
  84       * @return array An array of arrays describing users for each associated courseid and capability
  85       * @since  Moodle 2.4
  86       */
  87      public static function get_enrolled_users_with_capability($coursecapabilities, $options) {
  88          global $CFG, $DB;
  89          require_once($CFG->dirroot . "/user/lib.php");
  90  
  91          if (empty($coursecapabilities)) {
  92              throw new invalid_parameter_exception('Parameter can not be empty');
  93          }
  94          $params = self::validate_parameters(self::get_enrolled_users_with_capability_parameters(),
  95              array ('coursecapabilities' => $coursecapabilities,  'options'=>$options));
  96          $result = array();
  97          $userlist = array();
  98          $groupid        = 0;
  99          $onlyactive     = false;
 100          $userfields     = array();
 101          $limitfrom = 0;
 102          $limitnumber = 0;
 103          foreach ($params['options'] as $option) {
 104              switch ($option['name']) {
 105                  case 'groupid':
 106                      $groupid = (int)$option['value'];
 107                      break;
 108                  case 'onlyactive':
 109                      $onlyactive = !empty($option['value']);
 110                      break;
 111                  case 'userfields':
 112                      $thefields = explode(',', $option['value']);
 113                      foreach ($thefields as $f) {
 114                          $userfields[] = clean_param($f, PARAM_ALPHANUMEXT);
 115                      }
 116                      break;
 117                  case 'limitfrom' :
 118                      $limitfrom = clean_param($option['value'], PARAM_INT);
 119                      break;
 120                  case 'limitnumber' :
 121                      $limitnumber = clean_param($option['value'], PARAM_INT);
 122                      break;
 123              }
 124          }
 125  
 126          foreach ($params['coursecapabilities'] as $coursecapability) {
 127              $courseid = $coursecapability['courseid'];
 128              $course = $DB->get_record('course', array('id'=>$courseid), '*', MUST_EXIST);
 129              $coursecontext = context_course::instance($courseid);
 130              if (!$coursecontext) {
 131                  throw new moodle_exception('cannotfindcourse', 'error', '', null,
 132                          'The course id ' . $courseid . ' doesn\'t exist.');
 133              }
 134              if ($courseid == SITEID) {
 135                  $context = context_system::instance();
 136              } else {
 137                  $context = $coursecontext;
 138              }
 139              try {
 140                  self::validate_context($context);
 141              } catch (Exception $e) {
 142                  $exceptionparam = new stdClass();
 143                  $exceptionparam->message = $e->getMessage();
 144                  $exceptionparam->courseid = $params['courseid'];
 145                  throw new moodle_exception(get_string('errorcoursecontextnotvalid' , 'webservice', $exceptionparam));
 146              }
 147  
 148              if ($courseid == SITEID) {
 149                  require_capability('moodle/site:viewparticipants', $context);
 150              } else {
 151                  require_capability('moodle/course:viewparticipants', $context);
 152              }
 153              // The accessallgroups capability is needed to use this option.
 154              if (!empty($groupid) && groups_is_member($groupid)) {
 155                  require_capability('moodle/site:accessallgroups', $coursecontext);
 156              }
 157              // The course:enrolereview capability is needed to use this option.
 158              if ($onlyactive) {
 159                  require_capability('moodle/course:enrolreview', $coursecontext);
 160              }
 161  
 162              // To see the permissions of others role:review capability is required.
 163              require_capability('moodle/role:review', $coursecontext);
 164              foreach ($coursecapability['capabilities'] as $capability) {
 165                  $courseusers['courseid'] = $courseid;
 166                  $courseusers['capability'] = $capability;
 167  
 168                  list($enrolledsql, $enrolledparams) = get_enrolled_sql($coursecontext, $capability, $groupid, $onlyactive);
 169  
 170                  $sql = "SELECT u.* FROM {user} u WHERE u.id IN ($enrolledsql) ORDER BY u.id ASC";
 171  
 172                  $enrolledusers = $DB->get_recordset_sql($sql, $enrolledparams, $limitfrom, $limitnumber);
 173                  $users = array();
 174                  foreach ($enrolledusers as $courseuser) {
 175                      if ($userdetails = user_get_user_details($courseuser, $course, $userfields)) {
 176                          $users[] = $userdetails;
 177                      }
 178                  }
 179                  $enrolledusers->close();
 180                  $courseusers['users'] = $users;
 181                  $result[] = $courseusers;
 182              }
 183          }
 184          return $result;
 185      }

 288      /**
 289       * Get list of courses user is enrolled in (only active enrolments are returned).
 290       * Please note the current user must be able to access the course, otherwise the course is not included.
 291       *
 292       * @param int $userid
 293       * @return array of courses
 294       */
 295      public static function get_users_courses($userid) {
 296          global $USER, $DB;
 297  
 298          // Do basic automatic PARAM checks on incoming data, using params description
 299          // If any problems are found then exceptions are thrown with helpful error messages
 300          $params = self::validate_parameters(self::get_users_courses_parameters(), array('userid'=>$userid));
 301  
 302          $courses = enrol_get_users_courses($params['userid'], true, 'id, shortname, fullname, idnumber, visible,
 303                     summary, summaryformat, format, showgrades, lang, enablecompletion');
 304          $result = array();
 305  
 306          foreach ($courses as $course) {
 307              $context = context_course::instance($course->id, IGNORE_MISSING);
 308              try {
 309                  self::validate_context($context);
 310              } catch (Exception $e) {
 311                  // current user can not access this course, sorry we can not disclose who is enrolled in this course!
 312                  continue;
 313              }
 314  
 315              if ($userid != $USER->id and !has_capability('moodle/course:viewparticipants', $context)) {
 316                  // we need capability to view participants
 317                  continue;
 318              }
 319  
 320              list($enrolledsqlselect, $enrolledparams) = get_enrolled_sql($context);
 321              $enrolledsql = "SELECT COUNT('x') FROM ($enrolledsqlselect) enrolleduserids";
 322              $enrolledusercount = $DB->count_records_sql($enrolledsql, $enrolledparams);
 323  
 324              list($course->summary, $course->summaryformat) =
 325                  external_format_text($course->summary, $course->summaryformat, $context->id, 'course', 'summary', null);
 326  
 327              $result[] = array('id' => $course->id, 'shortname' => $course->shortname, 'fullname' => $course->fullname,
 328                  'idnumber' => $course->idnumber, 'visible' => $course->visible, 'enrolledusercount' => $enrolledusercount,
 329                  'summary' => $course->summary, 'summaryformat' => $course->summaryformat, 'format' => $course->format,
 330                  'showgrades' => $course->showgrades, 'lang' => $course->lang, 'enablecompletion' => $course->enablecompletion
 331                  );
 332          }
 333  
 334          return $result;
 335      }

 391       * Get course participants details
 392       *
 393       * @param int $courseid  course id
 394       * @param array $options options {
 395       *                                'name' => option name
 396       *                                'value' => option value
 397       *                               }
 398       * @return array An array of users
 399       */
 400      public static function get_enrolled_users($courseid, $options = array()) {
 401          global $CFG, $USER, $DB;
 402          require_once($CFG->dirroot . "/user/lib.php");
 403  
 404          $params = self::validate_parameters(
 405              self::get_enrolled_users_parameters(),
 406              array(
 407                  'courseid'=>$courseid,
 408                  'options'=>$options
 409              )
 410          );
 411          $withcapability = '';
 412          $groupid        = 0;
 413          $onlyactive     = false;
 414          $userfields     = array();
 415          $limitfrom = 0;
 416          $limitnumber = 0;
 417          foreach ($options as $option) {
 418              switch ($option['name']) {
 419              case 'withcapability':
 420                  $withcapability = $option['value'];
 421                  break;
 422              case 'groupid':
 423                  $groupid = (int)$option['value'];
 424                  break;
 425              case 'onlyactive':
 426                  $onlyactive = !empty($option['value']);
 427                  break;
 428              case 'userfields':
 429                  $thefields = explode(',', $option['value']);
 430                  foreach ($thefields as $f) {
 431                      $userfields[] = clean_param($f, PARAM_ALPHANUMEXT);
 432                  }
 433                  break;
 434              case 'limitfrom' :
 435                  $limitfrom = clean_param($option['value'], PARAM_INT);
 436                  break;
 437              case 'limitnumber' :
 438                  $limitnumber = clean_param($option['value'], PARAM_INT);
 439                  break;
 440              }
 441          }
 442  
 443          $course = $DB->get_record('course', array('id'=>$courseid), '*', MUST_EXIST);
 444          $coursecontext = context_course::instance($courseid, IGNORE_MISSING);
 445          if ($courseid == SITEID) {
 446              $context = context_system::instance();
 447          } else {
 448              $context = $coursecontext;
 449          }
 450          try {
 451              self::validate_context($context);
 452          } catch (Exception $e) {
 453              $exceptionparam = new stdClass();
 454              $exceptionparam->message = $e->getMessage();
 455              $exceptionparam->courseid = $params['courseid'];
 456              throw new moodle_exception('errorcoursecontextnotvalid' , 'webservice', '', $exceptionparam);
 457          }
 458  
 459          if ($courseid == SITEID) {
 460              require_capability('moodle/site:viewparticipants', $context);
 461          } else {
 462              require_capability('moodle/course:viewparticipants', $context);
 463          }
 464          // to overwrite this parameter, you need role:review capability
 465          if ($withcapability) {
 466              require_capability('moodle/role:review', $coursecontext);
 467          }
 468          // need accessallgroups capability if you want to overwrite this option
 469          if (!empty($groupid) && groups_is_member($groupid)) {
 470              require_capability('moodle/site:accessallgroups', $coursecontext);
 471          }
 472          // to overwrite this option, you need course:enrolereview permission
 473          if ($onlyactive) {
 474              require_capability('moodle/course:enrolreview', $coursecontext);
 475          }
 476  
 477          list($enrolledsql, $enrolledparams) = get_enrolled_sql($coursecontext, $withcapability, $groupid, $onlyactive);
 478          $ctxselect = ', ' . context_helper::get_preload_record_columns_sql('ctx');
 479          $ctxjoin = "LEFT JOIN {context} ctx ON (ctx.instanceid = u.id AND ctx.contextlevel = :contextlevel)";
 480          $enrolledparams['contextlevel'] = CONTEXT_USER;
 481          $sql = "SELECT u.* $ctxselect
 482                    FROM {user} u $ctxjoin
 483                   WHERE u.id IN ($enrolledsql)
 484                   ORDER BY u.id ASC";
 485          $enrolledusers = $DB->get_recordset_sql($sql, $enrolledparams, $limitfrom, $limitnumber);
 486          $users = array();
 487          foreach ($enrolledusers as $user) {
 488              context_helper::preload_from_record($user);
 489              if ($userdetails = user_get_user_details($user, $course, $userfields)) {
 490                  $users[] = $userdetails;
 491              }
 492          }
 493          $enrolledusers->close();
 494  
 495          return $users;
 496      }

 822       * Get list of course participants.
 823       *
 824       * @param int $courseid
 825       * @param text $withcapability
 826       * @param int $groupid
 827       * @param bool $onlyactive
 828       * @return array of course participants
 829       * @since Moodle 2.0
 830       * @deprecated Moodle 2.2 MDL-29106 - Please do not call this function any more.
 831       * @see core_enrol_external::get_enrolled_users()
 832       */
 833      public static function get_enrolled_users($courseid, $withcapability = null, $groupid = null, $onlyactive = false) {
 834          global $DB, $CFG, $USER;
 835  
 836          // Do basic automatic PARAM checks on incoming data, using params description
 837          // If any problems are found then exceptions are thrown with helpful error messages
 838          $params = self::validate_parameters(self::get_enrolled_users_parameters(), array(
 839              'courseid'=>$courseid,
 840              'withcapability'=>$withcapability,
 841              'groupid'=>$groupid,
 842              'onlyactive'=>$onlyactive)
 843          );
 844  
 845          $coursecontext = context_course::instance($params['courseid'], IGNORE_MISSING);
 846          if ($courseid == SITEID) {
 847              $context = context_system::instance();
 848          } else {
 849              $context = $coursecontext;
 850          }
 851  
 852          try {
 853              self::validate_context($context);
 854          } catch (Exception $e) {
 855              $exceptionparam = new stdClass();
 856              $exceptionparam->message = $e->getMessage();
 857              $exceptionparam->courseid = $params['courseid'];
 858              throw new moodle_exception('errorcoursecontextnotvalid' , 'webservice', '', $exceptionparam);
 859          }
 860  
 861          if ($courseid == SITEID) {
 862              require_capability('moodle/site:viewparticipants', $context);
 863          } else {
 864              require_capability('moodle/course:viewparticipants', $context);
 865          }
 866  
 867          if ($withcapability) {
 868              require_capability('moodle/role:review', $coursecontext);
 869          }
 870          if ($groupid && groups_is_member($groupid)) {
 871              require_capability('moodle/site:accessallgroups', $coursecontext);
 872          }
 873          if ($onlyactive) {
 874              require_capability('moodle/course:enrolreview', $coursecontext);
 875          }
 876  
 877          list($sqlparams, $params) =  get_enrolled_sql($coursecontext, $withcapability, $groupid, $onlyactive);
 878          $sql = "SELECT ue.userid, e.courseid, u.firstname, u.lastname, u.username, c.id as usercontextid
 879                    FROM {user_enrolments} ue
 880                    JOIN {enrol} e ON (e.id = ue.enrolid)
 881                    JOIN {user} u ON (ue.userid = u.id)
 882                    JOIN {context} c ON (u.id = c.instanceid AND contextlevel = " . CONTEXT_USER . ")
 883                    WHERE e.courseid = :courseid AND ue.userid IN ($sqlparams)
 884                    GROUP BY ue.userid, e.courseid, u.firstname, u.lastname, u.username, c.id";
 885          $params['courseid'] = $courseid;
 886          $enrolledusers = $DB->get_records_sql($sql, $params);
 887          $result = array();
 888          $isadmin = is_siteadmin($USER);
 889          $canviewfullnames = has_capability('moodle/site:viewfullnames', $context);
 890          foreach ($enrolledusers as $enrolleduser) {
 891              $profilimgurl = moodle_url::make_pluginfile_url($enrolleduser->usercontextid, 'user', 'icon', NULL, '/', 'f1');
 892              $profilimgurlsmall = moodle_url::make_pluginfile_url($enrolleduser->usercontextid, 'user', 'icon', NULL, '/', 'f2');
 893              $resultuser = array(
 894                  'courseid' => $enrolleduser->courseid,
 895                  'userid' => $enrolleduser->userid,
 896                  'fullname' => fullname($enrolleduser),
 897                  'profileimgurl' => $profilimgurl->out(false),
 898                  'profileimgurlsmall' => $profilimgurlsmall->out(false)
 899              );
 900              // check if we can return username
 901              if ($isadmin) {
 902                  $resultuser['username'] = $enrolleduser->username;
 903              }
 904              // check if we can return first and last name
 905              if ($isadmin or $canviewfullnames) {
 906                  $resultuser['firstname'] = $enrolleduser->firstname;
 907                  $resultuser['lastname'] = $enrolleduser->lastname;
 908              }
 909              $result[] = $resultuser;
 910          }
 911  
 912          return $result;
 913      }

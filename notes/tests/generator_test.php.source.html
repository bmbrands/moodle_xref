  36      /** Test create_instance method */
  37      public function test_create_instance() {
  38          global $DB;
  39          $this->resetAfterTest();
  40          $this->setAdminUser();
  41  
  42          $course = $this->getDataGenerator()->create_course();
  43          $user = $this->getDataGenerator()->create_user();
  44          $gen = $this->getDataGenerator()->get_plugin_generator('core_notes');
  45  
  46          $this->assertFalse($DB->record_exists('post', array('courseid' => $course->id)));
  47          $note = $gen->create_instance(array('courseid' => $course->id, 'userid' => $user->id));
  48          $this->assertEquals(1, $DB->count_records('post', array('courseid' => $course->id, 'userid' => $user->id)));
  49          $this->assertTrue($DB->record_exists('post', array('id' => $note->id)));
  50  
  51          $params = array('courseid' => $course->id, 'userid' => $user->id, 'publishstate' => NOTES_STATE_DRAFT);
  52          $note = $gen->create_instance($params);
  53          $this->assertEquals(2, $DB->count_records('post', array('courseid' => $course->id, 'userid' => $user->id)));
  54          $this->assertEquals(NOTES_STATE_DRAFT, $DB->get_field_select('post', 'publishstate', 'id = :id',
  55                  array('id' => $note->id)));
  56      }

  46      /**
  47       * Tests constructing and using date condition as part of tree.
  48       */
  49      public function test_in_tree() {
  50          global $SITE, $USER, $CFG;
  51          $this->resetAfterTest();
  52          $this->setAdminUser();
  53  
  54          // Set server timezone for test. (Important as otherwise the timezone
  55          // could be anything - this is modified by other unit tests, too.)
  56          $this->setTimezone('UTC');
  57  
  58          // SEt user to GMT+5.
  59          $USER->timezone = 5;
  60  
  61          // Construct tree with date condition.
  62          $time = strtotime('2014-02-18 14:20:00 GMT');
  63          $structure = (object)array('op' => '|', 'show' => true, 'c' => array(
  64                  (object)array('type' => 'date', 'd' => '>=', 't' => $time)));
  65          $tree = new \core_availability\tree($structure);
  66          $info = new \core_availability\mock_info();
  67  
  68          // Check if available (when not available).
  69          condition::set_current_time_for_test($time - 1);
  70          $information = '';
  71          $result = $tree->check_available(false, $info, true, $USER->id);
  72          $this->assertFalse($result->is_available());
  73          $information = $tree->get_result_information($info, $result);
  74  
  75          // Note: PM is normally upper-case, but an issue with PHP on Mac means
  76          // that on that platform, it is reported lower-case.
  77          $this->assertRegExp('~from.*18 February 2014, 7:20 (PM|pm)~', $information);
  78  
  79          // Check if available (when available).
  80          condition::set_current_time_for_test($time);
  81          $result = $tree->check_available(false, $info, true, $USER->id);
  82          $this->assertTrue($result->is_available());
  83          $information = $tree->get_result_information($info, $result);
  84          $this->assertEquals('', $information);
  85      }

 119          // Invalid ->t.
 120          $structure->t = 'got you all in check';
 121          try {
 122              $date = new condition($structure);

 147      /**
 148       * Tests the is_available() and is_available_to_all() functions.
 149       */
 150      public function test_is_available() {
 151          global $SITE, $USER;
 152  
 153          $time = strtotime('2014-02-18 14:50:10 GMT');
 154          $info = new \core_availability\mock_info();
 155  
 156          // Test with >=.
 157          $date = new condition((object)array('d' => '>=', 't' => $time));
 158          condition::set_current_time_for_test($time - 1);
 159          $this->assertFalse($date->is_available(false, $info, true, $USER->id));
 160          condition::set_current_time_for_test($time);
 161          $this->assertTrue($date->is_available(false, $info, true, $USER->id));
 162  
 163          // Test with <.
 164          $date = new condition((object)array('d' => '<', 't' => $time));
 165          condition::set_current_time_for_test($time);
 166          $this->assertFalse($date->is_available(false, $info, true, $USER->id));
 167          condition::set_current_time_for_test($time - 1);
 168          $this->assertTrue($date->is_available(false, $info, true, $USER->id));
 169  
 170          // Repeat this test with is_available_to_all() - it should be the same.
 171          $date = new condition((object)array('d' => '<', 't' => $time));
 172          condition::set_current_time_for_test($time);
 173          $this->assertFalse($date->is_available_for_all(false));
 174          condition::set_current_time_for_test($time - 1);
 175          $this->assertTrue($date->is_available_for_all(false));
 176      }

 178      /**
 179       * Tests the get_description and get_standalone_description functions.
 180       */
 181      public function test_get_description() {
 182          global $SITE, $CFG;
 183  
 184          $this->resetAfterTest();
 185          $this->setTimezone('UTC');
 186  
 187          $modinfo = get_fast_modinfo($SITE);
 188          $info = new \core_availability\mock_info();
 189          $time = strtotime('2014-02-18 14:55:01 GMT');
 190  
 191          // Test with >=.
 192          $date = new condition((object)array('d' => '>=', 't' => $time));
 193          $information = $date->get_description(true, false, $info);
 194          $this->assertRegExp('~after.*18 February 2014, 2:55 (PM|pm)~', $information);
 195          $information = $date->get_description(true, true, $info);
 196          $this->assertRegExp('~before.*18 February 2014, 2:55 (PM|pm)~', $information);
 197          $information = $date->get_standalone_description(true, false, $info);
 198          $this->assertRegExp('~from.*18 February 2014, 2:55 (PM|pm)~', $information);
 199          $information = $date->get_standalone_description(true, true, $info);
 200          $this->assertRegExp('~until.*18 February 2014, 2:55 (PM|pm)~', $information);
 201  
 202          // Test with <.
 203          $date = new condition((object)array('d' => '<', 't' => $time));
 204          $information = $date->get_description(true, false, $info);
 205          $this->assertRegExp('~before.*18 February 2014, 2:55 (PM|pm)~', $information);
 206          $information = $date->get_description(true, true, $info);
 207          $this->assertRegExp('~after.*18 February 2014, 2:55 (PM|pm)~', $information);
 208          $information = $date->get_standalone_description(true, false, $info);
 209          $this->assertRegExp('~until.*18 February 2014, 2:55 (PM|pm)~', $information);
 210          $information = $date->get_standalone_description(true, true, $info);
 211          $this->assertRegExp('~from.*18 February 2014, 2:55 (PM|pm)~', $information);
 212  
 213          // Test special case for dates that are midnight.
 214          $date = new condition((object)array('d' => '>=',
 215                  't' => strtotime('2014-03-05 00:00 GMT')));
 216          $information = $date->get_description(true, false, $info);
 217          $this->assertRegExp('~on or after.*5 March 2014([^0-9]*)$~', $information);
 218          $information = $date->get_description(true, true, $info);
 219          $this->assertRegExp('~before.*end of.*4 March 2014([^0-9]*)$~', $information);
 220          $information = $date->get_standalone_description(true, false, $info);
 221          $this->assertRegExp('~from.*5 March 2014([^0-9]*)$~', $information);
 222          $information = $date->get_standalone_description(true, true, $info);
 223          $this->assertRegExp('~until end of.*4 March 2014([^0-9]*)$~', $information);
 224  
 225          // In the 'until' case for midnight, it shows the previous day. (I.e.
 226          // if the date is 5 March 00:00, then we show it as available until 4
 227          // March, implying 'the end of'.)
 228          $date = new condition((object)array('d' => '<',
 229                  't' => strtotime('2014-03-05 00:00 GMT')));
 230          $information = $date->get_description(true, false, $info);
 231          $this->assertRegExp('~before end of.*4 March 2014([^0-9]*)$~', $information);
 232          $information = $date->get_description(true, true, $info);
 233          $this->assertRegExp('~on or after.*5 March 2014([^0-9]*)$~', $information);
 234          $information = $date->get_standalone_description(true, false, $info);
 235          $this->assertRegExp('~until end of.*4 March 2014([^0-9]*)$~', $information);
 236          $information = $date->get_standalone_description(true, true, $info);
 237          $this->assertRegExp('~from.*5 March 2014([^0-9]*)$~', $information);
 238      }

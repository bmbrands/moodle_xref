  50      public function i_add_user_to_group_members($userfullname, $groupname) {

  52          $userfullname = $this->getSession()->getSelectorsHandler()->xpathLiteral($userfullname);
  53  
  54          // Using a xpath liternal to avoid problems with quotes and double quotes.

  57          // We don't know the option text as it contains the number of users in the group.

  60          $groupoption = $this->find('xpath', $xpath);

  63  
  64          // Here we don't need to wait for the AJAX response.
  65          $this->find_button(get_string('adduserstogroup', 'group'))->click();

  68          $this->getSession()->wait(self::TIMEOUT * 1000, self::PAGE_READY_JS);
  69  
  70          // Getting the option and selecting it.
  71          $select = $this->find_field('addselect');

  73          $memberoption = $this->find('xpath', $xpath);
  74          $fulloption = $memberoption->getText();

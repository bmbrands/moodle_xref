  73  } elseif ($data = $editform->get_data()) {
  74  
  75      /// Allocate members from the selected role to groups

  77          case 'no':
  78          case 'random':
  79          case 'lastname':
  80              $orderby = 'lastname, firstname'; break;
  81          case 'firstname':
  82              $orderby = 'firstname, lastname'; break;
  83          case 'idnumber':
  84              $orderby = 'idnumber'; break;
  85          default:
  86              print_error('unknoworder');

  88      $source = array();

  91      }
  92      if ($data->groupingid) {
  93          $source['groupingid'] = $data->groupingid;
  94      }
  95      if ($data->groupid) {
  96          $source['groupid'] = $data->groupid;
  97      }

  99      // Display only active users if the option was selected or they do not have the capability to view suspended users.

 102      $users = groups_get_potential_members($data->courseid, $data->roleid, $source, $orderby, !empty($data->notingroup),
 103          $onlyactive);

 107          srand($data->seed);
 108          shuffle($users);

 114      if ($data->groupby == 'groups') {

 128                  $userpergrp = floor($usercnt/$numgrps);

 130          }
 131      }

 137          $groups[$i]['members'] = array();
 138          if ($data->allocateby == 'no') {

 140          }
 141          for ($j=0; $j<$userpergrp; $j++) {
 142              if (empty($users)) {

 146              $groups[$i]['members'][$user->id] = $user;
 147          }
 148      }

 151          for ($i=0; $i<$numgrps; $i++) {

 155              $user = array_shift($users);

 158      }

 163              $table->head  = array(get_string('groupscount', 'group', $numgrps));

 179                  $error = get_string('groupnameexists', 'group', $group['name']);

 182              }
 183              if ($data->allocateby != 'no') {
 184                  $unames = array();

 196      } else {

 199          $createdgroups = array();

 234          }

 239          if ($failed) {

 242              }
 243              if ($createdgrouping) {
 244                  groups_delete_grouping($createdgrouping);
 245              }
 246          } else {

 248          }
 249      }
 250  }

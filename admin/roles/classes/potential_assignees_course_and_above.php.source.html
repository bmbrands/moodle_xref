  27  /**
  28   * User selector subclass for the list of potential users on the assign roles page,
  29   * when we are assigning in a context at or above the course level. In this case we
  30   * show all the users in the system who do not already have the role.
  31   */
  32  class core_role_potential_assignees_course_and_above extends core_role_assign_user_selector_base {
  33      public function find_users($search) {
  34          global $DB;
  35  
  36          list($wherecondition, $params) = $this->search_sql($search, '');
  37  
  38          $fields      = 'SELECT ' . $this->required_fields_sql('');
  39          $countfields = 'SELECT COUNT(1)';
  40  
  41          $sql = " FROM {user}
  42                  WHERE $wherecondition
  43                        AND id NOT IN (
  44                           SELECT r.userid
  45                             FROM {role_assignments} r
  46                            WHERE r.contextid = :contextid
  47                                  AND r.roleid = :roleid)";
  48  
  49          list($sort, $sortparams) = users_order_by_sql('', $search, $this->accesscontext);
  50          $order = ' ORDER BY ' . $sort;
  51  
  52          $params['contextid'] = $this->context->id;
  53          $params['roleid'] = $this->roleid;
  54  
  55          if (!$this->is_validating()) {
  56              $potentialmemberscount = $DB->count_records_sql($countfields . $sql, $params);
  57              if ($potentialmemberscount > $this->maxusersperpage) {
  58                  return $this->too_many_results($search, $potentialmemberscount);
  59              }
  60          }
  61  
  62          $availableusers = $DB->get_records_sql($fields . $sql . $order, array_merge($params, $sortparams));
  63  
  64          if (empty($availableusers)) {
  65              return array();
  66          }
  67  
  68          if ($search) {
  69              $groupname = get_string('potusersmatching', 'core_role', $search);
  70          } else {
  71              $groupname = get_string('potusers', 'core_role');
  72          }
  73  
  74          return array($groupname => $availableusers);
  75      }
  76  }

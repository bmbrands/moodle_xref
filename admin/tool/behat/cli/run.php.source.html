 285   * Prints header from the first process.

 287   * @param array $processes list of processes to loop though.

 289  function print_process_start_info($processes) {
 290      $printed = false;
 291      // Keep looping though processes, till we get first process o/p.
 292      while (!$printed) {
 293          usleep(10000);
 294          foreach ($processes as $name => $process) {
 295              // Exit if any process has stopped.
 296              if (!$process->isRunning()) {
 297                  $printed = true;
 298                  break;
 299              }
 300  
 301              $op = explode(PHP_EOL, $process->getOutput());
 302              if (count($op) >= 3) {
 303                  foreach ($op as $line) {
 304                      if (trim($line) && (strpos($line, '.') !== 0)) {
 305                          echo $line . PHP_EOL;
 306                      }
 307                  }
 308                  $printed = true;
 309              }
 310          }
 311      }
 312  }

 314  /**
 315   * Loop though all processes and print combined o/p
 316   *
 317   * @param array $processes list of processes to loop though.
 318   * @param bool $stoponfail Stop all processes and exit if failed.
 319   * @return array list of exit codes from all processes.
 320   */
 321  function print_combined_run_output($processes, $stoponfail = false) {
 322      $exitcodes = array();
 323      $maxdotsonline = 70;
 324      $remainingprintlen = $maxdotsonline;
 325      $progresscount = 0;
 326      while (count($exitcodes) != count($processes)) {
 327          usleep(10000);
 328          foreach ($processes as $name => $process) {
 329              if ($process->isRunning()) {
 330                  $op = $process->getIncrementalOutput();
 331                  if (trim($op)) {
 332                      $update = preg_filter('#^\s*([FS\.\-]+)(?:\s+\d+)?\s*$#', '$1', $op);
 333                      // Exit process if anything fails.
 334                      if ($stoponfail && (strpos($update, 'F') !== false)) {
 335                          $process->stop(0);
 336                      }
 337  
 338                      $strlentoprint = strlen($update);
 339  
 340                      // If not enough dots printed on line then just print.
 341                      if ($strlentoprint < $remainingprintlen) {
 342                          echo $update;
 343                          $remainingprintlen = $remainingprintlen - $strlentoprint;
 344                      } else if ($strlentoprint == $remainingprintlen) {
 345                          $progresscount += $maxdotsonline;
 346                          echo $update ." " . $progresscount . PHP_EOL;
 347                          $remainingprintlen = $maxdotsonline;
 348                      } else {
 349                          while ($part = substr($update, 0, $remainingprintlen) > 0) {
 350                              $progresscount += $maxdotsonline;
 351                              echo $part . " " . $progresscount . PHP_EOL;
 352                              $update = substr($update, $remainingprintlen);
 353                              $remainingprintlen = $maxdotsonline;
 354                          }
 355                      }
 356                  }
 357              } else {
 358                  $exitcodes[$name] = $process->getExitCode();
 359                  if ($stoponfail && ($exitcodes[$name] != 0)) {
 360                      foreach ($processes as $l => $p) {
 361                          $exitcodes[$l] = -1;
 362                          $process->stop(0);
 363                      }
 364                  }
 365              }
 366          }
 367      }
 368  
 369      echo PHP_EOL;
 370      return $exitcodes;
 371  }

 373  /**
 374   * Loop though all processes and print combined o/p
 375   *
 376   * @param array $processes list of processes to loop though.
 377   * @param bool $verbose Show verbose output for each process.
 378   */
 379  function print_each_process_info($processes, $verbose = false) {
 380      foreach ($processes as $name => $process) {
 381          echo "**************** [" . $name . "] ****************" . PHP_EOL;
 382          if ($verbose) {
 383              echo $process->getOutput();
 384              echo $process->getErrorOutput();
 385          } else {
 386              $op = explode(PHP_EOL, $process->getOutput());
 387              foreach ($op as $line) {
 388                  // Don't print progress .
 389                  if (trim($line) && (strpos($line, '.') !== 0) && (strpos($line, 'Moodle ') !== 0) &&
 390                      (strpos($line, 'Server OS ') !== 0) && (strpos($line, 'Started at ') !== 0) &&
 391                      (strpos($line, 'Browser specific fixes ') !== 0)) {
 392                      echo $line . PHP_EOL;
 393                  }
 394              }
 395          }
 396          echo PHP_EOL;
 397      }

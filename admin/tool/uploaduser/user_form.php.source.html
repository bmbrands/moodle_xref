 315      /**
 316       * Form tweaks that depend on current data.
 317       */
 318      function definition_after_data() {
 319          $mform   = $this->_form;
 320          $columns = $this->_customdata['columns'];
 321  
 322          foreach ($columns as $column) {
 323              if ($mform->elementExists($column)) {
 324                  $mform->removeElement($column);
 325              }
 326          }
 327  
 328          if (!in_array('password', $columns)) {
 329              // password resetting makes sense only if password specified in csv file
 330              if ($mform->elementExists('uuforcepasswordchange')) {
 331                  $mform->removeElement('uuforcepasswordchange');
 332              }
 333          }
 334      }

 336      /**
 337       * Server side validation.
 338       */
 339      function validation($data, $files) {
 340          $errors = parent::validation($data, $files);
 341          $columns = $this->_customdata['columns'];
 342          $optype  = $data['uutype'];
 343  
 344          // detect if password column needed in file
 345          if (!in_array('password', $columns)) {
 346              switch ($optype) {
 347                  case UU_USER_UPDATE:
 348                      if (!empty($data['uupasswordold'])) {
 349                          $errors['uupasswordold'] = get_string('missingfield', 'error', 'password');
 350                      }
 351                      break;
 352  
 353                  case UU_USER_ADD_UPDATE:
 354                      if (empty($data['uupasswordnew'])) {
 355                          $errors['uupasswordnew'] = get_string('missingfield', 'error', 'password');
 356                      }
 357                      if  (!empty($data['uupasswordold'])) {
 358                          $errors['uupasswordold'] = get_string('missingfield', 'error', 'password');
 359                      }
 360                      break;
 361  
 362                  case UU_USER_ADDNEW:
 363                      if (empty($data['uupasswordnew'])) {
 364                          $errors['uupasswordnew'] = get_string('missingfield', 'error', 'password');
 365                      }
 366                      break;
 367                  case UU_USER_ADDINC:
 368                      if (empty($data['uupasswordnew'])) {
 369                          $errors['uupasswordnew'] = get_string('missingfield', 'error', 'password');
 370                      }
 371                      break;
 372               }
 373          }
 374  
 375          // look for other required data
 376          if ($optype != UU_USER_UPDATE) {
 377              $requiredusernames = useredit_get_required_name_fields();
 378              $missing = array();
 379              foreach ($requiredusernames as $requiredusername) {
 380                  if (!in_array($requiredusername, $columns)) {
 381                      $missing[] = get_string('missingfield', 'error', $requiredusername);;
 382                  }
 383              }
 384              if ($missing) {
 385                  $errors['uutype'] = implode('<br />',  $missing);
 386              }
 387              if (!in_array('email', $columns) and empty($data['email'])) {
 388                  $errors['email'] = get_string('requiredtemplate', 'tool_uploaduser');
 389              }
 390          }
 391          return $errors;
 392      }

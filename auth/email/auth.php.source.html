 135       * Confirm the new user as registered.
 136       *
 137       * @param string $username
 138       * @param string $confirmsecret
 139       */
 140      function user_confirm($username, $confirmsecret) {
 141          global $DB;
 142          $user = get_complete_user_data('username', $username);
 143  
 144          if (!empty($user)) {
 145              if ($user->auth != $this->authtype) {
 146                  return AUTH_CONFIRM_ERROR;
 147  
 148              } else if ($user->secret == $confirmsecret && $user->confirmed) {
 149                  return AUTH_CONFIRM_ALREADY;
 150  
 151              } else if ($user->secret == $confirmsecret) {   // They have provided the secret key to get in
 152                  $DB->set_field("user", "confirmed", 1, array("id"=>$user->id));
 153                  return AUTH_CONFIRM_OK;
 154              }
 155          } else {
 156              return AUTH_CONFIRM_ERROR;
 157          }
 158      }

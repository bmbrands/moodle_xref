  91      function loginpage_hook() {
  92          global $frm;
  93          global $CFG;
  94          global $SESSION, $OUTPUT, $PAGE;
  95  
  96          $site = get_site();
  97          $CASform = get_string('CASform', 'auth_cas');
  98          $username = optional_param('username', '', PARAM_RAW);
  99          $courseid = optional_param('courseid', 0, PARAM_INT);
 100  
 101          if (!empty($username)) {
 102              if (isset($SESSION->wantsurl) && (strstr($SESSION->wantsurl, 'ticket') ||
 103                                                strstr($SESSION->wantsurl, 'NOCAS'))) {
 104                  unset($SESSION->wantsurl);
 105              }
 106              return;
 107          }
 108  
 109          // Return if CAS enabled and settings not specified yet
 110          if (empty($this->config->hostname)) {
 111              return;
 112          }
 113  
 114          // If the multi-authentication setting is used, check for the param before connecting to CAS.
 115          if ($this->config->multiauth) {
 116  
 117              // If there is an authentication error, stay on the default authentication page.
 118              if (!empty($SESSION->loginerrormsg)) {
 119                  return;
 120              }
 121  
 122              $authCAS = optional_param('authCAS', '', PARAM_RAW);
 123              if ($authCAS == 'NOCAS') {
 124                  return;
 125              }
 126              // Show authentication form for multi-authentication.
 127              // Test pgtIou parameter for proxy mode (https connection in background from CAS server to the php server).
 128              if ($authCAS != 'CAS' && !isset($_GET['pgtIou'])) {
 129                  $PAGE->set_url('/login/index.php');
 130                  $PAGE->navbar->add($CASform);
 131                  $PAGE->set_title("$site->fullname: $CASform");
 132                  $PAGE->set_heading($site->fullname);
 133                  echo $OUTPUT->header();
 134                  include($CFG->dirroot.'/auth/cas/cas_form.html');
 135                  echo $OUTPUT->footer();
 136                  exit();
 137              }
 138          }
 139  
 140          // Connection to CAS server
 141          $this->connectCAS();
 142  
 143          if (phpCAS::checkAuthentication()) {
 144              $frm = new stdClass();
 145              $frm->username = phpCAS::getUser();
 146              $frm->password = 'passwdCas';
 147  
 148              // Redirect to a course if multi-auth is activated, authCAS is set to CAS and the courseid is specified.
 149              if ($this->config->multiauth && !empty($courseid)) {
 150                  redirect(new moodle_url('/course/view.php', array('id'=>$courseid)));
 151              }
 152  
 153              return;
 154          }
 155  
 156          if (isset($_GET['loginguest']) && ($_GET['loginguest'] == true)) {
 157              $frm = new stdClass();
 158              $frm->username = 'guest';
 159              $frm->password = 'guest';
 160              return;
 161          }

 164          if (!phpCAS::isAuthenticated()) {
 165              phpCAS::setLang($this->config->language);

 172       * Connect to the CAS (clientcas connection or proxycas connection)
 173       *

 175      function connectCAS() {
 176          global $CFG;
 177          static $connected = false;
 178  
 179          if (!$connected) {
 180              // Make sure phpCAS doesn't try to start a new PHP session when connecting to the CAS server.
 181              if ($this->config->proxycas) {
 182                  phpCAS::proxy($this->config->casversion, $this->config->hostname, (int) $this->config->port, $this->config->baseuri, false);
 183              } else {
 184                  phpCAS::client($this->config->casversion, $this->config->hostname, (int) $this->config->port, $this->config->baseuri, false);
 185              }
 186              $connected = true;
 187          }
 188  
 189          // If Moodle is configured to use a proxy, phpCAS needs some curl options set.
 190          if (!empty($CFG->proxyhost) && !is_proxybypass($this->config->hostname)) {
 191              phpCAS::setExtraCurlOption(CURLOPT_PROXY, $CFG->proxyhost);
 192              if (!empty($CFG->proxyport)) {
 193                  phpCAS::setExtraCurlOption(CURLOPT_PROXYPORT, $CFG->proxyport);
 194              }
 195              if (!empty($CFG->proxytype)) {
 196                  // Only set CURLOPT_PROXYTYPE if it's something other than the curl-default http
 197                  if ($CFG->proxytype == 'SOCKS5') {
 198                      phpCAS::setExtraCurlOption(CURLOPT_PROXYTYPE, CURLPROXY_SOCKS5);
 199                  }
 200              }
 201              if (!empty($CFG->proxyuser) and !empty($CFG->proxypassword)) {
 202                  phpCAS::setExtraCurlOption(CURLOPT_PROXYUSERPWD, $CFG->proxyuser.':'.$CFG->proxypassword);
 203                  if (defined('CURLOPT_PROXYAUTH')) {
 204                      // any proxy authentication if PHP 5.1
 205                      phpCAS::setExtraCurlOption(CURLOPT_PROXYAUTH, CURLAUTH_BASIC | CURLAUTH_NTLM);
 206                  }
 207              }
 208          }
 209  
 210          if ($this->config->certificate_check && $this->config->certificate_path){
 211              phpCAS::setCasServerCACert($this->config->certificate_path);
 212          } else {
 213              // Don't try to validate the server SSL credentials
 214              phpCAS::setNoCasServerValidation();
 215          }
 216      }

 402      /**
 403       * Returns true if user should be coursecreator.
 404       *
 405       * @param mixed $username    username (without system magic quotes)
 406       * @return boolean result
 407       */
 408      function iscreator($username) {
 409          if (empty($this->config->host_url) or (empty($this->config->attrcreators) && empty($this->config->groupecreators)) or empty($this->config->memberattribute)) {
 410              return false;
 411          }
 412  
 413          $extusername = core_text::convert($username, 'utf-8', $this->config->ldapencoding);
 414  
 415          // Test for group creator
 416          if (!empty($this->config->groupecreators)) {
 417              $ldapconnection = $this->ldap_connect();
 418              if ($this->config->memberattribute_isdn) {
 419                  if(!($userid = $this->ldap_find_userdn($ldapconnection, $extusername))) {
 420                      return false;
 421                  }
 422              } else {
 423                  $userid = $extusername;
 424              }
 425  
 426              $group_dns = explode(';', $this->config->groupecreators);
 427              if (ldap_isgroupmember($ldapconnection, $userid, $group_dns, $this->config->memberattribute)) {
 428                  return true;
 429              }
 430          }
 431  
 432          // Build filter for attrcreator
 433          if (!empty($this->config->attrcreators)) {
 434              $attrs = explode(';', $this->config->attrcreators);
 435              $filter = '(& ('.$this->config->user_attribute."=$username)(|";
 436              foreach ($attrs as $attr){
 437                  if(strpos($attr, '=')) {
 438                      $filter .= "($attr)";
 439                  } else {
 440                      $filter .= '('.$this->config->memberattribute."=$attr)";
 441                  }
 442              }
 443              $filter .= '))';
 444  
 445              // Search
 446              $result = $this->ldap_get_userlist($filter);
 447              if (count($result) != 0) {
 448                  return true;
 449              }
 450          }
 451  
 452          return false;
 453      }

 507       * Post logout hook.
 508       *
 509       * Note: this method replace the prelogout_hook method to avoid redirect to CAS logout
 510       * before the event userlogout being triggered.
 511       *
 512       * @param stdClass $user clone of USER object object before the user session was terminated

 514      public function postlogout_hook($user) {
 515          global $CFG;
 516          // Only redirect to CAS logout if the user is logged as a CAS user.
 517          if (!empty($this->config->logoutcas) && $user->auth == $this->authtype) {
 518              $backurl = !empty($this->config->logout_return_url) ? $this->config->logout_return_url : $CFG->wwwroot;
 519              $this->connectCAS();
 520              phpCAS::logoutWithRedirectService($backurl);

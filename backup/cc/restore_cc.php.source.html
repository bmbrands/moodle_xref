  99  function detect_requirements () {
 100  
 101      if (floor(phpversion()) >= 5) {
 102          $detected["php5"] = true;
 103      } else {
 104          $detected["php5"] = false;
 105      }
 106  
 107      $detected["xsl"] = extension_loaded('xsl');
 108      $detected['dom'] = extension_loaded('dom');
 109      $detected['libxml'] = extension_loaded('libxml');
 110      $detected['libxmlminversion'] = extension_loaded('libxml') && version_compare(LIBXML_DOTTED_VERSION, '2.6.30', '>=');
 111  
 112      return $detected;
 113  
 114  }

 116  function detect_cc_format ($xml_file) {
 117  
 118      $inpos = 0;

 120

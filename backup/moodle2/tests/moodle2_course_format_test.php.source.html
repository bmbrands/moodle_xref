 162      /**
 163       * Backs a course up and restores it.
 164       *
 165       * @param stdClass $srccourse Course object to backup
 166       * @param stdClass $dstcourse Course object to restore into
 167       * @return int ID of newly restored course
 168       */
 169      protected function backup_and_restore($srccourse, $dstcourse = null) {
 170          global $USER, $CFG;
 171  
 172          // Turn off file logging, otherwise it can't delete the file (Windows).
 173          $CFG->backup_file_logger_level = backup::LOG_NONE;
 174  
 175          // Do backup with default settings. MODE_IMPORT means it will just
 176          // create the directory and not zip it.
 177          $bc = new backup_controller(backup::TYPE_1COURSE, $srccourse->id,
 178                  backup::FORMAT_MOODLE, backup::INTERACTIVE_NO, backup::MODE_IMPORT,
 179                  $USER->id);
 180          $backupid = $bc->get_backupid();
 181          $bc->execute_plan();
 182          $bc->destroy();
 183  
 184          // Do restore to new course with default settings.
 185          if ($dstcourse !== null) {
 186              $newcourseid = $dstcourse->id;
 187          } else {
 188              $newcourseid = restore_dbops::create_new_course(
 189                  $srccourse->fullname, $srccourse->shortname . '_2', $srccourse->category);
 190          }
 191          $rc = new restore_controller($backupid, $newcourseid,
 192                  backup::INTERACTIVE_NO, backup::MODE_GENERAL, $USER->id,
 193                  backup::TARGET_NEW_COURSE);
 194  
 195          $this->assertTrue($rc->execute_precheck());
 196          $rc->execute_plan();
 197          $rc->destroy();
 198  
 199          return $newcourseid;
 200      }

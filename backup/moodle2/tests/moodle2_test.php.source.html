 368  
 369          // Add an assign with defined start date.
 370          $assign = $generator->create_module('assign', array('course' => $course->id,
 371                  'allowsubmissionsfromdate' => strtotime('7 Jan 2014 16:00 GMT')));
 372  
 373          // Do backup and restore.

 375  
 376          $modinfo = get_fast_modinfo($newcourseid);
 377  
 378          // Check forum dates are modified by the same amount as the course start.
 379          $newforums = $modinfo->get_instances_of('forum');
 380          $newforum = reset($newforums);
 381          $after = str_replace(
 382              array('DATE1', 'DATE2'),
 383              array(strtotime('3 Feb 2015 00:00 GMT'), strtotime('12 Feb 2015 00:00 GMT')),
 384              $availability);
 385          $this->assertEquals($after, $newforum->availability);
 386  
 387          // Check assign date.
 388          $newassigns = $modinfo->get_instances_of('assign');
 389          $newassign = reset($newassigns);

 470       * Backs a course up and restores it.
 471       *
 472       * @param stdClass $course Course object to backup
 473       * @param int $newdate If non-zero, specifies custom date for new course
 474       * @return int ID of newly restored course
 475       */
 476      protected function backup_and_restore($course, $newdate = 0) {
 477          global $USER, $CFG;
 478  
 479          // Turn off file logging, otherwise it can't delete the file (Windows).
 480          $CFG->backup_file_logger_level = backup::LOG_NONE;
 481  
 482          // Do backup with default settings. MODE_IMPORT means it will just
 483          // create the directory and not zip it.
 484          $bc = new backup_controller(backup::TYPE_1COURSE, $course->id,
 485                  backup::FORMAT_MOODLE, backup::INTERACTIVE_NO, backup::MODE_IMPORT,
 486                  $USER->id);
 487          $backupid = $bc->get_backupid();
 488          $bc->execute_plan();
 489          $bc->destroy();
 490  
 491          // Do restore to new course with default settings.
 492          $newcourseid = restore_dbops::create_new_course(
 493                  $course->fullname, $course->shortname . '_2', $course->category);
 494          $rc = new restore_controller($backupid, $newcourseid,
 495                  backup::INTERACTIVE_NO, backup::MODE_GENERAL, $USER->id,
 496                  backup::TARGET_NEW_COURSE);
 497          if ($newdate) {
 498              $rc->get_plan()->get_setting('course_startdate')->set_value($newdate);
 499          }
 500          $this->assertTrue($rc->execute_precheck());
 501          $rc->execute_plan();
 502          $rc->destroy();
 503  
 504          return $newcourseid;
 505      }

 507      /**
 508       * Duplicates a single activity within a course.
 509       *
 510       * This is based on the code from course/modduplicate.php, but reduced for
 511       * simplicity.
 512       *
 513       * @param stdClass $course Course object
 514       * @param int $cmid Activity to duplicate
 515       * @return int ID of new activity
 516       */
 517      protected function duplicate($course, $cmid) {
 518          global $USER;
 519  
 520          // Do backup.
 521          $bc = new backup_controller(backup::TYPE_1ACTIVITY, $cmid, backup::FORMAT_MOODLE,
 522                  backup::INTERACTIVE_NO, backup::MODE_IMPORT, $USER->id);
 523          $backupid = $bc->get_backupid();
 524          $bc->execute_plan();
 525          $bc->destroy();
 526  
 527          // Do restore.
 528          $rc = new restore_controller($backupid, $course->id,
 529                  backup::INTERACTIVE_NO, backup::MODE_IMPORT, $USER->id, backup::TARGET_CURRENT_ADDING);
 530          $this->assertTrue($rc->execute_precheck());
 531          $rc->execute_plan();
 532  
 533          // Find cmid.
 534          $tasks = $rc->get_plan()->get_tasks();
 535          $cmcontext = context_module::instance($cmid);

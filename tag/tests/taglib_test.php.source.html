 147      /**
 148       * Test the tag cleanup function used by the cron.
 149       */
 150      public function test_tag_cleanup() {
 151          global $DB;
 152  
 153          // Create some users.
 154          $users = array();
 155          for ($i = 0; $i < 10; $i++) {
 156              $users[] = $this->getDataGenerator()->create_user();
 157          }
 158  
 159          // Create a course to tag.
 160          $course = $this->getDataGenerator()->create_course();
 161          $context = context_course::instance($course->id);
 162  
 163          // Test clean up instances with tags that no longer exist.
 164          $tags = array();
 165          for ($i = 0; $i < 10; $i++) {
 166              $tags[] = $this->getDataGenerator()->create_tag(array('userid' => $users[0]->id));
 167          }
 168          // Create instances with the tags.
 169          foreach ($tags as $tag) {
 170              tag_assign('course', $course->id, $tag->id, 0, 0, 'core', $context->id);
 171          }
 172          // We should now have ten tag instances.
 173          $coursetaginstances = $DB->count_records('tag_instance', array('itemtype' => 'course'));
 174          $this->assertEquals(10, $coursetaginstances);
 175          // Delete four tags
 176          // Manual delete of tags is done as the function will remove the instances as well.
 177          $DB->delete_records('tag', array('id' => $tags[6]->id));
 178          $DB->delete_records('tag', array('id' => $tags[7]->id));
 179          $DB->delete_records('tag', array('id' => $tags[8]->id));
 180          $DB->delete_records('tag', array('id' => $tags[9]->id));
 181          // Clean up the tags.
 182          tag_cleanup();
 183          // Check that we now only have six tag_instance records left.
 184          $coursetaginstances = $DB->count_records('tag_instance', array('itemtype' => 'course'));
 185          $this->assertEquals(6, $coursetaginstances);
 186  
 187          // Test clean up with users that have been deleted.
 188          // Create a tag for this course.
 189          foreach ($users as $user) {
 190              tag_assign('user', $user->id, $tags[0]->id, 0, 0, 'core', $context->id);
 191          }
 192          $usertags = $DB->count_records('tag_instance', array('itemtype' => 'user'));
 193          $this->assertCount($usertags, $users);
 194          // Remove three students.
 195          // Using the proper function to delete the user will also remove the tags.
 196          $DB->update_record('user', array('id' => $users[4]->id, 'deleted' => 1));
 197          $DB->update_record('user', array('id' => $users[5]->id, 'deleted' => 1));
 198          $DB->update_record('user', array('id' => $users[6]->id, 'deleted' => 1));
 199          // Clean up the tags.
 200          tag_cleanup();
 201          $usertags = $DB->count_records('tag_instance', array('itemtype' => 'user'));
 202          $usercount = $DB->count_records('user', array('deleted' => 0));
 203          // Remove admin and guest from the count.
 204          $this->assertEquals($usertags, ($usercount - 2));
 205  
 206          // Test clean up where a course has been removed.
 207          // Delete the course. This also needs to be this way otherwise the tags are removed by using the proper function.
 208          $DB->delete_records('course', array('id' => $course->id));
 209          tag_cleanup();
 210          $coursetags = $DB->count_records('tag_instance', array('itemtype' => 'course'));
 211          $this->assertEquals(0, $coursetags);
 212  
 213          // Test clean up where a post has been removed.
 214          // Create default post.
 215          $post = new stdClass();
 216          $post->userid = $users[1]->id;
 217          $post->content = 'test post content text';
 218          $post->id = $DB->insert_record('post', $post);
 219          tag_assign('post', $post->id, $tags[0]->id, 0, 0, 'core', $context->id);
 220          // Add another one with a fake post id to be removed.
 221          tag_assign('post', 15, $tags[0]->id, 0, 0, 'core', $context->id);
 222          // Check that there are two tag instances.
 223          $posttags = $DB->count_records('tag_instance', array('itemtype' => 'post'));
 224          $this->assertEquals(2, $posttags);
 225          // Clean up the tags.
 226          tag_cleanup();
 227          // We should only have one entry left now.
 228          $posttags = $DB->count_records('tag_instance', array('itemtype' => 'post'));
 229          $this->assertEquals(1, $posttags);
 230      }

 232      /**
 233       * Test deleting a group of tag instances.
 234       */
 235      public function test_tag_bulk_delete_instances() {
 236          global $DB;
 237          // Setup.
 238          $user = $this->getDataGenerator()->create_user();
 239          $course = $this->getDataGenerator()->create_course();
 240          $context = context_course::instance($course->id);
 241  
 242          // Create some tag instances.
 243          for ($i = 0; $i < 10; $i++) {
 244              $tag = $this->getDataGenerator()->create_tag(array('userid' => $user->id));
 245              tag_assign('course', $course->id, $tag->id, 0, 0, 'core', $context->id);
 246          }
 247          // Get tag instances. tag name and rawname are required for the event fired in this function.
 248          $sql = "SELECT ti.*, t.name, t.rawname
 249                    FROM {tag_instance} ti
 250                    JOIN {tag} t ON t.id = ti.tagid";
 251          $taginstances = $DB->get_records_sql($sql);
 252          $this->assertCount(10, $taginstances);
 253          // Run the function.
 254          tag_bulk_delete_instances($taginstances);
 255          // Make sure they are gone.
 256          $instancecount = $DB->count_records('tag_instance');
 257          $this->assertEquals(0, $instancecount);
 258      }

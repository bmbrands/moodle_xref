  39      /**
  40       * Processes the message and sends a notification via airnotifier
  41       *
  42       * @param stdClass $eventdata the event data submitted by the message sender plus $eventdata->savedmessageid
  43       * @return true if ok, false if error
  44       */
  45      public function send_message($eventdata) {
  46          global $CFG;
  47          require_once($CFG->libdir . '/filelib.php');
  48  
  49          if (!empty($CFG->noemailever)) {
  50              // Hidden setting for development sites, set in config.php if needed.
  51              debugging('$CFG->noemailever active, no airnotifier message sent.', DEBUG_MINIMAL);
  52              return true;
  53          }
  54  
  55          // Skip any messaging suspended and deleted users.
  56          if ($eventdata->userto->auth === 'nologin' or
  57              $eventdata->userto->suspended or
  58              $eventdata->userto->deleted) {
  59              return true;
  60          }
  61  
  62          // Site id, to map with Moodle Mobile stored sites.
  63          $siteid = md5($CFG->wwwroot . $eventdata->userto->username);
  64  
  65          // Airnotifier can handle custom requests using processors (that are Airnotifier plugins).
  66          // In the extra parameter we indicate which processor to use and also additional data to be handled by the processor.
  67          // Here we clone the eventdata object because will be deleting/adding attributes.
  68          $extra = clone $eventdata;
  69  
  70          // Delete attributes that may content private information.
  71          if (!empty($eventdata->userfrom)) {
  72              $extra->userfromid = $eventdata->userfrom->id;
  73              $extra->userfromfullname = fullname($eventdata->userfrom);
  74              unset($extra->userfrom);
  75          }
  76          $extra->usertoid = $eventdata->userto->id;
  77          unset($extra->userto);
  78  
  79          $extra->processor       = 'moodle';
  80          $extra->site            = $siteid;
  81          $extra->date            = (!empty($eventdata->timecreated)) ? $eventdata->timecreated : time();
  82          $extra->notification    = (!empty($eventdata->notification)) ? 1 : 0;
  83  
  84          // Site name.
  85          $site = get_site();
  86          $extra->sitefullname = format_string($site->fullname);
  87          $extra->siteshortname = format_string($site->shortname);
  88  
  89          // We are sending to message to all devices.
  90          $airnotifiermanager = new message_airnotifier_manager();
  91          $devicetokens = $airnotifiermanager->get_user_devices($CFG->airnotifiermobileappname, $eventdata->userto->id);
  92  
  93          foreach ($devicetokens as $devicetoken) {
  94  
  95              if (!$devicetoken->enable) {
  96                  continue;
  97              }
  98  
  99              // Sending the message to the device.
 100              $serverurl = $CFG->airnotifierurl . ':' . $CFG->airnotifierport . '/api/v2/push/';
 101              $header = array('Accept: application/json', 'X-AN-APP-NAME: ' . $CFG->airnotifierappname,
 102                  'X-AN-APP-KEY: ' . $CFG->airnotifieraccesskey);
 103              $curl = new curl;
 104              $curl->setHeader($header);
 105  
 106              $params = array(
 107                  'device'    => $devicetoken->platform,
 108                  'token'     => $devicetoken->pushid,
 109                  'extra'     => $extra
 110              );
 111  
 112              // JSON POST raw body request.
 113              $resp = $curl->post($serverurl, json_encode($params));
 114          }
 115  
 116          return true;
 117      }

 124      public function config_form($preferences) {
 125          global $CFG, $OUTPUT, $USER, $PAGE;
 126  
 127          $systemcontext = context_system::instance();
 128          if (!has_capability('message/airnotifier:managedevice', $systemcontext)) {
 129              return get_string('nopermissiontomanagedevices', 'message_airnotifier');
 130          }
 131  
 132          if (!$this->is_system_configured()) {
 133              return get_string('notconfigured', 'message_airnotifier');
 134          } else {
 135  
 136              $PAGE->requires->css('/message/output/airnotifier/style.css');
 137  
 138              $airnotifiermanager = new message_airnotifier_manager();
 139              $devicetokens = $airnotifiermanager->get_user_devices($CFG->airnotifiermobileappname, $USER->id);
 140  
 141              if (!empty($devicetokens)) {
 142                  $output = '';
 143  
 144                  foreach ($devicetokens as $devicetoken) {
 145  
 146                      if ($devicetoken->enable) {
 147                          $hideshowiconname = 't/hide';
 148                          $dimmed = '';
 149                      } else {
 150                          $hideshowiconname = 't/show';
 151                          $dimmed = 'dimmed_text';
 152                      }
 153  
 154                      $hideshowicon = $OUTPUT->pix_icon($hideshowiconname, get_string('showhide', 'message_airnotifier'));
 155                      $name = "{$devicetoken->name} {$devicetoken->model} {$devicetoken->platform} {$devicetoken->version}";
 156                      $hideurl = new moodle_url('message/output/airnotifier/action.php',
 157                                      array('hide' => !$devicetoken->enable, 'deviceid' => $devicetoken->id,
 158                                          'sesskey' => sesskey()));
 159  
 160                      $output .= html_writer::start_tag('li', array('id' => $devicetoken->id,
 161                                                                      'class' => 'airnotifierdevice ' . $dimmed)) . "\n";
 162                      $output .= html_writer::label($name, 'deviceid-' . $devicetoken->id, array('class' => 'devicelabel ')) . ' ' .
 163                          html_writer::link($hideurl, $hideshowicon, array('class' => 'hidedevice', 'alt' => 'show/hide')) . "\n";
 164                      $output .= html_writer::end_tag('li') . "\n";
 165                  }
 166  
 167                  // Include the AJAX script to automatically trigger the action.
 168                  $airnotifiermanager->include_device_ajax();
 169  
 170                  $output = html_writer::tag('ul', $output, array('id' => 'airnotifierdevices'));
 171              } else {
 172                  $output = get_string('nodevices', 'message_airnotifier');
 173              }
 174              return $output;
 175          }
 176      }

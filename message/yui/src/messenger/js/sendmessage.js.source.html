 136      /**
 137       * Send the message to the user.
 138       *
 139       * @method sendMessage
 140       * @param  {String} message The message to be sent.
 141       */
 142      sendMessage: function(message) {
 143          if (this._sendLock) {
 144              // Do not proceed if the lock is active.
 145              return;
 146          }
 147  
 148          if (!message || !this._validateMessage(message)) {
 149              // Do not send falsy messages.
 150              return;
 151          }
 152  
 153          // Actually send the message.
 154          this._ioSend = Y.io(this.get('url'), {
 155              method: 'POST',
 156              data: {
 157                  sesskey: M.cfg.sesskey,
 158                  action: 'sendmessage',
 159                  userid: this.get('userid'),
 160                  message: message
 161              },
 162              on: {
 163                  start: function() {
 164                      var img = '<img alt="" role="presentation" src="' + M.util.image_url('i/loading_small', 'moodle') + '">';
 165                      this.setSendLock(true);
 166                      this.showNotice(img + ' ' + M.util.get_string('sendingmessage', 'core_message'));
 167                  },
 168                  success: function(id, response) {
 169                      var data = null;
 170  
 171                      try {
 172                          data = Y.JSON.parse(response.responseText);
 173                          if (data.error) {
 174                              this.hideNotice();
 175                              new M.core.ajaxException(data);
 176                              return;
 177                          }
 178                      } catch (e) {
 179                          this.hideNotice();
 180                          new M.core.exception(e);
 181                          return;
 182                      }
 183  
 184                      // Show a success message.
 185                      this.showNotice(M.util.get_string('messagesent', 'core_message'));
 186  
 187                      // Hide the dialog.
 188                      Y.later(1300, this, function() {
 189                          this.setSendLock(false);
 190                          this.hideNotice();
 191                          this.hide();
 192                      });
 193                  },
 194                  failure: function() {
 195                      this.setSendLock(false);
 196                      this.hideNotice();
 197                      new M.core.alert({
 198                          title: M.util.get_string('error', 'core'),
 199                          message: M.util.get_string('errorwhilesendingmessage', 'core_message')
 200                      });
 201                  }
 202              },
 203              context: this
 204          });
 205      },

 258      /**
 259       * Validates a message.
 260       *
 261       * @method _validateMessage
 262       * @param  {String} message A message to be validated.
 263       */
 264      _validateMessage: function(message) {
 265          var trimmed;
 266          if (!message) {
 267              return false;
 268          }
 269  
 270          // Basic validation.
 271          trimmed = message.replace(' ', '')
 272                           .replace('&nbsp;', '')
 273                           .replace(/(<br\s*\/?>(<\/br\s*\/?>)?)+/, '')
 274                           .trim();
 275  
 276          return trimmed.length > 1;
 277      }

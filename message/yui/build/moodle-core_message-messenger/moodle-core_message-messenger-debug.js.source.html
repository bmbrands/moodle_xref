 103      sendMessage: function(userid, fullname, e) {
 104          var Klass;
 105          if (!this._sendMessageDialog) {
 106              Klass = Y.namespace('M.core_message.messenger.sendMessage');
 107              this._sendMessageDialog = new Klass({
 108                  url: this.get('url')
 109              });

 113          this._sendMessageDialog.show(e);

 121      _setEvents: function() {
 122          var captureEvent = function(e) {
 123              var target = e.currentTarget,
 124                  userid = parseInt(target.getData('userid'), 10),
 125                  fullname = target.getData('fullname');
 126  
 127              if (!userid || !fullname) {
 128                  return;
 129              }
 130  
 131              // Pass the validation before preventing defaults.
 132              e.preventDefault();
 133              this.sendMessage(userid, fullname, e);
 134          };
 135  
 136          this._events.push(Y.delegate('click', captureEvent, 'body', SELECTORS.MANAGER.SENDMESSAGE, this));
 137          this._events.push(Y.one(Y.config.doc).delegate('key', captureEvent, 'space', SELECTORS.MANAGER.SENDMESSAGE, this));
 138      }

 301      /**
 302       * Send the message to the user.
 303       *
 304       * @method sendMessage
 305       * @param  {String} message The message to be sent.
 306       */
 307      sendMessage: function(message) {
 308          if (this._sendLock) {
 309              // Do not proceed if the lock is active.
 310              return;
 311          }
 312  
 313          if (!message || !this._validateMessage(message)) {
 314              // Do not send falsy messages.
 315              return;
 316          }
 317  
 318          // Actually send the message.
 319          this._ioSend = Y.io(this.get('url'), {
 320              method: 'POST',
 321              data: {
 322                  sesskey: M.cfg.sesskey,
 323                  action: 'sendmessage',
 324                  userid: this.get('userid'),
 325                  message: message
 326              },
 327              on: {
 328                  start: function() {
 329                      var img = '<img alt="" role="presentation" src="' + M.util.image_url('i/loading_small', 'moodle') + '">';
 330                      this.setSendLock(true);
 331                      this.showNotice(img + ' ' + M.util.get_string('sendingmessage', 'core_message'));
 332                  },
 333                  success: function(id, response) {
 334                      var data = null;
 335  
 336                      try {
 337                          data = Y.JSON.parse(response.responseText);
 338                          if (data.error) {
 339                              this.hideNotice();
 340                              new M.core.ajaxException(data);
 341                              return;
 342                          }
 343                      } catch (e) {
 344                          this.hideNotice();
 345                          new M.core.exception(e);
 346                          return;
 347                      }
 348  
 349                      // Show a success message.
 350                      this.showNotice(M.util.get_string('messagesent', 'core_message'));
 351  
 352                      // Hide the dialog.
 353                      Y.later(1300, this, function() {
 354                          this.setSendLock(false);
 355                          this.hideNotice();
 356                          this.hide();
 357                      });
 358                  },
 359                  failure: function() {
 360                      this.setSendLock(false);
 361                      this.hideNotice();
 362                      new M.core.alert({
 363                          title: M.util.get_string('error', 'core'),
 364                          message: M.util.get_string('errorwhilesendingmessage', 'core_message')
 365                      });
 366                  }
 367              },
 368              context: this
 369          });
 370      },

  57       *
  58       * {@link message_send()} does not support transaction, this function will simulate a message
  59       * sent from a user to another. We should stop using it once {@link message_send()} will support
  60       * transactions. This is not clean at all, this is just used to add rows to the table.
  61       *
  62       * @param stdClass $userfrom user object of the one sending the message.

  69

  71          $record->useridfrom = $userfrom->id;
  72          $record->useridto = $userto->id;
  73          $record->subject = 'No subject';
  74          $record->fullmessage = $message;

 566      /**
 567       * Test message_get_recent_conversations with a mixture of messages.
 568       *
 569       * @dataProvider message_get_recent_conversations_provider
 570       * @param array $usersdata The list of users to create for this test.
 571       * @param array $messagesdata The list of messages to create.
 572       * @param array $expectations The list of expected outcomes.
 573       */
 574      public function test_message_get_recent_conversations($usersdata, $messagesdata, $expectations) {
 575          global $DB;
 576  
 577          // Create all of the users.
 578          $users = array();
 579          foreach ($usersdata as $username) {
 580              $users[$username] = $this->getDataGenerator()->create_user(array('username' => $username));
 581          }
 582  
 583          $defaulttimecreated = time();
 584          foreach ($messagesdata as $messagedata) {
 585              $from       = $users[$messagedata['from']];
 586              $to         = $users[$messagedata['to']];
 587              $subject    = $messagedata['subject'];
 588  
 589              if (isset($messagedata['state']) && $messagedata['state'] == 'unread') {
 590                  $table = 'message';
 591                  $messageid = $this->send_fake_message($from, $to, $subject, FORMAT_PLAIN);
 592              } else {
 593                  // If there is no state, or the state is not 'unread', assume the message is read.
 594                  $table = 'message_read';
 595                  $messageid = message_post_message($from, $to, $subject, FORMAT_PLAIN);
 596              }
 597  
 598              $updatemessage = new stdClass();
 599              $updatemessage->id = $messageid;
 600              if (isset($messagedata['timecreated'])) {
 601                  $updatemessage->timecreated = $messagedata['timecreated'];
 602              } else if (isset($messagedata['timemodifier'])) {
 603                  $updatemessage->timecreated = $defaulttimecreated + $messagedata['timemodifier'];
 604              } else {
 605                  $updatemessage->timecreated = $defaulttimecreated;
 606              }
 607              $DB->update_record($table, $updatemessage);
 608          }
 609  
 610          foreach ($expectations as $username => $data) {
 611              // Get the recent conversations for the specified user.
 612              $user = $users[$username];
 613              $conversations = message_get_recent_conversations($user);
 614              foreach ($data as $expectation) {
 615                  $otheruser = $users[$expectation['with']];
 616                  $conversation = $conversations[$expectation['messageposition']];
 617                  $this->assertEquals($otheruser->id, $conversation->id);
 618                  $this->assertEquals($expectation['subject'], $conversation->smallmessage);
 619              }
 620          }
 621      }

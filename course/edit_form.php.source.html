 324      function definition_after_data() {
 325          global $DB;
 326  
 327          $mform = $this->_form;
 328  
 329          // add available groupings

 331              $options = array();
 332              if ($groupings = $DB->get_records('groupings', array('courseid'=>$courseid))) {
 333                  foreach ($groupings as $grouping) {
 334                      $options[$grouping->id] = format_string($grouping->name);
 335                  }
 336              }
 337              core_collator::asort($options);
 338              $gr_el =& $mform->getElement('defaultgroupingid');
 339              $gr_el->load($options);
 340          }
 341  
 342          // add course format options
 343          $formatvalue = $mform->getElementValue('format');
 344          if (is_array($formatvalue) && !empty($formatvalue)) {
 345              $courseformat = course_get_format((object)array('format' => $formatvalue[0]));

 347              $elements = $courseformat->create_edit_form_elements($mform);
 348              for ($i = 0; $i < count($elements); $i++) {

 355      /**
 356       * Validation.
 357       *
 358       * @param array $data
 359       * @param array $files
 360       * @return array the errors that were found
 361       */
 362      function validation($data, $files) {
 363          global $DB;
 364  
 365          $errors = parent::validation($data, $files);
 366  
 367          // Add field validation check for duplicate shortname.
 368          if ($course = $DB->get_record('course', array('shortname' => $data['shortname']), '*', IGNORE_MULTIPLE)) {
 369              if (empty($data['id']) || $course->id != $data['id']) {
 370                  $errors['shortname'] = get_string('shortnametaken', '', $course->fullname);
 371              }
 372          }
 373  
 374          // Add field validation check for duplicate idnumber.
 375          if (!empty($data['idnumber']) && (empty($data['id']) || $this->course->idnumber != $data['idnumber'])) {
 376              if ($course = $DB->get_record('course', array('idnumber' => $data['idnumber']), '*', IGNORE_MULTIPLE)) {
 377                  if (empty($data['id']) || $course->id != $data['id']) {
 378                      $errors['idnumber'] = get_string('courseidnumbertaken', 'error', $course->fullname);
 379                  }
 380              }
 381          }
 382  
 383          $errors = array_merge($errors, enrol_course_edit_validation($data, $this->context));
 384  
 385          $courseformat = course_get_format((object)array('format' => $data['format']));
 386          $formaterrors = $courseformat->edit_form_validation($data, $files, $errors);
 387          if (!empty($formaterrors) && is_array($formaterrors)) {
 388              $errors = array_merge($errors, $formaterrors);
 389          }
 390  
 391          return $errors;
 392      }

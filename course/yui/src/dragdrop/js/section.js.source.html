  52       /**
  53       * Apply dragdrop features to the specified selector or node that refers to section(s)

  58      setup_for_section: function(baseselector) {
  59          Y.Node.all(baseselector).each(function(sectionnode) {
  60              // Determine the section ID
  61              var sectionid = Y.Moodle.core_course.util.section.getId(sectionnode);
  62  
  63              // We skip the top section as it is not draggable
  64              if (sectionid > 0) {
  65                  // Remove move icons
  66                  var movedown = sectionnode.one('.' + CSS.RIGHT + ' a.' + CSS.MOVEDOWN);
  67                  var moveup = sectionnode.one('.' + CSS.RIGHT + ' a.' + CSS.MOVEUP);
  68  
  69                  // Add dragger icon
  70                  var title = M.util.get_string('movesection', 'moodle', sectionid);
  71                  var cssleft = sectionnode.one('.' + CSS.LEFT);
  72  
  73                  if ((movedown || moveup) && cssleft) {
  74                      cssleft.setStyle('cursor', 'move');
  75                      cssleft.appendChild(this.get_drag_handle(title, CSS.SECTIONHANDLE, 'icon', true));
  76  
  77                      if (moveup) {
  78                          moveup.remove();
  79                      }
  80                      if (movedown) {
  81                          movedown.remove();
  82                      }
  83  
  84                      // This section can be moved - add the class to indicate this to Y.DD.
  85                      sectionnode.addClass(CSS.SECTIONDRAGGABLE);
  86                  }
  87              }

 118      get_section_index: function(node) {
 119          var sectionlistselector = '.' + CSS.COURSECONTENT + ' ' + M.course.format.get_section_selector(Y),
 120              sectionList = Y.all(sectionlistselector),
 121              nodeIndex = sectionList.indexOf(node),
 122              zeroIndex = sectionList.indexOf(Y.one('#section-0'));
 123  
 124          return (nodeIndex - zeroIndex);

 127      drop_hit: function(e) {
 128          var drag = e.drag;
 129  
 130          // Get references to our nodes and their IDs.
 131          var dragnode = drag.get('node'),
 132              dragnodeid = Y.Moodle.core_course.util.section.getId(dragnode),
 133              loopstart = dragnodeid,
 134  
 135              dropnodeindex = this.get_section_index(dragnode),
 136              loopend = dropnodeindex;
 137  
 138          if (dragnodeid === dropnodeindex) {
 139              Y.log("Skipping move - same location moving " + dragnodeid + " to " + dropnodeindex, 'debug', 'moodle-course-dragdrop');
 140              return;
 141          }
 142  
 143          Y.log("Moving from position " + dragnodeid + " to position " + dropnodeindex, 'debug', 'moodle-course-dragdrop');
 144  
 145          if (loopstart > loopend) {
 146              // If we're going up, we need to swap the loop order
 147              // because loops can't go backwards.
 148              loopstart = dropnodeindex;
 149              loopend = dragnodeid;
 150          }
 151  
 152          // Get the list of nodes.
 153          drag.get('dragNode').removeClass(CSS.COURSECONTENT);
 154          var sectionlist = Y.Node.all(this.sectionlistselector);
 155  
 156          // Add a lightbox if it's not there.
 157          var lightbox = M.util.add_lightbox(Y, dragnode);
 158  
 159          // Handle any variables which we must pass via AJAX.
 160          var params = {},
 161              pageparams = this.get('config').pageparams,
 162              varname;
 163  
 164          for (varname in pageparams) {
 165              if (!pageparams.hasOwnProperty(varname)) {
 166                  continue;
 167              }
 168              params[varname] = pageparams[varname];
 169          }
 170  
 171          // Prepare request parameters
 172          params.sesskey = M.cfg.sesskey;
 173          params.courseId = this.get('courseid');
 174          params['class'] = 'section';
 175          params.field = 'move';
 176          params.id = dragnodeid;
 177          params.value = dropnodeindex;
 178  
 179          // Perform the AJAX request.
 180          var uri = M.cfg.wwwroot + this.get('ajaxurl');
 181          Y.io(uri, {
 182              method: 'POST',
 183              data: params,
 184              on: {
 185                  start: function() {
 186                      lightbox.show();
 187                  },
 188                  success: function(tid, response) {
 189                      // Update section titles, we can't simply swap them as
 190                      // they might have custom title
 191                      try {
 192                          var responsetext = Y.JSON.parse(response.responseText);
 193                          if (responsetext.error) {
 194                              new M.core.ajaxException(responsetext);
 195                          }
 196                          M.course.format.process_sections(Y, sectionlist, responsetext, loopstart, loopend);
 197                      } catch (e) {}
 198  
 199                      // Update all of the section IDs - first unset them, then set them
 200                      // to avoid duplicates in the DOM.
 201                      var index;
 202  
 203                      // Classic bubble sort algorithm is applied to the section
 204                      // nodes between original drag node location and the new one.
 205                      var swapped = false;
 206                      do {
 207                          swapped = false;
 208                          for (index = loopstart; index <= loopend; index++) {
 209                              if (Y.Moodle.core_course.util.section.getId(sectionlist.item(index - 1)) >
 210                                          Y.Moodle.core_course.util.section.getId(sectionlist.item(index))) {
 211                                  Y.log("Swapping " + Y.Moodle.core_course.util.section.getId(sectionlist.item(index - 1)) +
 212                                          " with " + Y.Moodle.core_course.util.section.getId(sectionlist.item(index)));
 213                                  // Swap section id.
 214                                  var sectionid = sectionlist.item(index - 1).get('id');
 215                                  sectionlist.item(index - 1).set('id', sectionlist.item(index).get('id'));
 216                                  sectionlist.item(index).set('id', sectionid);
 217  
 218                                  // See what format needs to swap.
 219                                  M.course.format.swap_sections(Y, index - 1, index);
 220  
 221                                  // Update flag.
 222                                  swapped = true;
 223                              }
 224                          }
 225                          loopend = loopend - 1;
 226                      } while (swapped);
 227  
 228                      window.setTimeout(function() {
 229                          lightbox.hide();
 230                      }, 250);
 231                  },
 232  
 233                  failure: function(tid, response) {
 234                      this.ajax_failure(response);
 235                      lightbox.hide();
 236                  }
 237              },
 238              context:this
 239          });
 240      }

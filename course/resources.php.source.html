  26  require_once('../config.php');
  27  require_once("$CFG->libdir/resourcelib.php");
  28  
  29  $id = required_param('id', PARAM_INT); // course id
  30  
  31  $course = $DB->get_record('course', array('id'=>$id), '*', MUST_EXIST);
  32  $PAGE->set_pagelayout('incourse');
  33  require_course_login($course, true);
  34  
  35  // get list of all resource-like modules
  36  $allmodules = $DB->get_records('modules', array('visible'=>1));
  37  $availableresources = array();
  38  foreach ($allmodules as $key=>$module) {
  39      $modname = $module->name;
  40      $libfile = "$CFG->dirroot/mod/$modname/lib.php";
  41      if (!file_exists($libfile)) {
  42          continue;
  43      }
  44      $archetype = plugin_supports('mod', $modname, FEATURE_MOD_ARCHETYPE, MOD_ARCHETYPE_OTHER);
  45      if ($archetype != MOD_ARCHETYPE_RESOURCE) {
  46          continue;
  47      }
  48  
  49      $availableresources[] = $modname;
  50  }
  51  
  52  // Triger view event.
  53  $event = \core\event\course_resources_list_viewed::create(array('context' => context_course::instance($course->id)));
  54  $event->set_legacy_logdata($availableresources);
  55  $event->add_record_snapshot('course', $course);
  56  $event->trigger();
  57  
  58  $strresources    = get_string('resources');
  59  $strname         = get_string('name');
  60  $strintro        = get_string('moduleintro');
  61  $strlastmodified = get_string('lastmodified');
  62  
  63  $PAGE->set_url('/course/resources.php', array('id' => $course->id));
  64  $PAGE->set_title($course->shortname.': '.$strresources);
  65  $PAGE->set_heading($course->fullname);
  66  $PAGE->navbar->add($strresources);
  67  echo $OUTPUT->header();
  68  
  69  $modinfo = get_fast_modinfo($course);
  70  $usesections = course_format_uses_sections($course->format);
  71  $cms = array();
  72  $resources = array();
  73  foreach ($modinfo->cms as $cm) {
  74      if (!in_array($cm->modname, $availableresources)) {
  75          continue;
  76      }
  77      if (!$cm->uservisible) {
  78          continue;
  79      }
  80      if (!$cm->has_view()) {
  81          // Exclude label and similar
  82          continue;
  83      }
  84      $cms[$cm->id] = $cm;
  85      $resources[$cm->modname][] = $cm->instance;
  86  }
  87  
  88  // preload instances
  89  foreach ($resources as $modname=>$instances) {
  90      $additionalfields = '';
  91      if (plugin_supports('mod', $modname, FEATURE_MOD_INTRO)) {
  92          $additionalfields = ',intro,introformat';
  93      }
  94      $resources[$modname] = $DB->get_records_list($modname, 'id', $instances, 'id', 'id,name'.$additionalfields);
  95  }
  96  
  97  if (!$cms) {
  98      notice(get_string('thereareno', 'moodle', $strresources), "$CFG->wwwroot/course/view.php?id=$course->id");
  99      exit;
 100  }
 101  
 102  $table = new html_table();
 103  $table->attributes['class'] = 'generaltable mod_index';
 104  
 105  if ($usesections) {
 106      $strsectionname = get_string('sectionname', 'format_'.$course->format);
 107      $table->head  = array ($strsectionname, $strname, $strintro);
 108      $table->align = array ('center', 'left', 'left');
 109  } else {
 110      $table->head  = array ($strlastmodified, $strname, $strintro);
 111      $table->align = array ('left', 'left', 'left');
 112  }
 113  
 114  $currentsection = '';
 115  foreach ($cms as $cm) {
 116      if (!isset($resources[$cm->modname][$cm->instance])) {
 117          continue;
 118      }
 119      $resource = $resources[$cm->modname][$cm->instance];
 120      $printsection = '';
 121      if ($usesections) {
 122          if ($cm->sectionnum !== $currentsection) {
 123              if ($cm->sectionnum) {
 124                  $printsection = get_section_name($course, $cm->sectionnum);
 125              }
 126              if ($currentsection !== '') {
 127                  $table->data[] = 'hr';
 128              }
 129              $currentsection = $cm->sectionnum;
 130          }
 131      }
 132  
 133      $extra = empty($cm->extra) ? '' : $cm->extra;
 134      $icon = '<img src="'.$cm->get_icon_url().'" class="activityicon" alt="'.$cm->get_module_type_name().'" /> ';
 135  
 136      if (isset($resource->intro) && isset($resource->introformat)) {
 137          $intro = format_module_intro($cm->modname, $resource, $cm->id);
 138      } else {
 139          $intro = '';
 140      }
 141  
 142      $class = $cm->visible ? '' : 'class="dimmed"'; // hidden modules are dimmed
 143      $table->data[] = array (
 144          $printsection,
 145          "<a $class $extra href=\"".$cm->url."\">".$icon.$cm->get_formatted_name()."</a>",
 146          $intro);
 147  }
 148  
 149  echo html_writer::table($table);
 150  
 151  echo $OUTPUT->footer();

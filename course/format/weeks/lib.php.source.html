  48       * Returns the display name of the given section that the course prefers.

  50       * @param int|stdClass $section Section object from database or just field section.section
  51       * @return string Display name that the course format prefers, e.g. "Topic 2"

  53      public function get_section_name($section) {
  54          $section = $this->get_section($section);
  55          if ((string)$section->name !== '') {
  56              // Return the name the user set.
  57              return format_string($section->name, true, array('context' => context_course::instance($this->courseid)));
  58          } else if ($section->section == 0) {
  59              // Return the general section.
  60              return get_string('section0name', 'format_weeks');
  61          } else {
  62              $dates = $this->get_section_dates($section);
  63  
  64              // We subtract 24 hours for display purposes.
  65              $dates->end = ($dates->end - 86400);
  66  
  67              $dateformat = get_string('strftimedateshort');
  68              $weekday = userdate($dates->start, $dateformat);
  69              $endweekday = userdate($dates->end, $dateformat);
  70              return $weekday.' - '.$endweekday;
  71          }
  72      }

  74      /**
  75       * The URL to use for the specified course (with section)
  76       *
  77       * @param int|stdClass $section Section object from database or just field course_sections.section
  78       *     if omitted the course view page is returned
  79       * @param array $options options for view URL. At the moment core uses:
  80       *     'navigation' (bool) if true and section has no separate page, the function returns null
  81       *     'sr' (int) used by multipage formats to specify to which section to return
  82       * @return null|moodle_url
  83       */
  84      public function get_view_url($section, $options = array()) {
  85          global $CFG;
  86          $course = $this->get_course();
  87          $url = new moodle_url('/course/view.php', array('id' => $course->id));
  88  
  89          $sr = null;
  90          if (array_key_exists('sr', $options)) {
  91              $sr = $options['sr'];
  92          }
  93          if (is_object($section)) {
  94              $sectionno = $section->section;
  95          } else {
  96              $sectionno = $section;
  97          }
  98          if ($sectionno !== null) {
  99              if ($sr !== null) {
 100                  if ($sr) {
 101                      $usercoursedisplay = COURSE_DISPLAY_MULTIPAGE;
 102                      $sectionno = $sr;
 103                  } else {
 104                      $usercoursedisplay = COURSE_DISPLAY_SINGLEPAGE;
 105                  }
 106              } else {
 107                  $usercoursedisplay = $course->coursedisplay;
 108              }
 109              if ($sectionno != 0 && $usercoursedisplay == COURSE_DISPLAY_MULTIPAGE) {
 110                  $url->param('section', $sectionno);
 111              } else {
 112                  if (empty($CFG->linkcoursesections) && !empty($options['navigation'])) {
 113                      return null;
 114                  }
 115                  $url->set_anchor('section-'.$sectionno);
 116              }
 117          }
 118          return $url;
 119      }

 136       * Loads all of the course sections into the navigation
 137       *
 138       * @param global_navigation $navigation
 139       * @param navigation_node $node The course node within the navigation
 140       */
 141      public function extend_course_navigation($navigation, navigation_node $node) {
 142          global $PAGE;
 143          // if section is specified in course/view.php, make sure it is expanded in navigation
 144          if ($navigation->includesectionnum === false) {
 145              $selectedsection = optional_param('section', null, PARAM_INT);
 146              if ($selectedsection !== null && (!defined('AJAX_SCRIPT') || AJAX_SCRIPT == '0') &&
 147                      $PAGE->url->compare(new moodle_url('/course/view.php'), URL_MATCH_BASE)) {
 148                  $navigation->includesectionnum = $selectedsection;
 149              }
 150          }
 151          parent::extend_course_navigation($navigation, $node);
 152  
 153          // We want to remove the general section if it is empty.
 154          $modinfo = get_fast_modinfo($this->get_course());
 155          $sections = $modinfo->get_sections();
 156          if (!isset($sections[0])) {
 157              // The general section is empty to find the navigation node for it we need to get its ID.
 158              $section = $modinfo->get_section_info(0);
 159              $generalsection = $node->get($section->id, navigation_node::TYPE_SECTION);
 160              if ($generalsection) {
 161                  // We found the node - now remove it.
 162                  $generalsection->remove();
 163              }
 164          }
 165      }

 170       * Used in course/rest.php

 174      function ajax_section_move() {
 175          global $PAGE;
 176          $titles = array();
 177          $current = -1;
 178          $course = $this->get_course();
 179          $modinfo = get_fast_modinfo($course);
 180          $renderer = $this->get_renderer($PAGE);
 181          if ($renderer && ($sections = $modinfo->get_section_info_all())) {
 182              foreach ($sections as $number => $section) {
 183                  $titles[$number] = $renderer->section_title($section, $course);
 184                  if ($this->is_section_current($section)) {
 185                      $current = $number;
 186                  }
 187              }
 188          }
 189          return array('sectiontitles' => $titles, 'current' => $current, 'action' => 'move');
 190      }

 217          static $courseformatoptions = false;
 218          if ($courseformatoptions === false) {
 219              $courseconfig = get_config('moodlecourse');
 220              $courseformatoptions = array(

 237              $sectionmenu = array();

 282       * Adds format options elements to the course/section edit form.
 283       *
 284       * This function is called from {@link course_edit_form::definition_after_data()}.
 285       *
 286       * @param MoodleQuickForm $mform form the elements are added to.
 287       * @param bool $forsection 'true' if this is a section edit form, 'false' if this is course edit form.
 288       * @return array array of references to the added form elements.
 289       */
 290      public function create_edit_form_elements(&$mform, $forsection = false) {
 291          $elements = parent::create_edit_form_elements($mform, $forsection);
 292  
 293          // Increase the number of sections combo box values if the user has increased the number of sections
 294          // using the icon on the course page beyond course 'maxsections' or course 'maxsections' has been
 295          // reduced below the number of sections already set for the course on the site administration course
 296          // defaults page.  This is so that the number of sections is not reduced leaving unintended orphaned
 297          // activities / resources.
 298          if (!$forsection) {
 299              $maxsections = get_config('moodlecourse', 'maxsections');
 300              $numsections = $mform->getElementValue('numsections');
 301              $numsections = $numsections[0];
 302              if ($numsections > $maxsections) {
 303                  $element = $mform->getElement('numsections');
 304                  for ($i = $maxsections+1; $i <= $numsections; $i++) {
 305                      $element->addOption("$i", $i);
 306                  }
 307              }
 308          }
 309          return $elements;
 310      }

 312      /**
 313       * Updates format options for a course
 314       *
 315       * In case if course format was changed to 'weeks', we try to copy options
 316       * 'coursedisplay', 'numsections' and 'hiddensections' from the previous format.
 317       * If previous course format did not have 'numsections' option, we populate it with the
 318       * current number of sections
 319       *
 320       * @param stdClass|array $data return value from {@link moodleform::get_data()} or array with data
 321       * @param stdClass $oldcourse if this function is called from {@link update_course()}
 322       *     this object contains information about the course before update
 323       * @return bool whether there were any changes to the options values
 324       */
 325      public function update_course_format_options($data, $oldcourse = null) {
 326          global $DB;
 327          $data = (array)$data;
 328          if ($oldcourse !== null) {
 329              $oldcourse = (array)$oldcourse;
 330              $options = $this->course_format_options();
 331              foreach ($options as $key => $unused) {
 332                  if (!array_key_exists($key, $data)) {
 333                      if (array_key_exists($key, $oldcourse)) {
 334                          $data[$key] = $oldcourse[$key];
 335                      } else if ($key === 'numsections') {
 336                          // If previous format does not have the field 'numsections'
 337                          // and $data['numsections'] is not set,
 338                          // we fill it with the maximum section number from the DB
 339                          $maxsection = $DB->get_field_sql('SELECT max(section) from {course_sections}
 340                              WHERE course = ?', array($this->courseid));
 341                          if ($maxsection) {
 342                              // If there are no sections, or just default 0-section, 'numsections' will be set to default
 343                              $data['numsections'] = $maxsection;
 344                          }
 345                      }
 346                  }
 347              }
 348          }
 349          $changed = $this->update_format_options($data);
 350          if ($changed && array_key_exists('numsections', $data)) {
 351              // If the numsections was decreased, try to completely delete the orphaned sections (unless they are not empty).
 352              $numsections = (int)$data['numsections'];
 353              $maxsection = $DB->get_field_sql('SELECT max(section) from {course_sections}
 354                          WHERE course = ?', array($this->courseid));
 355              for ($sectionnum = $maxsection; $sectionnum > $numsections; $sectionnum--) {
 356                  if (!$this->delete_section($sectionnum, false)) {
 357                      break;
 358                  }
 359              }
 360          }
 361          return $changed;
 362      }

 364      /**
 365       * Return the start and end date of the passed section
 366       *
 367       * @param int|stdClass|section_info $section section to get the dates for
 368       * @return stdClass property start for startdate, property end for enddate
 369       */
 370      public function get_section_dates($section) {
 371          $course = $this->get_course();
 372          if (is_object($section)) {
 373              $sectionnum = $section->section;
 374          } else {
 375              $sectionnum = $section;
 376          }
 377          $oneweekseconds = 604800;
 378          // Hack alert. We add 2 hours to avoid possible DST problems. (e.g. we go into daylight
 379          // savings and the date changes.
 380          $startdate = $course->startdate + 7200;
 381  
 382          $dates = new stdClass();
 383          $dates->start = $startdate + ($oneweekseconds * ($sectionnum - 1));
 384          $dates->end = $dates->start + $oneweekseconds;
 385  
 386          return $dates;
 387      }

 389      /**
 390       * Returns true if the specified week is current
 391       *
 392       * @param int|stdClass|section_info $section
 393       * @return bool
 394       */
 395      public function is_section_current($section) {
 396          if (is_object($section)) {
 397              $sectionnum = $section->section;
 398          } else {
 399              $sectionnum = $section;
 400          }
 401          if ($sectionnum < 1) {
 402              return false;
 403          }
 404          $timenow = time();
 405          $dates = $this->get_section_dates($section);
 406          return (($timenow >= $dates->start) && ($timenow < $dates->end));
 407      }
